1 /**
  2  * Backend module to provide logging that is aware of the current location of scene traversal.
  3  *
  4  * There are three "channels" of log message: error, warning, info and debug.
  5  *
  6  * Provides an interface on the backend context through which other backends may log messages.
  7  *
  8  * Provides an interface to scene nodes to allow them to log messages, as well as set and get the function
  9  * that actually processes messages on each channel. Those getters and setters are used by the SceneJS.logging node,
 10  * which may be distributed throughout a scene graph to cause messages to be processed in particular ways for different
 11  * parts of the graph.
 12  *
 13  * Messages are queued. Initially, each channel has no function set for it and will queue messages until a function is
 14  * set, at which point the queue flushes.  If the function is unset, subsequent messages will queue, then flush when a
 15  * function is set again. This allows messages to be logged before any SceneJS.logging node is visited.
 16  *
 17  * This backend is always the last to handle a RESET
 18  *
 19  *  @private
 20  *
 21  */
 22 var SceneJS_loggingModule = new (function() {
 23 
 24     var activeSceneId;
 25     var funcs = null;
 26     var queues = {};
 27     var indent = 0;
 28     var indentStr = "";
 29 
 30     /**
 31      * @private
 32      */
 33     function log(channel, message) {
 34         message = activeSceneId
 35                 ? indentStr + activeSceneId + ": " + message
 36                 : indentStr + message;
 37         var func = funcs ? funcs[channel] : null;
 38         if (func) {
 39             func(message);
 40         } else {
 41             var queue = queues[channel];
 42             if (!queue) {
 43                 queue = queues[channel] = [];
 44             }
 45             queue.push(message);
 46         }
 47     }
 48 
 49     /**
 50      * @private
 51      */
 52     function flush(channel) {
 53         var queue = queues[channel];
 54         if (queue) {
 55             var func = funcs ? funcs[channel] : null;
 56             if (func) {
 57                 for (var i = 0; i < queue.length; i++) {
 58                     func(queue[i]);
 59                 }
 60                 queues[channel] = [];
 61             }
 62         }
 63     }
 64 
 65     SceneJS_eventModule.onEvent(
 66             SceneJS_eventModule.SCENE_ACTIVATED, // Set default logging for scene root
 67             function(params) {
 68                 activeSceneId = params.sceneId;
 69 
 70                 var element = document.getElementById(SceneJS_webgl_DEFAULT_LOGGING_ID);
 71                 if (element) {
 72                     funcs = {
 73                         warn : function log(msg) {
 74                             element.innerHTML += "<p style=\"color:orange;\">" + msg + "</p>";
 75                         },
 76                         error : function log(msg) {
 77                             element.innerHTML += "<p style=\"color:darkred;\">" + msg + "</p>";
 78                         },
 79                         debug : function log(msg) {
 80                             element.innerHTML += "<p style=\"color:darkblue;\">" + msg + "</p>";
 81                         },
 82                         info : function log(msg) {
 83                             element.innerHTML += "<p style=\"color:darkgreen;\">" + msg + "</p>";
 84                         }
 85                     };
 86                 } else {
 87                     funcs = null;
 88                 }
 89             });
 90 
 91     SceneJS_eventModule.onEvent(
 92             SceneJS_eventModule.SCENE_DEACTIVATED, // Set default logging for scene root
 93             function() {
 94                 activeSceneId = null;
 95             });
 96 
 97     SceneJS_eventModule.onEvent(
 98             SceneJS_eventModule.RESET,
 99             function() {
100                 queues = {};
101                 funcs = null;
102             },
103             100000);  // Really low priority - must be reset last
104 
105     // @private
106     this.setIndent = function(_indent) {
107         indent = _indent;
108         var indentArray = [];
109         for (var i = 0; i < indent; i++) {
110             indentArray.push("----");
111         }
112         indentStr = indentArray.join("");
113     };
114 
115     // @private
116     this.error = function(msg) {
117         log("error", msg);
118     };
119 
120     // @private
121     this.warn = function(msg) {
122         log("warn", msg);
123     };
124 
125     // @private
126     this.info = function(msg) {
127         log("info", msg);
128     };
129 
130     // @private
131     this.debug = function(msg) {
132         log("debug", msg);
133     };
134 
135     // @private
136     this.getFuncs = function() {
137         return funcs;
138     };
139 
140     // @private
141     this.setFuncs = function(l) {
142         if (l) {
143             funcs = l;
144             for (var channel in queues) {
145                 flush(channel);
146             }
147         }
148     };
149 })();
