1 /**
  2  * Backend module for asynchronous process management.
  3  *
  4  * This module provides creation, destruction and query of SceneJS processes.
  5  *
  6  * This module maintains a separate group of processes for each active scene. When a scene is defined, it
  7  * will create a group for it, then whenever it is deactivated it will automatically reap all processes
  8  * in its group that have timed out.
  9  *
 10  *  @private
 11  */
 12 var SceneJS_processModule = new (function() {
 13 
 14     var time = (new Date()).getTime();          // System time
 15     var groups = {};                            // A process group for each existing scene
 16     var activeSceneId;                          // ID of currently-active scene
 17 
 18     SceneJS_eventModule.onEvent(
 19             SceneJS_eventModule.TIME_UPDATED,
 20             function(t) {
 21                 time = t;
 22             });
 23 
 24     SceneJS_eventModule.onEvent(// Scene defined, create new process group for it
 25             SceneJS_eventModule.SCENE_CREATED,
 26             function(params) {
 27                 var group = {   // IDEA like this
 28                     sceneId : params.sceneId,
 29                     processes: {} ,
 30                     numProcesses : 0
 31                 };
 32                 groups[params.sceneId] = group;
 33             });
 34 
 35     SceneJS_eventModule.onEvent(// Scene traversal begins
 36             SceneJS_eventModule.SCENE_ACTIVATED,
 37             function(params) {
 38                 activeSceneId = params.sceneId;
 39             });
 40 
 41     SceneJS_eventModule.onEvent(// Scene traversed - reap its dead and timed-out processes
 42             SceneJS_eventModule.SCENE_DEACTIVATED,
 43             function() {
 44                 var group = groups[activeSceneId];
 45                 var processes = group.processes;
 46                 for (var pid in processes) {
 47                     var process = processes[pid];
 48                     if (process) {
 49                         if (process.destroyed) {
 50                             processes[pid] = undefined;
 51                             group.numProcesses--;
 52                         } else {
 53                             var elapsed = time - process.timeStarted;
 54                             if ((process.timeoutSecs > -1) && (elapsed > (process.timeoutSecs * 1000))) {
 55 
 56                                 SceneJS_loggingModule.warn("Process timed out after " +
 57                                                            process.timeoutSecs +
 58                                                            " seconds: " + process.description);
 59 
 60                                 /* Process timed out - notify listeners
 61                                  */
 62                                 SceneJS_eventModule.fireEvent(SceneJS_eventModule.PROCESS_TIMED_OUT, {
 63                                     sceneId: activeSceneId,
 64                                     process: {
 65                                         id: process.id,
 66                                         timeStarted : process.timeStarted,
 67                                         description: process.description,
 68                                         timeoutSecs: process.timeoutSecs
 69                                     }
 70                                 });
 71 
 72                                 process.destroyed = true;
 73                                 processes[pid] = undefined;
 74                                 group.numProcesses--;
 75                                 if (process.onTimeout) {
 76                                     process.onTimeout();
 77                                 }
 78                             } else {
 79                                 process.timeRunning = elapsed;
 80                             }
 81                         }
 82                     }
 83                 }
 84                 activeSceneId = null;
 85             });
 86 
 87     SceneJS_eventModule.onEvent(// Scene destroyed - destroy its process group
 88             SceneJS_eventModule.SCENE_DESTROYED,
 89             function(params) {
 90                 groups[params.sceneId] = undefined;
 91             });
 92 
 93     SceneJS_eventModule.onEvent(// Framework reset - destroy all process groups
 94             SceneJS_eventModule.RESET,
 95             function(params) {
 96                 groups = {};
 97                 activeSceneId = null;
 98             });
 99 
100 
101     /**
102      *
103      * Creates a new asynchronous process for the currently active scene and returns a handle to it.
104      * The handle is actually an object containing live information on the process, which must
105      * not be modified.
106      *
107      * Example:
108      *
109      * createProcess({
110      *      description: "loading texture image",
111      *      timeoutSecs: 30,                         // 30 Seconds
112      *      onTimeout(function() {
113      *              alert("arrrg!!");
114      *          });
115      *
116      * @private
117      */
118     this.createProcess = function(cfg) {
119         if (!activeSceneId) {
120             SceneJS_errorModule.fatalError(new SceneJS.NoSceneActiveException("No scene active - can't create process"));
121         }
122         var group = groups[activeSceneId];
123         var i = 0;
124         while (true) {
125             var pid = activeSceneId + i++;
126             if (!group.processes[pid]) {
127 
128                 /* Register process
129                  */
130                 var process = {
131                     sceneId: activeSceneId,
132                     id: pid,
133                     timeStarted : time,
134                     timeRunning: 0,
135                     description : cfg.description || "",
136                     timeoutSecs : cfg.timeoutSecs || 30, // Thirty second default timout
137                     onTimeout : cfg.onTimeout
138                 };
139                 group.processes[pid] = process;
140                 group.numProcesses++;
141 
142                 /* Notify listeners
143                  */
144                 SceneJS_eventModule.fireEvent(SceneJS_eventModule.PROCESS_CREATED, {
145                     sceneId: activeSceneId,
146                     process: {
147                         id: process.id,
148                         timeStarted : process.timeStarted,
149                         description: process.description,
150                         timeoutSecs: process.timeoutSecs
151                     }
152                 });
153 
154                 return process;
155             }
156         }
157     };
158 
159     /**
160      * Destroys the given process, which is the object returned by the previous call to createProcess.
161      * Does not care if no scene is active, or if the process no longer exists or is dead.
162      *
163      * @private
164      */
165     this.killProcess = function(process) {
166         if (process) {
167             process.destroyed = true;
168 
169             /* Notify listeners
170              */
171             SceneJS_eventModule.fireEvent(SceneJS_eventModule.PROCESS_KILLED, {
172                 sceneId: activeSceneId,
173                 process: {
174                     id: process.id,
175                     timeStarted : process.timeStarted,
176                     description: process.description,
177                     timeoutSecs: process.timeoutSecs
178                 }
179             });
180         }
181     };
182 
183     /**
184      * Returns the number of living processes for either the scene of the given ID, or if
185      * no ID supplied, the active scene. If no scene is active, returns zero.
186      *
187      * @private
188      */
189     this.getNumProcesses = function(sceneId) {
190         var group = groups[sceneId];
191         if (!group) {
192             return 0;
193         }
194         return sceneId ? group.numProcesses : (activeSceneId ? groups[activeSceneId].numProcesses : 0);
195     };
196 
197     /**
198      * Returns all living processes for the given scene, which may be null, in which case this
199      * method will return the living processes for the currently active scene by default. An empty map
200      * will be returned if there is no scene active.
201      *
202      * Process info looks like this:
203      *
204      *      {   id: "xx",
205      *          timeStarted :   65765765765765,             // System time in milliseconds
206      *          timeRunning:    876870,                     // Elapsed time in milliseconds
207      *          description :   "loading texture image",
208      *          timeoutSecs :       30,                      // Timeout in milliseconds
209      *          onTimeout :     <function>                  // Function that will fire on timeoutSecs
210      *
211      * @private
212      */
213     this.getProcesses = function(sceneId) {
214         var group = groups[sceneId];
215         if (!group) {
216             return {};
217         }
218         return sceneId ? group.processes : (activeSceneId ? groups[activeSceneId].processes : {});
219     };
220 })();
