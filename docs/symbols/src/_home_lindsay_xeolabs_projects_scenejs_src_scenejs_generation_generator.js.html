1 /**
  2  *  @class A scene node that performs procedural scene generation by causing its child nodes to be rendered multiple times
  3  * in a loop within a scene traversal, while varying the data available to them in each loop.
  4  *
  5  *  <p>Each time a SceneJS.Generator loops over its children it creates a child data scope for them from the result of its
  6  *  configuration callback function, then repeats the process as long as the function returns something.</p>
  7  *
  8  *  <p>This node type must be configured dynamically therefore, in the SceneJS style, with a configuration function.</p>
  9  *
 10  *  <p>This node type is useful for procedurally generating scene subtrees. Its most common application would be
 11  *  to dynamically instance elements of primitive geometry to build complex objects.</p>
 12  *
 13  *  <p>Note that generator nodes can have a negative impact on performance, where they will often prevent subnodes from
 14  *  employing memoization strategies that fast scene graphs often depend upon. Use them carefully when high performance
 15  *  is desired in large scenes. The impact will depend on the type of subnode that receives the generated data.
 16  *  For example, inability to memoize will cascade downwards through  modelling transform node hierarchies since they
 17  *  will have to re-multiply matrices by dynamic parent modelling transforms etc.</p>
 18  *
 19  * <p><b>Example Usage</b></p><p>Below is a SceneJS.Generator that loops over its subgraph to create a ring of cubes, 45 degrees apart.</b></p><pre><code>
 20  * var g = new SceneJS.Generator(
 21  *        (function() {                        // Higher order function tracks the angle in closure
 22  *            var angle = 0;
 23  *            return function() {              // Generator function
 24  *                angle += 45.0;
 25  *                if (angle <= 360.0) {
 26  *                    return { angle: angle }; // Angle still less than 360, return config object
 27  *                } else {  // Reset the generator
 28  *                    angle = 0;               // Angle at max, reset and return nothing,
 29  *                }                            // causing loop to finish for this frame
 30  *            };
 31  *        })(),
 32  *
 33  *        new SceneJS.Rotate(function(data) {
 34  *            return { angle : data.get("angle"), y: 1.0 };
 35  *        },
 36  *                new SceneJS.Translate(function(data) {
 37  *                    return { x: 10.0 };
 38  *                },
 39  *
 40  *                new SceneJS.objects.cube()
 41  *            )
 42  *         )
 43  *   );
 44  * </pre></code>
 45  * @extends SceneJS.Node
 46  * @since Version 0.7.4
 47  * @constructor
 48  * Create a new SceneJS.Generator
 49  * @param {Object} [cfg] Static configuration object
 50  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 51  * @param {...SceneJS.Node} [childNodes] Child nodes
 52  */
 53 SceneJS.Generator = function(cfg) {
 54     SceneJS.Node.apply(this, arguments);
 55     this._nodeType = "generator";
 56 };
 57 
 58 SceneJS._inherit(SceneJS.Generator, SceneJS.Node);
 59 
 60 // @private
 61 SceneJS.Generator.prototype._render = function(traversalContext, data) {
 62     if (this._fixedParams) {
 63         throw SceneJS._errorModule.fatalError(
 64                 new SceneJS.errors.InvalidNodeConfigException
 65                         ("SceneJS.Generator may only be configured with a function"));
 66     }
 67     var params = this._getParams(data);
 68     while (params) {
 69         this._renderNodes(traversalContext, new SceneJS.Data(data, false, params));
 70         params = this._getParams(data);
 71     }
 72 };
 73 
 74 /** Factory function that returns a new {@link SceneJS.Generator} instance
 75  * @param {Object} [cfg] Static configuration object
 76  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 77  * @param {...SceneJS.Node} [childNodes] Child nodes
 78  * @returns {SceneJS.Generator}
 79  *  @since Version 0.7.2
 80  */
 81 SceneJS.generator = function() {
 82     var n = new SceneJS.Generator();
 83     SceneJS.Generator.prototype.constructor.apply(n, arguments);
 84     return n;
 85 };
