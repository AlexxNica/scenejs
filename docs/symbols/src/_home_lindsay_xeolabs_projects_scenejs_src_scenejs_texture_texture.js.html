1 /**
  2  * @class A scene node that defines one or more layers of texture to apply to all geometries within its subgraph that have UV coordinates.
  3  * @extends SceneJS.node
  4  * <p>Texture layers are applied to specified material reflection cooficients, and may be transformed.</p>
  5  * <p><b>Live Examples</b></p>
  6  * <li><a target = "other" href="http://bit.ly/scenejs-animated-texture">Animated Texture</a></li>
  7  * </ul>
  8  * <p>A cube wrapped with a material which specifies its base (diffuse) color coefficient, and a texture with
  9  * one layer which applies a texture image to that particular coefficient. The texture is also translated, scaled and
 10  * rotated, in that order. All the texture properties are specified here to show what they are. </p>
 11  *  <pre><code>
 12  * var subGraph =
 13  *       new SceneJS.Material({
 14  *           baseColor: { r: 1.0, g: 1.0, b: 1.0 }
 15  *       },
 16  *               new SceneJS.Texture({
 17  *                   layers: [
 18  *                       {
 19  *                           // Only the image URI is mandatory:
 20  *
 21  *                           uri:"http://scenejs.org/library/textures/misc/general-zod.jpg",
 22  *
 23  *                          // Optional params:
 24  *
 25  *                           minFilter: "linear",                   // Options are ”nearest”, “linear” (default), “nearestMipMapNearest”,
 26  *                                                                  //        ”nearestMipMapLinear” or “linearMipMapLinear”
 27  *                           magFilter: "linear",                   // Options are “nearest” or “linear” (default)
 28  *                           wrapS: "repeat",                       // Options are “clampToEdge” (default) or “repeat”
 29  *                           wrapT: "repeat",                       // Options are "clampToEdge” (default) or “repeat”
 30  *                           isDepth: false,                        // Options are false (default) or true
 31  *                           depthMode:"luminance"                  // (default)
 32  *                           depthCompareMode: "compareRToTexture", // (default)
 33  *                           depthCompareFunc: "lequal",            // (default)
 34  *                           flipY: false,                          // Options are true (default) or false
 35  *                           width: 1,
 36  *                           height: 1,
 37  *                           internalFormat:"lequal",               // (default)
 38  *                           sourceFormat:"alpha",                  // (default)
 39  *                           sourceType: "unsignedByte",            // (default)
 40  *                           applyTo:"baseColor",                   // Options so far are “baseColor” (default) or “diffuseColor”
 41  *
 42  *                           // Optional transforms - these can also be functions, as shown in the next example
 43  *
 44  *                           rotate: {      // Currently textures are 2-D, so only rotation about Z makes sense
 45  *                               z: 45.0
 46  *                           },
 47  *
 48  *                           translate : {
 49  *                               x: 10,
 50  *                               y: 0,
 51  *                               z: 0
 52  *                           },
 53  *
 54  *                           scale : {
 55  *                               x: 1,
 56  *                               y: 2,
 57  *                               z: 1
 58  *                           }
 59  *                       }
 60  *                   ]
 61  *               },
 62  *
 63  *               new SceneJS.objects.Cube()
 64  *           )
 65  *     );
 66  *  </code></pre>
 67  *
 68  * <p><b>Example 2</b></p>
 69  * <p>You can animate texture transformations - this example shows how the rotate, scale and translate properties
 70  * can be functions to take their values from the data scope, in this case created by a higher WithData node:</p>
 71  *  <pre><code>
 72  * var subGraph =
 73  *       new SceneJS.WithData({
 74  *           angle: 45.0   // Vary this value to rotate the texture
 75  *       },
 76  *               new SceneJS.Texture({
 77  *                   layers: [
 78  *                       {
 79  *                           uri:"http://scenejs.org/library/textures/misc/general-zod.jpg",
 80  *
 81  *                           rotate: function(data) {
 82  *                               return { z: data.get("angle") }
 83  *                           }
 84  *                       }
 85  *                   ]
 86  *               },
 87  *               new SceneJS.objects.Cube()
 88  *         )
 89  *   );
 90  *  </code></pre>
 91  * @constructor
 92  * Create a new SceneJS.texture
 93  * @param {Object} The config object or function, followed by zero or more child nodes
 94  */
 95 SceneJS.Texture = function() {
 96     SceneJS.Node.apply(this, arguments);
 97     this._nodeType = "texture";
 98     this._layers = null;
 99 };
100 
101 SceneJS._inherit(SceneJS.Texture, SceneJS.Node);
102 
103 // @private
104 SceneJS.Texture.prototype._getMatrix = function(translate, rotate, scale) {
105     var matrix = null;
106     var t;
107     if (translate) {
108         matrix = SceneJS._math_translationMat4v([ translate.x || 0, translate.y || 0, translate.z || 0]);
109     }
110     if (scale) {
111         t = SceneJS._math_scalingMat4v([ scale.x || 1, scale.y || 1, scale.z || 1]);
112         matrix = matrix ? SceneJS._math_mulMat4(matrix, t) : t;
113     }
114     if (rotate) {
115         if (rotate.x) {
116             t = SceneJS._math_rotationMat4v(rotate.x * 0.0174532925, [1,0,0]);
117             matrix = matrix ? SceneJS._math_mulMat4(matrix, t) : t;
118         }
119         if (rotate.y) {
120             t = SceneJS._math_rotationMat4v(rotate.y * 0.0174532925, [0,1,0]);
121             matrix = matrix ? SceneJS._math_mulMat4(matrix, t) : t;
122         }
123         if (rotate.z) {
124             t = SceneJS._math_rotationMat4v(rotate.z * 0.0174532925, [0,0,1]);
125             matrix = matrix ? SceneJS._math_mulMat4(matrix, t) : t;
126         }
127     }
128     return matrix;
129 };

131 /** Ready to create texture layer
132  *  @private

134 SceneJS.Texture._STATE_INITIAL = 0;

137  *  @private

142  *  @private

144 SceneJS.Texture._STATE_LOADED = 2;

147  * @private
148  */

152 // @private
153 SceneJS.Texture.prototype._init = function(params) {
154     this._layers = [];
155     if (!params.layers) {
156         throw new SceneJS.NodeConfigExpectedException(
157                 "SceneJS.Texture.layers is undefined");
158     }
159     for (var i = 0; i < params.layers.length; i++) {
160         var layerParam = params.layers[i];
161         if (!layerParam.uri) {
162             throw new SceneJS.NodeConfigExpectedException(
163                     "SceneJS.Texture.layers[" + i + "].uri is undefined");
164         }
165         if (layerParam.applyFrom) {
166             if (layerParam.applyFrom != "uv" &&
167                 layerParam.applyFrom != "uv2" &&
168                 layerParam.applyFrom != "normal" &&
169                 layerParam.applyFrom != "geometry") {
170                 throw SceneJS._errorModule.fatalError(
171                         new SceneJS.InvalidNodeConfigException(
172                                 "SceneJS.Texture.layers[" + i + "].applyFrom value is unsupported - " +
173                                 "should be either 'uv', 'uv2', 'normal' or 'geometry'"));
174             }
175         }
176         if (layerParam.applyTo) {
177             if (layerParam.applyTo != "baseColor" && // Colour map
178                 layerParam.applyTo != "diffuseColor") {
179                 throw SceneJS._errorModule.fatalError(
180                         new SceneJS.InvalidNodeConfigException(
181                                 "SceneJS.Texture.layers[" + i + "].applyTo value is unsupported - " +
182                                 "should be either 'baseColor', 'diffuseColor'"));
183             }
184         }
185         this._layers.push({
186             state : SceneJS.Texture._STATE_INITIAL,
187             process: null,                      // Image load process handle
188             image : null,                       // Initialised when state == IMAGE_LOADED
189             creationParams: layerParam,         // Create texture using this
190             texture: null,                      // Initialised when state == TEXTURE_LOADED
191             createMatrix : new (function() {
192                 var translate = layerParam.translate;
193                 var rotate = layerParam.rotate;
194                 var scale = layerParam.scale;
195                 var dynamic = ((translate instanceof Function) ||
196                                (rotate instanceof Function) ||
197                                (scale instanceof Function));
198                 var defined = dynamic || translate || rotate || scale;
199                 return function(data) {
200                     var matrix = null;
201                     if (defined && (dynamic || !matrix)) {
202                         matrix = SceneJS.Texture.prototype._getMatrix(
203                                 (translate instanceof Function) ? translate(data) : translate,
204                                 (rotate instanceof Function) ? rotate(data) : rotate,
205                                 (scale instanceof Function) ? scale(data) : scale);
206                     }
207                     return matrix;
208                 };
209             })(),
210             applyFrom: layerParam.applyFrom || "uv",
211             applyTo: layerParam.applyTo || "baseColor",
212             blendMode: layerParam.blendMode || "multiply"
213         });
214     }
215 };

217 SceneJS.Texture.prototype._render = function(traversalContext, data) {
218     if (!this._layers) { // One-shot dynamic config
219         this._init(this._getParams(data));
220     }
221 
222     /*-----------------------------------------------------
223      * On each render, update state of each texture layer
224      * and count how many are ready to apply
225      *-----------------------------------------------------*/
226 
227     var countLayersReady = 0;
228     for (var i = 0; i < this._layers.length; i++) {
229         var layer = this._layers[i];
230 
231         if (layer.state == SceneJS.Texture._STATE_LOADED) {
232             if (!SceneJS._textureModule.textureExists(layer.texture)) {  // Texture evicted from cache
233                 layer.state = SceneJS.Texture._STATE_INITIAL;
234 
235             }
236         }
237 
238 
239         switch (layer.state) {
240 
241             case SceneJS.Texture._STATE_LOADED: // Layer ready to apply
242                 countLayersReady++;
243                 break;
244 
245             case SceneJS.Texture._STATE_INITIAL: // Layer load to start
246                 layer.state = SceneJS.Texture._STATE_LOADING;
247                 (function(l) { // Closure allows this layer to receive results
248                     SceneJS._textureModule.createTexture(
249                             l.creationParams.uri,
250                             l.creationParams,
251 
252                             function(texture) { // Success
253                                 l.texture = texture;
254                                 l.state = SceneJS.Texture._STATE_LOADED;                                
255                             },
256 
257                             function() { // General error, probably 404
258                                 l.state = SceneJS.Texture._STATE_ERROR;
259                                 var message = "SceneJS.texture image load failed: " + l.creationParams.uri;
260                                 SceneJS._loggingModule.warn(message);
261                             },
262 
263                             function() { // Load aborted - user probably refreshed/stopped page
264                                 SceneJS._loggingModule.warn("SceneJS.texture image load aborted: " + l.creationParams.uri);
265                                 l.state = SceneJS.Texture._STATE_ERROR;
266                             });
267                 }).call(this, layer);
268                 break;
269 
270             case SceneJS.Texture._STATE_LOADING: // Layer still loading
271                 break;
272 
273             case SceneJS.Texture._STATE_ERROR: // Layer disabled
274                 break;
275         }
276     }
277 
278     /*------------------------------------------------
279      * Render this node
280      *-----------------------------------------------*/
281 
282     if (SceneJS._traversalMode == SceneJS.TRAVERSAL_MODE_PICKING) {
283         this._renderNodes(traversalContext, data);
284     } else {
285 
286         /* Fastest strategy is to allow the complete set of layers to load
287          * before applying any of them. There would be a huge performance penalty
288          * if we were to apply the incomplete set as layers are still loading -
289          * SceneJS._shaderModule would then have to generate a new shader for each new
290          * layer loaded, which would become redundant as soon as the next layer is loaded.
291          */
292 
293         if (countLayersReady == this._layers.length) {
294             var countPushed = 0;
295             for (var i = 0; i < this._layers.length; i++) {
296                 var layer = this._layers[i];
297 
298                 if (layer.state = SceneJS.Texture._STATE_LOADED) {
299                     SceneJS._textureModule.pushLayer(layer.texture, {
300                         applyFrom : layer.applyFrom,
301                         applyTo : layer.applyTo,
302                         blendMode : layer.blendMode,
303                         matrix: layer.createMatrix(data)
304                     });
305                     countPushed++;
306                 }
307             }
308             this._renderNodes(traversalContext, data);
309             SceneJS._textureModule.popLayers(countPushed);
310 
311         } else {
312             this._renderNodes(traversalContext, data);
313         }
314     }
315 };

318 /** Factory function that returns a new {@link SceneJS.Texture} instance
319  */
320 SceneJS.texture = function() {
321     var n = new SceneJS.Texture();
322     SceneJS.Texture.prototype.constructor.apply(n, arguments);
323     return n;
