1 /**
  2  * @class A scene node that defines one or more layers of texture to apply to all geometries within its subgraph that have UV coordinates.
  3  * @extends SceneJS.node
  4  * <p>Texture layers are applied to specified material reflection cooficients, and may be transformed.</p>
  5  * <p><b>Example 1</b></p>
  6  * <p>A cube wrapped with a material which specifies its base (diffuse) color coefficient, and a texture with
  7  * one layer which applies a texture image to that particular coefficient. The texture is also translated, scaled and
  8  * rotated, in that order. All the texture properties are specified here to show what they are. </p>
  9  *  <pre><code>
 10  * var subGraph =
 11  *       new SceneJS.Material({
 12  *           baseColor: { r: 1.0, g: 1.0, b: 1.0 }
 13  *       },
 14  *               new SceneJS.Texture({
 15  *                   layers: [
 16  *                       {
 17  *                           // Only the image URI is mandatory:
 18  *
 19  *                           uri:"http://scenejs.org/library/textures/misc/general-zod.jpg",
 20  *
 21  *                          // Optional params:
 22  *
 23  *                           minFilter: "linear",                   // Options are ”nearest”, “linear” (default), “nearestMipMapNearest”,
 24  *                                                                  //        ”nearestMipMapLinear” or “linearMipMapLinear”
 25  *                           magFilter: "linear",                   // Options are “nearest” or “linear” (default)
 26  *                           wrapS: "repeat",                       // Options are “clampToEdge” (default) or “repeat”
 27  *                           wrapT: "repeat",                       // Options are "clampToEdge” (default) or “repeat”
 28  *                           isDepth: false,                        // Options are false (default) or true
 29  *                           depthMode:"luminance"                  // (default)
 30  *                           depthCompareMode: "compareRToTexture", // (default)
 31  *                           depthCompareFunc: "lequal",            // (default)
 32  *                           flipY: false,                          // Options are true (default) or false
 33  *                           width: 1,
 34  *                           height: 1,
 35  *                           internalFormat:"lequal",               // (default)
 36  *                           sourceFormat:"alpha",                  // (default)
 37  *                           sourceType: "unsignedByte",            // (default)
 38  *                           applyTo:"baseColor",                   // Options so far are “baseColor” (default) or “diffuseColor”
 39  *
 40  *                           // Optional transforms - these can also be functions, as shown in the next example
 41  *
 42  *                           rotate: {      // Currently textures are 2-D, so only rotation about Z makes sense
 43  *                               z: 45.0
 44  *                           },
 45  *
 46  *                           translate : {
 47  *                               x: 10,
 48  *                               y: 0,
 49  *                               z: 0
 50  *                           },
 51  *
 52  *                           scale : {
 53  *                               x: 1,
 54  *                               y: 2,
 55  *                               z: 1
 56  *                           }
 57  *                       }
 58  *                   ]
 59  *               },
 60  *
 61  *               new SceneJS.objects.Cube()
 62  *           )
 63  *     );
 64  *  </code></pre>
 65  *
 66  * <p><b>Example 2</b></p>
 67  * <p>You can animate texture transformations - this example shows how the rotate, scale and translate properties
 68  * can be functions to take their values from the data scope, in this case created by a higher WithData node:</p>
 69  *  <pre><code>
 70  * var subGraph =
 71  *       new SceneJS.WithData({
 72  *           angle: 45.0   // Vary this value to rotate the texture
 73  *       },
 74  *               new SceneJS.Texture({
 75  *                   layers: [
 76  *                       {
 77  *                           uri:"http://scenejs.org/library/textures/misc/general-zod.jpg",
 78  *
 79  *                           rotate: function(data) {
 80  *                               return { z: data.get("angle") }
 81  *                           }
 82  *                       }
 83  *                   ]
 84  *               },
 85  *               new SceneJS.objects.Cube()
 86  *         )
 87  *   );
 88  *  </code></pre>
 89  * @constructor
 90  * Create a new SceneJS.texture
 91  * @param {Object} The config object or function, followed by zero or more child nodes
 92  */
 93 SceneJS.Texture = function() {
 94     SceneJS.Node.apply(this, arguments);
 95     this._nodeType = "texture";
 96     this._layers = null;
 97 };
 98 
 99 SceneJS._inherit(SceneJS.Texture, SceneJS.Node);

101 // @private
102 SceneJS.Texture.prototype._getMatrix = function(translate, rotate, scale) {
103     var matrix = null;
104     var t;
105     if (translate) {
106         matrix = SceneJS_math_translationMat4v([ translate.x || 0, translate.y || 0, translate.z || 0]);
107     }
108     if (scale) {
109         t = SceneJS_math_scalingMat4v([ scale.x || 1, scale.y || 1, scale.z || 1]);
110         matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
111     }
112     if (rotate) {
113         if (rotate.x) {
114             t = SceneJS_math_rotationMat4v(rotate.x * 0.0174532925, [1,0,0]);
115             matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
116         }
117         if (rotate.y) {
118             t = SceneJS_math_rotationMat4v(rotate.y * 0.0174532925, [0,1,0]);
119             matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
120         }
121         if (rotate.z) {
122             t = SceneJS_math_rotationMat4v(rotate.z * 0.0174532925, [0,0,1]);
123             matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
124         }
125     }
126     return matrix;
127 };

129 SceneJS.Texture.prototype._STATE_INITIAL = 0;            // Ready to get texture
130 SceneJS.Texture.prototype._STATE_IMAGE_LOADING = 2;      // Texture image load in progress
131 SceneJS.Texture.prototype._STATE_IMAGE_LOADED = 3;       // Texture image load completed
132 SceneJS.Texture.prototype._STATE_TEXTURE_CREATED = 4;    // Texture created
133 SceneJS.Texture.prototype._STATE_ERROR = -1;             // Image load or texture creation failed
134 
135 SceneJS.Texture.prototype._render = function(traversalContext, data) {
136     if (!this._layers) { // One-shot dynamic config
137         this._layers = [];
138         var params = this._getParams(data);
139         if (!params.layers) {
140             throw new SceneJS.NodeConfigExpectedException(
141                     "SceneJS.Texture.layers is undefined");
142         }
143         for (var i = 0; i < params.layers.length; i++) {
144             var layerParam = params.layers[i];
145             if (!layerParam.uri) {
146                 throw new SceneJS.NodeConfigExpectedException(
147                         "SceneJS.Texture.layers[" + i + "].uri is undefined");
148             }
149             if (layerParam.applyFrom) {
150                 if (layerParam.applyFrom != "uv" &&
151                     layerParam.applyFrom != "uv2" &&
152                     layerParam.applyFrom != "normal" &&
153                     layerParam.applyFrom != "geometry") {
154                     SceneJS_errorModule.fatalError(
155                             new SceneJS.InvalidNodeConfigException(
156                                     "SceneJS.Texture.layers[" + i + "].applyFrom value is unsupported - " +
157                                     "should be either 'uv', 'uv2', 'normal' or 'geometry'"));
158                 }
159             }
160             if (layerParam.applyTo) {
161                 if (layerParam.applyTo != "baseColor" && // Colour map
162                     layerParam.applyTo != "diffuseColor") {
163                     SceneJS_errorModule.fatalError(
164                             new SceneJS.InvalidNodeConfigException(
165                                     "SceneJS.Texture.layers[" + i + "].applyTo value is unsupported - " +
166                                     "should be either 'baseColor', 'diffuseColor'"));
167                 }
168             }
169 
170             this._layers.push({
171                 state : this._STATE_INITIAL,
172                 process: null,                      // Image load process handle
173                 image : null,                       // Initialised when state == IMAGE_LOADED
174                 creationParams: layerParam,         // Create texture using this
175                 texture: null,                      // Initialised when state == TEXTURE_LOADED
176                 createMatrix : new (function() {
177                     var translate = layerParam.translate;
178                     var rotate = layerParam.rotate;
179                     var scale = layerParam.scale;
180                     var dynamic = ((translate instanceof Function) ||
181                                    (rotate instanceof Function) ||
182                                    (scale instanceof Function));
183                     var defined = dynamic || translate || rotate || scale;
184                     return function(data) {
185                         var matrix = null;
186                         if (defined && (dynamic || !matrix)) {
187                             matrix = SceneJS.Texture.prototype._getMatrix(
188                                     (translate instanceof Function) ? translate(data) : translate,
189                                     (rotate instanceof Function) ? rotate(data) : rotate,
190                                     (scale instanceof Function) ? scale(data) : scale);
191                         }
192                         return matrix;
193                     };
194                 })(),
195                 applyFrom: layerParam.applyFrom || "uv",
196                 applyTo: layerParam.applyTo || "baseColor",
197                 blendMode: layerParam.blendMode || "multiply"
198             });
199         }
200     }
201 
202     /* Update state of each texture layer and
203      * count how many are created and ready to apply
204      */
205     var countLayersReady = 0;
206     for (var i = 0; i < this._layers.length; i++) {
207         var layer = this._layers[i];
208         if (layer.state == this._STATE_TEXTURE_CREATED) {
209             if (!SceneJS_textureModule.textureExists(layer.texture)) {  // Texture evicted from cache
210                 layer.state = this._STATE_INITIAL;
211             }
212         }
213         switch (layer.state) {
214             case this._STATE_TEXTURE_CREATED:
215                 countLayersReady++;
216                 break;
217 
218             case this._STATE_INITIAL:
219 
220                 /* Start loading image - in a new closure so that the right layer gets the process result.
221                  */
222                 (function(l) {
223                     var _this = this;
224                     l.state = this._STATE_IMAGE_LOADING;
225 
226                     /* Logging each image load slows things down a lot
227                      */
228                     // loggingBackend.getLogger().info("SceneJS.texture image loading: "
229                     //  + _layer.creationParams.uri);
230 
231                     SceneJS_textureModule.loadImage(
232                             l.creationParams.uri,
233                             function(_image) {
234 
235                                 /* Image loaded successfully. Note that this callback will
236                                  * be called in the idle period between render traversals (ie. scheduled by a
237                                  * setInterval), so we're not actually visiting this node at this point. We'll
238                                  * defer creation and application of the texture to the subsequent visit.
239                                  */
240                                 l.image = _image;
241                                 l.state = _this._STATE_IMAGE_LOADED;
242                             },
243 
244                         /* General error, probably a 404
245                          */
246                             function() {
247                                 l.state = _this._STATE_ERROR;
248                                 var message = "SceneJS.texture image load failed: " + l.creationParams.uri;
249                                 SceneJS_loggingModule.getLogger().warn(message);
250 
251                                 /* Currently recovering from failed texture load
252                                  */
253 
254                                
255                             },
256 
257                         /* Load aborted - eg. user stopped browser
258                          */
259                             function() {
260                                 SceneJS_loggingModule.getLogger().warn("SceneJS.texture image load aborted: " + l.creationParams.uri);
261                                 l.state = _this._STATE_ERROR;
262                             });
263                 }).call(this, layer);
264                 break;
265 
266             case this._STATE_IMAGE_LOADING:
267 
268                 /* Continue loading this texture layer
269                  */
270                 break;
271 
272             case this._STATE_IMAGE_LOADED:
273 
274                 /* Create this texture layer
275                  */
276                 layer.texture = SceneJS_textureModule.createTexture(layer.image, layer.creationParams);
277                 layer.state = this._STATE_TEXTURE_CREATED;
278                 countLayersReady++;
279                 break;
280 
281             case this._STATE_ERROR:
282 
283                 /* Give up on this texture layer, but we'll keep updating the others
284                  * to at least allow diagnostics to log
285                  */
286                 break;
287         }
288     }
289 
290     if (SceneJS._traversalMode == SceneJS.TRAVERSAL_MODE_PICKING) {
291         this._renderNodes(traversalContext, data);
292     } else {
293         if ((countLayersReady == this._layers.length)) { // All or none - saves on generating/destroying shaders
294             var countPushed = 0;
295             for (var i = 0; i < this._layers.length; i++) {
296                 var layer = this._layers[i];
297               //   if (layer.state == this._STATE_TEXTURE_CREATED) {
298                 SceneJS_textureModule.pushLayer(layer.texture, {
299                     applyFrom : layer.applyFrom,
300                     applyTo : layer.applyTo,
301                     blendMode : layer.blendMode,
302                     matrix: layer.createMatrix(data)
303                 });
304                 countPushed++;
305                 }
306            // }
307             this._renderNodes(traversalContext, data);
308             SceneJS_textureModule.popLayers(countPushed);
309         }
310     }
311 };

314 /** Function wrapper to support functional scene definition
315  */
316 SceneJS.texture = function() {
317     var n = new SceneJS.Texture();
318     SceneJS.Texture.prototype.constructor.apply(n, arguments);
319     return n;
320 };
