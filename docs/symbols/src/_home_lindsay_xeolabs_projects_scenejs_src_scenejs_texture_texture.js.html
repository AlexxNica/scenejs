  4     var utils = SceneJS.__texture = {   // Just one object in closure
  5 
  6         errorBackend : SceneJS._backends.getBackend("error"),
  7         textureBackend : SceneJS._backends.getBackend("texture"),
  8         loggingBackend : SceneJS._backends.getBackend("logging"),

 10         getMatrix: function(translate, rotate, scale) {
 11             var matrix = null;
 12             var t;
 13             if (translate) {
 14                 matrix = SceneJS_math_translationMat4v([ translate.x || 0, translate.y || 0, translate.z || 0]);
 15             }
 16             if (scale) {
 17                 t = SceneJS_math_scalingMat4v([ scale.x || 1, scale.y || 1, scale.z || 1]);
 18                 matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
 19             }
 20             if (rotate) {
 21                 if (rotate.x) {
 22                     t = SceneJS_math_rotationMat4v(rotate.x * 0.0174532925, [1,0,0]);
 23                     matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
 24                 }
 25                 if (rotate.y) {
 26                     t = SceneJS_math_rotationMat4v(rotate.y * 0.0174532925, [0,1,0]);
 27                     matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
 28                 }
 29                 if (rotate.z) {
 30                     t = SceneJS_math_rotationMat4v(rotate.z * 0.0174532925, [0,0,1]);
 31                     matrix = matrix ? SceneJS_math_mulMat4(matrix, t) : t;
 32                 }
 33             }
 34             return matrix;
 35         },
 36 
 37         STATE_INITIAL : 0,            // Ready to get texture
 38         STATE_IMAGE_LOADING : 2,      // Texture image load in progress
 39         STATE_IMAGE_LOADED : 3,       // Texture image load completed
 40         STATE_TEXTURE_CREATED : 4,    // Texture created
 41         STATE_ERROR : -1             // Image load or texture creation failed
 42     };

 44     SceneJS.texture = function() {
 45         var cfg = SceneJS._utils.getNodeConfig(arguments);
 46         var params;
 47         var layers = [];
 48         var matrix;
 49 
 50         return SceneJS._utils.createNode(
 51                 "texture",
 52                 cfg.children,
 53 
 54                 new (function() {
 55 
 56                     this._render = function(traversalContext, data) {
 57 
 58                         /* Node can be dynamically configured, but only once
 59                          */
 60                         if (!params) {
 61                             params = cfg.getParams(data);
 62 
 63                             if (!params.layers) {
 64                                 throw new SceneJS.exceptions.NodeConfigExpectedException(
 65                                         "SceneJS.texture.layers is undefined");
 66                             }
 67 
 68                             params.layers = params.layers || [];
 69 
 70                             /* Prepare texture layers from params
 71                              */
 72                             for (var i = 0; i < params.layers.length; i++) {
 73 
 74                                 var layerParam = params.layers[i];
 75 
 76                                 if (!layerParam.uri) {
 77                                     throw new SceneJS.exceptions.NodeConfigExpectedException(
 78                                             "SceneJS.texture.layers[" + i + "].uri is undefined");
 79                                 }
 80 
 81                                 if (layerParam.applyFrom) {
 82                                     if (layerParam.applyFrom != "uv" &&
 83                                         layerParam.applyFrom != "uv2" &&
 84                                         layerParam.applyFrom != "normal" &&
 85                                         layerParam.applyFrom != "geometry") {
 86 
 87                                         utils.errorBackend.fatalError(
 88                                                 new SceneJS.exceptions.InvalidNodeConfigException(
 89                                                         "SceneJS.texture.layers[" + i + "].applyFrom value is unsupported - " +
 90                                                         "should be either 'uv', 'uv2', 'normal' or 'geometry'"));
 91                                     }
 92                                 }
 93 
 94                                 if (layerParam.applyTo) {
 95                                     if (layerParam.applyTo != "baseColor" && // Colour map
 96                                         layerParam.applyTo != "diffuseColor") {
 97 
 98                                         utils.errorBackend.fatalError(
 99                                                 new SceneJS.exceptions.InvalidNodeConfigException(
100                                                         "SceneJS.texture.layers[" + i + "].applyTo value is unsupported - " +
101                                                         "should be either 'baseColor', 'diffuseColor'"));
102                                     }
103                                 }
104 
105                                 layers.push({
106                                     state : utils.STATE_INITIAL,
107                                     process: null,                  // Imageload process handle
108                                     image : null,                   // Initialised when state == IMAGE_LOADED
109                                     creationParams: layerParam,   // Create texture using this
110                                     texture: null,          // Initialised when state == TEXTURE_LOADED
111                                     createMatrix : new (function() {
112                                         var translate = layerParam.translate;
113                                         var rotate = layerParam.rotate;
114                                         var scale = layerParam.scale;
115                                         var dynamic = ((translate instanceof Function) ||
116                                                        (rotate instanceof Function) ||
117                                                        (scale instanceof Function));
118                                         var defined = dynamic || translate || rotate || scale;
119                                         return function(data) {
120                                             if (defined && (dynamic || !matrix)) {
121                                                 matrix = utils.getMatrix(
122                                                         (translate instanceof Function) ? translate(data) : translate,
123                                                         (rotate instanceof Function) ? rotate(data) : rotate,
124                                                         (scale instanceof Function) ? scale(data) : scale);
125                                             }
126                                             return matrix;
127                                         };
128                                     })(),
129                                     applyFrom: layerParam.applyFrom || "uv",
130                                     applyTo: layerParam.applyTo || "baseColor",
131                                     blendMode: layerParam.blendMode || "multiply"
132                                 });
133                             }
134                         }
135 
136                         /* Update state of each texture layer and
137                          * count how many are created and ready to apply
138                          */
139                         var countLayersReady = 0;
140 
141                         for (var i = 0; i < layers.length; i++) {
142                             var layer = layers[i];
143 
144                             /* Backend may evict texture when not recently used,
145                              * in which case we'll have to load it again
146                              */
147                             if (layer.state == utils.STATE_TEXTURE_CREATED) {
148                                 if (!utils.textureBackend.textureExists(layer.texture)) {
149                                     layer.state = utils.STATE_INITIAL;
150                                 }
151                             }
152 
153                             switch (layer.state) {
154                                 case utils.STATE_TEXTURE_CREATED:
155                                     countLayersReady++;
156                                     break;
157 
158                                 case utils.STATE_INITIAL:
159 
160                                     /* Start loading image for this texture layer.
161                                      *
162                                      * Do it in a new closure so that the right layer gets the process result.
163                                      */
164                                     (function(_layer) {
165                                         _layer.state = utils.STATE_IMAGE_LOADING;
166 
167                                         /* Logging each image load slows things down a lot
168                                          */
169                                         // utils.loggingBackend.getLogger().info("SceneJS.texture image loading: "
170                                         //  + _layer.creationParams.uri);
171 
172                                         utils.textureBackend.loadImage(
173                                                 _layer.creationParams.uri,
174                                                 function(_image) {
175 
176                                                     /* Image loaded successfully. Note that this callback will
177                                                      * be called in the idle period between render traversals (ie. scheduled by a
178                                                      * setInterval), so we're not actually visiting this node at this point. We'll
179                                                      * defer creation and application of the texture to the subsequent visit.
180                                                      */
181                                                     _layer.image = _image;
182                                                     _layer.state = utils.STATE_IMAGE_LOADED;
183                                                 },
184 
185                                             /* General error, probably a 404
186                                              */
187                                                 function() {
188                                                     _layer.state = utils.STATE_ERROR;
189                                                     var message = "SceneJS.texture image load failed: "
190                                                             + _layer.creationParams.uri;
191                                                     utils.loggingBackend.getLogger().warn(message);
192 
193                                                     /* Currently recovering from failed texture load
194                                                      */
195 
196                                                     // utils.errorBackend.error(
197                                                     //       new SceneJS.exceptions.ImageLoadFailedException(message));
198                                                 },
199 
200                                             /* Load aborted - eg. user stopped browser
201                                              */
202                                                 function() {
203                                                     utils.loggingBackend.getLogger().warn("SceneJS.texture image load aborted: "
204                                                             + _layer.creationParams.uri);
205                                                     _layer.state = utils.STATE_ERROR;
206                                                 });
207 
208 
209                                     })(layer);
210                                     break;
211 
212                                 case utils.STATE_IMAGE_LOADING:
213 
214                                     /* Continue loading this texture layer
215                                      */
216                                     break;
217 
218                                 case utils.STATE_IMAGE_LOADED:
219 
220                                     /* Create this texture layer
221                                      */
222                                     layer.texture = utils.textureBackend.createTexture(layer.image, layer.creationParams);
223                                     layer.state = utils.STATE_TEXTURE_CREATED;
224                                     countLayersReady++;
225                                     break;
226 
227                                 case utils.STATE_ERROR:
228 
229                                     /* Give up on this texture layer, but we'll keep updating the others
230                                      * to at least allow diagnostics to log
231                                      */
232                                     break;
233                             }
234                         }
235 
236                         if (SceneJS._utils.traversalMode == SceneJS._utils.TRAVERSAL_MODE_PICKING) {
237 
238                             /* Dont apply textures if picking
239                              */
240                             SceneJS._utils.visitChildren(cfg, traversalContext, data);
241 
242                         } else {
243 
244                             /** Render either all layers or none - saves on generating/destroying shaders
245                              */
246                             //  if ((countLayersReady == layers.length)) {
247                             var countPushed = 0;
248                             for (var i = 0; i < layers.length; i++) {
249                                 var layer = layers[i];
250                                 if (layer.state == utils.STATE_TEXTURE_CREATED) {
251                                     utils.textureBackend.pushLayer(layer.texture, {
252                                         applyFrom : layer.applyFrom,
253                                         applyTo : layer.applyTo,
254                                         blendMode : layer.blendMode,
255                                         matrix: layer.createMatrix(data)
256                                     });
257                                     countPushed++;
258                                 }
259                             }
260                             this._renderChildren(traversalContext, data);
261                             utils.textureBackend.popLayers(countPushed);
262                         }
263                     };
264                 })());
265     };
