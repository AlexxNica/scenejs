1 /**
  2  * @class A scene node that defines a viewing transform by specifing location of the eye position, the point being looked
  3  * at, and the direction of "up".
  4  * @extends SceneJS.Node
  5  * <p><b>Usage Example:</b></p><p>Defining perspective, specifying parameters that happen to be the default values</b></p><pre><code>
  6  * var l = new SceneJS.LookAt({
  7  *     eye : { x: 0.0, y: 10.0, z: -15 },
  8  *    look : { y:1.0 },
  9  *    up : { y: 1.0 },
 10  *
 11  * // .. Child nodes ...
 12  *
 13  * </pre></code>
 14  *
 15  * @constructor
 16  * Create a new SceneJS.LookAt
 17  * @param {Object} cfg  Config object or function, followed by zero or more child nodes
 18  */
 19 SceneJS.LookAt = function() {
 20     SceneJS.Node.apply(this, arguments);
 21     this._nodeType = "lookat";
 22     this._mat = null;
 23     this._xform = null;
 24 
 25     this._eyeX = 0;
 26     this._eyeY = 0;
 27     this._eyeZ = 1;
 28 
 29     this._lookX = 0;
 30     this._lookY = 0;
 31     this._lookZ = 0;
 32 
 33     this._upX = 0;
 34     this._upY = 1;
 35     this._upZ = 0;
 36 
 37     if (this._fixedParams) {
 38         this._init(this._getParams());
 39     }
 40 };
 41 
 42 SceneJS._inherit(SceneJS.LookAt, SceneJS.Node);
 43 
 44 /** Sets the eye position.
 45  * Don't allow this position to be the same as the position being looked at.
 46  *
 47  * @param {Object} eye - Eg. { x: 0.0, y: 10.0, z: -15 }
 48  * @returns {SceneJS.LookAt} this
 49  */
 50 SceneJS.LookAt.prototype.setEye = function(eye) {
 51     this._eyeX = eye.x || 0;
 52     this._eyeY = eye.y || 0;
 53     this._eyeZ = eye.z || 1;
 54     this._memoLevel = 0;
 55     return this;
 56 };
 57 
 58 /** Returns the eye position.
 59  *
 60  * @returns {Object} Eye position - Eg. { x: 0.0, y: 10.0, z: -15 }
 61  */
 62 SceneJS.LookAt.prototype.getEye = function() {
 63     return {
 64         x: this._eyeX,
 65         y: this._eyeY,
 66         z: this._eyeZ
 67     };
 68 };
 69 
 70 /** Sets the point being looked at.
 71  * Don't allow this point to be the same as the eye position.
 72  *
 73  * @param {Object} look - Eg. { x: 0.0, y: 2.0, z: 0.0 }
 74  * @returns {SceneJS.LookAt} this
 75  */
 76 SceneJS.LookAt.prototype.setLook = function(look) {
 77     this._lookX = look.x || 0;
 78     this._lookY = look.y || 0;
 79     this._lookZ = look.z || 0;
 80     this._memoLevel = 0;
 81     return this;
 82 };
 83 
 84 /** Returns the position being looked at.
 85  * @returns {Object} Point looked at - Eg. { x: 0.0, y: 2.0, z: 0.0 }
 86  */
 87 SceneJS.LookAt.prototype.getLook = function() {
 88     return {
 89         x: this._lookX,
 90         y: this._lookY,
 91         z: this._lookZ
 92     };
 93 };
 94 
 95 /** Sets the "up" vector - the direction that is considered "upwards".
 96  *
 97  * @param {Object} up - Eg. { x: 0.0, y: 1.0, z: 0.0 }
 98  * @returns {SceneJS.LookAt} this
 99  */
100 SceneJS.LookAt.prototype.setUp = function(up) {
101     var x = up.x || 0;
102     var y = up.y || 0;
103     var z = up.z || 0;
104     if (x + y + z == 0) {
105         throw SceneJS._errorModule.fatalError(
106                 new SceneJS.InvalidNodeConfigException(
107                         "SceneJS.lookAt up vector is zero length - at least one of its x,y and z components must be non-zero"));
108     }
109     this._upX = x;
110     this._upY = y;
111     this._upZ = z;
112     this._memoLevel = 0;
113     return this;
114 };
115 
116 
117 /** Returns the "up" vector - the direction that is considered "upwards".
118  *
119  * @returns {Object} Up vector - Eg. { x: 0.0, y: 1.0, z: 0.0 }
120  */
121 SceneJS.LookAt.prototype.getUp = function() {
122     return {
123         x: this._upX,
124         y: this._upY,
125         z: this._upZ
126     };
127 };
128 
129 SceneJS.LookAt.prototype._init = function(params) {
130     if (params.eye) {
131         this.setEye(params.eye);
132     }
133     if (params.look) {
134         this.setLook(params.look);
135     }
136     if (params.up) {
137         this.setUp(params.up);
138     }
139 };
140 
141 SceneJS.LookAt.prototype._render = function(traversalContext, data) {
142     if (this._memoLevel == 0) {
143         if (!this._fixedParams) {
144             this._init(this._getParams(data));
145         } else {
146             this._memoLevel = 1;
147         }
148         this._mat = SceneJS._math_lookAtMat4c(
149                 this._eyeX, this._eyeY, this._eyeZ,
150                 this._lookX, this._lookY, this._lookZ,
151                 this._upX, this._upY, this._upZ);
152     }
153     var superXform = SceneJS._modelViewTransformModule.getTransform();
154     if (this._memoLevel < 2) {
155         var tempMat = SceneJS._math_mulMat4(superXform.matrix, this._mat);
156         this._xform = {
157             type: "lookat",
158             matrix: tempMat,
159             lookAt : {
160                 eye: { x: this._eyeX, y: this._eyeY, z: this._eyeZ },
161                 look: { x: this._lookX, y: this._lookY, z: this._lookZ },
162                 up:  { x: this._upX, y: this._upY, z: this._upZ }
163             },
164             fixed: superXform.fixed && this._fixedParams
165         };
166         if (this._memoLevel == 1 && superXform.fixed && !SceneJS._instancingModule.instancing()) {   // Bump up memoization level if space fixed
167             this._memoLevel = 2;
168         }
169     }
170     SceneJS._modelViewTransformModule.setTransform(this._xform);
171     this._renderNodes(traversalContext, data);
172     SceneJS._modelViewTransformModule.setTransform(superXform);
173 };

175 /** Factory function that returns a new {@link SceneJS.LookAt} instance
176  */
177 SceneJS.lookAt = function() {
178     var n = new SceneJS.LookAt();
179     SceneJS.LookAt.prototype.constructor.apply(n, arguments);
180     return n;
181 };
