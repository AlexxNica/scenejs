1 /**
  2  * This module encapsulates shading behind an event API.
  3  *
  4  * By listening to XXX_UPDATED events, this module tracks various elements of scene state, such as WebGL settings,
  5  * texture layers, lighting, current material properties etc.
  6  *
  7  * On a SHADER_ACTIVATE event it will compose and activate a shader taylored to the current scene state
  8  * (ie. where the shader has variables and routines for the current lights, materials etc), then fire a
  9  * SHADER_ACTIVATED event when the shader is ready for business.
 10  *
 11  * Other modules will then handle the SHADER_ACTIVATED event by firing XXXXX_EXPORTED events parameterised with
 12  * resources that they want loaded into the shader. This module then handles those by loading their parameters into
 13  * the shader.
 14  *
 15  * The module will avoid constant re-generation of shaders by caching each of them against a hash code that it
 16  * derives from the current collective scene state; on a SHADER_ACTIVATE event, it will attempt to reuse a shader
 17  * cached for the hash of the current scene state.
 18  *
 19  * Shader allocation and LRU cache eviction is mediated by SceneJS_memoryModule.
 20  *  @private
 21  */
 22 var SceneJS_shaderModule = new (function() {
 23 
 24     var time = (new Date()).getTime();      // For LRU caching
 25 
 26     /* Resources contributing to shader
 27      */
 28     var canvas;                             // Currently active canvas
 29     var rendererState;                      // WebGL settings state
 30     var lights = [];                        // Current lighting state
 31     var material = {};                      // Current material state
 32     var fog = null;                         // Current fog
 33     var textureLayers = [];                 // Texture layers are pushed/popped to this as they occur
 34     var geometry = null;                    // Current geometry
 35 
 36     /* Hash codes identifying states of contributing resources
 37      */
 38     var rendererHash = "";
 39     var fogHash = "";
 40     var lightsHash = "";
 41     var textureHash = "";
 42     var geometryHash = "";
 43 
 44     /* Shader programs
 45      */
 46     var programs = {};                      // Program cache
 47     var activeProgram = null;               // Currently active program
 48 
 49     /* Combined hash from those of contributing resources, to identify shaders
 50      */
 51     var sceneHash;

109     SceneJS_eventModule.onEvent(
110             SceneJS_eventModule.RENDERER_EXPORTED,
111             function(_rendererState) {
112 
113                 /* Default ambient material colour is taken from canvas clear colour
114                  */
115                 var clearColor = _rendererState.clearColor;
116                 activeProgram.setUniform("uAmbient",
117                         clearColor

122     SceneJS_eventModule.onEvent(
123             SceneJS_eventModule.TEXTURES_UPDATED,
124             function(stack) {
125                 textureLayers = stack;
126                 sceneHash = null;
127 
128                 /* Build texture hash
129                  */
130                 var hash = [];
131                 for (var i = 0; i < stack.length; i++) {
132                     var layer = textureLayers[i];
133                     hash.push("/");
134                     hash.push(layer.params.applyFrom);
135                     hash.push("/");
136                     hash.push(layer.params.applyTo);
137                     hash.push("/");
138                     hash.push(layer.params.blendMode);
139                     if (layer.params.matrix) {
140                         hash.push("/anim");
141                     }
142                 }
143                 textureHash = hash.join("");
144             });
145 
146     SceneJS_eventModule.onEvent(
147             SceneJS_eventModule.TEXTURES_EXPORTED,
148             function(stack) {
149                 for (var i = 0; i < stack.length; i++) {
150                     var layer = stack[i];
151                     activeProgram.bindTexture("uSampler" + i, layer.texture, i);
152                     if (layer.params.matrixAsArray) {
153                         activeProgram.setUniform("uLayer" + i + "Matrix", layer.params.matrixAsArray);
154                     }
155                 }
156             });

158     SceneJS_eventModule.onEvent(
159             SceneJS_eventModule.LIGHTS_UPDATED,
160             function(l) {
161                 lights = l;
162                 sceneHash = null;
163 
164                 /* Build lights hash
165                  */
166                 var hash = [];
167                 for (var i = 0; i < lights.length; i++) {
168                     var light = lights[i];
169                     hash.push(light._type);
170                     if (light._specular) {
171                         hash.push("s");
172                     }
173                     if (light._diffuse) {
174                         hash.push("d");
175                     }
176                 }
177                 lightsHash = hash.join("");
178             });

180     SceneJS_eventModule.onEvent(
181             SceneJS_eventModule.LIGHTS_EXPORTED,
182             function(_lights) {
183                 for (var i = 0; i < _lights.length; i++) {
184                     var light = _lights[i];
185                     activeProgram.setUniform("uLightColor" + i, light._color);
186                     activeProgram.setUniform("uLightDiffuse" + i, light._diffuse);
187                     if (light._type == "dir") {
188                         activeProgram.setUniform("uLightDir" + i, light._viewDir);
189                     } else {
190                         if (light._type == "point") {
191                             activeProgram.setUniform("uLightPos" + i, light._viewPos);
192                         }
193                         if (light._type == "spot") {
194                             activeProgram.setUniform("uLightPos" + i, light._viewPos);
195                             activeProgram.setUniform("uLightDir" + i, light._viewDir);
196                             activeProgram.setUniform("uLightSpotCosCutOff" + i, light._spotCosCutOff);
197                             activeProgram.setUniform("uLightSpotExp" + i, light._spotExponent);
198                         }
199                         activeProgram.setUniform("uLightAttenuation" + i,
200                                 [
201                                     light._constantAttenuation,
202                                     light._linearAttenuation,
203                                     light._quadraticAttenuation
204                                 ]);
205                     }
206                 }
207             });

273                 geometryHash = ([

279     SceneJS_eventModule.onEvent(
280             SceneJS_eventModule.GEOMETRY_EXPORTED,
281             function(geo) {
282                 if (geo.vertexBuf) {
283                     activeProgram.bindFloatArrayBuffer("aVertex", geo.vertexBuf);
284                 }
285                 if (geo.normalBuf) {
286                     activeProgram.bindFloatArrayBuffer("aNormal", geo.normalBuf);
287                 }
288                 if (textureLayers.length > 0 && rendererState.enableTexture2D) {
289                     if (geo.uvBuf) {
290                         activeProgram.bindFloatArrayBuffer("aUVCoord", geo.uvBuf);
291                     }
292                     if (geo.uvBuf2) {

294                     }

298     SceneJS_eventModule.onEvent(
299             SceneJS_eventModule.SHADER_ACTIVATE, // Request to activate a shader
300             function() {
301                 activateProgram();
302             });
303 
304     SceneJS_memoryModule.registerEvictor(
305             function() {
306                 var earliest = time;
307                 var programToEvict;
308                 for (var hash in programs) {
309                     if (hash) {
310                         var program = programs[hash];
311 
312                         /* Avoiding eviction of shader just used,
313                          * currently in use, or likely about to use
314                          */
315                         if (program.lastUsed < earliest && program.hash != sceneHash) {
316                             programToEvict = program;
317                             earliest = programToEvict.lastUsed;
318                         }
319                     }
320                 }
321                 if (programToEvict) { // Delete LRU program's shaders and deregister program
322                     //  SceneJS_loggingModule.info("Evicting shader: " + hash);
323                     programToEvict.destroy();
324                     programs[programToEvict.hash] = null;
325                     return true;
326                 }
327                 return false;   // Couldnt find suitable program to delete
328             });

330     /**
331      * @private
332      */
333     function activateProgram() {
334         if (!canvas) {
335             SceneJS_errorModule.fatalError(new SceneJS.exceptions.NoCanvasActiveException("No canvas active"));
336         }
337 
338         if (!sceneHash) {
339             generateHash();
340         }
341 
342         if (!activeProgram || activeProgram.hash != sceneHash) {
343             if (activeProgram) {
344                 canvas.context.flush();
345                 activeProgram.unbind();
346                 activeProgram = null;
347                 SceneJS_eventModule.fireEvent(SceneJS_eventModule.SHADER_DEACTIVATED);
348             }
349 
350             if (!programs[sceneHash]) {
351                 SceneJS_loggingModule.info("Creating shader: '" + sceneHash + "'");
352                 var vertexShaderSrc = composeVertexShader();
353                 var fragmentShaderSrc = composeFragmentShader();
354                 SceneJS_memoryModule.allocate(
355                         "shader",
356                         function() {
357                             try {
358                                 programs[sceneHash] = new SceneJS_webgl_Program(
359                                         sceneHash,
360                                         time,
361                                         canvas.context,
362                                         [vertexShaderSrc],
363                                         [fragmentShaderSrc],
364                                         SceneJS_loggingModule);
365                                 SceneJS_loggingModule.info("OK - Created shader: '" + sceneHash + "'");
366                             } catch (e) {
367                                 SceneJS_loggingModule.debug("Vertex shader:");
368                                 SceneJS_loggingModule.debug(getShaderLoggingSource(vertexShaderSrc.split(";")));
369                                 SceneJS_loggingModule.debug("Fragment shader:");
370                                 SceneJS_loggingModule.debug(getShaderLoggingSource(fragmentShaderSrc.split(";")));
371                                 SceneJS_errorModule.fatalError(e);
372                             }
373                         });
374             }
375             activeProgram = programs[sceneHash];
376             activeProgram.lastUsed = time;
377             activeProgram.bind();
378             SceneJS_eventModule.fireEvent(SceneJS_eventModule.SHADER_ACTIVATED);
379         }
380 
381         SceneJS_eventModule.fireEvent(SceneJS_eventModule.SHADER_RENDERING);
382     }

384     /**
385      * Generates a shader hash code from current rendering state.
386      * @private
387      */
388     function generateHash() {
389         if (SceneJS._utils.traversalMode == SceneJS._utils.TRAVERSAL_MODE_PICKING) {
390             sceneHash = ([canvas.canvasId, "picking"]).join(";");
391         } else {
392             sceneHash = ([canvas.canvasId, rendererHash, fogHash, lightsHash, textureHash, geometryHash]).join(";");
393         }
394         //      SceneJS_loggingModule.debug("Scene shading hash:" + sceneHash);
395     }

397     /**
398      * @private
399      */
400     function getShaderLoggingSource(src) {
401         var src2 = [];
402         for (var i = 0; i < src.length; i++) {
403             var padding = (i < 10) ? "   " : ((i < 100) ? "  " : (i < 1000 ? " " : ""));
404             src2.push(i + padding + ": " + src[i]);
405         }
406         return src2.join("<br/>");
407     }

425     /**
426      * Composes a vertex shader script for rendering mode in current scene state
427      * @private
428      */
429     function composePickingVertexShader() {
430         return [
431             "attribute vec3 aVertex;",
432             "uniform mat4 uMMatrix;",
433             "uniform mat4 uVMatrix;",
434             "uniform mat4 uPMatrix;",
435             "void main(void) {",
436             "  gl_Position = uPMatrix * (uVMatrix * (uMMatrix * vec4(aVertex, 1.0)));",
437             "}"
438         ].join("\n");
439     }

441     /**
442      * Composes a fragment shader script for rendering mode in current scene state
443      * @private
444      */
445     function composePickingFragmentShader() {
446         var g = parseFloat(Math.round((10 + 1) / 256) / 256);
447         var r = parseFloat((10 - g * 256 + 1) / 256);
448         var src = [
449             "uniform vec3 uColor;",
450             "void main(void) {",
451 
452             "gl_FragColor = vec4(" + (r.toFixed(17)) + ", " + (g.toFixed(17)) + ",1.0,1.0);",
453 
454             //      "    gl_FragColor = vec4(uColor.rgb, 1.0);  ",
455             "}"
456         ].join("\n");
457 
458         return src;
459     }

461     /**
462      * @private
463      */
464     function composeRenderingVertexShader() {
465 
466         var texturing = textureLayers.length > 0 && rendererState.enableTexture2D && (geometry.uvBuf || geometry.uvBuf2);
467         var lighting = (lights.length > 0 && geometry.normalBuf);
468 
469         var src = ["\n"];
470         src.push("attribute vec3 aVertex;");                // World coordinates
471 
472         if (lighting) {
473             src.push("attribute vec3 aNormal;");            // Normal vectors
474             src.push("uniform mat4 uMNMatrix;");            // Model normal matrix
475             src.push("uniform mat4 uVNMatrix;");            // View normal matrix
476 
477             src.push("varying vec3 vNormal;");              // Output view normal vector
478             src.push("varying vec3 vEyeVec;");              // Output view eye vector
479 
480             for (var i = 0; i < lights.length; i++) {
481                 var light = lights[i];
482                 if (light._type == "dir") {
483                     src.push("uniform vec3 uLightDir" + i + ";");
484                 }
485                 if (light._type == "point") {
486                     src.push("uniform vec4 uLightPos" + i + ";");
487                 }
488                 if (light._type == "spot") {
489                     src.push("uniform vec4 uLightPos" + i + ";");
490                 }
491 
492                 src.push("varying vec3 vLightVec" + i + ";");
493                 src.push("varying float vLightDist" + i + ";");
494             }
495         }
496 
497         if (texturing) {
498             if (geometry.uvBuf) {
499                 src.push("attribute vec2 aUVCoord;");      // UV coords
500             }
501             if (geometry.uvBuf2) {
502                 src.push("attribute vec2 aUVCoord2;");     // UV2 coords
503             }
504         }
505         src.push("uniform mat4 uMMatrix;");                // Model matrix
506         src.push("uniform mat4 uVMatrix;");                // View matrix
507         src.push("uniform mat4 uPMatrix;");                 // Projection matrix
508 
509         src.push("varying vec4 vViewVertex;");
510 
511         if (texturing) {
512             if (geometry.uvBuf) {
513                 src.push("varying vec2 vUVCoord;");
514             }
515             if (geometry.uvBuf2) {
516                 src.push("varying vec2 vUVCoord2;");
517             }
518         }
519 
520         src.push("void main(void) {");
521         if (lighting) {
522             src.push("  vec4 tmpVNormal = uVNMatrix * (uMNMatrix * vec4(aNormal, 1.0)); ");
523             src.push("  vNormal = normalize(tmpVNormal.xyz);");
524         }
525         src.push("  vec4 tmpVertex = uVMatrix * (uMMatrix * vec4(aVertex, 1.0)); ");
526         src.push("  vViewVertex = tmpVertex;");
527         src.push("  gl_Position = uPMatrix * vViewVertex;");
528 
529         src.push("  vec3 tmpVec;");
530 
531         if (lighting) {
532             for (var i = 0; i < lights.length; i++) {
533                 var light = lights[i];
534                 if (light._type == "dir") {
535                     src.push("tmpVec = -uLightDir" + i + ";");
536                 }
537                 if (light._type == "point") {
538                     src.push("tmpVec = -(uLightPos" + i + ".xyz - tmpVertex.xyz);");
539                     src.push("vLightDist" + i + " = length(tmpVec);");          // Distance from light to vertex
540                 }
541                 if (light._type == "spot") {
542                     src.push("tmpVec = -(uLightPos" + i + ".xyz - tmpVertex.xyz);");
543                     src.push("vLightDist" + i + " = length(tmpVec);");          // Distance from light to vertex
544 
545                 }
546                 src.push("vLightVec" + i + " = tmpVec;");                   // Vector from light to vertex
547 
548             }
549             src.push("vEyeVec = normalize(-vViewVertex.xyz);");
550         }
551 
552         if (texturing) {
553             if (geometry.uvBuf) {
554                 src.push("vUVCoord = aUVCoord;");
555             }
556             if (geometry.uvBuf2) {
557                 src.push("vUVCoord2 = aUVCoord2;");
558             }
559         }
560         src.push("}");
561         //      SceneJS_loggingModule.info(getShaderLoggingSource(src));
562         return src.join("\n");
563     }

566     /**
567      * @private
568      */
569     function composeRenderingFragmentShader() {
570         var texturing = textureLayers.length > 0 && rendererState.enableTexture2D && (geometry.uvBuf || geometry.uvBuf2);
571         var lighting = (lights.length > 0 && geometry.normalBuf);
572 
573         var src = ["\n"];
574 
575         src.push("varying vec4 vViewVertex;");              // View-space vertex
576 
577         if (texturing) {
578             if (geometry.uvBuf) {
579                 src.push("varying vec2 vUVCoord;");
580             }
581             if (geometry.uvBuf2) {
582                 src.push("varying vec2 vUVCoord2;");
583             }
584 
585             for (var i = 0; i < textureLayers.length; i++) {
586                 var layer = textureLayers[i];
587                 src.push("uniform sampler2D uSampler" + i + ";");
588                 if (layer.params.matrix) {
589                     src.push("uniform mat4 uLayer" + i + "Matrix;");
590                 }
591             }
592         }
593 
594         src.push("uniform vec3  uMaterialBaseColor;");
595         src.push("uniform float uMaterialAlpha;");
596 
597         if (lighting) {
598             src.push("varying vec3 vNormal;");                  // View-space normal
599             src.push("varying vec3 vEyeVec;");                  // Direction of view-space vertex from eye
600 
601             src.push("uniform vec3  uAmbient;");                         // Scene ambient colour - taken from clear colour
602             src.push("uniform float uMaterialEmit;");
603 
604             src.push("uniform vec3  uMaterialSpecularColor;");
605             src.push("uniform float uMaterialSpecular;");
606             src.push("uniform float uMaterialShine;");
607 
608             for (var i = 0; i < lights.length; i++) {
609                 var light = lights[i];
610                 src.push("uniform vec3  uLightColor" + i + ";");
611                 if (light._type == "point") {
612                     src.push("uniform vec4   uLightPos" + i + ";");
613                 }
614                 if (light._type == "dir") {
615                     src.push("uniform vec3   uLightDir" + i + ";");
616                 }
617                 if (light._type == "spot") {
618                     src.push("uniform vec4   uLightPos" + i + ";");
619                     src.push("uniform vec3   uLightDir" + i + ";");
620                     src.push("uniform float  uLightSpotCosCutOff" + i + ";");
621                     src.push("uniform float  uLightSpotExp" + i + ";");
622                 }
623                 src.push("uniform vec3  uLightAttenuation" + i + ";");
624                 src.push("varying vec3  vLightVec" + i + ";");         // Vector from light to vertex
625                 src.push("varying float vLightDist" + i + ";");        // Distance from light to vertex
626             }
627         }
628 
629         /* Fog uniforms
630          */
631         if (fog && fog.mode != "disabled") {
632             src.push("uniform vec3  uFogColor;");
633             src.push("uniform float uFogDensity;");
634             src.push("uniform float uFogStart;");
635             src.push("uniform float uFogEnd;");
636         }
637 
638         src.push("void main(void) {");
639         src.push("  vec3    color   = uMaterialBaseColor;");
640         src.push("  float   alpha   = uMaterialAlpha;");
641 
642         if (lighting) {
643             src.push("  vec3    ambientValue=uAmbient;");
644             src.push("  float   emit    = uMaterialEmit;");
645 
646             src.push("  vec4    normalmap = vec4(vNormal,0.0);");
647             src.push("  float   specular=uMaterialSpecular;");
648             src.push("  vec3    specularColor=uMaterialSpecularColor;");
649             src.push("  float   shine=uMaterialShine;");
650             src.push("  float   attenuation = 1.0;");
651         }
652 
653         if (texturing) {
654             src.push("  vec4    texturePos;");
655             src.push("  vec2    textureCoord=vec2(0.0,0.0);");
656 
657             for (var i = 0; i < textureLayers.length; i++) {
658                 var layer = textureLayers[i];
659 
660                 /* Texture input
661                  */
662                 if (layer.params.applyFrom == "normal" && lighting) {
663                     if (geometry.normalBuf) {
664                         src.push("texturePos=vec4(vNormal.xyz, 1.0);");
665                     } else {
666                         SceneJS_loggingModule.warn("Texture layer applyFrom='normal' but geometry has no normal vectors");
667                         continue;
668                     }
669                 }
670                 if (layer.params.applyFrom == "uv") {
671                     if (geometry.uvBuf) {
672                         src.push("texturePos = vec4(vUVCoord.s, vUVCoord.t, 1.0, 1.0);");
673                     } else {
674                         SceneJS_loggingModule.warn("Texture layer applyTo='uv' but geometry has no UV coordinates");
675                         continue;
676                     }
677                 }
678                 if (layer.params.applyFrom == "uv2") {
679                     if (geometry.uvBuf2) {
680                         src.push("texturePos = vec4(vUVCoord2.s, vUVCoord2.t, 1.0, 1.0);");
681                     } else {
682                         SceneJS_loggingModule.warn("Texture layer applyTo='uv2' but geometry has no UV2 coordinates");
683                         continue;
684                     }
685                 }
686 
687                 /* Texture matrix
688                  */
689                 if (layer.params.matrixAsArray) {
690                     src.push("textureCoord=(uLayer" + i + "Matrix * texturePos).xy;");
691                 } else {
692                     src.push("textureCoord=texturePos.xy;");
693                 }
694 
695                 /* Texture output
696                  */
697                 if (layer.params.applyTo == "baseColor") {
698                     if (layer.params.blendMode == "multiply") {
699                         src.push("color  = color * texture2D(uSampler" + i + ", vec2(textureCoord.x, 1.0 - textureCoord.y)).rgb;");
700                     } else {
701                         src.push("color  = color + texture2D(uSampler" + i + ", vec2(textureCoord.x, 1.0 - textureCoord.y)).rgb;");
702                     }
703                 }
704             }
705         }
706 
707         if (lighting) {
708             src.push("  vec3    lightValue      = uAmbient;");
709             src.push("  vec3    specularValue   = vec3(0.0, 0.0, 0.0);");
710 
711             src.push("  vec3    lightVec;");
712             src.push("  float   dotN;");
713             src.push("  float   spotFactor;");
714             src.push("  float   pf;");
715 
716             for (var i = 0; i < lights.length; i++) {
717                 var light = lights[i];
718                 src.push("lightVec = normalize(vLightVec" + i + ");");
719 
720                 /* Point Light
721                  */
722                 if (light._type == "point") {
723                     src.push("dotN = max(dot(vNormal,lightVec),0.0);");
724                     src.push("if (dotN > 0.0) {");
725                     src.push("  attenuation = 1.0 / (" +
726                              "  uLightAttenuation" + i + "[0] + " +
727                              "  uLightAttenuation" + i + "[1] * vLightDist" + i + " + " +
728                              "  uLightAttenuation" + i + "[2] * vLightDist" + i + " * vLightDist" + i + ");");
729                     if (light._diffuse) {
730                         src.push("  lightValue += dotN *  uLightColor" + i + " * attenuation;");
731                     }
732                     if (light._specular) {
733                         src.push("specularValue += attenuation * specularColor * uLightColor" + i +
734                                  " * specular  * pow(max(dot(reflect(lightVec, vNormal), vEyeVec),0.0), shine);");
735                     }
736                     src.push("}");
737                 }
738 
739                 /* Directional Light
740                  */
741                 if (light._type == "dir") {
742                     src.push("dotN = max(dot(vNormal,lightVec),0.0);");
743                     if (light._diffuse) {
744                         src.push("lightValue += dotN * uLightColor" + i + ";");
745                     }
746                     if (light._specular) {
747                         src.push("specularValue += specularColor * uLightColor" + i +
748                                  " * specular  * pow(max(dot(reflect(lightVec, vNormal),normalize(vEyeVec)),0.0), shine);");
749                     }
750                 }
751 
752                 /* Spot light
753                  */
754                 if (light._type == "spot") {
755                     src.push("spotFactor = max(dot(normalize(uLightDir" + i + "), lightVec));");
756                     src.push("if ( spotFactor > 20) {");
757                     src.push("  spotFactor = pow(spotFactor, uLightSpotExp" + i + ");");
758                     src.push("  dotN = max(dot(vNormal,normalize(lightVec)),0.0);");
759                     src.push("      if(dotN>0.0){");
760 
761                     //                            src.push("          attenuation = spotFactor / (" +
762                     //                                     "uLightAttenuation" + i + "[0] + " +
763                     //                                     "uLightAttenuation" + i + "[1] * vLightDist" + i + " + " +
764                     //                                     "uLightAttenuation" + i + "[2] * vLightDist" + i + " * vLightDist" + i + ");");
765                     src.push("          attenuation = 1;");
766 
767                     if (light._diffuse) {
768                         src.push("lightValue +=  dotN * uLightColor" + i + " * attenuation;");
769                     }
770                     if (light._specular) {
771                         src.push("specularValue += attenuation * specularColor * uLightColor" + i +
772                                  " * specular  * pow(max(dot(reflect(normalize(lightVec), vNormal),normalize(vEyeVec)),0.0), shine);");
773                     }
774 
775                     src.push("      }");
776                     src.push("}");
777                 }
778             }
779             src.push("if (emit>0.0) lightValue = vec3(1.0, 1.0, 1.0);");
780             src.push("vec4 fragColor = vec4(specularValue.rgb + color.rgb * (emit+1.0) * lightValue.rgb, alpha);");
781         } else {
782 
783             /* No lighting
784              */
785             src.push("vec4 fragColor = vec4(color.rgb, alpha);");
786         }
787 
788         /* Fog
789          */
790         if (fog && fog.mode != "disabled") {
791             src.push("float fogFact=1.0;");
792             if (fog.mode == "exp") {
793                 src.push("fogFact=clamp(pow(max((uFogEnd - length(-vViewVertex.xyz)) / (uFogEnd - uFogStart), 0.0), 2.0), 0.0, 1.0);");
794             } else if (fog.mode == "linear") {
795                 src.push("fogFact=clamp((uFogEnd - length(-vViewVertex.xyz)) / (uFogEnd - uFogStart), 0.0, 1.0);");
796             }
797             src.push("gl_FragColor = fragColor * fogFact + vec4(uFogColor, 1) * (1.0 - fogFact);");
798         } else {
799             src.push("gl_FragColor = fragColor;");
800         }
801 
802         // src.push("gl_FragColor = vec4(1.0, 1.0, 1.0, 1.0);");
803         src.push("}");
804 
805         // SceneJS_loggingModule.info(getShaderLoggingSource(src));
806         return src.join("\n");
807     }
