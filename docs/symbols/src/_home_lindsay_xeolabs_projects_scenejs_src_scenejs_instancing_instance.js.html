1 /**
  2  * @class Instantiates the subtree of a target {@link SceneJS.Symbol} at the node's location within the scene graph.
  3  *
  4  * The flexible <a href="http://scenejs.wikispaces.com/instancing+algorithm">Instancing Algorithm</a> also permits recursive
  5  * instantiation, where target {@link SceneJS.Symbol}s may define further instances of other target {@link SceneJS.Symbol}s,
  6  * and so on. Instances may also be parameterised using the data flow capabilities provided by the
  7  * {@link SceneJS.WithData} and {@link SceneJS.WithConfigs} nodes.</p>
  8  *
  9  * <p>When a {@link SceneJS.Symbol} has been rendered prior to the Instance during the current scene traversal, then the
 10  * Instance can instantiate it with a URI that which will walk the SIDs (sub-identifiers) of previously rendered
 11  * nodes to resolve the {@link SceneJS.Symbol}. Here's the most trivial case - see how the fragment URI of the instance
 12  * maps to the {@link SceneJS.Symbol}'s SID.<p>
 13  * <pre><code>
 14  * new SceneJS.Symbol({ sid: "myBox" },
 15  *    new SceneJS.objects.Cube()
 16  * ),
 17  *
 18  * new SceneJS.Instance( { uri: "myBox" })
 19  * </code></pre>
 20  * <p>Another case, in which an Instance resolves a Symbol that is within a namespace defined by the SID of a parent
 21  * node:</p>
 22  * <pre><code>
 23  * new SceneJS.Node({ sid: "mySymbols" },
 24  *     new SceneJS.Symbol({ sid: "myBox" },
 25  *         new SceneJS.objects.Cube()
 26  *     )
 27  * ),
 28  *
 29  * new SceneJS.Instance( { uri: "mySymbols/myBox" })
 30  * </code></pre>
 31  * <p>Think of these nested SIDs as directories, where the URI fragment part works works the same way as a directory
 32  * path. An absolute fragment path begins with a '/' - in the following example, if the node with the "moreSymbols" SID
 33  * is defined at the top level, we may then reference it with an absolute path:</p>
 34  * <pre><code>
 35  * new SceneJS.Node({ sid: "moreSymbols" },
 36  *     new SceneJS.Symbol({ sid: "myOtherBox" },
 37  *         new SceneJS.objects.Cube()
 38  *     )
 39  * ),
 40  *
 41  * new SceneJS.Instance( { uri: "/moreSymbols/myOtherBox" })
 42  * </code></pre>
 43  *
 44  * <h2>States and Events</h2>
 45  * <p>A SceneJS.Instance has four states which it transitions through during it's lifecycle, as described below. After
 46  * it transitions into each state, it will fire an event - see {@link SceneJS.Node}. Also, while in {@link #STATE_RENDERING},
 47  * it can provide its target {@link SceneJS.Symbol} node via {@link #getTarget}.<p>
 48  *
 49  * @events
 50  * @extends SceneJS.Node
 51  * @constructor
 52  * Creates a new SceneJS.Instance
 53  *  @param {Object} [cfg] Static configuration object
 54  * @param {String} cfg.uri URI of file to load
 55  * @param {int} [cfg.timeoutSecs] Timeout - falls back on any loadTimoutSecs that was configured on the {@link SceneJS.Scene}
 56  * at the root of the scene graph, or the default 180 seconds if none configured there
 57  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 58  * @param {...SceneJS.Node} [childNodes] Child nodes
 59  */
 60 SceneJS.Instance = function() {
 61     SceneJS.Node.apply(this, arguments);
 62     this._nodeType = "instance";
 63     this._uri = null;
 64     this._symbol = null;
 65     this._state = SceneJS.Instance.STATE_INITIAL;
 66     if (this._fixedParams) {
 67         this._init(this._getParams());
 68     }
 69 };
 70 
 71 SceneJS._inherit(SceneJS.Instance, SceneJS.Node);
 72 
 73 /**
 74  * Initial state of a SceneJS.Instance, in which it has not been rendered yet and thus not attempted to resolve its
 75  * target {@link SceneJS.Symbol} yet.
 76  * @const
 77  */
 78 SceneJS.Instance.STATE_INITIAL = 0;
 79 
 80 /**
 81  * State of a SceneJS.Instance in which instantiation has failed. This condition might be temporary (eg. the target
 82  * {@link SceneJS.Symbol} has just not been rendered yet for some reason), so a SceneJS.Instance will then try
 83  * instancing again when next rendered, transitioning to {@link STATE_RENDERING} when that succeeds.
 84  * @const
 85  */
 86 SceneJS.Instance.STATE_ERROR = -1;
 87 
 88 /**
 89  * State of a SceneJS.Instance in which it will attempt to instantiate its target when next rendered. This is when it
 90  * is ready to attempt aquisition of its target {@link SceneJS.Symbol}. From here, it will transition to
 91  * {@link #STATE_RENDERING} if that succeeds, otherwise it will transition to {@link #STATE_ERROR}.
 92  * @const
 93  */
 94 SceneJS.Instance.STATE_READY = 2;
 95 
 96 
 97 /**
 98  * State of an SceneJS.Instance in which it is currently rendering its target {@link SceneJS.Symbol}. While in
 99  * this state, you can obtain the target {@link SceneJS.Symbol} through {@link #getTarget}. From this
100  * state, the SceneJS.Instance will transition back to {@link #STATE_READY} once it has completed rendering the target.
101  * @const
102  */
103 SceneJS.Instance.STATE_RENDERING = 3;
104 
105 /**
106  * Returns the node's current state. Possible states are {@link #STATE_INITIAL},
107  * {@link #STATE_READY}, {@link #STATE_ERROR} and {@link #STATE_RENDERING}.
108  * @returns {int} The state
109  */
110 SceneJS.Instance.prototype.getState = function() {
111     return this._state;
112 };
113 
114 // @private
115 SceneJS.Instance.prototype._init = function(params) {
116     if (params.uri) {
117         this._uri = params.uri;
118         this._state = SceneJS.Instance.STATE_READY;
119         this._mustExist = params.mustExist;
120     }
121 };
122 
123 // @private
124 SceneJS.Instance.prototype._render = function(traversalContext, data) {
125     if (!this._fixedParams) {
126         this._init(this._getParams(data));
127     }
128     if (this._uri) {
129         this._instanceSymbol(this._uri, traversalContext, data);
130     } else {
131         throw SceneJS._errorModule.fatalError(
132                 new SceneJS.errors.InvalidNodeConfigException(
133                         "SceneJS.Instance uri property not defined"));
134     }
135 };
136 
137 // @private
138 SceneJS.Instance.prototype._changeState = function(newState, exception) {
139     var oldState = this._state;
140     this._state = newState;
141     if (this._listeners["state-changed"]) { // Optimisation
142         this._fireEvent("state-changed", { oldState: oldState, newState: newState, exception : exception });
143     }
144 };

146 /* Returns a traversal context for traversal of the children of the given target node.
147  *
148  * If this Instance has children then it will have a callback that will render them after the last of
149  * the target's sub-nodes have rendered, as effectively the children of that last node. The callback will
150  * create a traversal context for the sub-nodes that will:
151  *
152  * - initially flag the traversal as inside the right fringe if the there are more than one child
153  * - pass on any callback that was passed in on the traversal context to this Instance
154  * - pass on any WithConfigs configs that were passed in on the traversal context to this Instance
155  *
156  * @private
157  */
158 SceneJS.Instance.prototype._createTargetTraversalContext = function(traversalContext, target) {
159     var callback;
160     this._superCallback = traversalContext.callback;
161     var _this = this;
162     if (!this._callback) {
163         this._callback = function(traversalContext, data) {
164             var subTraversalContext = {
165                 callback : _this._superCallback,
166                 insideRightFringe : _this._children.length > 1,
167                 configs: traversalContext.configs,
168                 configsModes: traversalContext.configsModes
169             };
170             _this._renderNodes(subTraversalContext, data);
171         };
172     }
173     return {
174         callback: this._callback,
175         insideRightFringe:  target._children.length > 1,
176         configs: traversalContext.configs,
177         configsModes: traversalContext.configsModes
178     };
179 };
180 
181 /**
182  * Recursively renders the subtrees (child nodes) of the Instance in left-to-right, depth-first order. As the recursion
183  * descends, the traversalContext tracks whether the current node is inside the right fringe of the right-most subtree.
184  * As soon as the current node is at the right fringe and has no children, then it is the last node to render among all
185  * the sub-nodes (the "terminal node"). If a callback is then present on the traversalContext, then that means that this
186  * Instance is actually within a subtree of a Symbol node that is being instanced by another super-Instance. The
187  * callback is then invoked, which causes the traversal of the super-Instance's subtree as if it were a child of the
188  * terminal node.
189  *
190  * @param traversalContext
191  * @param data
192  * @private
193  */
194 SceneJS.Instance.prototype._renderNodes = function(traversalContext, data) {
195     var numChildren = this._children.length;
196     var child;
197     var childConfigs;
198     var configUnsetters;
199 
200     if (numChildren == 0) {
201 
202         /* Instance has no child nodes - render super-Instance's child nodes
203          * through callback if one is passed in
204          */
205         if (traversalContext.callback) {
206             traversalContext.callback(traversalContext, data);
207         }
208 
209     } else {
210 
211         /* Instance has child nodes - last node in Instance's subtree will invoke
212          * the callback, if any (from within its SceneJS.Node#_renderNodes)
213          */
214         for (var i = 0; i < numChildren; i++) {
215             child = this._children[i];
216 
217             childConfigs = traversalContext.configs;
218             configUnsetters = null;
219 
220             if (childConfigs && child._sid) {
221                 childConfigs = traversalContext.configs["#" + child._sid];
222                 if (childConfigs) {
223                     configUnsetters = this._setConfigs(childConfigs, child);
224                 }
225             }
226 
227             var childTraversalContext = {
228                 insideRightFringe : (i < numChildren - 1),
229                 callback : traversalContext.callback,
230                 configs : childConfigs || traversalContext.configs,
231                 configsModes : traversalContext.configsModes
232             };
233 
234             child._render.call(child, childTraversalContext, data);
235 
236             if (configUnsetters) {
237                 this._unsetConfigs(configUnsetters);
238             }
239         }
240     }
241 };

243 /** Instances a Symbol that is currently defined after being rendered prior to this Instance
244  *
245  * @private
246  * @param symbolSIDPath Path to Symbol, relative to this Instance
247  * @param traversalContext
248  * @param data
249  */
250 SceneJS.Instance.prototype._instanceSymbol = function(symbolSIDPath, traversalContext, data) {
251     this._symbol = SceneJS._instancingModule.acquireInstance(symbolSIDPath);
252     if (!this._symbol) {
253         //SceneJS._loggingModule.info("SceneJS.Instance could not find SceneJS.Symbol to instance: '" + symbolSIDPath + "'");
254         if (this._mustExist) {
255             SceneJS._instancingModule.releaseInstance();
256             throw SceneJS._errorModule.fatalError(
257                     new SceneJS.errors.SymbolNotFoundException
258                             ("SceneJS.Instance could not find SceneJS.Symbol to instance: '" + symbolSIDPath + "'"));
259         }
260         this._changeState(SceneJS.Instance.STATE_ERROR);
261     } else {
262         this._changeState(SceneJS.Instance.STATE_RENDERING);
263         this._symbol._renderNodes(this._createTargetTraversalContext(traversalContext, this._symbol), data);
264         SceneJS._instancingModule.releaseInstance();
265         this._changeState(SceneJS.Instance.STATE_READY);
266         this._symbol = null;
267     }
268 };

271 /**
272  * While in {@link #STATE_RENDERING}, returns the target {@link SceneJS.Symbol} currently being rendered.
273  * @returns {SceneJS.Symbol} Target symbol
274  */
275 SceneJS.Instance.prototype.getTarget = function() {
276     if (this.state != SceneJS.Instance.STATE_RENDERING) {
277         return null;
278     }
279     return this._symbol;
280 };

282 /**
283  * Returns the URI on which the Instance looks for its target {@link SceneJS.Symbol}

285 SceneJS.Instance.prototype.getURI = function() {
286     return this._uri;

289 /** Factory function that returns a new {@link SceneJS.Instance} instance
290  *  @param {Object} [cfg] Static configuration object
291  * @param {String} cfg.uri URI of file to load
292  * @param {int} [cfg.timeoutSecs] Timeout - falls back on any loadTimoutSecs that was configured on the {@link SceneJS.Scene}
293  * at the root of the scene graph, or the default 180 seconds if none configured there
294  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
295  * @param {...SceneJS.Node} [childNodes] Child nodes
296  * @returns {SceneJS.Instance}
297  */
298 SceneJS.instance = function() {
299     var n = new SceneJS.Instance();
300     SceneJS.Instance.prototype.constructor.apply(n, arguments);
301     return n;
302 };
