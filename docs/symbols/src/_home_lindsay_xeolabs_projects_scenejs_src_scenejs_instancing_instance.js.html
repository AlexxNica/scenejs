1 /**
  2  * @class A scene node that instantiates a previously-defined SceneJS.Symbol node.
  3  *
  4  * <p>When rendered, a SceneJS.Symbol node registers its contents against a specified name, without rendering the
  5  * content. Then when SceneJS then finds a SceneJS.Instance node that refers to the registered name,
  6  * it will instantiate the Symbol's child nodes as if they were children of the Instance node.</p>
  7  *
  8  * <p><b>Live Examples</b></p>
  9  * <ul><li><a target = "other" href="http://bit.ly/9d8wLu">Example 1</a></li></ul>
 10  * <p><b>Example Usage</b></p><p>Here we're defining a Symbol in a Name, then instantiating it three times with
 11  * Instance nodes to show variations on how an Instance node can refer to a Symbol, relative to a namespace
 12  * created by a Name node:</b></p><pre><code>
 13  *
 14  *  // Define a "teapot" symbol inside a namespace.
 15  *  // Note that the Name is not mandatory.
 16  *
 17  *  SceneJS.name({name: "mySymbols"},
 18  *
 19  *       SceneJS.symbol({ name: "teapot" },
 20  *           SceneJS.objects.teapot()
 21  *       ),
 22  *
 23  *       // Instance the teapot Symbol from inside the namespace.
 24  *       // See how the symbol reference is relative, where it
 25  *       // does not start with a '/'.
 26  *
 27  *       SceneJS.instance({name: "teapot" })
 28  * ),
 29  *
 30  * // Instance the teapot Symbol again, from outside the namespace
 31  *
 32  * SceneJS.instance({name: "mySymbols/teapot"}),
 33  *
 34  * // Instance the teapot Symbol one more time from outside the
 35  * // namespace to show how an absolute path can be specified to
 36  * // the Symbol
 37  *
 38  * SceneJS.instance({name: "/mySymbols/teapot"})
 39  *
 40  * // ...
 41  * </pre></code>
 42  *  @extends SceneJS.Node
 43  * @constructor
 44  * Create a new SceneJS.Instance
 45  * @param {Object} config  Config object or function, followed by zero or more child nodes
 46  */
 47 SceneJS.Instance = function() {
 48     SceneJS.Node.apply(this, arguments);
 49     this._nodeType = "instance";
 50     this._name = null;
 51     if (this._fixedParams) {
 52         this._init(this._getParams());
 53     }
 54 };
 55 
 56 SceneJS._inherit(SceneJS.Instance, SceneJS.Node);
 57 
 58 /**
 59  Sets the name of the symbol to instance. The name will be "unnamed" if none is specified.
 60  @function setName
 61  @param {string} name
 62  @returns {SceneJS.Instance} This instance node
 63  */
 64 SceneJS.Instance.prototype.setName = function(name) {
 65     this._name = name || "unnamed";
 66     return this;
 67 };
 68 
 69 /**
 70  Returns the name of the instanced symbol. The name will be "unnamed" if none is specified.
 71  @function {string} getName
 72  @returns {string} The name
 73  */
 74 SceneJS.Instance.prototype.getName = function() {
 75     return this._name;
 76 };
 77 
 78 // @private
 79 SceneJS.Instance.prototype._init = function(params) {
 80     if (params.name) {
 81         this.setName(params.name);
 82     }
 83 };
 84 
 85 // @private
 86 SceneJS.Instance.prototype._render = function(traversalContext, data) {
 87     if (!this._fixedParams) {
 88         this._init(this._getParams(data));
 89     }
 90     var nameNode = SceneJS_instancingModule.acquireInstance(this._name);
 91     if (!nameNode) {
 92         SceneJS_errorModule.fatalError(
 93                 new SceneJS.SymbolNotFoundException
 94                         ("SceneJS.Instance could not find SceneJS.Symbol to instance: '" + this._name + "'"));
 95     } else {
 96         nameNode._renderNodes(traversalContext, data);
 97         SceneJS_instancingModule.releaseInstance();
 98     }
 99 };
100 
101 /** Returns a new SceneJS.Instance instance
102  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Instance constructor
103  * @returns {SceneJS.Instance}
104  */
105 SceneJS.instance = function() {
106     var n = new SceneJS.Instance();
107     SceneJS.Instance.prototype.constructor.apply(n, arguments);
108     return n;
109 };
