1 /**
  2  * Scene node that specifies the current viewing volume and transform matrix
  3  */
  4 
  5 
  6 SceneJS.frustum = function() {
  7     var cfg = SceneJS.utils.getNodeConfig(arguments);
  8     var backend = SceneJS._backends.getBackend('viewprojection');
  9 
 10     return SceneJS._utils.createNode(
 11             "frustum",
 12             cfg.children,
 13 
 14             new (function() {
 15 
 16                 var transform;
 17 
 18                 this._render = function(traversalContext, data) {
 19                     if (!transform || cfg.fixed) {    // Memoize matrix if node config is constant
 20                         var params = cfg.getParams(data);
 21                         var volume = {
 22                             xmin: params.left || -1.0,
 23                             xmax: params.right || 1.0,
 24                             ymin: params.bottom || -1.0,
 25                             ymax: params.top || 1.0,
 26                             zmin: params.near || 0.1,
 27                             zmax: params.far || 1000.0
 28                         };
 29                         var tempMat = SceneJS_math_frustumMatrix4(
 30                                 volume.xmin,
 31                                 volume.xmax,
 32                                 volume.ymin,
 33                                 volume.ymax,
 34                                 volume.zmin,
 35                                 volume.zmax
 36                                 );
 37                         transform = {
 38                             matrix: tempMat,
 39                             frustum : new SceneJS_math_Frustum(tempMat),
 40                             volume: volume
 41                         };
 42                     }
 43                     var prevTransform = backend.getTransform();
 44                     backend.setTransform(transform);
 45                     this._renderChildren(traversalContext, data);
 46                     backend.setTransform(prevTransform);
 47                 };
 48             })());
 49 };
