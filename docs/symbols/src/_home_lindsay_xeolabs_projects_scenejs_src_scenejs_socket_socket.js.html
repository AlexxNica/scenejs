1 /**
  2  * @class Binds its subgraph to a WebSocket, providing a server with the ability to add, remove and manipulate content within the subgraph
  3  *
  4  * <p>The SceneJS.Socket node enables a server to dynamically participate in the construction, destruction and
  5  * configuration of its subgraph. It binds the subgraph to a WebSocket through which it exchanges JSON message objects
  6  * with a server.</p>
  7  *
  8  * <h2>Message Format</h2>
  9  * <p>Incoming messages from a server are either error responses, like that shown below, or configuration maps
 10  * exactly like those specified with a {@link SceneJS.WithConfigs}. On receipt of a configuration map, a Socket
 11  * automatically applies the map to its subgraph in the same way that a {@link SceneJS.WithConfigs} does.</p>
 12  * <p>An error response object has two parts: the error code, either a string or number, and a message.</p>
 13  * <p>Below is an example of an HTTP 404 error response:</p>
 14  * <pre><code>
 15  * {
 16  *     error : 404,
 17  *     body  : "Could not find asset 'foobar'
 18  * }
 19  * </code></pre>
 20  *
 21  *
 22  * <p>The outgoing message format is not part of the SceneJS specification and is whatever JSON objects the server on
 23  * the other end expects.</p>
 24  *
 25  * <h2>Message Queues</h2>
 26  * <p>Messages can be configured on the Socket to send when the connection first opens, or enqueued at any
 27  * time with {@link #addMessage} to send when the socket is next rendered while a connection is open.</p>
 28  *
 29  * <p>Incoming messages are also queued - each time the Socket is rendered while a connection is open, it dequeues and
 30  * processes the next incoming message before sending all queued outgoing messages.</p>
 31  *
 32  * <p>You can have multiple Sockets scattered throughout a scene graph, and may even nest them. This enables you to do some
 33  * fancy things, like have "live assets" that are controlled by the servers that they are downloaded from. Imagine a
 34  * subgraph defining an airplane, that relies on its server to manage its complex physics computations, for example.</p>
 35  *
 36  *  <p><b>Example Usage</b></p><p>Below is a Socket that connects to a server on the local host. The Socket starts off
 37  * in {@link #STATE_INITIAL}. Then as soon as it is rendered, it transitions to {@link #STATE_CONNECTING} and tries to
 38  * open the connection. When successful, it sends the optional specified messages and transitions to {@link #STATE_OPENED}.
 39  * The server may respond with an error (described further below) or a configuration map. If the response is a
 40  * configuration map, the Socket would then apply that to its sub-nodes.
 41  * On error, the Socket will transition to {@link #STATE_ERROR} and remain in that state, with connection closed. If
 42  * the connection ever closes, the Socket will attempt to re-open it when next rendered.</p> 
 43  * <pre><code>
 44  * new SceneJS.Socket({
 45  *
 46  *       // Location of server
 47  *
 48  *       uri: "ws://127.0.0.1:8888/",
 49  *
 50  *       // Messages to send as soon as the socket is first opened
 51  *
 52  *       messages: [
 53  *          {
 54  *               myParam  : "foo",
 55  *               myParam2 : "bar"
 56  *          },
 57  *
 58  *          // Next message ...
 59  *
 60  *      ],
 61  *
 62  *      listeners: {
 63  *         "state-changed" : {
 64  *              fn: function(theNode, params) {
 65  *                  switch (params.newState) {
 66  *                      case SceneJS.Socket.STATE_CONNECTING:
 67  *
 68  *                          // Socket attempting to open connection with server at specified URI
 69  *
 70  *                          alert("STATE_CONNECTING");
 71  *                          break;
 72  *
 73  *                      case SceneJS.Socket.STATE_OPEN:
 74  *
 75  *                          // Server connection opened, messages sent
 76  *
 77  *                          alert("STATE_OPEN");
 78  *                          break;
 79  *
 80  *                      case SceneJS.Socket.STATE_CLOSED:
 81  *
 82  *                          // Connection closed OK
 83  *
 84  *                          alert("STATE_CLOSED");
 85  *                          break;
 86  *
 87  *                      case SceneJS.Socket.STATE_ERROR:
 88  *
 89  *                          // Error opening connection, or server error response
 90  *
 91  *                          alert("STATE_ERROR: " + params.exception.message);
 92  *                          break;
 93  *                   }
 94  *               }
 95  *           }
 96  *       },
 97  *
 98  *       // Child nodes that will receive configs returned by socket's server peer
 99  *)
100  *
101  * </code></pre>
102 
103  */
104 SceneJS.Socket = function() {
105     SceneJS.Node.apply(this, arguments);
106     this._nodeType = "socket";
107     this._uri = null;
108     this._autoOpen = true;
109     this._socketId = null;
110     this._outMessages = [];
111     this._state = SceneJS.Socket.STATE_INITIAL;
112     this._configsModes = {
113         strictProperties : true,
114         strictNodes : false
115     };
116     if (this._fixedParams) {
117         this._init(this._getParams());
118     }
119 };
120 
121 SceneJS._inherit(SceneJS.Socket, SceneJS.Node);
122 
123 /** Initial state of Socket when not rendered yet.
124  */
125 SceneJS.Socket.STATE_INITIAL = 0;
126 
127 /** State of Socket when it has attempted to open a connection with a server and awaiting response. Socket tries to open
128  * the connection as soon as it is rendered and enters this state.
129  */
130 SceneJS.Socket.STATE_CONNECTING = 2;
131 
132 /** State of Socket when connection is open with server.
133  */
134 SceneJS.Socket.STATE_OPEN = 3;
135 
136 /**
137  * State of Socket in which connection is closed. From here it will attempt to r-open when next rendered.
138  */
139 SceneJS.Socket.STATE_CLOSED = 4;
140 
141 /** State of Socket in which an error occured, either due to failure to open connection, or as signalled by the server.
142  */
143 SceneJS.Socket.STATE_ERROR = -1;
144 
145 /** Enqeues an outgoing message to send when the Socket is next rendered while a connection is open.
146  *
147  * @param message
148  * @returns {this}
149  */
150 SceneJS.Socket.prototype.addMessage = function(message) {
151     this._outMessages.push(message);
152     return this;
153 };
154 
155 /** Clears outgoing message queue - messages are not sent.
156  * @returns {this}
157  */
158 SceneJS.Socket.prototype.removeMessages = function() {
159     this._outMessages = [];
160     return this;
161 };
162 
163 // @private
164 SceneJS.Socket.prototype._init = function(params) {
165     this._uri = params.uri;
166     if (params.messages) {
167         this._outMessages = params.messages;
168     }
169 };
170 
171 // @private
172 SceneJS.Socket.prototype._render = function(traversalContext, data) {
173     if (!this._fixedParams) {
174         this._init(this._getParams(data));
175     }
176     if (!this._uri) {
177         throw SceneJS._errorModule.fatalError(
178                 new SceneJS.errors.InvalidNodeConfigException("SceneJS.Socket uri property not defined"));
179     }
180 
181     /* Socket can close after lack of use, in which case
182      * the node would have to re-open it
183      */
184     if (this._state == SceneJS.Socket.STATE_OPEN) {
185         if (!SceneJS._SocketModule.acquireSocket(this._socketId)) {
186             this._changeState(SceneJS.Socket.STATE_CLOSED);
187         }
188     }
189 
190     if (this._state == SceneJS.Socket.STATE_OPEN) { // Still open, and socket was acquired above
191 
192         /* Process next incoming message then send pending outgoing messages
193          */
194         var _self = this;
195         var message = SceneJS._SocketModule.getNextMessage(
196                 function(exception) { // onerror
197                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
198                     SceneJS._errorModule.error(exception);
199                 });
200         if (message) {
201             if (message.error) {
202                 SceneJS._errorModule.error(
203                         new SceneJS.errors.SocketServerErrorException(
204                                 "SceneJS.Socket server responded with error: " + message.error + ", " + message.body));
205             } else if (message.body) {
206                 // alert(message.body);
207                 traversalContext = {
208                     appendix : traversalContext.appendix,
209                     insideRightFringe: this._children.length > 1,
210                     configs : message.body,
211                     configsModes : this._configsModes // TODO configsModes in message?
212                 };
213                 data = new SceneJS.Data(data, this._fixedParams, this._data);
214             }
215         }
216         this._sendMessages();
217         this._renderNodes(traversalContext, data);
218         SceneJS._SocketModule.releaseSocket();
219     } else {
220         if (this._state == SceneJS.Socket.STATE_INITIAL || this._state == SceneJS.Socket.STATE_CLOSED) {
221 
222             this._changeState(SceneJS.Socket.STATE_CONNECTING);
223             (function(_self) { // Closure allows this node to receive results
224                 SceneJS._SocketModule.openSocket({ uri: _self._uri },
225                         function(socketId) { // onopen
226                             _self._socketId = socketId;
227                             _self._changeState(SceneJS.Socket.STATE_OPEN);
228                         },
229                         function() { // onclose
230                             _self._changeState(SceneJS.Socket.STATE_CLOSED);
231                         },
232                         function(exception) { // onerror
233                             _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
234                             SceneJS._errorModule.error(exception);
235                         });
236             })(this);
237         }
238         if (this._state == this._STATE_ERROR) { // Socket disabled - TODO: retry?
239         }
240         this._renderNodes(traversalContext, data); // We're assuming socket wont open instantly, ie. during this node visit
241     }
242 };
243 
244 // @private
245 SceneJS.Socket.prototype._changeState = function(newState, params) {
246     params = params || {};
247     params.oldState = this._state;
248     params.newState = newState;
249     this._state = newState;
250     if (this._listeners["state-changed"]) { // Optimisation
251         this._fireEvent("state-changed", params);
252     }
253 };
254 
255 // @private
256 SceneJS.Socket.prototype._sendMessages = function() {
257     if (this._outMessages.length > 0) {
258         var _self = this;
259         SceneJS._SocketModule.sendMessages(
260                 this._outMessages,
261                 function(exception) { // onerror
262                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
263                 },
264                 function() {  // onsuccess
265                     this._outMessages = [];
266                 });
267 
268     }
269 };

271 /** Factory function that returns a new {@link SceneJS.Socket} instance
272  * @param {Object} [cfg] Static configuration object
273 
274  * @returns {SceneJS.Socket}
275  * @since Version 0.7.6
276  */
277 SceneJS.socket = function() {
278     var n = new SceneJS.Socket();
279     SceneJS.Socket.prototype.constructor.apply(n, arguments);
280     return n;
