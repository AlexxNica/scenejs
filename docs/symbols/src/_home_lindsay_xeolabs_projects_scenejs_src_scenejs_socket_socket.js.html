1 /**
  2  * @class Binds its subgraph to a WebSocket, providing a server with the ability to add, remove and manipulate content within the subgraph
  3  *
  4  * <p>The SceneJS.Socket node enables a server to dynamically participate in the construction, destruction and
  5  * configuration of its subgraph. It binds the subgraph to a WebSocket through which it exchanges JSON message objects
  6  * with a server.</p>
  7  *
  8  * <h2>Message Format</h2>
  9  * <p>Incoming messages from a server are either error responses, like that shown below, or configuration maps
 10  * exactly like those specified with a {@link SceneJS.WithConfigs}. On receipt of a configuration map, a Socket
 11  * automatically applies the map to its subgraph in the same way that a {@link SceneJS.WithConfigs} does.</p>
 12  * <p>An error response object has two parts: the error code, either a string or number, and a message.</p>
 13  *
 14  * <p>Below is an example of an HTTP 404 error server response:</p>
 15  * <pre><code>
 16  * {
 17  *     error : 404,
 18  *     body  : "Could not find asset 'foobar'
 19  * }
 20  * </code></pre>
 21  *
 22  * <p>Below is an example of a server response containing subnode configurations:</p>
 23  * <pre><code>
 24  * {
 25  *     body : {
 26  *          configs: {
 27  *              // ...
 28  *          }
 29  *     }
 30  * }
 31  * </code></pre>
 32  *
 33  * <p>The outgoing message format is not part of the SceneJS specification and is whatever JSON objects the server on
 34  * the other end expects.</p>
 35  *
 36  * <h2>Message Queues</h2>
 37  * <p>Messages can be configured on the Socket to send when the connection first opens, or enqueued at any
 38  * time with {@link #addMessage} to send when the socket is next rendered while a connection is open.</p>
 39  *
 40  * <p>Incoming messages are also queued - each time the Socket is rendered while a connection is open, it dequeues and
 41  * processes the next incoming message before sending all queued outgoing messages.</p>
 42  *
 43  * <p>You can have multiple Sockets scattered throughout a scene graph, and may even nest them. This enables you to do some
 44  * fancy things, like have "live assets" that are controlled by the servers that they are downloaded from. Imagine a
 45  * subgraph defining an airplane, that relies on its server to manage its complex physics computations, for example.</p>
 46  *
 47  *  <p><b>Example Usage</b></p><p>Below is a Socket that connects to a server on the local host. The Socket starts off
 48  * in {@link #STATE_INITIAL}. Then as soon as it is rendered, it transitions to {@link #STATE_CONNECTING} and tries to
 49  * open the connection. When successful, it sends the optional specified messages and transitions to {@link #STATE_OPENED}.
 50  * The server may respond with an error (described further below) or a configuration map. If the response is a
 51  * configuration map, the Socket would then apply that to its sub-nodes.
 52  * On error, the Socket will transition to {@link #STATE_ERROR} and remain in that state, with connection closed. If
 53  * the connection ever closes, the Socket will attempt to re-open it when next rendered.</p>
 54  * <pre><code>
 55  * new SceneJS.Socket({
 56  *
 57  *       // Location of server
 58  *
 59  *       uri: "ws://127.0.0.1:8888/",
 60  *
 61  *       // Messages to send as soon as the socket is first opened
 62  *
 63  *       messages: [
 64  *          {
 65  *               myParam  : "foo",
 66  *               myParam2 : "bar"
 67  *          },
 68  *
 69  *          // Next message ...
 70  *
 71  *      ],
 72  *
 73  *      listeners: {
 74  *         "state-changed" : {
 75  *              fn: function(theNode, params) {
 76  *                  switch (params.newState) {
 77  *                      case SceneJS.Socket.STATE_CONNECTING:
 78  *
 79  *                          // Socket attempting to open connection with server at specified URI
 80  *
 81  *                          alert("STATE_CONNECTING");
 82  *                          break;
 83  *
 84  *                      case SceneJS.Socket.STATE_OPEN:
 85  *
 86  *                          // Server connection opened, messages sent
 87  *
 88  *                          alert("STATE_OPEN");
 89  *                          break;
 90  *
 91  *                      case SceneJS.Socket.STATE_CLOSED:
 92  *
 93  *                          // Connection closed OK
 94  *
 95  *                          alert("STATE_CLOSED");
 96  *                          break;
 97  *
 98  *                      case SceneJS.Socket.STATE_ERROR:
 99  *
100  *                          // Error opening connection, or server error response
101  *
102  *                          alert("STATE_ERROR: " + params.exception.message);
103  *                          break;
104  *                   }
105  *               }
106  *           }
107  *       },
108  *
109  *       // Child nodes that will receive configs returned by socket's server peer
110  *)
111  *
112  * </code></pre>
113 
114  */
115 SceneJS.Socket = function() {
116     SceneJS.Node.apply(this, arguments);
117     this._nodeType = "socket";
118     this._uri = null;
119     this._autoOpen = true;
120     this._socketId = null;
121     this._outMessages = [];
122     this._state = SceneJS.Socket.STATE_INITIAL;
123     this._configsModes = {
124         strictProperties : true,
125         strictNodes : false
126     };
127     if (this._fixedParams) {
128         this._init(this._getParams());
129     }
130 };
131 
132 SceneJS._inherit(SceneJS.Socket, SceneJS.Node);
133 
134 /** Initial state of Socket when not rendered yet.
135  */
136 SceneJS.Socket.STATE_INITIAL = 0;
137 
138 /** State of Socket when it has attempted to open a connection with a server and awaiting response. Socket tries to open
139  * the connection as soon as it is rendered and enters this state.
140  */
141 SceneJS.Socket.STATE_CONNECTING = 2;
142 
143 /** State of Socket when connection is open with server.
144  */
145 SceneJS.Socket.STATE_OPEN = 3;
146 
147 /**
148  * State of Socket in which connection is closed. From here it will attempt to r-open when next rendered.
149  */
150 SceneJS.Socket.STATE_CLOSED = 4;
151 
152 /** State of Socket in which an error occured, either due to failure to open connection, or as signalled by the server.
153  */
154 SceneJS.Socket.STATE_ERROR = -1;
155 
156 /** Enqeues an outgoing message to send when the Socket is next rendered while a connection is open.
157  *
158  * @param message
159  * @returns {this}
160  */
161 SceneJS.Socket.prototype.addMessage = function(message) {
162     this._outMessages.unshift(message);
163     return this;
164 };
165 
166 /** Clears outgoing message queue - messages are not sent.
167  * @returns {this}
168  */
169 SceneJS.Socket.prototype.removeMessages = function() {
170     this._outMessages = [];
171     return this;
172 };
173 
174 // @private
175 SceneJS.Socket.prototype._init = function(params) {
176     this._uri = params.uri;
177     if (params.messages) {
178         this._outMessages = params.messages.reverse();
179     }
180 };
181 
182 // @private
183 SceneJS.Socket.prototype._render = function(traversalContext, data) {
184     if (!this._fixedParams) {
185         this._init(this._getParams(data));
186     }
187     if (!this._uri) {
188         throw SceneJS._errorModule.fatalError(
189                 new SceneJS.errors.InvalidNodeConfigException("SceneJS.Socket uri property not defined"));
190     }
191 
192     /* Socket can close after lack of use, in which case
193      * the node would have to re-open it
194      */
195     if (this._state == SceneJS.Socket.STATE_OPEN) {
196         if (!SceneJS._SocketModule.acquireSocket(this._socketId)) {
197             this._changeState(SceneJS.Socket.STATE_CLOSED);
198         }
199     }
200 
201     if (this._state == SceneJS.Socket.STATE_OPEN) { // Still open, and socket was acquired above
202 
203         /* Process next incoming message then send pending outgoing messages
204          */
205         var _self = this;
206         SceneJS._SocketModule.getNextMessage(
207 
208             /* Error
209              */
210                 function(exception) { // onerror
211                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
212                     SceneJS._errorModule.error(exception);
213                 },
214 
215             /* OK
216              */
217                 function(messageBody) {
218 
219                     if (messageBody.configs) {
220 
221                         /* Configuration message
222                          */
223                         traversalContext = {
224                             appendix : traversalContext.appendix,
225                             insideRightFringe: _self._children.length > 1,
226                             configs : _self._preprocessConfigs(messageBody.configs),
227                             configsModes : _self._configsModes // TODO configsModes in message?
228                         };
229                         data = new SceneJS.Data(data, _self._fixedParams, this._data);
230 
231                     } else {
232 
233                         /* TODO: handle other message types
234                          */
235                         SceneJS._errorModule.error(
236                                 new SceneJS.errors.SocketServerErrorException(
237                                         "SceneJS.Socket server responded with unrecognised message: " + JSON.stringify(messageBody)));
238                     }
239                 });
240         this._sendMessages();
241         this._renderNodes(traversalContext, data);
242         SceneJS._SocketModule.releaseSocket();
243     } else {
244         if (this._state == SceneJS.Socket.STATE_INITIAL || this._state == SceneJS.Socket.STATE_CLOSED) {
245 
246             this._changeState(SceneJS.Socket.STATE_CONNECTING);
247             (function(_self) { // Closure allows this node to receive results
248                 SceneJS._SocketModule.openSocket({ uri: _self._uri },
249                         function(socketId) { // onopen
250                             _self._socketId = socketId;
251                             _self._changeState(SceneJS.Socket.STATE_OPEN);
252                         },
253                         function() { // onclose
254                             _self._changeState(SceneJS.Socket.STATE_CLOSED);
255                         },
256                         function(exception) { // onerror
257                             _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
258                             SceneJS._errorModule.error(exception);
259                         });
260             })(this);
261         }
262         if (this._state == this._STATE_ERROR) { // Socket disabled - TODO: retry?
263         }
264         this._renderNodes(traversalContext, data); // We're assuming socket wont open instantly, ie. during this node visit
265     }
266 };

270 SceneJS.Socket.prototype._preprocessConfigs = function(configs) {
271     var configAction;
272     var funcName;
273     var newConfigs = {};
274     for (var key in configs) {
275         if (configs.hasOwnProperty(key)) {
276             key = key.replace(/^\s*/, "").replace(/\s*$/, "");    // trim
277             if (key.length > 0) {
278                 configAction = key.substr(0, 1);
279                 if (configAction != "#") {  // Property reference
280                     if (configAction == "+") {
281                         funcName = "add" + key.substr(1, 1).toUpperCase() + key.substr(2);
282                     } else if (configAction == "-") {
283                         funcName = "remove" + key.substr(1, 1).toUpperCase() + key.substr(2);
284                     } else {
285                         funcName = "set" + key.substr(0, 1).toUpperCase() + key.substr(1);
286                     }
287                     newConfigs[funcName] = {
288                         isFunc : true,
289                         value : configs[key]
290                     };
291 
292                 } else {
293                     newConfigs[key.substr(1)] = this._preprocessConfigs(configs[key]);
294                 }
295             }
296         }
297     }
298     return newConfigs;
299 };
300 
301 // @private
302 SceneJS.Socket.prototype._changeState = function(newState, params) {
303     params = params || {};
304     params.oldState = this._state;
305     params.newState = newState;
306     this._state = newState;
307     if (this._listeners["state-changed"]) {
308         this._fireEvent("state-changed", params);
309     }
310 };
311 
312 SceneJS.Socket.prototype._onMessage = function(message) {
313     if (this._listeners["msg-received"]) {
314         this._fireEvent("msg-received", message);
315     }
316 };
317 
318 // @private
319 SceneJS.Socket.prototype._sendMessages = function() {
320     if (this._outMessages.length > 0) {
321         var _self = this;
322         SceneJS._SocketModule.sendMessages(
323                 this._outMessages,
324                 function(exception) { // onerror
325                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
326                 },
327                 function() {  // onsuccess
328                     this._outMessages = [];
329                 });
330 
331     }
332 };
