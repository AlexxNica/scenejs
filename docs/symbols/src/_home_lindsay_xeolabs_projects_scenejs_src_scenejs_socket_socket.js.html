1 /**
  2  * @class Binds its subgraph to a WebSocket, providing a server with the ability to add, remove and manipulate content within the subgraph
  3  *
  4  * <p>The SceneJS.Socket node enables a server to dynamically participate in the construction, destruction and
  5  * configuration of its subgraph. It binds the subgraph to a WebSocket through which it exchanges JSON message objects
  6  * with a server, sending messages on events within its subgraph, while receiving messages containing updates to apply
  7  * to its subnodes.</p>
  8  * <p>When a SceneJS.Socket processes incoming messages, it works in a similar manner to a {@link SceneJS.WithConfigs}
  9  * node, pushing configurations from the message body down into the setter methods of identified nodes in the subgraph.</p>
 10  *
 11  * <p>While the format of incoming messages must be a {@link SceneJS.WithConfigs} configuration object, <b>the outgoing
 12  * message format is not part of the SceneJS specification</b> and can be whatever JSON objects the server on the other
 13  * end expects. As for when messages are sent by the scene, that can be triggered by SceneJS node listeners, eg. where a
 14  * Socket node can be configured to send messages as it transitions through the various stages of its life cycle.</p>
 15  *
 16  * <p>You can have multiple Sockets scattered through a scene graph, and may even nest them. This enables you to do some
 17  * fancy things, like have "live assets" that are controlled by the servers that they are downloaded from. Imagine a
 18  * subgraph defining an airplane, that relies on its server to manage its complex physics computations, for example.</p>
 19  *
 20  * <h2>Message Protocol</h2>
 21  *
 22  * <pre><code>
 23  * {
 24  *     error : {String} or undefined when no error,
 25  *     body  : {
 26  *         /* Response from server - either a JSON SceneJS subgraph or a configuration map
 27  *     }
 28  * }
 29  * </code></pre>
 30  */
 31 SceneJS.Socket = function() {
 32     SceneJS.Node.apply(this, arguments);
 33     this._nodeType = "socket";
 34     this._uri = null;
 35     this._autoOpen = true;
 36     this._socketId = null;
 37     this._outMessages = [];
 38     this._state = SceneJS.Socket.STATE_INITIAL;
 39     this._configsModes = {
 40         strictProperties : true,
 41         strictNodes : false
 42     };
 43     if (this._fixedParams) {
 44         this._init(this._getParams());
 45     }
 46 };
 47 
 48 SceneJS._inherit(SceneJS.Socket, SceneJS.Node);

 52 SceneJS.Socket.STATE_INITIAL = 0;
 53 
 54 /**
 55  */
 56 SceneJS.Socket.STATE_CONNECTING = 2;
 57 
 58 /**
 59  */
 60 SceneJS.Socket.STATE_OPEN = 3;
 61 
 62 /**
 63  * State of Socket in which it re-open is pending after it has been closed by SceneJS or server, perhaps due to
 64  * lack of recent use. When next rendered, the Socket will then attempt to reopen and transition to
 65  * {@link #STATE_CONNECTING}.
 66  * @const
 67  */
 68 SceneJS.Socket.STATE_CLOSED = 4;
 69 
 70 /**
 71  */
 72 SceneJS.Socket.STATE_ERROR = -1;
 73

 75 /** Enqeues an outgoing message to send when the Socket is next rendered.
 76  *
 77  * @param message
 78  * @returns {this}
 79  */
 80 SceneJS.Socket.prototype.addMessage = function(message) {
 81     this._outMessages.push(message);
 82     return this;
 83 };
 84 
 85 /** Clears outgoing message queue.
 86  * @returns {this}
 87  */
 88 SceneJS.Socket.prototype.removeMessages = function() {
 89     this._outMessages = [];
 90     return this;
 91 };
 92 
 93 // @private
 94 SceneJS.Socket.prototype._init = function(params) {
 95     this._uri = params.uri;
 96     if (params.messages) {
 97         this._outMessages = params.messages;
 98     }
 99 };
100 
101 // @private
102 SceneJS.Socket.prototype._render = function(traversalContext, data) {
103     if (!this._fixedParams) {
104         this._init(this._getParams(data));
105     }
106     if (!this._uri) {
107         throw SceneJS._errorModule.fatalError(
108                 new SceneJS.InvalidNodeConfigException("SceneJS.Socket uri property not defined"));
109     }
110 
111     /* Socket can close after lack of use, in which case
112      * the node would have to re-open it
113      */
114     if (this._state == SceneJS.Socket.STATE_OPEN) {
115         if (!SceneJS._SocketModule.acquireSocket(this._socketId)) {
116             this._changeState(SceneJS.Socket.STATE_CLOSED);
117         }
118     }
119 
120     if (this._state == SceneJS.Socket.STATE_OPEN) { // Still open, and socket was acquired above
121 
122         /* Process next incoming message then send pending outgoing messages
123          */
124         var _self = this;
125         var message = SceneJS._SocketModule.getNextMessage(
126                 function(exception) { // onerror
127                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
128                     SceneJS._errorModule.error(exception);
129                 });
130         if (message) {
131             if (message.error) {
132                 SceneJS._errorModule.error(
133                         new SceneJS.SocketServerErrorException(
134                                 "SceneJS.Socket server responded with error: " + message.error + ", " + message.body));
135             } else if (message.body) {
136                 // alert(message.body);
137                 traversalContext = {
138                     appendix : traversalContext.appendix,
139                     insideRightFringe: this._children.length > 1,
140                     configs : message.body,
141                     configsModes : this._configsModes // TODO configsModes in message?
142                 };
143                 data = new SceneJS.Data(data, this._fixedParams, this._data);
144             }
145         }
146         this._sendMessages();
147         this._renderNodes(traversalContext, data);
148         SceneJS._SocketModule.releaseSocket();
149     } else {
150         if (this._state == SceneJS.Socket.STATE_INITIAL || this._state == SceneJS.Socket.STATE_CLOSED) {
151 
152             this._changeState(SceneJS.Socket.STATE_CONNECTING);
153             (function(_self) { // Closure allows this node to receive results
154                 SceneJS._SocketModule.openSocket({ uri: _self._uri },
155                         function(socketId) { // onopen
156                             _self._socketId = socketId;
157                             _self._changeState(SceneJS.Socket.STATE_OPEN);
158                         },
159                         function() { // onclose
160                             _self._changeState(SceneJS.Socket.STATE_CLOSED);
161                         },
162                         function(exception) { // onerror
163                             _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
164                             SceneJS._errorModule.error(exception);
165                         });
166             })(this);
167         }
168         if (this._state == this._STATE_ERROR) { // Socket disabled - TODO: retry?
169         }
170         this._renderNodes(traversalContext, data); // We're assuming socket wont open instantly, ie. during this node visit
171     }
172 };
173 
174 // @private
175 SceneJS.Socket.prototype._changeState = function(newState, params) {
176     params = params || {};
177     params.oldState = this._state;
178     params.newState = newState;
179     this._state = newState;
180     if (this._listeners["state-changed"]) { // Optimisation
181         this._fireEvent("state-changed", params);
182     }
183 };
184 
185 // @private
186 SceneJS.Socket.prototype._sendMessages = function() {
187     if (this._outMessages.length > 0) {
188         var _self = this;
189         SceneJS._SocketModule.sendMessages(
190                 this._outMessages,
191                 function(exception) { // onerror
192                     _self._changeState(SceneJS.Socket.STATE_ERROR, { exception: exception });
193                 },
194                 function() {  // onsuccess
195                     this._outMessages = [];
196                 });
197 
198     }
199 };
