  2  @class SceneJS.node
  3  <p>The basic scene node type, providing the ability to connect nodes into parent-child relationships to form scene graphs.</p>
  4  @constructor
  5  Create a new SceneJS.node
  6  @param {SceneJS.node, ...} arguments Zero or more child nodes
  7  */
  8 SceneJS._nodeCount = 0;

 10 SceneJS.Node = function() {
 11     this._nodeType = "node";
 12     this._nodeId = "" + SceneJS._nodeCount++;
 13     this._children = [];
 14     this._fixedParams = true;
 15     this._parent = null;
 16 
 17     /* Used by many node types to track the level at which they can
 18      * memoise internal state. When rendered, a node increments
 19      * this each time it discovers that it can cache more state, so that
 20      * it knows not to recompute that state when next rendered.
 21      * Since internal state is usually dependent on the states of higher
 22      * nodes, this is reset whenever the node is attached to a new
 23      * parent.
 24      *
 25      * private
 26      */
 27     this._memoLevel = 0;
 28 
 29     /* Configure from variable args
 30      */
 31     if (arguments.length > 0) {
 32         for (var i = 0; i < arguments.length; i++) {
 33             var arg = arguments[i];
 34             if (arg._render) {                      // arg is node
 35                 this._children.push(arg);
 36             } else if (i == 0) {                    // arg is config
 37                 if (arg instanceof Function) {      // arg is config function
 38                     this._getParams = arg;
 39                     this._fixedParams = false;
 40                 } else {
 41                     var config = arg;               // arg is config object
 42                     this._fixedParams = true;
 43                     for (var key in arg) {
 44                         if (this._fixedParams && arg[key] instanceof Function) {
 45                             this._fixedParams = false;
 46                         }
 47                     }
 48                     this._getParams = function() {  // Wrap with function to return config object
 49                         return config;
 50                     };
 51                 }
 52             } else {
 53                 SceneJS_errorModule.fatalError(new SceneJS.exceptions.InvalidNodeConfigException
 54                         ("Invalid node parameters - config should be first, IE. node(config, node, node,...)"));
 55             }
 56         }
 57         if (!this._getParams) {
 58             this._getParams = function() {
 59                 return {
 60                 };
 61             };
 62         }
 63     } else {
 64         if (!this._getParams) {
 65             this._getParams = function() {
 66                 return {};
 67             };
 68         }
 69     }
 70 };

 74 SceneJS.Node.prototype._renderNodes = function(traversalContext, data) {
 75     var child;
 76     var len = this._children.length;
 77     if (len) {
 78         for (var i = 0; i < len; i++) {
 79             child = this._children[i];
 80             child._render.call(child, { // Traversal context
 81                 appendix : traversalContext.appendix,
 82                 insideRightFringe: traversalContext.insideRightFringe || (i < len - 1)
 83             }, data);
 84         }
 85     } else {
 86 
 87         /* Leaf node - if on right fringe of tree then
 88          * render appended nodes
 89          */
 90         if (traversalContext.appendix && (!traversalContext.insideRightFringe)) {
 91             len = traversalContext.appendix.length;
 92             for (var i = 0; i < len; i++) {
 93                 child = traversalContext.appendix[i];
 94                 child._render.call(child, { // Traversal context
 95                     appendix : null,
 96                     insideRightFringe: (i < len - 1)
 97                 }, data);
 98             }
 99         }
100     }
101 };

108 SceneJS.Node.prototype._toString2 = function(strList, indent) {
109     for (var i = 0; i < indent; i++) {
110         strList.push("  ");
111     }
112     strList.push(this._nodeType + ":" + this._nodeId);
113 
114     for (var i = 0; i < this._children.length; i++) {
115         this._children[i]._toString2(strList, indent);
116     }
117 };

119 SceneJS.Node.prototype._toString = function() {
120     var strList = [];
121     this._toString2(strList, 0);
122     return strList.join("\n");
123 };

191 /** Inserts a child node

194  * @return {SceneJS.node} The child node
195  */
196 SceneJS.Node.prototype.insertNode = function(node, i) {
197     if (node._parent != null) {
198     }
199     if (i == undefined || i <= 0) {
200         this._children.unshift(node);
201     } else if (i >= this._children.length) {
