1 /**
  2  * @class Data scope that is passed as the single argument to the callback function that many scene node classes may be
  3  * dynamically configured through.
  4  * <p>These are created whenever data is generated within a scene graph, to transport the data down to sub-nodes.</p>
  5  * <p>Methods and nodes that create instances of these include {@link SceneJS.Scene#render}, {@link SceneJS.WithData} and
  6  * {@link SceneJS.ScalarInterpolator}.</p>.
  7  * <p><b>Example:</b></p><p>The example below shows how nested creation of these will form a linked chain of data scopes.
  8  * The outer {@link SceneJS.WithData} node creates one SceneJS.Data with "sizeX" and "sizeY" properties, then the inner
  9  * {@link SceneJS.WithData} chains another SceneJS.Data to that, containing a "sizeZ" property. The dynamic config
 10  * callback on the {@link SceneJS.Scale} node then hunts up the chain to get each of the properties for the
 11  * configuration object it generates.</b></p><pre><code>
 12  *
 13  * var wd new SceneJS.WithData({
 14  *          sizeX: 5,
 15  *          sizeY: 6
 16  *      },
 17  *      new SceneJS.Translate({ x: 100 },
 18  *
 19  *          var wd new SceneJS.WithData({
 20  *              sizeZ: 2
 21  *          },
 22  *          new SceneJS.Scale(function(data) {        // Here's our SceneJS.Data object
 23  *                   return {
 24  *                       x: data.get("sizeX"),
 25  *                       y: data.get("sizeY"),
 26  *                       z: data.get("sizeZ")
 27  *                   }
 28  *          },
 29  *
 30  *              new SceneJS.objects.Cube()
 31  *          )
 32  *      )
 33  *  )
 34  * </code></pre>
 35  *
 36  */
 37 SceneJS.Data = function(_parent, _fixed, _data) {
 38     this._parent = _parent;
 39     this._data = _data || {};
 40     this._fixed = _fixed || (_parent ? _parent._fixed : false);
 41 
 42     /** Hunts up the data scope chain to get the property with the given key, getting it off the
 43      * first data scope that has it.
 44      * @param {String} key Name of property
 45      * @returns {Object} The property
 46      */
 47     this.get = function(key) {
 48         var value = this._data[key];
 49         if ((value == 0) || value) {
 50             return value;
 51         }
 52         if (!this._parent) {
 53             return null;
 54         }
 55         return this._parent.get(key);
 56     };
 57 
 58     /**
 59      * Returns true if all data on the scope chain is fixed, ie. will not change between scene graph traversals.
 60      * @returns {boolean}
 61      */
 62     this.isFixed = function() {
 63         return this._fixed;
 64     };
 65 };
