1 /**
  2  * @class A scene node that asynchronously loads JavaScript content for its subgraph from a server.
  3  * <p>A scene node can load content cross-domain if neccessary. This node is configured with the
  4  * location of a JavaScript file containing a SceneJS definition of the subgraph. When first visited during scene
  5  * traversal, it will begin the load and allow traversal to continue at its next sibling node. When on a subsequent
  6  * visit its subgraph has been loaded, it will then allow traversal to descend into that subgraph to render it.</p>
  7  * <p>You can monitor loads by registering "process-started" and "process-killed" listeners with SceneJS.onEvent().</p>
  8  * <p><b>Live Examples</b></p>
  9  * <li><a target = "other" href="http://bit.ly/scenejs-asset-load">Example 1</a></li>
 10  * </ul>
 11  * <p><b>Usage Example</b></p><p>The SceneJS.Load node shown below loads a fragment of JavaScript-defined scene
 12  * definition cross-domain, via the JSONP proxy located by the <b>uri</b> property on the SceneJS.Scene node.</b></p>
 13  * <pre><code>
 14  * var exampleScene = new SceneJS.Scene({
 15  *
 16  *       // JSONP proxy location - needed only for cros-domain load
 17  *       proxy:"http://scenejs.org/cgi-bin/jsonp_proxy.pl" });
 18  *
 19  *       new SceneJS.Load({
 20  *                 uri:"http://foo.com/my-asset.js"
 21  *            })
 22  *  );
 23  *  </pre></code>
 24  * @extends SceneJS.Node
 25  *  @constructor
 26  *  Create a new SceneJS.Load
 27  *  @param {Object} cfg  Config object or function, followed by zero or more child nodes
 28  */
 29 SceneJS.Load = function() {
 30     SceneJS.Node.apply(this, arguments);
 31     this._nodeType = "load";
 32     this._uri = null;
 33     this._assetParams = null;
 34     this._parser = null;
 35     this._assetNode = null;
 36     this._handle = null;
 37     this._state = SceneJS.Load.prototype._STATE_INITIAL;
 38     //    if (this._fixedParams) {
 39     //        this._init(this._getParams());
 40     //    }
 41 };
 42 
 43 SceneJS._inherit(SceneJS.Load, SceneJS.Node);
 44 
 45 // @private
 46 SceneJS.Load.prototype._STATE_ERROR = -1;         // Asset load or texture creation failed
 47 
 48 // @private
 49 SceneJS.Load.prototype._STATE_INITIAL = 0;        // Ready to start load
 50 
 51 // @private
 52 SceneJS.Load.prototype._STATE_LOADING = 1;        // Load in progress
 53 
 54 // @private
 55 SceneJS.Load.prototype._STATE_LOADED = 2;         // Load completed
 56 
 57 // @private
 58 SceneJS.Load.prototype._STATE_ATTACHED = 3;       // Subgraph integrated
 59 
 60 // @private
 61 SceneJS.Load.prototype._init = function(params) {
 62     if (!params.uri) {
 63         SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
 64                 ("SceneJS.Load parameter expected: uri"));
 65     }
 66     this._uri = params.uri;
 67     this._serverParams = params.serverParams;
 68     this._parser = params.parser;
 69 };
 70 
 71 // @private
 72 SceneJS.Load.prototype._visitSubgraph = function(data) {
 73     var traversalContext = {
 74         appendix : this._children
 75     };
 76     this._assetNode._render.call(this._assetNode, traversalContext, data);
 77 };
 78 
 79 // @private
 80 SceneJS.Load.prototype._parse = function(data, onError) {
 81     if (!data._render) {
 82         onError(data.error || "unknown server error");
 83         return null;
 84     } else {
 85         return data;
 86     }
 87 };
 88 
 89 // @private
 90 SceneJS.Load.prototype._render = function(traversalContext, data) {
 91     if (!this._uri) {
 92         this._init(this._getParams(data));
 93     }
 94     if (this._state == this._STATE_ATTACHED) {
 95         if (!SceneJS_loadModule.getAsset(this._handle)) { // evicted from cache - must reload
 96             this._state = this._STATE_INITIAL;
 97         }
 98     }
 99     switch (this._state) {
100         case this._STATE_ATTACHED:
101             this._visitSubgraph(data);
102             break;
103 
104         case this._STATE_LOADING:
105             break;
106 
107         case this._STATE_LOADED:
108             SceneJS_loadModule.assetLoaded(this._handle);  // Finish loading - kill process
109             this._state = this._STATE_ATTACHED;
110             this._visitSubgraph(data);
111             break;
112 
113         case this._STATE_INITIAL:
114             this._state = this._STATE_LOADING;
115 
116             /* Asset not currently loaded or loading - load it
117              */
118             var _this = this;
119             this._handle = SceneJS_loadModule.loadAsset(// Process killed automatically on error or abort
120                     this._uri,
121                     this._serverParams || {
122                         format: "scenejs"
123                     },
124                     this._parser || this._parse,
125                     function(asset) { // Success
126                         _this._assetNode = asset;   // Asset is wrapper created by SceneJS.createNode
127                         _this._state = _this._STATE_LOADED;
128                     },
129                     function() { // onTimeout
130                         _this._state = _this._STATE_ERROR;
131                         SceneJS_errorModule.error(
132                                 new SceneJS.LoadTimeoutException("Load timed out - uri: " + _this._uri));
133                     },
134                     function(e) { // onError - SceneJS_loadModule has killed process
135                         _this._state = _this._STATE_ERROR;
136                         e.message = "Load failed - " + e.message + " - uri: " + _this._uri;
137                         SceneJS_errorModule.error(e);
138                     });
139             break;
140 
141         case this._STATE_ERROR:
142             break;
143     }
144 };
145 
146 /** Returns a new SceneJS.Load instance
147  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Load constructor
148  * @returns {SceneJS.Load}
149  */
150 SceneJS.load = function() {
151     var n = new SceneJS.Load();
152     SceneJS.Load.prototype.constructor.apply(n, arguments);
153     return n;
154 };
