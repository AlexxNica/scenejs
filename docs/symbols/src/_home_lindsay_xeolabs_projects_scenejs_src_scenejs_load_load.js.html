1 /** Asynchronously loads a subgraph cross-domain. This node is configured with the location of a file
  2  * that describes the subgraph. When first visited, it will start the load, then finish the load and
  3  * integrate the subgraph on a future visit.
  4  *
  5  * Can be configured with child nodes to visit while the load is in progress, which will be replaced by
  6  * the loaded subgraph.
  7  */
  8 SceneJS.load = function() {
  9     var cfg = SceneJS._utils.getNodeConfig(arguments);
 10 
 11     var params;
 12 
 13     var backend = SceneJS._backends.getBackend("load");
 14     var logging = SceneJS._backends.getBackend("logging");
 15     var errorBackend = SceneJS._backends.getBackend("error");
 16     var assetNode;
 17     var handle;
 18 
 19     const STATE_INITIAL = 0;        // Ready to start load
 20     const STATE_LOADING = 2;        // Load in progress
 21     const STATE_LOADED = 3;         // Load completed
 22     const STATE_ATTACHED = 4;       // Subgraph integrated
 23     const STATE_ERROR = -1;         // Asset load or texture creation failed
 24 
 25     var state = STATE_INITIAL;
 26 
 27     function sceneJSParser(data, onError) {
 28         if (!data.___isSceneJSNode) {
 29             onError(data.error || "unknown server error");
 30             return null;
 31         } else {
 32             return data;
 33         }
 34     }
 35 
 36     function visitSubgraph(params, data) {
 37         var traversalContext = {
 38             appendix : cfg.children
 39         };
 40         if (params) { // Parameters for asset - supply in a new child data
 41             var childData = SceneJS._utils.newScope(data, cfg.fixed);
 42             for (var key in params.params) {
 43                 childData.put(key, params.params[key]);
 44             }            
 45             assetNode._render.call(assetNode, traversalContext, childData);
 46         } else {
 47             assetNode._render.call(assetNode, traversalContext, data);
 48         }
 49     }
 50 
 51     return SceneJS._utils.createNode(
 52             "load",
 53             cfg.children,
 54 
 55             new (function() {
 56 
 57                 this._render = function(traversalContext, data) {
 58 
 59                     if (!params) {
 60                         params = cfg.getParams(data);
 61                         if (!params.uri) {
 62                             errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException
 63                                     ("Scene definiton error - mandatory SceneJS.load parameter missing: uri"));
 64                         }
 65                     }
 66 
 67                     if (state == STATE_ATTACHED) {
 68                         if (!backend.getAsset(handle)) {
 69                             state = STATE_INITIAL;
 70                         }
 71                     }
 72 
 73                     switch (state) {
 74                         case STATE_ATTACHED:
 75                             visitSubgraph(params.params, data);
 76                             break;
 77 
 78                         case STATE_LOADING:
 79                             break;
 80 
 81                         case STATE_LOADED:
 82                             backend.assetLoaded(handle);  // Finish loading - kill process
 83                             state = STATE_ATTACHED;
 84                             visitSubgraph(params.params, data);
 85                             break;
 86 
 87                         case STATE_INITIAL:
 88                             state = STATE_LOADING;
 89 
 90                             /* Asset not currently loaded or loading - load it
 91                              */
 92                             handle = backend.loadAsset(// Process killed automatically on error or abort
 93                                     params.uri,
 94                                     params.serverParams || {
 95                                         format: "scenejs"
 96                                     },
 97                                     params.parser || sceneJSParser,
 98                                     function(asset) { // Success
 99                                         assetNode = asset;   // Asset is wrapper created by SceneJS._utils.createNode
100                                         state = STATE_LOADED;
101                                     },
102                                     function() { // onTimeout
103                                         state = STATE_ERROR;
104                                         errorBackend.error(
105                                                 new SceneJS.exceptions.AssetLoadTimeoutException(
106                                                         "SceneJS.load timed out - uri: " + params.uri));
107                                     },
108                                     function(msg) { // onError - backend has killed process
109                                         state = STATE_ERROR;
110                                         errorBackend.error("SceneJS.load failed - " + msg + " - uri: " + params.uri);
111                                     });
112                             break;
113 
114                         case STATE_ERROR:
115                             break;
116                     }
117                 };
118             })());
119 };
