1 /**
  2  * @class A scene node that asynchronously loads its subgraph from a server.
  3  * <p>This basic SceneJS.Load type loads a JavaScript file that decribes content through the SceneJS API.
  4  * Subclasses of this (such as {@link SceneJS.LoadCollada} extend it to load other file types.</p>
  5  * <p>The node has six states it can be in, which may be queried with {@link #getState}:</p>
  6  * <ul><li>{@link #STATE_INITIAL} in which it is about to make an asynchronous request for the content.</li>
  7  * <li>{@link #STATE_LOADING} in which it has made the request and is awaiting the response, which it will receive
  8  * at any time, either within a scene render or between scene renders. In this state it has nothing to render
  9  * and will pass scene traversal on to its next sibling.</li>
 10  * <li>{@link #STATE_LOADED} in which it has successfully received its content and parsed it into a subgraph.
 11  * From here the node will transition to {@link #STATE_RENDERED} as soon as the subgraph has been rendered
 12  * and traversal has arrived back at this node on the way back up the graph. It may instead transition back
 13  * to {@link #STATE_INITIAL} as soon as it has not been rendered in a while and has then had its subgraph destroyed
 14  * by SceneJS to reclaim memory.</li>
 15  * <li>{@link #STATE_RENDERED} in which it has successfully received its content, parsed it into a subgraph, and been rendered
 16  * at least once. From here the node will transition back to {@link #STATE_INITIAL} as soon as it has not been rendered in
 17  * a while and has then had its subgraph destroyed by SceneJS to reclaim memory.</li>
 18  * <li>{@link #STATE_FREED} in which it has been dormant too long and SceneJS has reclaimed memory by destroying
 19  * the subgraph. When next rendered, it will then repeat its load request and transition
 20  * to {@link #STATE_LOADING}.</li>
 21  * <li>{@link #STATE_ERROR} if SceneJS has permanently inactivated the node after load failed</li></ul>
 22  * <p><b>Loading Cross-Domain</b></p>
 23  * <p>When SceneJS is configured with a JSONP strategy via {@link SceneJS.setJSONPStrategy}, you can perform the
 24  * load cross-domain. Otherwise, the URL of the content must be at the same domain as the scene definition's JavaScript file
 25  * in order to not violate the browser' same-domain security policy.
 26  * <p><b>Live Examples</b></p>
 27  * <li><a target = "other" href="http://bit.ly/scenejs-asset-load">Example 1</a></li>
 28  * </ul>
 29  * <p><b>Usage Example</b></p><p>The SceneJS.Load node shown below loads a fragment of JavaScript-defined scene
 30  * definition cross-domain, via the JSONP proxy located by the <b>loadProxy</b> property on the {@link SceneJS.Scene} node.</b></p>
 31  * <pre><code>
 32  * var exampleScene = new SceneJS.Scene({
 33  *
 34  *       // JSONP proxy location - needed only for cros-domain load
 35  *
 36  *       loadProxy:"http://scenejs.org/cgi-bin/jsonp_proxy.pl" });
 37  *
 38  *       new SceneJS.Load({
 39  *                 uri : "http://foo.com/my-asset.js",
 40  *
 41  *              // Optional timeout - falls back on any loadTimoutSecs that was configured on the
 42  *              // SceneJS.Scene at the root of the scene graph, or the default 180 seconds if
 43  *              // none configured there
 44  *
 45  *                 loadTimeoutSecs : 180
 46  *            })
 47  *  );
 48  *  </pre></code>
 49  *
 50  * @events
 51  * @extends SceneJS.Node
 52  * @constructor
 53  * Creates a new SceneJS.Load
 54  *  @param {Object} [cfg] Static configuration object
 55  * @param {String} cfg.uri URI of file to load
 56  * @param {int} [cfg.timeoutSecs] Timeout - falls back on any loadTimoutSecs that was configured on the {@link SceneJS.Scene}
 57  * at the root of the scene graph, or the default 180 seconds if none configured there
 58  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 59  * @param {...SceneJS.Node} [childNodes] Child nodes
 60  */
 61 SceneJS.Load = function() {
 62     SceneJS.Node.apply(this, arguments);
 63     this._nodeType = "load";
 64     this._uri = null;
 65     this._format = "json";
 66     this._assetParams = null;
 67     this._parser = null;
 68     this._assetNode = null;
 69     this._handle = null;
 70     this._state = SceneJS.Load.STATE_INITIAL;
 71     this._loadTimoutSecs = null;  // default while null
 72 };

 76  * @event
 77  * Fires when the node changes state
 78  * @param @param {Object} e
 79  * @param {Boolean} [e.withIce=false]
 80  */
 81 SceneJS._inherit(SceneJS.Load, SceneJS.Node);

 84 /**
 85  * State in which load has failed. This can be either due to HTTP error (eg. HTTP 404) or
 86  * failure to parse the response. The node will now be permanently inactive and stuck in this state.
 87  * @const
 88  */
 89 SceneJS.Load.STATE_ERROR = -1;
 90 
 91 /**
 92  * State in which load is pending because node has not been rendered yet and hence has
 93  * not made its load request yet. When next rendered, it will then make its load request
 94  * and transition to {@link #STATE_LOADING}.
 95  * @const
 96  */
 97 SceneJS.Load.STATE_INITIAL = 0;
 98 
 99 /**
100  * State in which node is awaiting a response to its load request. When the response arrives (asyncronously, ie.
101  * either during or between scene renders), the node will then transition to  either {@link #STATE_LOADED}
102  * or {@link #STATE_ERROR}, depending on whether or not it successfully parses the response. If the
103  * response does not arrive or parse within the timeout period (180 seconds by default, unless configured) it will
104  * transition to {@link STATE_ERROR}.
105  * @const
106  */
107 SceneJS.Load.STATE_LOADING = 1;
108 
109 /**
110  * State in which node has successfully received its content and parsed it into a subgraph.
111  * From here the node will transition to {@link #STATE_RENDERED} as soon as the subgraph has been rendered
112  * and traversal has arrived back at this node on the way back up the graph.
113  * @const
114  */
115 SceneJS.Load.STATE_LOADED = 2;
116 
117 /**
118  * State in which node has successfully received its content, parsed it into a subgraph, and been rendered
119  * at least once. From here the node will transition back to {@link #STATE_INITIAL} as soon as it has not been rendered in
120  * a while and has then had its subgraph destroyed by SceneJS to reclaim memory.
121  * @const
122  */
123 SceneJS.Load.STATE_RENDERED = 3;
124 
125 /**
126  * State in which load is pending after node has had its subgraph destroyed by SceneJS to reclaim memory after
127  * not having been rendered for some time (Eg. is within a {@link SceneJS.Boundary} that has
128  * not intersected the view frustum for a while, or perhaps within a {@link SceneJS.Selector} that has
129  * not selected it lately). When next rendered, it will then repeat its load request and transition
130  * to {@link #STATE_LOADING}.
131  * @const
132  */
133 SceneJS.Load.STATE_FREED = 4;
134 
135 /**
136  * Returns the node's current state. Possible states are {@link #STATE_INITIAL}, {@link #STATE_LOADING},
137  * {@link #STATE_LOADED}, {@link #STATE_FREED} and {@link #STATE_ERROR}.
138  * @returns {int} The state
139  */
140 SceneJS.Load.prototype.getState = function() {
141     return this._state;
142 };
143 
144 // @private
145 SceneJS.Load.prototype._init = function(params) {
146     if (!params.uri) {
147         throw SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
148                 ("SceneJS.Load parameter expected: uri"));
149     }
150     this._uri = params.uri;
151     this._loadTimeoutSecs = params._loadTimeoutSecs;
152 };
153 
154 // @private
155 SceneJS.Load.prototype._visitSubgraph = function(data) {
156 
157     /* After completion of the right-to-left, depth-first traversal of the Load node's imported subgraph,
158      * we'll render as an appendum the explicitly defined children of the Load node. So we'll configure
159      * the traversal context (which tracks various states during graph traversal) with the appended children.
160      * See _renderNodes for more details on how all that works.
161      */
162     var traversalContext = {
163         appendix : this._children // Load node's children become those of right-most leaf node
164     };
165     this._assetNode._render.call(this._assetNode, traversalContext, data);
166 };
167 
168 /**
169  * Parser callback to convert the response content into a SceneJS subgraph. For this SceneJS.Load base class,
170  * the format of the data could be either raw JavaScript in the case of a same-domain load, or a pre-evaluated
171  * SceneJS.Node or subtype for a cross-domain JSONP request.
172  * @private
173  */
174 SceneJS.Load.prototype._parse = function(cfg) {
175     if (cfg.data._render) {  // Cross-domain JSONP
176         return cfg.data;
177     }
178     if (data.error) {
179         cfg.onError(cfg.data.error);
180     }
181     try {
182         var evalData = eval(cfg.data);  // Same-domain
183         if (evalData.error) {           // TODO: This is specific to the old SceneJS proxy - remove? 
184             cfg.onError(evalData.error);
185         }
186         return data;
187     } catch (e) {
188         cfg.onError("Error parsing response: " + e);
189     }
190 };
191 
192 // @private
193 SceneJS.Load.prototype._changeState = function(newState) {
194     var oldState = this._state;
195     this._state = newState;
196     this._fireEvent("state-changed", { oldState: oldState, newState: newState });
197 };
198 
199 // @private
200 SceneJS.Load.prototype._render = function(traversalContext, data) {
201     if (!this._uri) {     // One-shot dynamic config
202         this._init(this._getParams(data));
203     }
204     if (this._state == SceneJS.Load.STATE_LOADED || this._state == SceneJS.Load.STATE_RENDERED) {
205         if (!SceneJS_loadModule.getAsset(this._handle)) { // evicted from cache - must reload
206             this._changeState(SceneJS.Load.STATE_FREED);
207         }
208     }
209 
210     switch (this._state) {
211 
212         case SceneJS.Load.STATE_RENDERED:
213             this._visitSubgraph(data);
214             break;
215 
216         case SceneJS.Load.STATE_LOADED:
217             this._visitSubgraph(data);
218             this._changeState(SceneJS.Load.STATE_RENDERED);
219             break;
220 
221         case SceneJS.Load.STATE_LOADING:
222             break;
223 
224         case SceneJS.Load.STATE_INITIAL:
225         case SceneJS.Load.STATE_FREED:
226             var _this = this;
227             this._changeState(SceneJS.Load.STATE_LOADING);
228             this._handle = SceneJS_loadModule.loadAsset(// Process killed automatically on error or abort
229                     this._uri,
230                     this._format,
231                     this._loadTimeoutSecs, // default when null
232                     this._serverParams || {
233                         format: "scenejs"
234                     },
235                     this._parse,
236                     function(asset) { // Success
237                         _this._assetNode = asset;
238                         SceneJS_loadModule.assetLoaded(_this._handle);
239                         _this._changeState(SceneJS.Load.STATE_LOADED);
240                     },
241                     function() { // onTimeout
242                         _this._changeState(SceneJS.Load.STATE_ERROR);
243                         SceneJS_errorModule.error(
244                                 new SceneJS.LoadTimeoutException("Load timed out - uri: " + _this._uri));
245                     },
246                     function(e) { // onError - SceneJS_loadModule has killed process
247                         _this._changeState(SceneJS.Load.STATE_ERROR);
248                         e.message = "Load failed - " + e.message + " - uri: " + _this._uri;
249                         SceneJS_errorModule.error(e);
250                     });
251             break;
252 
253         case SceneJS.Load.STATE_ERROR:
254             break;
255     }
256 };

258 /** Factory function that returns a new {@link SceneJS.Load} instance
259  *  @param {Object} [cfg] Static configuration object
260  * @param {String} cfg.uri URI of file to load
261  * @param {int} [cfg.timeoutSecs] Timeout - falls back on any loadTimoutSecs that was configured on the {@link SceneJS.Scene}
262  * at the root of the scene graph, or the default 180 seconds if none configured there
263  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
264  * @param {...SceneJS.Node} [childNodes] Child nodes
265  * @returns {SceneJS.Load}
266  */
267 SceneJS.load = function() {
268     var n = new SceneJS.Load();
269     SceneJS.Load.prototype.constructor.apply(n, arguments);
270     return n;
271 };
