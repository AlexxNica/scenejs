1 /**
  2  * @class A scene node that marks its subgraph as a "symbol" which can be then instanced with SceneJS.Instance nodes.
  3  *
  4  * <p>This node type is useful for keeping scene size small, while also simplifying editing of a scene; when you edit
  5  * content within a Symbol, all instances of the Symbol update to reflect the edits.</p>
  6  *
  7  * <p>When rendered, SceneJS registers this node against its specified name and prevents SceneJS from traversing
  8  * into its subgraph. The content defined within the Symbol will therefore only be rendered when it is instanced.
  9  * The registered name will be actually be the concatenation of the specified name with the namespace formed by
 10  * any enclosing SceneJS.Name nodes. When SceneJS then finds a SceneJS.Instance node that refers to the registered name,
 11  * it will instantiate the Symbol's child nodes as if they were children of the Instance node.</p>
 12  *
 13  * <p>Beware potential performance penalties for using Symbols and Instances. Within every subgraph, SceneJS
 14  * internally memoises whatever state it determines will not change between scene traversals. SceneJS may therefore be
 15  * restricted in what state it can memoise within a Symbol's subgraph when it is likely to be dynamically affected by
 16  * the different scene locations it which it is instanced.</p>
 17  *
 18  * <p><b>Live Examples</b></p>
 19  * <ul><li><a target = "other" href="http://bit.ly/9d8wLu">Example 1</a></li></ul>
 20  * <p><b>Example Usage</b></p><p>Here we're defining a Symbol in a Name, then instantiating it three times with
 21  * Instance nodes to show variations on how an Instance node can refer to a Symbol, relative to a namespace
 22  * created by a Name node:</b></p><pre><code>
 23  *
 24  *  // Define a "teapot" symbol inside a namespace.
 25  *  // Note that the Name is not mandatory.
 26  *
 27  *  SceneJS.name({name: "mySymbols"},
 28  *
 29  *       SceneJS.symbol({ name: "teapot" },
 30  *           SceneJS.objects.teapot()
 31  *       ),
 32  *
 33  *       // Instance the teapot Symbol from inside the namespace.
 34  *       // See how the symbol reference is relative, where it
 35  *       // does not start with a '/'.
 36  *
 37  *       SceneJS.instance({name: "teapot" })
 38  * ),
 39  *
 40  * // Instance the teapot Symbol again, from outside the namespace
 41  *
 42  * SceneJS.instance({name: "mySymbols/teapot"}),
 43  *
 44  * // Instance the teapot Symbol one more time from outside the
 45  * // namespace to show how an absolute path can be specified to
 46  * // the Symbol
 47  *
 48  * SceneJS.instance({name: "/mySymbols/teapot"})
 49  *
 50  * // ...
 51  * </pre></code>
 52  *  @extends SceneJS.Node
 53  * @constructor
 54  * Create a new SceneJS.Symbol
 55  * @param {Object} config  Config object or function, followed by zero or more child nodes
 56  */
 57 SceneJS.Symbol = function() {
 58     SceneJS.Node.apply(this, arguments);
 59     this._name = "unnamed";
 60     if (this._fixedParams) {
 61         this._init(this._getParams());
 62     }
 63 };
 64 
 65 SceneJS._inherit(SceneJS.Symbol, SceneJS.Node);
 66 
 67 /**
 68  Sets the symbol name. When the given value is undefined, the name will default to "unnamed".
 69  @function setName
 70  @param {string} name
 71  @returns {SceneJS.Symbol} This symbol node
 72  */
 73 SceneJS.Symbol.prototype.setName = function(name) {
 74     this._name = name || "unnamed";
 75     return this;
 76 };
 77 
 78 /**
 79  * Returns the symbol name. The name will be "unnamed" if none is specified.
 80  * @function {string} getName
 81  * @returns {string} The symbol name
 82  */
 83 SceneJS.Symbol.prototype.getName = function() {
 84     return this._name;
 85 };
 86 
 87 // @private
 88 SceneJS.Symbol.prototype._init = function(params) {
 89     if (params.name) {
 90         this.setName(params.name);
 91     }
 92 };
 93 
 94 // @private
 95 SceneJS.Symbol.prototype._render = function(traversalContext, data) {
 96     if (!this._fixedParams) {
 97         this._init(this._getParams(data));
 98     }
 99     SceneJS_instancingModule.createSymbol(this._name, this);
100 };
101 
102 /** Returns a new SceneJS.Symbol symbol
103  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Symbol constructor
104  * @returns {SceneJS.Symbol}
105  */
106 SceneJS.symbol = function() {
107     var n = new SceneJS.Symbol();
108     SceneJS.Symbol.prototype.constructor.apply(n, arguments);
109     return n;
110 };
