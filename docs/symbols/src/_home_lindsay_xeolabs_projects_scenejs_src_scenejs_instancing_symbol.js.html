1 /**
  2  * @class A scene node that marks its subgraph as a "symbol" which can be then instanced with {@link SceneJS.Instance} nodes.
  3  *
  4  * <p>This node type is useful for keeping scene size small, while also simplifying editing of a scene; when you edit
  5  * content within a Symbol, all instances of the Symbol update to reflect the edits.</p>
  6  *
  7  * <p>When rendered, SceneJS registers this node against its sub-identifier (SID) and prevents SceneJS from traversing
  8  * into its subgraph. The content defined within the Symbol will therefore only be rendered when it is instanced.
  9  * The registered identity will be actually be the concatenation of the SID with the namespace formed by the SIDs
 10  * any enclosing nodes. When SceneJS then finds a {@link SceneJS.Instance} node with a URL that refers to the
 11  * registered identity, it will instantiate the Symbol's child nodes as if they were children of the {@link SceneJS.Instance}
 12  *  node.</p>
 13  *
 14  * <p>Beware potential performance penalties for using Symbols and {@link SceneJS.Instances}. Within every subgraph, SceneJS
 15  * internally memoises whatever state it determines will not change between scene traversals. SceneJS may therefore be
 16  * restricted in what state it can memoise within a Symbol's subgraph when it is likely to be dynamically affected by
 17  * the different scene locations it which it is instanced.</p>
 18  *
 19  * <p><b>Live Examples</b></p>
 20  * <ul><li><a target = "other" href="http://bit.ly/scenejs-view-select">Example 1 - Instancing and Branching - Switchable Viewpoint</a></li></ul>
 21  * <p><b>Example Usage</b></p><p>Here we're defining a Symbol in a {@link SceneJS.Node}, then instantiating it three times with
 22  * {@link SceneJS.Instance} nodes to show variations on how an {@link SceneJS.Instance} node can refer to a Symbol, relative to
 23  * a namespace created by a {@link SceneJS.Node}:</b></p><pre><code>
 24  * var scene = new SceneJS.Scene(
 25  *
 26  *      // ...
 27  *
 28  *      // Define a "teapot" symbol inside a namespace.
 29  *
 30  *      new SceneJS.Node({ sid: "mySymbols"},
 31  *
 32  *          new SceneJS.Symbol({ sid: "teapot" },
 33  *              new SceneJS.objects.Teapot()
 34  *          ),
 35  *
 36  *          // Instance the teapot Symbol from inside the namespace.
 37  *          // See how the symbol reference is relative, where it
 38  *          // does not start with a '/'.
 39  *
 40  *          new SceneJS.Instance({ uri: "teapot" })
 41  *      ),
 42  *
 43  *      // Instance the teapot Symbol again, from outside the namespace
 44  *
 45  *      new SceneJS.Instance({ uri: "mySymbols/teapot"}),
 46  *
 47  *      // Instance the teapot Symbol one more time from outside the
 48  *      // namespace to show how an absolute path can be specified to
 49  *      // the Symbol
 50  *
 51  *      new SceneJS.Instance({ uri: "/mySymbols/teapot" })
 52  *
 53  *      // ...
 54  * );
 55  * </pre></code>
 56  *  @extends SceneJS.Node
 57  * @since Version 0.7.4
 58  * @constructor
 59  * Create a new SceneJS.Symbol
 60  * @param {Object} [cfg] Static configuration object
 61  * @param {String} [cfg.name="unnamed"]
 62  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 63  * @param {...SceneJS.Node} [childNodes] Child nodes
 64  * @returns {SceneJS.Instance}
 65  */
 66 SceneJS.Symbol = function() {
 67     SceneJS.Node.apply(this, arguments);
 68     this._nodeType = "symbol";
 69 };
 70 
 71 SceneJS._inherit(SceneJS.Symbol, SceneJS.Node);
 72 
 73 // @private
 74 SceneJS.Symbol.prototype._render = function(traversalContext, data) {
 75     if (!this._sid) {
 76         throw SceneJS._errorModule.fatalError(new SceneJS.NodeConfigExpectedException
 77                 ("SceneJS.Symbol parameter expected: sid"));
 78     }
 79     this._sidPath = SceneJS._instancingModule.getName();  // Path to this Symbol, without this Symbol's SID
 80     SceneJS._instancingModule.createSymbol(this._sid, this);
 81 };
 82 
 83 
 84 /**  Factory function that returns a new {@link SceneJS.Symbol}
 85  * @param {Object} [cfg] Static configuration object
 86  * @param {String} [cfg.name="unnamed"]
 87  * @param {...SceneJS.Node} [childNodes] Child nodes
 88  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 89  * @returns {SceneJS.Symbol}
 90  * @since Version 0.7.4
 91  */
 92 SceneJS.symbol = function() {
 93     var n = new SceneJS.Symbol();
 94     SceneJS.Symbol.prototype.constructor.apply(n, arguments);
 95     return n;
 96 };
