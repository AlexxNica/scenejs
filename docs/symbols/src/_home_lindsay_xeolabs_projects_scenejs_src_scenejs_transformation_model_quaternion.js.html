1 /**
  2  * @class Scene node that provides quaternion-based rotation transformation.
  3 
  4  * <p>This node provides a convenient way to define a 3D rotation that can be rotated continually on any axis without
  5  * gimbal lock or significant numeric instability.</p>
  6  * <p><b>Example 1</b></p><p>Below is a Quaternion that defines a rotation for the nodes in its subgraph. Since these are
  7  * used for rotation in SceneJS, we'll think of a quaternion in the same terms as the parameters of a
  8  * {@link SceneJS.Rotate} node. This example's  <em>x, y, z</em> and <em>angle</em> parameters define the base
  9  * rotation that this quaternion starts off at, in this case none, then the optional <em>rotations</em> array
 10  * defines a sequence of rotations to apply on top of that. Finally, we apply one more rotation to the Quaternion node
 11  * instance through its {@link #rotate} method. </p>
 12  * </p><pre><code>
 13  * var q = new SceneJS.Quaternion({
 14  *
 15  *         // "Base" rotation
 16  *
 17  *         x : 0.0, y : 0.0, z : 0.0, angle : 0.0,      // No rotation
 18  *
 19  *         // Sequence of rotations to apply on top of the base rotation
 20  *
 21  *         rotations: [
 22  *                 { x : 0, y : 0, z : 1, angle : 45 }, // Rotate 45 degrees about Z the axis
 23  *                 { x : 1, y : 0, z : 0, angle : 20 }, // Rotate 20 degrees about X the axis
 24  *                 { x : 0, y : 1, z : 0, angle : 90 }, // Rotate 90 degrees about Y the axis
 25  *              ]
 26  *          },
 27  *
 28  *          // .. Child nodes ...
 29  *     });
 30  *
 31  * // rotate one more time, 15 degrees about the Z axis
 32  *
 33  * q.rotate({ x : 0, y : 0, z : 1, angle : 15 });
 34  * </pre></code>
 35  * <p>Quaternions are designed to be animated. Typically, we would dynamically provide rotation updates to them at
 36  * render time, either from an interpolator node or human interaction.</p>
 37  * <p><b>Example 2</b></p><p>Below is a Quaternion that is dynamically configured with a callback that applies
 38  * rotation updates that are injected into the scene each time it is rendered. Note that the callback does not define
 39  * the base rotation - if it did, then the Quaternion would reset before each new rotation is applied, rather than
 40  * accumulate the rotations as intended. Note that the rotations could well ge generated by mouse drags to simulate
 41  * a trackball.</p>
 42  * <pre><code>
 43  *  var exampleScene = new SceneJS.Scene(
 44  *
 45  *      // ... sibling nodes
 46  *
 47  *      new SceneJS.Quaternion(
 48  *            function(data) {
 49  *                      return {
 50  *                         rotations: data.get("rotations");
 51  *                      };
 52  *           },
 53  *
 54  *           // ... chld nodes ...
 55  *      )
 56  *
 57  *      //... sibling nodes
 58  * );
 59  *
 60  * exampleScene.render({ rotations: [ { x : 0, y : 0, z : 1, angle : 45 } ] });
 61  * exampleScene.render({ rotations: [ { x : 1, y : 0, z : 0, angle : 20 } ] });
 62  * exampleScene.render({ rotations: [ { x : 0, y : 1, z : 0, angle : 90 } ] });
 63  * </code></pre>
 64  * @extends SceneJS.Node
 65  * @constructor
 66  * Create a new SceneJS.Quaternion
 67  * @param {Object} [cfg] Static configuration object
 68  * @param {float} [cfg.x=0.0] Base rotation vector X axis
 69  * @param {float} [cfg.y=0.0] Base rotation vector Y axis
 70  * @param {float} [cfg.z=0.0] Base rotation vector Z axis
 71  * @param {float} [cfg.angle=0.0] Base rotation angle in degrees
 72  * @param {[{x:float, y:float, z:float, angle:float}]} [cfg.rotations=[]] Sequence of rotations to apply on top of the base rotation
 73  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function that can return the same signature as the static config
 74  * @param {...SceneJS.Node} [childNodes] Child nodes
 75  */
 76 SceneJS.Quaternion = function() {
 77     SceneJS.Node.apply(this, arguments);
 78     this._nodeType = "quaternion";
 79     this._mat = null;
 80     this._xform = null;
 81     this._q = SceneJS_math_identityQuaternion();
 82     if (this._fixedParams) {
 83         this._init(this._getParams());
 84     }
 85 };
 86 
 87 SceneJS._inherit(SceneJS.Quaternion, SceneJS.Node);
 88 
 89 /** Sets the base rotation. Resets the quaternion when you supply no arguments.
 90  *
 91  * @param {Object} [q={ x: 0, y: 0, z: 0, angle: 0 }] Rotation vector and angle in degrees
 92  * @param {float} [q.x=0.0] Rotation vector X axis
 93  * @param {float} [q.y=0.0] Rotation vector Y axis
 94  * @param {float} [q.z=0.0] Rotation vector Z axis
 95  * @param {float} [q.angle=0.0] Rotation angle in degrees
 96  * @returns {SceneJS.Quaternion} this
 97  */
 98 SceneJS.Quaternion.prototype.setRotation = function(q) {
 99     q = q || {};
100     this._q = SceneJS_math_angleAxisQuaternion(q.x || 0, q.y || 0, q.z || 0, q.angle || 0);
101     this._memoLevel = 0;
102     return this;
103 };
104 
105 /** Applies a rotation to the quaternion
106  *
107  * @param {Object} [q={ x: 0, y: 0, z: 0, angle: 0 }] Rotation vector and angle in degrees
108  * @param {float} [q.x=0.0] Rotation vector X axis
109  * @param {float} [q.y=0.0] Rotation vector Y axis
110  * @param {float} [q.z=0.0] Rotation vector Z axis
111  * @param {float} [q.angle=0.0] Rotation angle in degrees
112  * @returns {SceneJS.Quaternion} this
113  */
114 SceneJS.Quaternion.prototype.rotate = function(q) {
115     this._q = SceneJS_math_mulQuaternions(SceneJS_math_angleAxisQuaternion(q.x || 0, q.y || 0, q.z || 0, q.angle || 0), this._q);
116     this._memoLevel = 0;
117     return this;
118 };
119 
120 SceneJS.Quaternion.prototype._init = function(params) {
121     if (params.q) {
122         this.setRotation(params.q);
123     }
124     if (params.rotations) {
125         for (var i = 0; i < params.rotations.length; i++) {
126             this.rotate(params.rotations[i]);
127         }
128     }
129 };
130 
131 SceneJS.Quaternion.prototype._render = function(traversalContext, data) {
132     if (this._memoLevel == 0) {
133         if (!this._fixedParams) {
134             this._init(this._getParams(data));
135         } else {
136             this._memoLevel = 1;
137         }
138         this._mat = SceneJS_math_newMat4FromQuaternion(this._q);
139     }
140     var superXform = SceneJS_modelTransformModule.getTransform();
141     if (this._memoLevel < 2) {
142         var instancing = SceneJS_instancingModule.instancing();
143         var tempMat = SceneJS_math_mulMat4(superXform.matrix, this._mat);
144         this._xform = {
145             localMatrix: this._mat,
146             matrix: tempMat,
147             fixed: superXform.fixed && this._fixedParams && !instancing
148         };
149         if (this._memoLevel == 1 && superXform.fixed && !instancing) {   // Bump up memoization level if model-space fixed
150             this._memoLevel = 2;
151         }
152     }
153     SceneJS_modelTransformModule.setTransform(this._xform);
154     this._renderNodes(traversalContext, data);
155     SceneJS_modelTransformModule.setTransform(superXform);
156 };
157 
158 /** Function wrapper to support functional scene definition
159  */
160 SceneJS.quaternion = function() {
161     var n = new SceneJS.Quaternion();
162     SceneJS.Quaternion.prototype.constructor.apply(n, arguments);
163     return n;
164 };
