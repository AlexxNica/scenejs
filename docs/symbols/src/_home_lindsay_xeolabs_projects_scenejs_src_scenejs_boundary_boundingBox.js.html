1 /**
  2  * @class A scene node that specifies the spatial boundaries of scene graph subtrees to support visibility and
  3  * level-of-detail culling.
  4  *
  5  * <p>The subgraphs of these are only traversed when the boundary intersect the current view frustum. When this node
  6  * is within the subgraph of a {@link SceneJS.Locality} node, it the boundary must also intersect the inner radius of the Locality.
  7  * the outer radius of the Locality is used internally by SceneJS to support content staging strategies.</p> 
  8  *  
  9  * <p>When configured with a projected size threshold for each child, they can also function as level-of-detail (LOD) selectors.</p>
 10  * <p><b>Live Demo</b></p>
 11  * <ul><li><a target = "other" href="http://bit.ly/scenejs-lod-boundingbox-example">Level of Detail Example</a></li></ul>
 12  *  <p><b>Example 1.</b></p><p>This BoundingBox is configured to work as a level-of-detail selector. The 'levels'
 13  * property specifies thresholds for the boundary's projected size, each corresponding to one of the node's children,
 14  * such that the child corresponding to the threshold imediately below the boundary's current projected size is only one
 15  * currently traversable.</p><p>This boundingBox will select exactly one of its child nodes to render for its current projected size, where the
 16  * levels parameter specifies for each child the size threshold above which the child becomes selected. No child is
 17  * selected (nothing is drawn) when the projected size is below the lowest level.</p>
 18  * <pre><code>
 19  * var bb = new SceneJS.BoundingBox({
 20  *          xmin: -2,
 21  *          ymin: -2,
 22  *          zmin: -2,
 23  *          xmax:  2,
 24  *          ymax:  2,
 25  *          zmax:  2,
 26  *
 27  *           // Levels are optional - acts as regular
 28  *          // frustum-culling bounding box when not specified
 29  *
 30  *          levels: [
 31  *             10,
 32  *             200,
 33  *             400,
 34  *             600
 35  *         ]
 36  *     },
 37  *
 38  *     // When size > 10px, draw a cube
 39  *
 40  *     new SceneJS.objects.Cube(),
 41  *
 42  *     // When size > 200px,  draw a low-detail sphere
 43  *
 44  *     new SceneJS.objects.Sphere({
 45  *         radius: 1,
 46  *         slices:10,
 47  *         rings:10
 48  *     }),
 49  *
 50  *     // When size > 400px, draw a medium-detail sphere
 51  *
 52  *     new SceneJS.objects.Sphere({
 53  *         radius: 1,
 54  *         slices:20,
 55  *         rings:20
 56  *     }),
 57  *
 58  *     // When size > 600px, draw a high-detail sphere
 59  *
 60  *     new SceneJS.objects.Sphere({
 61  *         radius: 1,
 62  *         slices:120,
 63  *         rings:120
 64  *     })
 65  * )
 66  * </code></pre>
 67  * @extends SceneJS.Node
 68  * @constructor
 69  * Create a new SceneJS.boundingBox
 70  * @param {Object} config Configuration object, followed by zero or more child nodes
 71  */
 72 SceneJS.BoundingBox = function() {
 73     SceneJS.Node.apply(this, arguments);
 74     this._nodeType="bounding-box";
 75     this._xmin = 0;
 76     this._ymin = 0;
 77     this._zmin = 0;
 78     this._xmax = 0;
 79     this._ymax = 0;
 80     this._zmax = 0;
 81     this._levels = null;
 82     this._states = [];
 83     this._objectsCoords = null;  // Six object-space vertices for memo level 1
 84     this._viewBox = null;         // Axis-aligned view-space box for memo level 2
 85     if (this._fixedParams) {
 86         this._init(this._getParams());
 87     }
 88 };
 89 
 90 SceneJS._inherit(SceneJS.BoundingBox, SceneJS.Node);

258 // @private
259 SceneJS.BoundingBox.prototype._init = function(params) {
260     this._xmin = params.xmin || 0;
261     this._ymin = params.ymin || 0;
262     this._zmin = params.zmin || 0;
263     this._xmax = params.xmax || 0;
264     this._ymax = params.ymax || 0;
265     this._zmax = params.zmax || 0;
266     if (params.levels) {
267         if (params.levels.length != this._children.length) {
268             SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
269                     ("SceneJS.boundingBox levels property should have a value for each child node"));
270         }
271 
272         for (var i = 1; i < params.levels.length; i++) {
273             if (params.levels[i - 1] >= params.levels[i]) {
274                 SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
275                         ("SceneJS.boundingBox levels property should be an ascending list of unique values"));
276             }
277         }
278         this._levels = params.levels;
279     }
280 };

282 // @private
283 SceneJS.BoundingBox.prototype._render = function(traversalContext, data) {
284     if (this._memoLevel == 0) {
285         if (!this._fixedParams) {
286             this._init(this._getParams(data));
287         } else {
288             this._memoLevel = 1;
289         }
290         var modelTransform = SceneJS_modelTransformModule.getTransform();
291         if (!modelTransform.identity) {
292 
293             /* Model transform exists
294              */
295             this._objectCoords = [
296                 [this._xmin, this._ymin, this._zmin],
297                 [this._xmax, this._ymin, this._zmin],
298                 [this._xmax, this._ymax, this._zmin],
299                 [this._xmin, this._ymax, this._zmin],
300                 [this._xmin, this._ymin, this._zmax],
301                 [this._xmax, this._ymin, this._zmax],
302                 [this._xmax, this._ymax, this._zmax],
303                 [this._xmin, this._ymax, this._zmax]
304             ];
305         } else {
306 
307             /* No model transform
308              */
309             this._viewBox = {
310                 min: [this._xmin, this._ymin, this._zmin],
311                 max: [this._xmax, this._ymax, this._zmax]
312             };
313             this._memoLevel = 2;
314         }
315     }
316 
317     if (this._memoLevel < 2) {
318         var modelTransform = SceneJS_modelTransformModule.getTransform();
319         this._viewBox = new SceneJS_math_Box3().fromPoints(
320                 SceneJS_math_transformPoints3(
321                         modelTransform.matrix,
322                         this._objectCoords)
323                 );
324         if (modelTransform.fixed && this._memoLevel == 1 && (!SceneJS_instancingModule.instancing())) {
325             this._objectCoords = null;
326             this._memoLevel = 2;
327         }
328     }
329     if (SceneJS_localityModule.testAxisBoxIntersectOuterRadius(this._viewBox)) {
330         if (SceneJS_localityModule.testAxisBoxIntersectInnerRadius(this._viewBox)) {
331             var result = SceneJS_frustumModule.testAxisBoxIntersection(this._viewBox);
332             switch (result) {
333                 case SceneJS_math_INTERSECT_FRUSTUM:  // TODO: GL clipping hints
334                 case SceneJS_math_INSIDE_FRUSTUM:
335                     if (this._levels) { // Level-of-detail mode
336                         var size = SceneJS_frustumModule.getProjectedSize(this._viewBox);
337                         for (var i = this._levels.length - 1; i >= 0; i--) {
338                             if (this._levels[i] <= size) {
339                                 var state = this._states[i];
340                                 this._renderNode(i, traversalContext, data);
341                                 return;
342                             }
343                         }
344                     } else {
345                         this._renderNodes(traversalContext, data);
346                     }
347                     break;
348 
349                 case SceneJS_math_OUTSIDE_FRUSTUM:
350                     break;
351             }
352         } else {
353 
354             /* Allow content staging for subgraph
355              */
356 
357             // TODO:
358 
359             this._renderNodes(traversalContext, data);
360         }
361     }
362 };
