 98                     this._render = function(traversalContext, data) {
 99                         if (!cfg.fixed || !(box || objectCoords)) {
100                             var params = cfg.getParams(data);
101 
102                             params.xmin = params.xmin || 0;
103                             params.ymin = params.ymin || 0;
104                             params.zmin = params.zmin || 0;
105                             params.xmax = params.xmax || 0;
106                             params.ymax = params.ymax || 0;
107                             params.zmax = params.zmax || 0;
108 
109                             var modelTransform = modelTransformBackend.getTransform();
110                             if (modelTransform.identity) {  // No model transform
111                                 box = {
112                                     min: [params.xmin, params.ymin, params.zmin],
113                                     max: [params.xmax, params.ymax, params.zmax]
114                                 };
115                             } else {                        // Model transform either fixed or dynamic
116                                 objectCoords = [
117                                     [params.xmin, params.ymin, params.zmin],
118                                     [params.xmax, params.ymin, params.zmin],

127                             }
128                             if (params.levels) {
129                                 if (params.levels.length != cfg.children.length) {
130                                     errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException

133 
134                                 for (var i = 1; i < params.levels.length; i++) {
135                                     if (params.levels[i - 1] >= params.levels[i]) {
136                                         errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException

139                                     states.push(STAGE_REMOTE);
140                                 }
141                                 levels = params.levels;
142                             }
143                         }
144

146                             var modelTransform = modelTransformBackend.getTransform();

152 
153                             if (modelTransform.fixed) {
154                                 objectCoords = null;    // Ensures we don't recreate axis box next time
155                             }
156                         }
157 
158                         if (localityBackend.testAxisBoxIntersectOuterRadius(box)) {
159 
160                             if (localityBackend.testAxisBoxIntersectInnerRadius(box)) {
161 
162                                 var result = backend.testAxisBoxIntersection(box);
163                                 window.countTeapots++;
164                                 switch (result) {
165                                     case SceneJS_math_INTERSECT_FRUSTUM:  // TODO: GL clipping hints
166 
167                                     case SceneJS_math_INSIDE_FRUSTUM:
168 
169                                         if (levels) { // Level-of-detail mode
170 
171                                             var size = backend.getProjectedSize(box);
172                                             for (var i = levels.length - 1; i >= 0; i--) {
173                                                 if (levels[i] <= size) {
174                                                     var state = states[i];
175                                                     this._renderChild(i, traversalContext, data);
176                                                     return;
177                                                 }
178                                             }
179                                         } else {
180                                             this._renderChildren(traversalContext, data);
181                                         }
