1 /**
  2  * Backend module that defines SceneJS events and provides an interface on the backend context through which
  3  * backend modules can fire and subscribe to them.
  4  *
  5  * Events are actually somewhat more like commands; they are always synchronous, and are often used to decouple the
  6  * transfer of data between backends, request events in response, and generally trigger some immediate action.
  7  *
  8  * Event subscription can optionally be prioritised, to control the order in which the subscriber will be notified of
  9  * a given event relative to other suscribers. This is useful, for example, when a backend must be the first to handle
 10  * an INIT, or the last to handle a RESET.
 11  *
 12  * @private
 13  */
 14 var SceneJS_eventModule = new (function() {
 15 
 16     this.ERROR = 0;
 17     this.INIT = 1;                           // SceneJS framework initialised
 18     this.RESET = 2;                          // SceneJS framework reset
 19     this.TIME_UPDATED = 3;                   // System time updated
 20     this.SCENE_CREATED = 4;                  // Scene has just been created
 21     this.SCENE_RENDERING = 5;                // Scene about to be traversed
 22     this.SCENE_RENDERED = 6;              // Scene just been completely traversed
 23     this.SCENE_DESTROYED = 7;                // Scene just been destroyed
 24     this.RENDERER_UPDATED = 8;                // Current WebGL context has been updated to the given state
 25     this.RENDERER_EXPORTED = 9;               // Export of the current WebGL context state
 26     this.CANVAS_ACTIVATED = 10;
 27     this.CANVAS_DEACTIVATED = 11;
 28     this.VIEWPORT_UPDATED = 12;
 29     this.GEOMETRY_UPDATED = 13;
 30     this.GEOMETRY_EXPORTED = 14;
 31     this.MODEL_TRANSFORM_UPDATED = 15;
 32     this.MODEL_TRANSFORM_EXPORTED = 16;
 33     this.PROJECTION_TRANSFORM_UPDATED = 17;
 34     this.PROJECTION_TRANSFORM_EXPORTED = 18;
 35     this.VIEW_TRANSFORM_UPDATED = 19;
 36     this.VIEW_TRANSFORM_EXPORTED = 20;
 37     this.LIGHTS_UPDATED = 21;
 38     this.LIGHTS_EXPORTED = 22;
 39     this.MATERIAL_UPDATED = 23;
 40     this.MATERIAL_EXPORTED = 24;
 41     this.TEXTURES_UPDATED = 25;
 42     this.TEXTURES_EXPORTED = 26;
 43     this.SHADER_ACTIVATE = 27;
 44     this.SHADER_ACTIVATED = 28;
 45     this.SHADER_RENDERING = 29;
 46     this.SHADER_DEACTIVATED = 30;
 47     this.FOG_UPDATED = 31;
 48     this.FOG_EXPORTED = 32;
 49     this.NAME_UPDATED = 33;
 50     this.PROCESS_CREATED = 34;
 51     this.PROCESS_KILLED = 35;
 52     this.PROCESS_TIMED_OUT = 36;
 53     this.LOGGING_ELEMENT_ACTIVATED = 37;
 54 
 55     /* Priority queue for each type of event
 56      */
 57     var events = new Array(37);
 58 
 59     /**
 60      * Registers a handler for the given event
 61      *
 62      * The handler can be registered with an optional priority number which specifies the order it is
 63      * called among the other handler already registered for the event.
 64      *
 65      * So, with n being the number of commands registered for the given event:
 66      *
 67      * (priority <= 0)      - command will be the first called
 68      * (priority >= n)      - command will be the last called
 69      * (0 < priority < n)   - command will be called at the order given by the priority
 70      * @private
 71      * @param type Event type - one of the values in SceneJS_eventModule
 72      * @param command - Handler function that will accept whatever parameter object accompanies the event
 73      * @param priority - Optional priority number (see above)
 74      */
 75     this.addListener = function(type, command, priority) {
 76         var list = events[type];
 77         if (!list) {
 78             list = [];
 79             events[type] = list;
 80         }
 81         var handler = {
 82             command: command,
 83             priority : (priority == undefined) ? list.length : priority
 84         };
 85         for (var i = 0; i < list.length; i++) {
 86             if (list[i].priority > handler.priority) {
 87                 list.splice(i, 0, handler);
 88                 return;
 89             }
 90         }
 91         list.push(handler);
 92     };
 93 
 94     /**
 95      * @private
 96      */
 97     this.fireEvent = function(type, params) {
 98         var list = events[type];
 99         if (list) {
100             if (!params) {
101                 params = {};
102             }
103             for (var i = 0; i < list.length; i++) {
104                 list[i].command(params);
105             }
106         }
107     };
108 })();

112 /** <p>Adds a listener to be notified when a given event occurs within SceneJS.</p>
113  * <p><b>Supported events</b></p>
114  * <p><b><em>error</em></b></p><p>An error has occurred either while defining or rendering a scene. These can be either fatal,
115  * or errors that SceneJS can recover from.</p><p>Example:</p><pre><code>
116  * SceneJS.addListener("error", function(e) {
117  *     if (e.exception.message) {
118  *         alert("Error: " + e.exception.message);
119  *     } else {
120  *         alert("Error: " + e.exception);
121  *     }
122  *  });
123  * </pre></code>
124  *
125  * <p><b><em>reset</em></b></p><p>The SceneJS framework has been reset, where all {@link SceneJS.Scene} instances have
126  * been destroyed and resources held for them freed.</p><p>Example:</p><pre><code>
127  *  SceneJS.addListener(
128  *      "reset",
129  *      function(e) {
130  *          alert("SceneJS has been reset");
131  *      });
132  * </pre></code>
133 
134  * <p><b><em>scene-created</em></b></p><p>A {@link SceneJS.Scene} has been defined.</p><p>Example:</p><pre><code>
135  *  SceneJS.addListener(
136  *      "scene-created",
137  *      function(e) {
138  *          alert("A new Scene has been created - scene ID: " + e.sceneId);
139  *      });
140  * </pre></code>
141  *
142  * <p><b><em>scene-rendering</em></b></p><p>Traversal (render) of a {@link SceneJS.Scene} has just begun.</p><p>Example:</p><pre><code>
143  *  SceneJS.addListener(
144  *      "scene-rendering",
145  *      function(e) {
146  *          alert("Rendering of a new Scene has just begun - scene ID: " + e.sceneId);
147  *      });
148  * </pre></code>
149  *
150  * <p><b><em>canvas-activated</em></b></p><p>A canvas has just been activated for a {@link SceneJS.Scene}, where that
151  * node is about to start rendering to it. This will come right after a "scene-rendering" event, which will indicate which
152  * {@link SceneJS.Scene} is the one about to do the rendering.</p><p>Example:</p><pre><code>
153  *  SceneJS.addListener(
154  *      "canvas-activated",
155  *      function(e) {
156  *          var canvas = e.canvas;
157  *          var context = e.context;
158  *          var canvasId = e.canvasId;
159  *          alert("Canvas is about to be rendered to : " + canvasId);
160  *      });
161  * </pre></code>
162  *
163  * <p><b><em>process-created</em></b></p><p>An asynchronous process has started somewhere among the nodes wtihin a
164  * {@link SceneJS.Scene}. Processes track the progress of tasks such as the loading of remotely-stored content by
165  * {@link SceneJS.Instance} nodes. This event is particularly useful to monitor for content loading. </p>
166  * <p>Example:</p><pre><code>
167  *  SceneJS.addListener(
168  *      "process-created",
169  *      function(e) {
170  *          var sceneId = e.sceneId;
171  *          var processId = e.process.id;
172  *          var timeStarted = e.process.timeStarted;
173  *          var description = e.process.description;
174  *          var timeoutSecs = e.process.timeoutSecs;
175  *
176  *          // ...
177  *      });
178  * </pre></code>
179  *
180  * <p><b><em>process-timed-out</em></b></p><p>An asynchronous process has timed out. This will be followed by
181  * a "process-killed" event.</p><p>Example:</p><pre><code>
182  *  SceneJS.addListener(
183  *      "process-timed-out",
184  *      function(e) {
185  *          var sceneId = e.sceneId;
186  *          var processId = e.process.id;
187  *          var timeStarted = e.process.timeStarted;
188  *          var description = e.process.description;
189  *          var timeoutSecs = e.process.timeoutSecs;
190  *
191  *          // ...
192  *      });
193  * </pre></code>
194  *
195  * <p><b><em>process-killed</em></b></p><p>An asynchronous process has finished.</p><p>Example:</p><pre><code>
196  *  SceneJS.addListener(
197  *      "process-killed",
198  *      function(e) {
199  *          var sceneId = e.sceneId;
200  *          var processId = e.process.id;
201  *          var timeStarted = e.process.timeStarted;
202  *          var description = e.process.description;
203  *          var timeoutSecs = e.process.timeoutSecs;
204  *
205  *          // ...
206  *      });
207  * </pre></code>
208  *
209  * <p><b><em>scene-rendered</em></b></p><p>A render of a {@link SceneJS.Scene} has completed.</p><p>Example:</p><pre><code>
210  *  SceneJS.addListener(
211  *      "scene-rendered",
212  *      function(e) {
213  *          alert("Traversal completed for Scene - scene ID: " + e.sceneId);
214  *      });
215  * </pre></code>
216  *
217  * <p><b><em>scene-destroyed</em></b></b></p><p>A SceneJS.Scene traversal has been destroyed.</p><p>Example:</p><pre><code>
218  *  SceneJS.addListener(
219  *      "scene-destroyed",
220  *      function(e) {
221  *          alert("Scene has been destroyed - scene ID: " + e.sceneId);
222  *      });
223  * </pre></code>
224  * @param name Event name
225  * @param func Callback function
226  */
227 SceneJS.addListener = function(name, func) {
228     switch (name) {
229 
230         /**
231          * @event error
232          * Fires when the data cache has changed in a bulk manner (e.g., it has been sorted, filtered, etc.) and a
233          * widget that is using this Store as a Record cache should refresh its view.
234          * @param {Store} this
235          */
236         case "error" : SceneJS_eventModule.addListener(
237                 SceneJS_eventModule.ERROR,
238                 function(params) {
239                     func({
240                         exception: params.exception,
241                         fatal: params.fatal
242                     });
243                 });
244             break;
245 
246         case "reset" : SceneJS_eventModule.addListener(
247                 SceneJS_eventModule.RESET,
248                 function() {
249                     func();
250                 });
251             break;
252 
253         case "scene-created" : SceneJS_eventModule.addListener(
254                 SceneJS_eventModule.SCENE_CREATED,
255                 function(params) {
256                     func({
257                         sceneId : params.sceneId
258                     });
259                 });
260             break;
261 
262         case "scene-rendering" : SceneJS_eventModule.addListener(
263                 SceneJS_eventModule.SCENE_RENDERING,
264                 function(params) {
265                     func({
266                         sceneId : params.sceneId
267                     });
268                 });
269             break;
270 
271         case "canvas-activated" : SceneJS_eventModule.addListener(
272                 SceneJS_eventModule.CANVAS_ACTIVATED,
273                 function(params) {
274                     func({
275                         canvas: params.canvas
276                     });
277                 });
278             break;
279 
280         case "process-created" : SceneJS_eventModule.addListener(
281                 SceneJS_eventModule.PROCESS_CREATED,
282                 function(params) {
283                     func(params);
284                 });
285             break;
286 
287         case "process-timed-out" : SceneJS_eventModule.addListener(
288                 SceneJS_eventModule.PROCESS_TIMED_OUT,
289                 function(params) {
290                     func(params);
291                 });
292             break;
293 
294         case "process-killed" : SceneJS_eventModule.addListener(
295                 SceneJS_eventModule.PROCESS_KILLED,
296                 function(params) {
297                     func(params);
298                 });
299             break;
300 
301         case "scene-rendered" : SceneJS_eventModule.addListener(
302                 SceneJS_eventModule.SCENE_RENDERED,
303                 function(params) {
304                     func({
305                         sceneId : params.sceneId
306                     });
307                 });
308             break;
309 
310         case "scene-destroyed" : SceneJS_eventModule.addListener(
311                 SceneJS_eventModule.SCENE_DESTROYED,
312                 function(params) {
313                     func({
314                         sceneId : params.sceneId
315                     });
316                 });
317             break;
318 
319         default:
320             throw "SceneJS.addListener - this event type not supported: '" + name + "'";
321     }
322 };
323 
324 /** @deprecated - use {@link #addListener} instead. 
325  */
326 SceneJS.onEvent = SceneJS.addListener;
