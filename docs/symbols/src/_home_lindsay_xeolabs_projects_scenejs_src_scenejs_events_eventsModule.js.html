1 /**
  2  * Backend module that defines SceneJS events and provides an interface on the backend context through which
  3  * backend modules can fire and subscribe to them.
  4  *
  5  * Events are actually somewhat more like commands; they are always synchronous, and are often used to decouple the
  6  * transfer of data between backends, request events in response, and generally trigger some immediate action.
  7  *
  8  * Event subscription can optionally be prioritised, to control the order in which the subscriber will be notified of
  9  * a given event relative to other suscribers. This is useful, for example, when a backend must be the first to handle
 10  * an INIT, or the last to handle a RESET.
 11  *
 12  * @private
 13  */
 14 var SceneJS_eventModule = new (function() {
 15 
 16     this.ERROR = 0;
 17     this.INIT = 1;                           // SceneJS framework initialised
 18     this.RESET = 2;                          // SceneJS framework reset
 19     this.TIME_UPDATED = 3;                   // System time updated
 20     this.SCENE_CREATED = 4;                  // Scene has just been created
 21     this.SCENE_ACTIVATED = 5;                // Scene about to be traversed
 22     this.SCENE_DEACTIVATED = 6;              // Scene just been completely traversed
 23     this.SCENE_DESTROYED = 7;                // Scene just been destroyed
 24     this.RENDERER_UPDATED = 8;                // Current WebGL context has been updated to the given state
 25     this.RENDERER_EXPORTED = 9;               // Export of the current WebGL context state
 26     this.CANVAS_ACTIVATED = 10;
 27     this.CANVAS_DEACTIVATED = 11;
 28     this.VIEWPORT_UPDATED = 12;
 29     this.GEOMETRY_UPDATED = 13;
 30     this.GEOMETRY_EXPORTED = 14;
 31     this.MODEL_TRANSFORM_UPDATED = 15;
 32     this.MODEL_TRANSFORM_EXPORTED = 16;
 33     this.PROJECTION_TRANSFORM_UPDATED = 17;
 34     this.PROJECTION_TRANSFORM_EXPORTED = 18;
 35     this.VIEW_TRANSFORM_UPDATED = 19;
 36     this.VIEW_TRANSFORM_EXPORTED = 20;
 37     this.LIGHTS_UPDATED = 21;
 38     this.LIGHTS_EXPORTED = 22;
 39     this.MATERIAL_UPDATED = 23;
 40     this.MATERIAL_EXPORTED = 24;
 41     this.TEXTURES_UPDATED = 25;
 42     this.TEXTURES_EXPORTED = 26;
 43     this.SHADER_ACTIVATE = 27;
 44     this.SHADER_ACTIVATED = 28;
 45     this.SHADER_RENDERING = 29;
 46     this.SHADER_DEACTIVATED = 30;
 47     this.FOG_UPDATED = 31;
 48     this.FOG_EXPORTED = 32;
 49     this.NAME_UPDATED = 33;
 50     this.PROCESS_CREATED = 34;
 51     this.PROCESS_KILLED = 35;
 52     this.PROCESS_TIMED_OUT = 36;
 53 
 54     /* Priority queue for each type of event
 55      */
 56     var events = new Array(37);
 57 
 58     /**
 59      * Registers a handler for the given event
 60      *
 61      * The handler can be registered with an optional priority number which specifies the order it is
 62      * called among the other handler already registered for the event.
 63      *
 64      * So, with n being the number of commands registered for the given event:
 65      *
 66      * (priority <= 0)      - command will be the first called
 67      * (priority >= n)      - command will be the last called
 68      * (0 < priority < n)   - command will be called at the order given by the priority
 69      * @private
 70      * @param type Event type - one of the values in SceneJS_eventModule
 71      * @param command - Handler function that will accept whatever parameter object accompanies the event
 72      * @param priority - Optional priority number (see above)
 73      */
 74     this.onEvent = function(type, command, priority) {
 75         var list = events[type];
 76         if (!list) {
 77             list = [];
 78             events[type] = list;
 79         }
 80         var handler = {
 81             command: command,
 82             priority : (priority == undefined) ? list.length : priority
 83         };
 84         for (var i = 0; i < list.length; i++) {
 85             if (list[i].priority > handler.priority) {
 86                 list.splice(i, 0, handler);
 87                 return;
 88             }
 89         }
 90         list.push(handler);
 91     };
 92 
 93     /**
 94      * @private
 95      */
 96     this.fireEvent = function(type, params) {
 97         var list = events[type];
 98         if (list) {
 99             if (!params) {
100                 params = {};
101             }
102             for (var i = 0; i < list.length; i++) {
103                 list[i].command(params);
104             }
105         }
106     };
107 })();

110 /** <p>Adds a listener to be notified when a given event occurs within SceneJS.</p>
111  * <p><b>Supported events</b></p>
112  * <p><b><em>error</em></b></p><p>An error has occurred either while defining or rendering a scene. These can be either fatal,
113  * or errors that SceneJS can recover from.</p><p>Example:</p><pre><code>
114  * SceneJS.onEvent("error", function(e) {
115  *     if (e.exception.message) {
116  *         alert("Error: " + e.exception.message);
117  *     } else {
118  *         alert("Error: " + e.exception);
119  *     }
120  *  });
121  * </pre></code>
122  *
123  * <p><b><em>reset</em></b></p><p>The SceneJS framework has been reset, where all SceneJS.Scene instances are destroyed and resources
124  * held for them freed.</p><p>Example:</p><pre><code>
125  *  SceneJS.onEvent(
126  *      "reset",
127  *      function(e) {
128  *          alert("SceneJS has been reset");
129  *      });
130  * </pre></code>
131 
132  * <p><b><em>scene-created</em></b></p><p>A SceneJS.Scene has been defined.</p><p>Example:</p><pre><code>
133  *  SceneJS.onEvent(
134  *      "scene-created",
135  *      function(e) {
136  *          alert("A new Scene has been created - scene ID: " + e.sceneId);
137  *      });
138  * </pre></code>
139  *
140  * <p><b><em>scene-activated</em></b></p><p>Traversal (render) of a SceneJS.Scene has just begun.</p><p>Example:</p><pre><code>
141  *  SceneJS.onEvent(
142  *      "scene-activated",
143  *      function(e) {
144  *          alert("Rendering of a new Scene has just begun - scene ID: " + e.sceneId);
145  *      });
146  * </pre></code>
147  *
148  * <p><b><em>canvas-activated</em></b></p><p>A canvas has just been activated for a Scene, where the Scene is about to start
149  * rendering to it. This will come right after a "scene-activated" event, which will indicate which Scene is the one
150  * about to do the rendering.</p><p>Example:</p><pre><code>
151  *  SceneJS.onEvent(
152  *      "canvas-activated",
153  *      function(e) {
154  *          var canvas = e.canvas;
155  *          var context = e.context;
156  *          var canvasId = e.canvasId;
157  *          alert("Canvas is about to be rendered to : " + canvasId);
158  *      });
159  * </pre></code>
160  *
161  * <p><b><em>process-created</em></b></p><p>An asynchronous process has started within a Scene. Processes track the progress of
162  * tasks such as the loading of remotely-stored content by SceneJS.Load and SceneJS.LoadCollada nodes. This event is
163  * paticularly useful to monitor for content loading. Since loads are triggered in one scene traversal, and then loaded
164  * content is integrated during a subsequent traversal, you might listen for this along with "process-killed" on a
165  * non-animated scene to ensure that a final scene image is rendered once all loads have completed and content
166  * integrated.</p><p>Example:</p><pre><code>
167  *  SceneJS.onEvent(
168  *      "process-created",
169  *      function(e) {
170  *          var sceneId = e.sceneId;
171  *          var processId = e.process.id;
172  *          var timeStarted = e.process.timeStarted;
173  *          var description = e.process.description;
174  *          var timeoutSecs = e.process.timeoutSecs;
175  *
176  *          // ...
177  *      });
178  * </pre></code>
179  *
180  * <p><b><em>process-timed-out</em></b></p><p>An asynchronous process has timed out. This will be followed by a "process-killed" event.</p><p>Example:</p><pre><code>
181  *  SceneJS.onEvent(
182  *      "process-timed-out",
183  *      function(e) {
184  *          var sceneId = e.sceneId;
185  *          var processId = e.process.id;
186  *          var timeStarted = e.process.timeStarted;
187  *          var description = e.process.description;
188  *          var timeoutSecs = e.process.timeoutSecs;
189  *
190  *          // ...
191  *      });
192  * </pre></code>
193  *
194  * <p><b><em>process-killed</em></b></p><p>An asynchronous process has finished.</p><p>Example:</p><pre><code>
195  *  SceneJS.onEvent(
196  *      "process-killed",
197  *      function(e) {
198  *          var sceneId = e.sceneId;
199  *          var processId = e.process.id;
200  *          var timeStarted = e.process.timeStarted;
201  *          var description = e.process.description;
202  *          var timeoutSecs = e.process.timeoutSecs;
203  *
204  *          // ...
205  *      });
206  * </pre></code>
207  *
208  * <p><b><em>scene-deactivated</em></b></p><p>A SceneJS.Scene traversal has completed.</p><p>Example:</p><pre><code>
209  *  SceneJS.onEvent(
210  *      "scene-created",
211  *      function(e) {
212  *          alert("Traversal completed for Scene - scene ID: " + e.sceneId);
213  *      });
214  * </pre></code>
215  *
216  * <p><b><em>scene-destroyed</em></b></b></p><p>A SceneJS.Scene traversal has been destroyed.</p><p>Example:</p><pre><code>
217  *  SceneJS.onEvent(
218  *      "scene-destroyed",
219  *      function(e) {
220  *          alert("Scene has been destroyed - scene ID: " + e.sceneId);
221  *      });
222  * </pre></code>
223  * @param name Event name
224  * @param func Callback function
225  */
226 SceneJS.onEvent = function(name, func) {
227     switch (name) {
228 
229         /**
230          * @event error
231          * Fires when the data cache has changed in a bulk manner (e.g., it has been sorted, filtered, etc.) and a
232          * widget that is using this Store as a Record cache should refresh its view.
233          * @param {Store} this
234          */
235         case "error" : SceneJS_eventModule.onEvent(
236                 SceneJS_eventModule.ERROR,
237                 function(params) {
238                     func({
239                         exception: params.exception,
240                         fatal: params.fatal
241                     });
242                 });
243             break;
244 
245         case "reset" : SceneJS_eventModule.onEvent(
246                 SceneJS_eventModule.RESET,
247                 function() {
248                     func();
249                 });
250             break;
251 
252         case "scene-created" : SceneJS_eventModule.onEvent(
253                 SceneJS_eventModule.SCENE_CREATED,
254                 function(params) {
255                     func({
256                         sceneId : params.sceneId
257                     });
258                 });
259             break;
260 
261         case "scene-activated" : SceneJS_eventModule.onEvent(
262                 SceneJS_eventModule.SCENE_ACTIVATED,
263                 function(params) {
264                     func({
265                         sceneId : params.sceneId
266                     });
267                 });
268             break;
269 
270         case "canvas-activated" : SceneJS_eventModule.onEvent(
271                 SceneJS_eventModule.CANVAS_ACTIVATED,
272                 function(params) {
273                     func({
274                         canvas: params.canvas
275                     });
276                 });
277             break;
278 
279         case "process-created" : SceneJS_eventModule.onEvent(
280                 SceneJS_eventModule.PROCESS_CREATED,
281                 function(params) {
282                     func(params);
283                 });
284             break;
285 
286         case "process-timed-out" : SceneJS_eventModule.onEvent(
287                 SceneJS_eventModule.PROCESS_TIMED_OUT,
288                 function(params) {
289                     func(params);
290                 });
291             break;
292 
293         case "process-killed" : SceneJS_eventModule.onEvent(
294                 SceneJS_eventModule.PROCESS_KILLED,
295                 function(params) {
296                     func(params);
297                 });
298             break;
299 
300         case "scene-deactivated" : SceneJS_eventModule.onEvent(
301                 SceneJS_eventModule.SCENE_DEACTIVATED,
302                 function(params) {
303                     func({
304                         sceneId : params.sceneId
305                     });
306                 });
307             break;
308 
309         case "scene-destroyed" : SceneJS_eventModule.onEvent(
310                 SceneJS_eventModule.SCENE_DESTROYED,
311                 function(params) {
312                     func({
313                         sceneId : params.sceneId
314                     });
315                 });
316             break;
317 
318         default:
319             throw "SceneJS.onEvent - this event type not supported: '" + name + "'";
320     }
321 };
