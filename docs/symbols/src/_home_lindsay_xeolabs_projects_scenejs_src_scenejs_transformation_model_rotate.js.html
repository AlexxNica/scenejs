1 /**
  2  * @class SceneJS.Rotate
  3  * @extends SceneJS.Node
  4  * <p>Scene node that applies a model-space rotation transform to the nodes within its subgraph.</p><p>The rotation
  5  * is described as a vector about which the rotation occurs, along with the angle or rotation in degrees.</p>
  6  * <p><b>Example</b></p><p>A cube rotated 45 degrees about its Y axis.</b></p><pre><code>
  7  * var rotate = new SceneJS.Rotate({
  8  *       angle: 45.0,    // Angle in degrees
  9  *       x: 0.0,         // Rotation vector points along positive Y axis
 10  *       y: 1.0,
 11  *       z: 0.0
 12  *   },
 13  *
 14  *      new SceneJS.objects.Cube()
 15  * )
 16  * </pre></code>
 17  * @constructor
 18  * Create a new SceneJS.Rotate
 19  * @param {Object} config  Config object or function, followed by zero or more child nodes
 20  */
 21 SceneJS.Rotate = function() {
 22     SceneJS.Node.apply(this, arguments);
 23     this._nodeType = "rotate";
 24     this._mat = null;
 25     this._xform = null;
 26     this._angle = 0;
 27     this._x = 0;
 28     this._y = 0;
 29     this._z = 1;
 30     if (this._fixedParams) {
 31         this._init(this._getParams());
 32     }
 33 };
 34 
 35 SceneJS._utils.inherit(SceneJS.Rotate, SceneJS.Node);

 41 SceneJS.Rotate.prototype.setAngle = function(angle) {
 42     this._angle = angle || 0;

 44     return this;

 54 /**
 55  * Sets the rotation axis vector. The vector must not be of zero length.
 56  * @param {object} xyz The vector - eg. {x: 0, y: 1, z: 0}
 57  * @returns {SceneJS.Rotate} this
 58  */
 59 SceneJS.Rotate.prototype.setXYZ = function(xyz) {
 60     var x = xyz.x || 0;
 61     var y = xyz.y || 0;
 62     var z = xyz.z || 0;
 63     //    if (x + y + z == 0) {
 64     //        SceneJS_errorModule.fatalError(
 65     //                new SceneJS.exceptions.IllegalRotateConfigException(
 66     //                        "SceneJS.rotate vector is zero - at least one of properties x,y and z must be non-zero"));
 67     //    }
 68     this._x = x;
 69     this._y = y;
 70     this._z = z;
 71     this._memoLevel = 0;
 72     return this;
 73 };

143 SceneJS.Rotate.prototype._init = function(params) {
144     if (params.angle) {
145         this.setAngle(params.angle);
146     }
147     this.setXYZ({x : params.x, y: params.y, z: params.z });
148 };
149 
150 SceneJS.Rotate.prototype._render = function(traversalContext, data) {
151     if (this._memoLevel == 0) {
152         if (!this._fixedParams) {
153             this._init(this._getParams(data));
154         } else {
155             this._memoLevel = 1;
156         }
157         this._mat = SceneJS_math_rotationMat4v(this._angle * Math.PI / 180.0, [this._x, this._y, this._z]);
158     }
159     var superXform = SceneJS_modelTransformModule.getTransform();
160     if (this._memoLevel < 2) {
161         var tempMat = SceneJS_math_mulMat4(superXform.matrix, this._mat);
162         this._xform = {
163             localMatrix: this._mat,
164             matrix: tempMat,
165             fixed: superXform.fixed && this._fixedParams
166         };
167         if (this._memoLevel == 1 && superXform.fixed) {   // Bump up memoization level if model-space fixed
168             this._memoLevel = 2;
169         }
170     }
171     SceneJS_modelTransformModule.setTransform(this._xform);
172     this._renderNodes(traversalContext, data);
173     SceneJS_modelTransformModule.setTransform(superXform);
174 };

176 /** Function wrapper to support functional scene definition

178 SceneJS.rotate = function() {
