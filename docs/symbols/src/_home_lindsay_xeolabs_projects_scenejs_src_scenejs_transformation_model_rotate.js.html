1 /**
  2  * Scene node that specifies a modelling-space rotation for the nodes in its subgraph.
  3  */
  4 SceneJS.rotate = function() {
  5     var cfg = SceneJS._utils.getNodeConfig(arguments);
  6 
  7     var modelTransformBackend = SceneJS._backends.getBackend('model-transform');
  8     var errorBackend = SceneJS._backends.getBackend('error');
  9 
 10     /* Memoization levels
 11      */
 12     const NO_MEMO = 0;              // No memoization, assuming that node's configuration is dynamic
 13     const FIXED_CONFIG = 1;         // Node config is fixed, memoizing local object-space matrix
 14     const FIXED_MODEL_SPACE = 2;    // Both node config and model-space are fixed
 15 
 16     return SceneJS._utils.createNode(
 17             "rotate",
 18             cfg.children,
 19             
 20             new (function() {
 21                 var _memoLevel = NO_MEMO;
 22                 var _mat;
 23                 var xform;
 24 
 25                 var _angle = 0;
 26                 var _x = 0;
 27                 var _y = 0;
 28                 var _z = 1;
 29 
 30                 /** Sets the rotation angle
 31                  * @param {double} angle Rotation angle in degrees
 32                  * @returns This node
 33                  */
 34                 this.setAngle = function(angle) {
 35                     _angle = angle || 0;
 36                     _memoLevel = NO_MEMO;
 37                     return this;
 38                 };
 39 
 40                 /** Returns the angle of rotation
 41                  * @returns {double} The angle in degrees
 42                  */
 43                 this.getAngle = function() {
 44                     return _angle;
 45                 };
 46 
 47                 /**
 48                  * Sets the vector about which rotation occurs. The elements of the vector must not all be zero.
 49                  * @param {object} vector The vector - eg. {x: 0, y: 1, z: 0}
 50                  * @returns This node
 51                  */
 52                 this.setXYZ = function(xyz) {
 53                     var x = xyz.x || 0;
 54                     var y = xyz.y || 0;
 55                     var z = xyz.z || 0;
 56                     if (x + y + z == 0) {
 57                         errorBackend.fatalError(
 58                                 new SceneJS.exceptions.IllegalRotateConfigException(
 59                                         "SceneJS.rotate vector is zero - at least one of properties x,y and z must be non-zero"));
 60                     }
 61                     _x = x;
 62                     _y = y;
 63                     _z = z;
 64                     _memoLevel = NO_MEMO;
 65                     return this;
 66                 };
 67 
 68                 /** Returns the vector about which rotation occurs
 69                  * @returns {object} the vector, eg. {x: 0, y: 1, z: 0}
 70                  */
 71                 this.getXYZ = function() {
 72                     return {
 73                         x: _x,
 74                         y: _y,
 75                         z: _z
 76                     };
 77                 };
 78 
 79                 this.setX = function(x) {
 80                     _x = x;
 81                     _memoLevel = NO_MEMO;
 82                     return this;
 83                 };
 84 
 85                 this.getX = function() {
 86                     return _x;
 87                 };
 88 
 89                 this.setY = function(y) {
 90                     _y = y;
 91                     _memoLevel = NO_MEMO;
 92                     return this;
 93                 };
 94 
 95                 this.getY = function() {
 96                     return _y;
 97                 };
 98 
 99                 this.setZ = function(z) {
100                     _z = z;
101                     _memoLevel = NO_MEMO;
102                     return this;
103                 };
104 
105                 this.getZ = function() {
106                     return _z;
107                 };
108 
109                 this._init = function(params) {
110                     if (params.angle) {
111                         this.setAngle(params.angle);
112                     }
113                     this.setXYZ({x : params.x, y: params.y, z: params.z });
114                 };
115 
116                 if (cfg.fixed) {
117                     this._init(cfg.getParams());
118                 }
119 
120                 this._render = function(traversalContext, data) {
121                     if (_memoLevel == NO_MEMO) {
122                         if (!cfg.fixed) {
123                             this._init(cfg.getParams(data));
124                         } else {
125                             _memoLevel = FIXED_CONFIG;
126                         }
127                         _mat = SceneJS_math_rotationMat4v(_angle * Math.PI / 180.0, [_x, _y, _z]);
128                     }
129                     var superXform = modelTransformBackend.getTransform();
130                     if (_memoLevel < FIXED_MODEL_SPACE) {
131                         var tempMat = SceneJS_math_mulMat4(superXform.matrix, _mat);
132                         xform = {
133                             localMatrix: _mat,
134                             matrix: tempMat,
135                             fixed: superXform.fixed && cfg.fixed
136                         };
137                         if (_memoLevel == FIXED_CONFIG && superXform.fixed) {   // Bump up memoization level if model-space fixed
138                             _memoLevel = FIXED_MODEL_SPACE;
139                         }
140                     }
141                     modelTransformBackend.setTransform(xform);
142                     this._renderChildren(traversalContext, data);
143                     modelTransformBackend.setTransform(superXform);
144                 };
145             })());
146 };
