1 /**
  2  * @class A scene node that defines a view of the nodes within its subgraph.
  3  *
  4  * <h2>Position and Orientation</h2>
  5  * <p>A Camera is oriented such that the local +X is to the right, the lens looks down the local -Z axis, and the top
  6  * points up the +Y axis. Its orientation and location may be transformed by defining it within transform nodes. The example
  7  * below defines a perspective Camera that is positioned using using a {@link SceneJS.LookAt}:</p><pre><code>
  8  * var exampleScene = new SceneJS.Scene({ ... },
  9  *
 10  *    // Viewing transform specifies eye position, looking
 11  *    // at the origin by default
 12  *
 13  *    SceneJS.lookAt({
 14  *           eye : { x: 0.0, y: 10.0, z: -10 },
 15  *           look : { y:1.0 },
 16  *           up : { y: 1.0 }
 17  *        },
 18  *
 19  *        new SceneJS.Camera({
 20  *              optics: {
 21  *                 type   : "perspective",
 22  *                 fovy   : 60.0,           // Horizontal field of view in degrees
 23  *                 aspect : 1.0,            // Aspect ratio of the field of view
 24  *                 near   : 0.10,           // Distance of the near clipping plane
 25  *                 far    : 10000.0         // Distance of the far clipping plane
 26  *              }
 27  *           },
 28  *
 29  *           // ... child nodes
 30  *        )
 31  *     )
 32  * )
 33  * </pre></code>
 34  *
 35  * <h2>Optics</h2>
 36  * <p>As you saw in the above example, a Camera has an <em>optics</em> property that defines the way that it projects light to
 37  * form the view. Supported types are described below.</p>
 38  *
 39  * <p><b>Perspective </b></p><p>Perspective projection embodies the appearance of objects relative to their
 40  * distance from the view point. It implicitly defines a frustum that embodies the view volume. The example below sets
 41  * the default properties for a projection:</p><pre><code>
 42  * var p = new SceneJS.Camera({
 43  *       optics: {
 44  *           type   : "perspective",
 45  *           fovy   : 60.0,           // Horizontal field of view in degrees
 46  *           aspect : 1.0,            // Aspect ratio of the field of view
 47  *           near   : 0.10,           // Distance of the near clipping plane
 48  *           far    : 10000.0         // Distance of the far clipping plane
 49  *       },
 50  *
 51  *       // ... child nodes
 52  * )
 53  * </pre></code>
 54  *
 55  * <p><b>Frustum</b></p><p>Frustum projection is effectively the same as perspective, providing you with the ability
 56  * to explicitly set the view frustum, which can be useful if you want it to be asymmetrical. The example below sets
 57  * the default properties for a frustum:</p><pre><code>
 58  * var p = new SceneJS.Camera({
 59  *       optics: {
 60  *           type   : "frustum",
 61  *           left   : -0.02,
 62  *           bottom : -0.02,
 63  *           near   :  0.1,
 64  *           right  :  0.02,
 65  *           top    :  0.02,
 66  *           far    :  1000.0
 67  *       },
 68  *
 69  *       // ... child nodes
 70  * )
 71  * </pre></code>
 72  *
 73  * <p><b>Ortho</b></p><p>Orthographic, or parallel, projections consist of those that involve no perspective correction.
 74  * There is no adjustment for distance from the camera made in these projections, meaning objects on the screen
 75  * will appear the same size no matter how close or far away they are. The example below specifies the default view
 76  * volume for orthographic projection:</p><pre><code>
 77  * var p = new SceneJS.Camera({
 78  *       optics: {
 79  *           type   : "ortho",
 80  *           left : -1.0,
 81  *           right : 1.0,
 82  *           bottom : -1.0,
 83  *           top : 1.0,
 84  *           near : 0.1,
 85  *           far : 1000.0
 86  *    },
 87  *
 88  *    // ... child nodes
 89  * )
 90  * </pre></code>
 91  *
 92  * @extends SceneJS.Node
 93  * @constructor
 94  * Create a new SceneJS.Camera
 95  * @param {Object} cfg  Config object or function, followed by zero or more child nodes
 96  */
 97 SceneJS.Camera = function() {
 98     SceneJS.Node.apply(this, arguments);
 99     this._nodeType = "camera";
100     this._optics = {
101         type: "perspective",
102         fovy : 60.0,
103         aspect : 1.0,
104         near : 0.10,
105         far : 5000.0
106     };
107     if (this._fixedParams) {
108         this._init(this._getParams());
109     }
110 };
111 
112 SceneJS._inherit(SceneJS.Camera, SceneJS.Node);
113 
114 /**
115  * Sets projection properties on the camera.
116  *
117  * @param {Object} optics Projection properties
118  * @returns {SceneJS.Camera} this
119  */
120 SceneJS.Camera.prototype.setOptics = function(optics) {
121     if (optics.type == "ortho") {
122         this._optics = {
123             type: optics.type,
124             left : optics.left || -1.0,
125             bottom : optics.bottom || -1.0,
126             near : optics.near || 0.1,
127             right : optics.right || 1.00,
128             top : optics.top || 1.0,
129             far : optics.far || 5000.0
130         };
131     } else if (optics.type == "frustum") {
132         this._optics = {
133             type: optics.type,
134             left : optics.left || -1.0,
135             bottom : optics.bottom || -1.0,
136             near : optics.near || 0.1,
137             right : optics.right || 1.00,
138             top : optics.top || 1.0,
139             far : optics.far || 5000.0
140         };
141     } else  if (optics.type == "perspective") {
142         this._optics = {
143             type: optics.type,
144             fovy : optics.fovy || 60.0,
145             aspect: optics.aspect || 1.0,
146             near : optics.near || 0.1,
147             far : optics.far || 5000.0
148         };
149     } else if (!optics.type) {
150         throw SceneJS._errorModule.fatalError(
151                 new SceneJS.errors.InvalidNodeConfigException(
152                         "SceneJS.Camera configuration invalid: optics type not specified - " +
153                         "supported types are 'perspective', 'frustum' and 'ortho'"));
154     } else {
155         throw SceneJS._errorModule.fatalError(
156                 new SceneJS.errors.InvalidNodeConfigException(
157                         "SceneJS.Camera configuration invalid: optics type not supported - " +
158                         "supported types are 'perspective', 'frustum' and 'ortho'"));
159     }
160     this._memoLevel = 0;
161     return this;
162 };
163 
164 /**
165  * Gets the camera's projection properties
166  * @returns {Object} Projection properties
167  */
168 SceneJS.Camera.prototype.getOptics = function() {
169     var optics = {};
170     for (var key in this._optics) {
171         if (this._optics.hasOwnProperty(key)) {
172             optics[key] = this._optics[key];
173         }
174     }
175     return optics;
176 };
177 
178 // Override
179 SceneJS.Camera.prototype._render = function(traversalContext, data) {
180     if (this._memoLevel == 0) {
181         if (!this._fixedParams) {
182             this._init(this._getParams(data));
183         } else {
184             this._memoLevel = 1;
185         }
186 
187         if (this._optics.type == "ortho") {
188             this._transform = {
189                 type: this._optics.type,
190                 matrix:SceneJS._math_orthoMat4c(
191                         this._optics.left,
192                         this._optics.right,
193                         this._optics.bottom,
194                         this._optics.top,
195                         this._optics.near,
196                         this._optics.far)
197             };
198         } else if (this._optics.type == "frustum") {
199             this._transform = {
200                 type: this._optics.type,
201                 matrix:SceneJS._math_frustumMatrix4(
202                         this._optics.left,
203                         this._optics.right,
204                         this._optics.bottom,
205                         this._optics.top,
206                         this._optics.near,
207                         this._optics.far)
208             };
209         } else if (this._optics.type == "perspective") {
210             this._transform = {
211                 type: this._optics.type,
212                 matrix:SceneJS._math_perspectiveMatrix4(
213                         this._optics.fovy * Math.PI / 180.0,
214                         this._optics.aspect,
215                         this._optics.near,
216                         this._optics.far)
217             };
218         }
219     }
220     var prevTransform = SceneJS._projectionModule.getTransform();
221     SceneJS._projectionModule.setTransform(this._transform);
222     this._renderNodes(traversalContext, data);
223     SceneJS._projectionModule.setTransform(prevTransform);
224 };
225 
226 // @private
227 SceneJS.Camera.prototype._init = function(params) {
228     if (params.optics) {
229         this.setOptics(params.optics);
230     }
231 };

233 /** Factory function that returns a new {@link SceneJS.Camera} instance
234  * @param {Arguments} args Variable arguments that are passed to the {@link SceneJS.Camera} constructor
235  * @returns {SceneJS.Camera}
236  */
237 SceneJS.camera = function() {
238     var n = new SceneJS.Camera();
239     SceneJS.Camera.prototype.constructor.apply(n, arguments);
240     return n;
241 };
