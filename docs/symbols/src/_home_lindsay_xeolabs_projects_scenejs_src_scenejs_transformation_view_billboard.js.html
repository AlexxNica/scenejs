  4 SceneJS.billboard = function() {

  6     var vxfBackend = SceneJS._backends.getBackend('view-transform');

  8 
  9     return SceneJS._utils.createNode(
 10             "billboard",
 11             cfg.children,
 12 
 13             new (function() {
 14                 this._render = function(traversalContext, data) {
 15 
 16                     var viewXform = vxfBackend.getTransform();
 17                     var modelXform = vxfBackend.getTransform();
 18 
 19                     var mat = SceneJS_math_billboardMat(viewXform.matrix);
 20 
 21                     var xform = {
 22                         localMatrix: mat,
 23                         matrix: SceneJS_math_mulMat4(modelXform.matrix, mat),
 24                         fixed: false
 25                     };
 26                     mxfBackend.setTransform(xform);

 28                     mxfBackend.setTransform(modelXform);
 29                 };
 30             })());
