1 /**
  2  * Scene node that selects which of its children are active.
  3  */
  4 SceneJS.selector = function() {
  5     var errorBackend = SceneJS._error;
  6     var cfg = SceneJS._utils.getNodeConfig(arguments);
  7 
  8     /* Augment the basic node type
  9      */
 10     return (function($) {
 11 
 12         var selection = 0;
 13 
 14         $._setSelection = function(index) {
 15 
 16         };
 17 
 18         $._render = function(traversalContext, data) {
 19             var params = cfg.getParams(data); // Don't memoize because selection is probably dynamic
 20             if (!params.selection) {
 21                 errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException
 22                         ("SceneJS.selection node mandatory property missing: selection"));
 23             } else {
 24                 var max = cfg.children.length;
 25                 var i = params.selection.length;
 26                 while (--i) {
 27                     if (i < 0 || i >= max) {
 28                         errorBackend.fatalError(new SceneJS.exceptions.InvalidNodeConfigException
 29                                 ("SceneJS.selection selection property out of range"));
 30                     }
 31                     $._renderNodes(i, traversalContext, data);
 32                 }
 33             }
 34         };
 35         return $;
 36     })(SceneJS.node.apply(this, arguments));
 37 };
