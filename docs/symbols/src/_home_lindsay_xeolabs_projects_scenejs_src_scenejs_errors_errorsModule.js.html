1 /**
  2  * Backend module that provides single point through which exceptions may be raised
  3  *
  4  * @private
  5  */
  6 var SceneJS_errorModule = new (function() {
  7 
  8     SceneJS_eventModule.onEvent(
  9             SceneJS_eventModule.SCENE_ACTIVATED,
 10             function() {
 11                 var time = (new Date()).getTime();
 12                 SceneJS_eventModule.fireEvent(SceneJS_eventModule.TIME_UPDATED, time);
 13             });
 14 
 15     // @private
 16     this.fatalError = function(e) {
 17         SceneJS_eventModule.fireEvent(SceneJS_eventModule.ERROR, {
 18             exception: e,
 19             fatal: true
 20         });
 21         throw e;
 22     };
 23 
 24     // @private
 25     this.error = function(e) {
 26         SceneJS_eventModule.fireEvent(SceneJS_eventModule.ERROR, {
 27             exception: e,
 28             fatal: false
 29         });
 30     };
 31 })();
