1 /**
  2  * @class A scene node that applies a model-space translate transform to the nodes within its subgraph.
  3  * @extends SceneJS.Node
  4  * <p><b>Example</b></p><p>A cube translated along the X axis.</b></p><pre><code>
  5  * var translate = new SceneJS.Translate({
  6  *       x: 5.0,
  7  *       y: 0.0,
  8  *       z: 0.0
  9  *   },
 10  *
 11  *      new SceneJS.objects.Cube()
 12  * )
 13  * </pre></code>
 14  * @constructor
 15  * Create a new SceneJS.Translate
 16  * @param {Object} config  Config object or function, followed by zero or more child nodes
 17  */
 18 SceneJS.Translate = function() {
 19     SceneJS.Node.apply(this, arguments);
 20     this._nodeType = "translate";
 21     this._mat = null;
 22     this._xform = null;
 23     this._x = 0;
 24     this._y = 0;
 25     this._z = 1;
 26     if (this._fixedParams) {
 27         this._init(this._getParams());
 28     }
 29 };
 30 
 31 SceneJS._inherit(SceneJS.Translate, SceneJS.Node);
 32 
 33 /**
 34  * Sets the translation vector
 35  * @param {object} xyz The vector - eg. {x: 0, y: 1, z: 0}
 36  * @returns {SceneJS.Translate} this
 37  */
 38 SceneJS.Translate.prototype.setXYZ = function(xyz) {
 39     var x = xyz.x || 0;
 40     var y = xyz.y || 0;
 41     var z = xyz.z || 0;
 42     this._x = x;
 43     this._y = y;
 44     this._z = z;
 45     this._memoLevel = 0;
 46     return this;
 47 };
 48 
 49 /** Returns the translation vector
 50  * @returns {Object} the vector, eg. {x: 0, y: 1, z: 0}
 51  */
 52 SceneJS.Translate.prototype.getXYZ = function() {
 53     return {
 54         x: this._x,
 55         y: this._y,
 56         z: this._z
 57     };
 58 };

117 SceneJS.Translate.prototype._init = function(params) {
118     this.setXYZ({x : params.x, y: params.y, z: params.z });
119 };
120 
121 SceneJS.Translate.prototype._render = function(traversalContext, data) {
122     if (this._memoLevel == 0) {
123         if (!this._fixedParams) {
124             this._init(this._getParams(data));
125         } else {
126             this._memoLevel = 1;
127         }
128         this._mat = SceneJS_math_translationMat4v([this._x, this._y, this._z]);
129     }
130     var superXform = SceneJS_modelTransformModule.getTransform();
131     if (this._memoLevel < 2) {
132         var instancing = SceneJS_instancingModule.instancing();
133         var tempMat = SceneJS_math_mulMat4(superXform.matrix, this._mat);
134         this._xform = {
135             localMatrix: this._mat,
136             matrix: tempMat,
137             fixed: superXform.fixed && this._fixedParams && !instancing
138         };
139         if (this._memoLevel == 1 && superXform.fixed && !instancing) {   // Bump up memoization level if model-space fixed
140             this._memoLevel = 2;
141         }
142     }
143     SceneJS_modelTransformModule.setTransform(this._xform);
144     this._renderNodes(traversalContext, data);
145     SceneJS_modelTransformModule.setTransform(superXform);
146 };

148 /** Function wrapper to support functional scene definition
149  */
150 SceneJS.translate = function() {
151     var n = new SceneJS.Translate();
152     SceneJS.Translate.prototype.constructor.apply(n, arguments);
153     return n;
154 };
