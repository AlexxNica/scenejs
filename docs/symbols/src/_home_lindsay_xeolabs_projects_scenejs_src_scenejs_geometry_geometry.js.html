  2  * @class A scene node that defines an element of geometry.
  3  * <p><b>Live Examples</b></p>
  4  * <ul><li><a target = "other" href="http://bit.ly/scenejs-geometry-example">Example 1</a></li></ul>
  5  * <p><b>Example Usage</b></p><p>Definition of a cube, with normals and UV texture coordinates, with coordinates shown here only for the first face:</b></p><pre><code>
  6  * var g = new SceneJS.Geometry({
  7  *
  8  *        // Optional geometry type ID. If some other Geometry node with this type has previously
  9  *        // been rendered in the scene graph then this Geometry will just re-use the geometry
 10  *        // (IE. vertex buffers etc.) that were created by it.
 11  *
 12  *        type: "cube_5_5_5",   // Optional
 13  *
 14  *        // Mandatory primitive type - "points", "lines", "line-loop", "line-strip", "triangles",
 15  *        // "triangle-strip" or "triangle-fan".
 16  *
 17  *        primitive: "triangles",
 18  *
 19  *        // Mandatory 3D positions - eight for our cube, each one spaining three array elements for X,Y and Z
 20  *
 21  *        positions : [
 22  *
 23  *            // Front cube face - vertices 0,1,2,3
 24  *
 25  *            5, 5, 5,
 26  *            -5, 5, 5,
 27  *            -5,-5, 5,
 28  *            5,-5, 5,
 29  *
 30  *            //...
 31  *        ],
 32  *
 33  *        // Optional normal vectors, one for each vertex. If you omit these, then cube will not be shaded.
 34  *
 35  *        normals : [
 36  *
 37  *            // Vertices 0,1,2,3
 38  *
 39  *            0, 0, -1,
 40  *            0, 0, -1,
 41  *            0, 0, -1,
 42  *            0, 0, -1,
 43  *
 44  *            //...
 45  *        ],
 46  *
 47  *        // Optional 2D texture coordinates corresponding to the 3D positions defined above -
 48  *        // eight for our cube, each one spanning two array elements for X and Y. If you omit these, then the cube
 49  *        // will never be textured.
 50  *
 51  *        uv : [
 52  *
 53  *            // Vertices 0,1,2,3
 54  *
 55  *            5, 5,
 56  *            0, 5,
 57  *            0, 0,
 58  *            5, 0,
 59  *
 60  *            // ...
 61  *        ],
 62  *
 63  *        // Optional coordinates for a second UV layer - just to illustrate their availability
 64  *
 65  *        uv2 : [
 66  *
 67  *        ],
 68  *
 69  *        // Mandatory indices - these organise the positions, normals and uv texture coordinates into geometric
 70  *        // primitives in accordance with the "primitive" parameter, in this case a set of three indices for each triangle.
 71  *        // Note that each triangle in this example is specified in counter-clockwise winding order. You can specify them in
 72  *        // clockwise order if you configure the SceneJS.renderer node's frontFace property as "cw", instead of the
 73  *        // default "ccw".
 74  *
 75  *        indices : [
 76  *
 77  *            // Vertices 0,1,2,3
 78  *
 79  *            0, 1, 2,
 80  *            0, 2, 3,
 81  *
 82  *            // ...
 83  *        ]
 84  * });

 86  * @extends SceneJS.Node
 87  * @since Version 0.7.4
 88  * @constructor
 89  * Create a new SceneJS.Geometry
 90  * @param {Object} [cfg] Static configuration object
 91  * @param {String} cfg.type Optional geometry type - Geometry nodes with same value of this will share the same vertex buffers
 92  * @param {String} cfg.primitive The primitive type - "points", "lines", "line-loop", "line-strip", "triangles", "triangle-strip" or "triangle-fan"
 93  * @param {double[]} cfg.positions Flattened array of 3D coordinates, three elements each
 94  * @param {double[]} [cfg.normals = []] Flattened array of 3D vertex normal vectors, three elements each
 95  * @param {double[]} [cfg.uv = []] Flattened array of 2D UV-space coordinates for the first texture layer, two elements each
 96  * @param {double[]} [cfg.uv2 = []] Flattened array of 2D UV-space coordinates for the second texture layer, two elements each
 97  * @param {int[]} cfg.indices Flattened array of indices to index the other arrays per the specified primitive type
 98  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 99  * @param {...SceneJS.Node} [childNodes] Child nodes
100  * @since Version 0.7.3
101  */
102 SceneJS.Geometry = function() {
103     SceneJS.Node.apply(this, arguments);
104     this._nodeType = "geometry";
105     this._geo = null;  // Holds geometry when configured as given arrays
106     this._create = null; // Callback to create geometry
107     this._type = null; // Optional geometry type ID
108     this._handle = null; // Handle to created geometry
109 };
110 
111 SceneJS._inherit(SceneJS.Geometry, SceneJS.Node);
112 
113 SceneJS.Geometry.prototype._init = function(params) {
114     this._type = params.type;       // Optional - can be null
115     if (params.create instanceof Function) {
116         this._create = params.create;
117     } else {
118         this._geo = {
119             positions : params.positions || [],
120             normals : params.normals || [],
121             colors : params.colors || [],
122             indices : params.indices || [],
123             uv : params.uv || [],
124             primitive : params.primitive || "triangles"
125         };
126     }
127 };
128 
129 // @private
130 SceneJS.Geometry.prototype._render = function(traversalContext, data) {
131     if (!this._geo && !this._create) { // Dynamically configured
132         this._init(this._getParams(data));
133     }
134     if (this._handle) { // Was created before - test if not evicted since
135         if (!SceneJS._geometryModule.testGeometryExists(this._handle)) {
136             this._handle = null;
137         }
138     }
139     if (!this._handle) { // Either not created yet or has been evicted
140         if (this._create) { // Use callback to create
141             this._handle = SceneJS._geometryModule.createGeometry(this._type, this._create());
142         } else { // Or supply arrays
143             this._handle = SceneJS._geometryModule.createGeometry(this._type, this._geo);
144         }
145     }
146     SceneJS._geometryModule.drawGeometry(this._handle);
147     this._renderNodes(traversalContext, data);
148 };

150 /** Factory function that returns a new {@link SceneJS.Geometry} instance
151  * @param {Object} [cfg] Static configuration object
152  * @param {String} cfg.primitive The primitive type - "points", "lines", "line-loop", "line-strip", "triangles", "triangle-strip" or "triangle-fan"
153  * @param {double[]} cfg.positions Flattened array of 3D coordinates, three elements each
154  * @param {double[]} [cfg.normals = []] Flattened array of 3D vertex normal vectors, three elements each
155  * @param {double[]} [cfg.uv = []] Flattened array of 2D UV-space coordinates for the first texture layer, two elements each
156  * @param {double[]} [cfg.uv2 = []] Flattened array of 2D UV-space coordinates for the second texture layer, two elements each
157  * @param {int[]} cfg.indices  Flattened array of indices to index the other arrays per the specified primitive type
158  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
159  * @param {...SceneJS.Node} [childNodes] Child nodes
160  * @returns {SceneJS.Geometry}
161  * @since Version 0.7.1
162  */
163 SceneJS.geometry = function() {
164     var n = new SceneJS.Geometry();
165     SceneJS.Geometry.prototype.constructor.apply(n, arguments);
166     return n;
167 };
