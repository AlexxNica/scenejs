1 SceneJS._utils.ns("SceneJS.geometry");
  2 
  3 /**
  4  * An element of geometry
  5  */
  6 (function() {
  7 
  8     /**
  9      * Defines an element of geometry within a scene.
 10      * The geometry within these node types is not modifiable.
 11      * @constructor
 12      */
 13     SceneJS.geometry = function() {
 14 
 15         var cfg = SceneJS._utils.getNodeConfig(arguments);
 16         var geometryBackend = SceneJS._backends.getBackend('geometry');
 17 
 18         return SceneJS._utils.createNode(
 19                 "geometry",
 20                 cfg.children,
 21 
 22                 new (function() {
 23 
 24                     var params;
 25                     var type;
 26                     var create;
 27                     var geo = {};
 28 
 29                     this._render = function(traversalContext, data) {
 30 
 31                         /* Dynamic config only happens first time
 32                          */
 33                         if (!params) {
 34                             params = cfg.getParams(data);
 35                             if (params.create instanceof Function) {
 36 
 37                                 /* Create must not be a dynamic config function!
 38                                  */
 39                                 create = params.create;
 40                             } else {
 41                                 geo = {
 42                                     positions : SceneJS._utils.getParam(params.positions, data) || [],
 43                                     normals : SceneJS._utils.getParam(params.normals, data) || [],
 44                                     colors : SceneJS._utils.getParam(params.colors, data) || [],
 45                                     indices : SceneJS._utils.getParam(params.indices, data) || [],
 46                                     uv : SceneJS._utils.getParam(params.uv, data) || [],
 47                                     primitive : SceneJS._utils.getParam(params.primitive, data) || "triangles"
 48                                 };
 49                             }
 50                         }
 51 
 52                         /* Check geometry not evicted
 53                          */
 54                         if (type) {
 55                             if (!geometryBackend.testGeometryExists(type)) {
 56                                 type = null;
 57                             }
 58                         }
 59 
 60                         /* type is null if geometry evicted or not yet defined
 61                          */
 62                         if (!type) {
 63                             if (create) {
 64 
 65                                 /* Type generated if null
 66                                  */
 67                                 type = geometryBackend.createGeometry(params.type, create()); // Lazy-create geometry through callback
 68                             } else {
 69                                 type = geometryBackend.createGeometry(params.type, geo);
 70                             }
 71                         }
 72                         geometryBackend.drawGeometry(type);
 73                         this._renderChildren(traversalContext, data);
 74                     };
 75                 })());
 76     };
 77 })();
