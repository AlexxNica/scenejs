1 /**
  2  * @class A scene node that defines an element of geometry.
  3  * <p><b>Live Examples</b></p>
  4  * <ul><li><a target = "other" href="http://bit.ly/scenejs-geometry-example">Example 1</a></li></ul>
  5  * <p><b>Example Usage</b></p><p>Definition of a cube, with normals and UV texture coordinates, with coordinates shown here only for the first face:</b></p><pre><code>
  6  * var g = new SceneJS.Geometry({
  7  *
  8  *        // Mandatory primitive type - "points", "lines", "line-loop", "line-strip", "triangles",
  9  *        // "triangle-strip" or "triangle-fan".
 10  *
 11  *        primitive: "triangles",
 12  *
 13  *        // Mandatory vertices - eight for our cube, each one spaining three array elements for X,Y and Z
 14  *
 15  *        positions : [
 16  *
 17  *            // Front cube face - vertices 0,1,2,3
 18  *
 19  *            5, 5, 5,
 20  *            -5, 5, 5,
 21  *            -5,-5, 5,
 22  *            5,-5, 5,
 23  *
 24  *            //...
 25  *        ],
 26  *
 27  *        // Optional normal vectors, one for each vertex. If you omit these, then cube will not be shaded.
 28  *
 29  *        normals : [
 30  *
 31  *            // Vertices 0,1,2,3
 32  *
 33  *            0, 0, -1,
 34  *            0, 0, -1,
 35  *            0, 0, -1,
 36  *            0, 0, -1,
 37  *
 38  *            //...
 39  *        ],
 40  *
 41  *        // Optional 2D texture coordinates corresponding to the 3D positions defined above -
 42  *        // eight for our cube, each one spanning two array elements for X and Y. If you omit these, then the cube
 43  *        // will never be textured.
 44  *
 45  *        uv : [
 46  *
 47  *            // Vertices 0,1,2,3
 48  *
 49  *            5, 5,
 50  *            0, 5,
 51  *            0, 0,
 52  *            5, 0,
 53  *
 54  *            // ...
 55  *        ],
 56  *
 57  *        // Optional coordinates for a second UV layer - just to illustrate their availability
 58  *
 59  *        uv2 : [
 60  *
 61  *        ],
 62  *
 63  *        // Mandatory indices - these organise the positions, normals and uv texture coordinates into geometric
 64  *        // primitives in accordance with the "primitive" parameter, in this case a set of three indices for each triangle.
 65  *        // Note that each triangle in this example is specified in counter-clockwise winding order. You can specify them in
 66  *        // clockwise order if you configure the SceneJS.renderer node's frontFace property as "cw", instead of the
 67  *        // default "ccw".
 68  *
 69  *        indices : [
 70  *
 71  *            // Vertices 0,1,2,3
 72  *
 73  *            0, 1, 2,
 74  *            0, 2, 3,
 75  *
 76  *            // ...
 77  *        ]
 78  * });
 79  *  </pre></code>
 80  * @extends SceneJS.Node
 81  * @constructor
 82  * Create a new SceneJS.Geometry
 83  * @param {Object} config The config object, followed by zero or more child nodes
 84  */
 85 SceneJS.Geometry = function() {
 86     SceneJS.Node.apply(this, arguments);
 87     this._nodeType = "geometry";
 88     this._geo = null;  // Holds geometry when configured as given arrays
 89     this._create = null; // Callback to create geometry
 90     this._type = null; // Optional geometry type ID
 91     this._handle = null; // Handle to created geometry
 92 };
 93 
 94 SceneJS._inherit(SceneJS.Geometry, SceneJS.Node);
 95 
 96 // @private
 97 SceneJS.Geometry.prototype._render = function(traversalContext, data) {
 98     if (!this._geo && !this._create) {
 99 
100         /* This is a dynamically-configured node
101          */
102         var params = this._getParams(data);
103         if (this._init) {
104 
105             /* Subclass implements an init method which will set
106              * _geo or _create on this node
107              */
108             this._init(params);
109         } else {
110 
111             /* Geometry provided in params
112              */
113             this._type = params.type;                  // Optional - can be null
114             if (params.create instanceof Function) {
115                 this._create = params.create;
116             } else {
117                 this._geo = {
118                     positions : params.positions || [],
119                     normals : params.normals || [],
120                     colors : params.colors || [],
121                     indices : params.indices || [],
122                     uv : params.uv || [],
123                     primitive : params.primitive || "triangles"
124                 };
125             }
126         }
127         this._handle = null;
128     }
129     if (this._handle) { // Was created before - test if not evicted since
130         if (!SceneJS_geometryModule.testGeometryExists(this._handle)) {
131             this._handle = null;
132         }
133     }
134     if (!this._handle) { // Either not created yet or has been evicted
135         if (this._create) { // Use callback to create
136             this._handle = SceneJS_geometryModule.createGeometry(this._type, this._create());
137         } else { // Or supply arrays
138             this._handle = SceneJS_geometryModule.createGeometry(this._type, this._geo);
139         }
140     }
141     SceneJS_geometryModule.drawGeometry(this._handle);
142     this._renderNodes(traversalContext, data);
143 };

145 /** Returns a new SceneJS.Geometry instance
146  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Geometry constructor
147  * @returns {SceneJS.Geometry}
148  */
149 SceneJS.geometry = function() {
150     var n = new SceneJS.Geometry();
151     SceneJS.Geometry.prototype.constructor.apply(n, arguments);
152     return n;
153 };
