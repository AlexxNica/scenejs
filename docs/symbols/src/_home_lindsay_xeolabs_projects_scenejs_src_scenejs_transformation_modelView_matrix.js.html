1 /**
  2  * @class A scene node that defines a 4x4 matrix to transform the nodes within its subgraph.
  3  * @extends SceneJS.Node
  4  * <p><b>Example</b></p><p>A cube translated along the X, Y and Z axis.</b></p><pre><code>
  5  * var mat = new SceneJS.Matrix({
  6  *       elements : [
  7  *              1, 0, 0, 10,
  8  *              0, 1, 0, 5,
  9  *              0, 0, 1, 3,
 10  *              0, 0, 0, 1
 11  *          ]
 12  *   },
 13  *
 14  *      new SceneJS.objects.Cube()
 15  * )
 16  * </pre></code>
 17  * @constructor
 18  * Create a new SceneJS.Matrix
 19  * @param {Object} config  Config object or function, followed by zero or more child nodes
 20  */
 21 SceneJS.Matrix = function() {
 22     SceneJS.Node.apply(this, arguments);
 23     this._nodeType = "matrix";
 24     this._mat = SceneJS._math_identityMat4();
 25     this._xform = null;
 26     if (this._fixedParams) {
 27         this._init(this._getParams());
 28     }
 29 };
 30 
 31 SceneJS._inherit(SceneJS.Matrix, SceneJS.Node);
 32 
 33 /**
 34  * Sets the matrix elements
 35  * @param {Array} elements One-dimensional array of matrix elements
 36  * @returns {SceneJS.Matrix} this
 37  */
 38 SceneJS.Matrix.prototype.setElements = function(elements) {
 39     if (!elements) {
 40         throw SceneJS._errorModule.fatalError(new SceneJS.InvalidNodeConfigException("SceneJS.Matrix elements undefined"));
 41     }
 42     if (elements.length != 16) {
 43         throw SceneJS._errorModule.fatalError(new SceneJS.InvalidNodeConfigException("SceneJS.Matrix elements should number 16"));
 44     }
 45     for (var i = 0; i < 16; i++) {
 46         this._mat[i] = elements[i];
 47     }
 48     this._memoLevel = 0;
 49     return this;
 50 };
 51 
 52 /** Returns the matrix elements
 53  * @returns {Object} One-dimensional array of matrix elements
 54  */
 55 SceneJS.Matrix.prototype.getElements = function() {
 56     var elements = new Array(16);
 57     for (var i = 0; i < 16; i++) {
 58         elements[i] = this._mat[i];
 59     }
 60     return elements;
 61 };
 62 
 63 SceneJS.Matrix.prototype._init = function(params) {
 64     if (params.elements) {
 65         this.setElements(params.elements);
 66     }
 67 };
 68 
 69 SceneJS.Matrix.prototype._render = function(traversalContext, data) {
 70     if (this._memoLevel == 0) {
 71         if (!this._fixedParams) {
 72             this._init(this._getParams(data));
 73         } else {
 74             this._memoLevel = 1;
 75         }
 76     }
 77     var superXform = SceneJS._modelViewTransformModule.getTransform();
 78     if (this._memoLevel < 2) {
 79         var instancing = SceneJS._instancingModule.instancing();
 80 
 81         /* When building a view transform, apply the inverse of the matrix
 82          * to correctly transform the SceneJS.Camera
 83          */
 84         var mat = SceneJS._modelViewTransformModule.isBuildingViewTransform()
 85                 ? SceneJS._math_inverseMat4(this._mat)
 86                 : this._mat;
 87         
 88         var tempMat = SceneJS._math_mulMat4(superXform.matrix, mat);
 89         this._xform = {
 90             localMatrix: this._mat,
 91             matrix: tempMat,
 92             fixed: superXform.fixed && this._fixedParams && !instancing
 93         };
 94         if (this._memoLevel == 1 && superXform.fixed && !instancing) {   // Bump up memoization level if model-space fixed
 95             this._memoLevel = 2;
 96         }
 97     }
 98     SceneJS._modelViewTransformModule.setTransform(this._xform);
 99     this._renderNodes(traversalContext, data);
100     SceneJS._modelViewTransformModule.setTransform(superXform);
101 };
102 
103 /** Factory function that returns a new {@link SceneJS.Matrix} instance
104  */
105 SceneJS.matrix = function() {
106     var n = new SceneJS.Matrix();
107     SceneJS.Matrix.prototype.constructor.apply(n, arguments);
108     return n;
109 };
