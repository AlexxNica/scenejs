1 /**
  2  * Scene node that routes messages logged by nodes in its subgraph to be written to a DIV element in the HTML document.
  3  * If you omit this node in your scene, SceneJS by default will look for a DIV with ID "_scenejs-default-logging" and
  4  * if that is found, will write logging there.
  5  *
  6  * @class SceneJS.loggingToPage
  7  * @extends SceneJS.logging
  8  */
  9 
 10 SceneJS.loggingToPage = function() {
 11     var cfg = SceneJS._utils.getNodeConfig(arguments);
 12 
 13     function findElement(elementId) {
 14         var element;
 15         if (!elementId) {
 16             elementId = SceneJS_webgl_DEFAULT_LOGGING_ID;
 17             element = document.getElementById(elementId);
 18             if (!element) {
 19                 SceneJS_errorModule.fatalError(new SceneJS.exceptions.PageLoggingElementNotFoundException
 20                         ("SceneJS.loggingToPage config 'elementId' omitted and no default element found with ID '"
 21                                 + SceneJS_webgl_DEFAULT_LOGGING_ID + "'"));
 22             }
 23         } else {
 24             element = document.getElementById(elementId);
 25             if (!element) {
 26                 elementId = SceneJS_webgl_DEFAULT_LOGGING_ID;
 27                 element = document.getElementById(elementId);
 28                 if (!element) {
 29                     SceneJS_errorModule.fatalError(
 30                             new SceneJS.exceptions.PageLoggingElementNotFoundException
 31                                     ("SceneJS.loggingToPage config 'elementId' does not match any elements in page and no " +
 32                                      "default element found with ID '" + SceneJS_webgl_DEFAULT_LOGGING_ID + "'"));
 33                 }
 34             }
 35         }
 36         return element;
 37     }
 38 
 39     /* Augment the basic node type
 40      */
 41     return (function($) {
 42 
 43         var funcs;
 44 
 45         $._render = function(traversalContext, data) {
 46             if (!funcs) {
 47                 var params = cfg.getParams();
 48                 var element = findElement(params.elementId);
 49                 funcs = {
 50                     warn : function (msg) {
 51                         element.innerHTML += "<p style=\"color:orange;\">" + msg + "</p>";
 52                     },
 53                     error : function (msg) {
 54                         element.innerHTML += "<p style=\"color:darkred;\">" + msg + "</p>";
 55                     },
 56                     debug : function (msg) {
 57                         element.innerHTML += "<p style=\"color:darkblue;\">" + msg + "</p>";
 58                     },
 59                     info : function (msg) {
 60                         element.innerHTML += "<p style=\"color:darkgreen;\">" + msg + "</p>";
 61                     }
 62                 };
 63             }
 64             var prevFuncs = SceneJS_loggingModule.getFuncs();
 65             SceneJS_loggingModule.setFuncs(funcs);
 66             $._renderNodes(traversalContext, data);
 67             SceneJS_loggingModule.setFuncs(prevFuncs);
 68         };
 69         return $;
 70     })(SceneJS.node.apply(this, arguments));
 71 };
