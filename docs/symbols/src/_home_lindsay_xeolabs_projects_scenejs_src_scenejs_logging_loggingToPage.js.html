1 /**
  2  * @class A scene node that allows you to intercept SceneJS logging at selected points within your
  3  * scene graph and write it to a specified DIV tag.
  4  * <p><b>Example Usage</b></p><p>Routing all logging within a given subgraph to alert popups:</b></p><pre><code>
  5  * var node = new SceneJS.Node(
  6  *
  7  *    // ...some nodes ...
  8  *
  9  *    new SceneJS.LoggingToPage({
 10  *           elementId: "myLoggingDiv"
 11  *       },
 12  *
 13  *       // ... sub-nodes we are in interesting
 14  *       //        intercepting logs from here ...
 15  *    )
 16  * );
 17  *
 18  * </pre></code>
 19  * @extends SceneJS.Node
 20  * @constructor
 21  * Create a new SceneJS.LoggingToPage
 22  * @param {Object} config  Config object or function, followed by zero or more child nodes
 23  */
 24 SceneJS.LoggingToPage = function() {
 25     SceneJS.Node.apply(this, arguments);
 26     this._nodeType = "logging-to-page";
 27     this._elementId = null;
 28     this._element = null;
 29     this._funcs = null; // Lazily initialised
 30     if (this._fixedParams) {
 31         this._init(this._getParams());
 32     }
 33 };
 34 
 35 /** ID ("_scenejs-default-logging") of element to which SceneJS will log to if found.
 36  * When no SceneJS.LoggingToPage is specified, or a SceneJS.LoggingToPage is
 37  * specified with no element ID.
 38  */
 39 SceneJS.LoggingToPage.DEFAULT_LOGGING_DIV_ID = "_scenejs-default-logging";
 40 
 41 SceneJS._inherit(SceneJS.LoggingToPage, SceneJS.Node);
 42 
 43 /**
 44  * Sets target DIV ID. If no value is given then ID will default to the value of
 45  * SceneJS.LoggingToPage.DEFAULT_LOGGING_DIV_ID.
 46  *
 47  * @param {string} elementId - ID of target DIV
 48  * @returns {SceneJS.LoggingToPage} This logging node
 49  * @throws {SceneJS.errors.DocumentElementNotFoundException} Element not found in document
 50  */
 51 SceneJS.LoggingToPage.prototype.setElementId = function(elementId) {
 52     if (elementId) {
 53         var element = document.getElementById(elementId);
 54         if (!element) {
 55             throw SceneJS._errorModule.fatalError(new SceneJS.errors.DocumentElementNotFoundException
 56                     ("SceneJS.LoggingToPage cannot find document element with ID '" + elementId + "'"));
 57         }
 58         this._elementId = elementId;
 59         this._element = element;
 60     } else {
 61         this._elementId = SceneJS.LoggingToPage.DEFAULT_LOGGING_DIV_ID;
 62         this._element = document.getElementById(SceneJS.LoggingToPage.DEFAULT_LOGGING_DIV_ID);
 63     }
 64     return this;
 65 };
 66 
 67 /** Returns the target DIV ID. If none has been given then the value will be that of
 68  * SceneJS.LoggingToPage.DEFAULT_LOGGING_DIV_ID.
 69  *
 70  * @returns {string} The DIV ID
 71  */
 72 SceneJS.LoggingToPage.prototype.getElementId = function() {
 73     return this._elementId;
 74 };
 75 
 76 // @private
 77 SceneJS.LoggingToPage.prototype._init = function(params) {
 78     this.setElementId(params.elementId);
 79 };
 80 
 81 // @private
 82 SceneJS.LoggingToPage.prototype._render = function(traversalContext, data) {
 83     if (!this._fixedParams) {
 84         this._init(this._getParams(data));
 85     }
 86     if (!this._funcs && this._element) {
 87         var element = this._element;
 88         this._funcs = {
 89             warn : function (msg) {
 90                 element.innerHTML += "<p style=\"color:orange;\">" + msg + "</p>";
 91             },
 92             error : function (msg) {
 93                 element.innerHTML += "<p style=\"color:darkred;\">" + msg + "</p>";
 94             },
 95             debug : function (msg) {
 96                 element.innerHTML += "<p style=\"color:darkblue;\">" + msg + "</p>";
 97             },
 98             info : function (msg) {
 99                 element.innerHTML += "<p style=\"color:darkgreen;\">" + msg + "</p>";
100             }
101         };
102     }
103     if (this._element) {
104         var prevFuncs = SceneJS._loggingModule.getFuncs();
105         SceneJS._loggingModule.setFuncs(this._funcs);
106         this._renderNodes(traversalContext, data);
107         SceneJS._loggingModule.setFuncs(prevFuncs);
108     } else {
109         this._renderNodes(traversalContext, data);
110     }
111 };
112 
113 /** Returns a new SceneJS.LoggingToPage instance
114  * @param {Arguments} args Variable arguments that are passed to the SceneJS.LoggingToPage constructor
115  * @returns {SceneJS.LoggingToPage}
116  */
117 SceneJS.loggingToPage = function() {
118     var n = new SceneJS.LoggingToPage();
119     SceneJS.LoggingToPage.prototype.constructor.apply(n, arguments);
120     return n;
121 };
