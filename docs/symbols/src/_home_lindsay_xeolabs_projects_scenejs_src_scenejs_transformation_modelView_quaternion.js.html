1 /**
  2  * @class Scene node that provides a quaternion-encoded rotation.
  3  *
  4  * <p>This node provides a convenient way to define a 3D rotation that can be rotated continually on any axis without
  5  * gimbal lock or significant numeric instability.</p>
  6  * <p><b>Example 1</b></p><p>Below is a Quaternion created from an "axis-angle" representation given as an axis to
  7  * rotate about, along with an angle in degrees. The optional <em>rotations</em> parameter defines a sequence of
  8  * rotations to then rotate the quaternion by. Finally, we apply one more rotation to the Quaternion node instance
  9  * through its {@link #rotate} method. </p>
 10  * </p><pre><code>
 11  * var q = new SceneJS.Quaternion({
 12  *
 13  *         // "Base" rotation
 14  *
 15  *         x : 0.0, y : 0.0, z : 0.0, angle : 0.0,      // No rotation, sets identity quaternion
 16  *
 17  *         // Sequence of rotations to apply on top of the base rotation
 18  *
 19  *         rotations: [
 20  *                 { x : 0, y : 0, z : 1, angle : 45 }, // Rotate 45 degrees about Z the axis
 21  *                 { x : 1, y : 0, z : 0, angle : 20 }, // Rotate 20 degrees about X the axis
 22  *                 { x : 0, y : 1, z : 0, angle : 90 }, // Rotate 90 degrees about Y the axis
 23  *              ]
 24  *          },
 25  *
 26  *          // .. Child nodes ...
 27  *     });
 28  *
 29  * // rotate one more time, 15 degrees about the Z axis
 30  *
 31  * q.rotate({ x : 0, y : 0, z : 1, angle : 15 });
 32  * </pre></code>
 33  * <p>Quaternions are designed to be animated. Typically, we would dynamically provide rotation updates to them at
 34  * render time, either from an interpolator node or human interaction.</p>
 35  * <p><b>Example 2</b></p><p>Below is a Quaternion that is dynamically configured with a callback that applies
 36  * rotation updates that are injected into the scene each time it is rendered. Note that the callback does not define
 37  * the base rotation - if it did, then the Quaternion would reset before each new rotation is applied, rather than
 38  * accumulate the rotations as intended. Note that the rotations could be generated by mouse drags to simulate
 39  * a trackball.</p>
 40  * <pre><code>
 41  *  var exampleScene = new SceneJS.Scene(
 42  *
 43  *      // ... sibling nodes
 44  *
 45  *      new SceneJS.Quaternion(
 46  *            function(data) {
 47  *                      return {
 48  *                         rotations: data.get("rotations");
 49  *                      };
 50  *           },
 51  *
 52  *           // ... chld nodes ...
 53  *      )
 54  *
 55  *      //... sibling nodes
 56  * );
 57  *
 58  * exampleScene.render({ rotations: [ { x : 0, y : 0, z : 1, angle : 45 } ] });
 59  * exampleScene.render({ rotations: [ { x : 1, y : 0, z : 0, angle : 20 } ] });
 60  * exampleScene.render({ rotations: [ { x : 0, y : 1, z : 0, angle : 90 } ] });
 61  * </code></pre>
 62  * @extends SceneJS.Node
 63  * @constructor
 64  * Create a new SceneJS.Quaternion
 65  * @param {Object} [cfg] Static configuration object
 66  * @param {float} [cfg.x=0.0] Base rotation vector X axis
 67  * @param {float} [cfg.y=0.0] Base rotation vector Y axis
 68  * @param {float} [cfg.z=0.0] Base rotation vector Z axis
 69  * @param {float} [cfg.angle=0.0] Base rotation angle in degrees
 70  * @param {[{x:float, y:float, z:float, angle:float}]} [cfg.rotations=[]] Sequence of rotations to apply on top of the base rotation
 71  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function that can return the same signature as the static config
 72  * @param {...SceneJS.Node} [childNodes] Child nodes
 73  */
 74 SceneJS.Quaternion = function() {
 75     SceneJS.Node.apply(this, arguments);
 76     this._nodeType = "quaternion";
 77     this._mat = null;
 78     this._xform = null;
 79     this._q = SceneJS._math_identityQuaternion();
 80     if (this._fixedParams) {
 81         this._init(this._getParams());
 82     }
 83 };
 84 
 85 SceneJS._inherit(SceneJS.Quaternion, SceneJS.Node);

132 /**
133  * Sets the quaternion properties in terms of a rotation axis and an angle in degrees.
134  * This method resets the quaternion to the identity quaternion when you supply no arguments.
135  *
136  * @param {Object} [q={ x: 0, y: 0, z: 0, angle: 1  }] Rotation vector and angle in degrees
137  * @param {float} [q.x=0.0] Rotation vector X axis
138  * @param {float} [q.y=0.0] Rotation vector Y axis
139  * @param {float} [q.z=0.0] Rotation vector Z axis
140  * @param {float} [q.angle=0.0] Rotation angle in degrees
141  * @returns {SceneJS.Quaternion} this
142  */
143 SceneJS.Quaternion.prototype.setRotation = function(q) {
144     q = q || {};
145     this._q = SceneJS._math_angleAxisQuaternion(q.x || 0, q.y || 0, q.z || 0, q.angle || 0);
146     this._memoLevel = 0;
147     return this;
148 };
149 
150 /** Returns the quaternion properties in terms of a rotation axis and an angle in degrees.
151  *
152  * @returns {{ x: float, y: float, z: float, angle: float }} Quaternion properties as rotation axis and angle
153  */
154 SceneJS.Quaternion.prototype.getRotation = function() {
155     return SceneJS._math_angleAxisFromQuaternion(this._q);
156 };
157 
158 /**
159  * Applies a rotation to the quaternion. This effectively rotates the quaternion by another quaternion
160  * that is defined in terms of a rotation axis and angle in degrees.
161  *
162  * @param {Object} [q={ x: 0, y: 0, z: 0, angle: 0 }] Rotation vector and angle in degrees
163  * @param {float} [q.x=0.0] Rotation vector X axis
164  * @param {float} [q.y=0.0] Rotation vector Y axis
165  * @param {float} [q.z=0.0] Rotation vector Z axis
166  * @param {float} [q.angle=0.0] Rotation angle in degrees
167  * @returns {SceneJS.Quaternion} this
168  */
169 SceneJS.Quaternion.prototype.rotate = function(q) {
170     this._q = SceneJS._math_mulQuaternions(SceneJS._math_angleAxisQuaternion(q.x || 0, q.y || 0, q.z || 0, q.angle || 0), this._q);
171     this._memoLevel = 0;
172     return this;
173 };

179 SceneJS.Quaternion.prototype.getMatrix = function() {

184 /** Normalises the quaternion.
185  *
186  * @returns {SceneJS.Quaternion} this
187  */
188 SceneJS.Quaternion.prototype.normalize = function() {
189     this._q = SceneJS._math_normalizeQuaternion(this._q);
190     this._memoLevel = 0;
191     return this;

194 SceneJS.Quaternion.prototype._init = function(params) {
195     if (params.x || params.y || params.x || params.angle || params.w) {
196         this.setRotation(params);
197     }
198     if (params.rotations) {
199         for (var i = 0; i < params.rotations.length; i++) {
200             this.rotate(params.rotations[i]);
201         }
202     }
203 };
204 
205 SceneJS.Quaternion.prototype._render = function(traversalContext, data) {
206     if (this._memoLevel == 0) {
207         if (!this._fixedParams) {
208             this._init(this._getParams(data));
209         } else {
210             this._memoLevel = 1;
211         }
212         this._mat = SceneJS._math_newMat4FromQuaternion(this._q);
213     }
214     var superXform = SceneJS._modelViewTransformModule.getTransform();
215     if (this._memoLevel < 2) {
216         var instancing = SceneJS._instancingModule.instancing();
217         var tempMat = SceneJS._math_mulMat4(superXform.matrix, this._mat);
218         this._xform = {
219             localMatrix: this._mat,
220             matrix: tempMat,
221             fixed: superXform.fixed && this._fixedParams && !instancing
222         };
223         if (this._memoLevel == 1 && superXform.fixed && !instancing) {   // Bump up memoization level if model-space fixed
224             this._memoLevel = 2;
225         }
226     }
227     SceneJS._modelViewTransformModule.setTransform(this._xform);
228     this._renderNodes(traversalContext, data);
229     SceneJS._modelViewTransformModule.setTransform(superXform);
230 };

232 /**Factory function that returns a new {@link SceneJS.Quaternion} instance
233  */
234 SceneJS.quaternion = function() {
235     var n = new SceneJS.Quaternion();
236     SceneJS.Quaternion.prototype.constructor.apply(n, arguments);
237     return n;
