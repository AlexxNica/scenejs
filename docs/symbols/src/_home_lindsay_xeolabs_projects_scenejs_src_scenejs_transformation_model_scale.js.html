1 /**
  2  * Scaling modelling transform node
  3  */
  4 (function() {
  5     var modelTransformBackend = SceneJS._backends.getBackend('model-transform');
  6 
  7     /* Memoization levels
  8      */
  9     const NO_MEMO = 0;              // No memoization, assuming that node's configuration is dynamic
 10     const FIXED_CONFIG = 1;         // Node config is fixed, memoizing local object-space matrix
 11     const FIXED_MODEL_SPACE = 2;    // Both node config and model-space are fixed, memoizing axis-aligned volume
 12 
 13     SceneJS.scale = function() {
 14         var cfg = SceneJS._utils.getNodeConfig(arguments);
 15 
 16         return SceneJS._utils.createNode(
 17                 "scale",
 18                 cfg.children,
 19 
 20                 new (function() {
 21                     this._memoLevel = NO_MEMO;
 22                     this._mat = null;
 23                     this._xform = null;
 24 
 25                     this._x = 0;
 26                     this._y = 0;
 27                     this._z = 1;
 28 
 29                     this.setX = function(x) {
 30                         this._x = x;
 31                         this._memoLevel = NO_MEMO;
 32                         return this;
 33                     };
 34 
 35                     this.getX = function() {
 36                         return this._x;
 37                     };
 38 
 39                     this.setY = function(y) {
 40                         this._y = y;
 41                         this._memoLevel = NO_MEMO;
 42                         return this;
 43                     };
 44 
 45                     this.getY = function() {
 46                         return this._y;
 47                     };
 48 
 49                     this.setZ = function(z) {
 50                         this._z = z;
 51                         this._memoLevel = NO_MEMO;
 52                         return this;
 53                     };
 54 
 55                     this.getZ = function() {
 56                         return this._z;
 57                     };
 58 
 59                     this.setXYZ = function(xyz) {
 60                         this._x = xyz.x == undefined ? 1 : xyz.x;
 61                         this._y = xyz.y == undefined ? 1 : xyz.y;
 62                         this._z = xyz.z == undefined ? 1 : xyz.z;
 63                         this._memoLevel = NO_MEMO;
 64                         return this;
 65                     };
 66 
 67                     this.getXYZ = function() {
 68                         return {
 69                             x: this._x,
 70                             y: this._y,
 71                             z: this._z
 72                         };
 73                     };
 74 
 75                     this._init = function(params) {
 76                         this.setXYZ({x : params.x, y: params.y, z: params.z });
 77                     };
 78 
 79                     if (cfg.fixed) {
 80                         this._init(cfg.getParams());
 81                     }
 82 
 83                     this._render = function(traversalContext, data) {
 84                         if (this._memoLevel == NO_MEMO) {
 85                             if (!cfg.fixed) {
 86                                 this._init(cfg.getParams(data));
 87                             } else {
 88                                 this._memoLevel = FIXED_CONFIG;
 89                             }
 90                             this._mat = SceneJS_math_scalingMat4v([this._x, this._y, this._z]);
 91                         }
 92                         var superXform = modelTransformBackend.getTransform();
 93                         if (this._memoLevel < FIXED_MODEL_SPACE) {
 94                             var tempMat = SceneJS_math_mulMat4(superXform.matrix, this._mat);
 95                             this._xform = {
 96                                 localMatrix: this._mat,
 97                                 matrix: tempMat,
 98                                 fixed: superXform.fixed && cfg.fixed
 99                             };
100                             if (this._memoLevel == FIXED_CONFIG && superXform.fixed) {   // Bump up memoization level if model-space fixed
101                                 this._memoLevel = FIXED_MODEL_SPACE;
102                             }
103                         }
104                         modelTransformBackend.setTransform(this._xform);
105                         this._renderChildren(traversalContext, data);
106                         modelTransformBackend.setTransform(superXform);
107                     };
108                 })());
109     };
110 })();
