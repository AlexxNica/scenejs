1 /**
  2  * @class A scene branch node that selects which among its children are currently active.
  3  *
  4  * <p>This node is useful for dynamically controlling traversal within a scene graph.</p>
  5  * <p><b>Live Examples</b></p>
  6  * <ul><li><a target = "other" href="http://bit.ly/9d8wLu">Example 1</a></li></ul>
  7  * <p><b>Example Usage 1</b></p><p>This selector will allow only child nodes at indices 0 and 2 to be rendered,
  8  * which are the teapot and sphere. Child 1, a cube, is not selected and therefore won't be rendered.</p><pre><code>
  9  * var s = new SceneJS.Selector({ selection: [0, 2]},
 10  *
 11  *      new SceneJS.objects.teapot(),   // Child 0
 12  *
 13  *      new SceneJS.objects.cube(),     // Child 1
 14  *
 15  *      new SceneJS.objects.sphere())   // Child 2
 16  *
 17  * // The selection can be set between traversals.
 18  * // Here we'll select all child nodes:
 19  *
 20  * s.setSelection([0,1,2]);
 21  *
 22  * </pre></code>
 23  * <p><b>Example Usage 2</b></p><p>A more advanced example - the selector in this example switches between three
 24  * viewpoints of the scene content. The content is instanced within each child of the Selector using Instance and Symbol
 25  * nodes. When we render the scene, we can pass in the selection.</p><pre><code>
 26  *
 27  * var myScene = new SceneJS.Scene({ ... },
 28  *
 29  *       new SceneJS.symbol({ name: "theScene" },
 30  *           new SceneJS.objects.Teapot()
 31  *       ),
 32  *
 33  *       new SceneJS.Selector(
 34  *               function(data) {   // Child index as a dynamic config
 35  *                   return {
 36  *                       selection: [data.get("activeCamera")]  // Selection
 37  *                   };
 38  *               },
 39  *
 40  *           new SceneJS.LookAt({ eye : { z: 10.0 } },
 41  *                new SceneJS.Instance({ name: "theScene"})),
 42  *
 43  *           new SceneJS.LookAt({ eye : { x: 10.0 }},
 44  *                new SceneJS.Instance({ name: "theScene"})),
 45  *
 46  *           new SceneJS.LookAt({ eye : { x: -5.0, y: 5, z: 5 }},
 47  *                new SceneJS.Instance({ name: "theScene" })
 48  *           )
 49  *       )
 50  *   );
 51  *
 52  * myScene.render({ activeCamera: 0 });  // Render scene for first viewpoint 
 53  * myScene.render({ activeCamera: 1 });  // Once more for second viewpoint
 54  *
 55  * </pre></code>
 56  *
 57  *
 58  * @extends SceneJS.Node
 59  * @constructor
 60  * Create a new SceneJS.Selector
 61  * @param {Object} config  Config object or function, followed by zero or more child nodes
 62  */
 63 SceneJS.Selector = function() {
 64     SceneJS.Node.apply(this, arguments);
 65     this._selection = [];
 66     if (this._fixedParams) {
 67         this._init(this._getParams());
 68     }
 69 };
 70 
 71 SceneJS._inherit(SceneJS.Selector, SceneJS.Node);
 72 
 73 /**
 74  Sets the indices of selected children. When the value is undefined or an empty array, then no children will be selected.
 75  @function setSelection
 76  @param {int []} selection
 77  @returns {SceneJS.Selector} This Selector node
 78  */
 79 SceneJS.Selector.prototype.setSelection = function(selection) {
 80     this._selection = selection;
 81     return this;
 82 };
 83 
 84 /**
 85  * Returns the indices of the selected child. The result will be an empty array if none are currently selected.
 86  * @function {int []} getSelection
 87  * @returns {int []} Array containing indices of selected children.
 88  */
 89 SceneJS.Selector.prototype.getSelection = function() {
 90     var selection = new Array(this._selection.length);
 91     for (var i = 0; i < this._selection; i++) {
 92         selection[i] = this._selection[i];
 93     }
 94     return selection;
 95 };
 96 
 97 // @private
 98 SceneJS.Selector.prototype._init = function(params) {
 99     if (params.selection) {
100         this.setSelection(params.selection);
101     }
102 };
103 
104 // @private
105 SceneJS.Selector.prototype._render = function(traversalContext, data) {
106     if (!this._fixedParams) {
107         this._init(this._getParams(data));
108     }
109     if (this._selection.length) {
110         var children = [];
111         for (var i = 0; i < this._selection.length; i++) {
112             var j = this._selection[i];
113             if (0 <= j && j < this._children.length) {
114                 children.push(this._children[j]);
115             }
116         }
117         this._renderNodes(traversalContext, data, children);
118     }
119 };
120 
121 /** Returns a new SceneJS.Selector Selector
122  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Selector constructor
123  * @returns {SceneJS.Selector}
124  */
125 SceneJS.selector = function() {
126     var n = new SceneJS.Selector();
127     SceneJS.Selector.prototype.constructor.apply(n, arguments);
128     return n;
129 };
