1 /**
  2  * @class A scene branch node that selects which among its children are currently active.
  3  *
  4  * <p>This node is useful for dynamically controlling traversal within a scene graph.</p>
  5 
  6  * <p><b>Example Usage 1</b></p><p>This selector will allow only child nodes at indices 0 and 2 to be rendered,
  7  * which are the teapot and sphere. Child 1, a cube, is not selected and therefore won't be rendered.</p><pre><code>
  8  * var s = new SceneJS.Selector({ selection: [0, 2]},
  9  *
 10  *      new SceneJS.objects.teapot(),   // Child 0
 11  *
 12  *      new SceneJS.objects.cube(),     // Child 1
 13  *
 14  *      new SceneJS.objects.sphere())   // Child 2
 15  *
 16  * s.setSelection([0,1,2]);  // Select all three child nodes
 17  *
 18  * </pre></code>
 19  * <p><b>Example Usage 2</b></p><p>A more advanced example - the selector in this example switches between three
 20  * viewpoints of the scene content. The content is instanced within each child of the Selector using Instance and Symbol
 21  * nodes. When we render the scene, we can pass in the selection.</p><pre><code>
 22  * var myScene = new SceneJS.Scene({ ... },
 23  *
 24  *       new SceneJS.symbol({ name: "theScene" },
 25  *           new SceneJS.objects.Teapot()
 26  *       ),
 27  *
 28  *       new SceneJS.Selector(
 29  *               function(data) {   // Child index as a dynamic config
 30  *                   return {
 31  *                       selection: [data.get("activeCamera")]  // Selection
 32  *                   };
 33  *               },
 34  *
 35  *           new SceneJS.LookAt({ eye : { z: 10.0 } },
 36  *                new SceneJS.Instance({ name: "theScene"})),
 37  *
 38  *           new SceneJS.LookAt({ eye : { x: 10.0 }},
 39  *                new SceneJS.Instance({ name: "theScene"})),
 40  *
 41  *           new SceneJS.LookAt({ eye : { x: -5.0, y: 5, z: 5 }},
 42  *                new SceneJS.Instance({ name: "theScene" })
 43  *           )
 44  *       )
 45  *   );
 46  *
 47  * myScene.render({ activeCamera: 0 });  // Render scene for first viewpoint
 48  * myScene.render({ activeCamera: 1 });  // Once more for second viewpoint
 49  *
 50  * </pre></code>
 51  *
 52  *
 53  * @extends SceneJS.Node
 54  * @constructor
 55  * Create a new SceneJS.Selector
 56  * @param {Object} config  Config object or function, followed by zero or more child nodes
 57  */
 58 SceneJS.Selector = function() {
 59     SceneJS.Node.apply(this, arguments);
 60     this._nodeType = "selector";
 61     this._selection = [];
 62     if (this._fixedParams) {
 63         this._init(this._getParams());
 64     }
 65 };
 66 
 67 SceneJS._inherit(SceneJS.Selector, SceneJS.Node);
 68 
 69 /**
 70  Sets the indices of selected children. When the value is undefined or an empty array, then no children will be selected.
 71  @function setSelection
 72  @param {int []} selection
 73  @returns {SceneJS.Selector} This Selector node
 74  */
 75 SceneJS.Selector.prototype.setSelection = function(selection) {
 76     selection = selection || [];
 77     this._selection = selection;
 78     return this;
 79 };
 80 
 81 /**
 82  * Returns the indices of the selected child. The result will be an empty array if none are currently selected.
 83  * @function {int []} getSelection
 84  * @returns {int []} Array containing indices of selected children.
 85  */
 86 SceneJS.Selector.prototype.getSelection = function() {
 87     var selection = new Array(this._selection.length);
 88     for (var i = 0; i < this._selection.length; i++) {
 89         selection[i] = this._selection[i];
 90     }
 91     return selection;
 92 };
 93 
 94 // @private
 95 SceneJS.Selector.prototype._init = function(params) {
 96     if (params.selection) {
 97         this.setSelection(params.selection);
 98     }
 99 };
100 
101 // @private
102 SceneJS.Selector.prototype._render = function(traversalContext, data) {
103     if (!this._fixedParams) {
104        this._init( this._getParams(data));
105     }
106     if (this._selection.length) {
107         var children = [];
108         for (var i = 0; i < this._selection.length; i++) {
109             var j = this._selection[i];
110             if (0 <= j && j < this._children.length) {
111                 children.push(this._children[j]);
112             }
113         }
114         this._renderNodes(traversalContext, data, children);
115     }
116 };
117 
118 /** Factory function that returns a new {@link SceneJS.Selector} instance
119  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Selector constructor
120  * @returns {SceneJS.Selector}
121  */
122 SceneJS.selector = function() {
123     var n = new SceneJS.Selector();
124     SceneJS.Selector.prototype.constructor.apply(n, arguments);
125     return n;
126 };
