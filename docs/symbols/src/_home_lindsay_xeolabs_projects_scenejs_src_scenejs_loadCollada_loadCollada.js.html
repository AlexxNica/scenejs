1 /**
  2  * @class A scene node that asynchronously loads its subgraph from a COLLADA file.
  3  * <p>This node is configured with the location of the COLLADA file. When first visited during scene traversal, it will
  4  * begin the load and allow traversal to continue at its next sibling node. When on a subsequent visit its subgraph has
  5  * been loaded, it will then allow traversal to descend into that subgraph to render it.</p>
  6  * <p><b>Loading Cross-Domain</b></p>
  7  * <p>When SceneJS is configured with a JSONP strategy via {@link SceneJS.setJSONPStrategy}, you can perform the
  8  * load cross-domain. Otherwise, the URL of the content must be at the same domain as the scene definition's JavaScript file
  9  * in order to not violate the browser' same-domain security policy.
 10  * <p><b>Monitoring Load Progress</b></p>
 11  * <p>You can monitor loads by registering "process-started" and "process-killed" listeners with {@link SceneJS.addListener()}.</p>
 12  * <p><b>Live Examples</b></p>
 13  * <li><a target = "other" href="http://bit.ly/scenejs-collada-load-seymour">Seymour Plane</a></li>
 14  * <li><a target = "other" href="http://bit.ly/scenejs-tron-tank">Tron Tank</a></li>
 15  * </ul>
 16  * <p><b>Usage Example</b></p><p>The SceneJS.LoadCollada node shown below loads a target asset from
 17  * the <node> with ID "propeller" in a Collada file "airplane.dae" stored on the local domain.</b></p>
 18  * <pre><code>
 19  * var exampleScene = new SceneJS.Scene({
 20  *
 21  *       new SceneJS.LoadCollada({
 22  *                  uri: "airplane.dae",
 23  *                  node: "propeller"
 24  *            })
 25  *  );
 26  *  </pre></code>
 27  *
 28  * @extends SceneJS.Load
 29  */
 30 
 31 SceneJS.LoadCollada = function() {
 32     SceneJS.Load.apply(this, arguments);
 33     this._nodeType = "load-collada";
 34     this._format = "xml";
 35     this._metadata = null;
 36 };
 37 
 38 SceneJS._inherit(SceneJS.LoadCollada, SceneJS.Load);
 39 
 40 // @private
 41 SceneJS.LoadCollada.prototype._init = function(params) {
 42     if (!params.uri) {   // Already checked in SceneJS.Load - do it again here for explicit error message
 43         throw SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
 44                 ("SceneJS.LoadCollada parameter expected: uri"));
 45     }
 46     SceneJS.Load.prototype._init.call(this, params);
 47     this._uri = params.uri;
 48     this._serverParams = {
 49         format: "xml"
 50     };
 51     var options = {
 52         loadCameras: params.loadCameras,
 53         loadLights: params.loadLights,
 54         createBoundingBoxes : params.createBoundingBoxes == undefined ? true : params.createBoundingBoxes,
 55         showBoundingBoxes : params.showBoundingBoxes
 56     };
 57     if (params.loadCameras && SceneJS_modelViewTransformModule.isCameraActive()) {
 58         SceneJS_loggingModule.warn("SceneJS.LoadCollada will not load cameras because it is defined within a SceneJS.Camera");
 59     }
 60     var _this = this;
 61     this._parse = function(cfg) { // Override default SceneJS.Load._parse
 62         var result = SceneJS_colladaParserModule.parse({
 63             assetId: cfg.assetId,  // Used as base for IDs generated for various nodes
 64             uri:params.uri, // Used in paths to texture images
 65             xml:cfg.data,
 66             //params.nodes || [], // Optional cherry-picked asset
 67             rootId: params.node, // Optional cherry-picked asset
 68             options: options
 69         });
 70         _this._metadata = result.meta;
 71         return result.root;
 72     };
 73 };

 75 /**
 76  * Returns an object (owned by the {@link SceneJS.LoadCollada} node) through you can access the configuration of various nodes
 77  * (such as lights and cameras) that the {@link SceneJS.LoadCollada} has created from the Collada file, along with some
 78  * read-only contributor data on the file.
 79  *
 80  * <p>You can walk the object to find out what nodes were created, and set properties within it to configure their state
 81  * when they are next rendered. Each camera and light will have a unique ID. The object provides a property through
 82  * which you can select which camera is currently active (see below).</p>
 83  *
 84  * <p>Various sub-objects provide data to dynamic configuration functions on the nodes (see how those functions work
 85  * in {@link SceneJS.Node}). Therefore, the object works as efficient read/write access to their render-time
 86  * configuration.</p>
 87  *
 88  * <p>Since the object is generated while creating nodes as the Collada file is loaded, this function will only return it when the
 89  * {@link SceneJS.LoadCollada}'s state is {@link SceneJS.Load.STATE_LOADED} or {@link SceneJS.Load.STATE_RENDERED}, which
 90  * you can check with a prior call to {@link SceneJS.Load#getState}.</p>
 91  *
 92  * <p>Shown below is the config object's general structure. When cameras are loaded, it contains an array of camera configurations,
 93  * each with properties pertaining to the camera type, plus a configuration for a default camera. When lights are loaded,
 94  * it contains an array of light configurations, again with pertenant properties for each of those. When caneras are loaded,
 95  * the activeCamera property may be set to a camera's ID to activate it.</p>
 96  *
 97  * <pre><code>
 98  * {
 99  *   contributor: {
100  *       author        : "Jane Doe",
101  *       authoringTool : "",
102  *       comments      : String,
103  *       copyright     : String,
104  *       sourceData    : String
105  *   },
106  *
107  *   // Set this to the ID of one of the cameras (see below) to activate it.
108  *
109  *   // When undefined
110  *
111  *   activeCamera: String,
112  *
113  *   // A configuration for each <camera> parsed:
114  *
115  *   cameras: [
116  *       {
117  *           // Uniquely identifies camera in .dae file
118  *
119  *           id: String,
120  *
121  *
122  *           // Camera projection configuration. There are two types, each of which
123  *           // will have different properties.
124  *
125  *           optics: {
126  *
127  *               type: String,       // "ortho" or "perspective"
128  *
129  *
130  *               // Properties when optics type is "ortho":
131  *
132  *               left: Number,
133  *               right: Number,
134  *               bottom: Number,
135  *               top: Number,
136  *               near: Number,
137  *               far: Number,
138  *
139  *
140  *               // Properties when optics type is "perspective":
141  *
142  *               fovy: Number,
143  *               aspect: Number,
144  *               near: Number,
145  *               far: Number
146  *           },
147  *
148  *
149  *           // When the camera is instanced by a <lookat> transform, there will
150  *           // also be a lookat property:
151  *
152  *           lookat: {
153  *               eye: {
154  *                   x: Number,
155  *                   y: Number,
156  *                   z: Number
157  *               },
158  *
159  *               look: {
160  *                   x: Number,
161  *                   y: Number,
162  *                   z: Number
163  *               },
164  *
165  *               up: {
166  *                   x: Number,
167  *                   y: Number,
168  *                   z: Number
169  *               }
170  *           }
171  *       }
172  *   ],
173  *
174  *   // The default camera which will be active when no <camera> elements
175  *   // were loaded, or when activeCamera does not resolve to any of the
176  *   // loaded cameras' IDs:
177  *
178  *   defaultCamera : {
179  *       lookat: {
180  *           eye: { x: 0, y: 0, z: 1},
181  *           look: { x: 0, y: 0, z: 0 },
182  *           up: { x: 0, y: 1, z: 0}
183  *       },
184  *       optics: {
185  *           type: "perspective",
186  *           fovy : 60.0,
187  *           aspect : 1.0,
188  *           near : 0.10,
189  *           far : 10000.0
190  *       }
191  *   },
192  *
193  *   // A configuration for each <light>. Each light configuration
194  *   // identifies it's type, for which it will have the pertenant
195  *   // properties.
196  *
197  *   lights: [
198  *
199  *       // A point light:
200  *
201  *       {
202  *           type: "point",
203  *           pos: { x: Number, y: Number, z: Number },
204  *           color: { r : Number, g: Number, b: Number },
205  *           constantAttenuation : Number,
206  *           linearAttenuation : Number,
207  *           quadraticAttenuation : Number
208  *       },
209  *
210  *       // A directional light:
211  *
212  *       {
213  *           type: "dir",
214  *           dir: { x: Number, y: Number, z: Number },
215  *           color: { r : Number, g: Number, b: Number }
216  *       }
217  *   ]
218  * }
219  * </pre></code>
220  * @returns {int} The state
221  */
222 SceneJS.LoadCollada.prototype.getConfig = function() {
223     return this._metadata;
224 };

226 /** Factory function that returns a new {@link SceneJS.LoadCollada} instance
227  * @param {Arguments} args Variable arguments that are passed to the SceneJS.LoadCollada constructor
228  * @returns {SceneJS.LoadCollada}
229  */
230 SceneJS.loadCollada = function() {
231     var n = new SceneJS.LoadCollada();
232     SceneJS.LoadCollada.prototype.constructor.apply(n, arguments);
233     return n;
234 };
