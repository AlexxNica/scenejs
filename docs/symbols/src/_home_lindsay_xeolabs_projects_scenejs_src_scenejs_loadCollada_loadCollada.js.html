1 /**
  2  * @class A scene node that asynchronously loads its subgraph from a COLLADA file.
  3  * <p>This node is configured with the location of the COLLADA file, which it can load-cross domain when a proxy is
  4  * provided. When first visited during scene traversal, it will
  5  * begin the load and allow traversal to continue at its next sibling node. When on a subsequent visit its subgraph has
  6  * been loaded, it will then allow traversal to descend into that subgraph to render it.</p>
  7  * <p>You can monitor loads by registering "process-started" and "process-killed" listeners with SceneJS.onEvent().</p>
  8  * <p><b>Usage Example</b></p><p>The SceneJS.LoadCollada node shown below loads a target asset cross-domain, from
  9  * the <node> with ID "propeller" in a Collada file "airplane.dae" stored on a server at "foo.com". The transfer is
 10  * routed via the JSONP proxy located by the <b>uri</b> property on the SceneJS.Scene node.</b></p>
 11  * <pre><code>
 12  * var exampleScene = new SceneJS.Scene({
 13  *
 14  *       // JSONP proxy location - needed only for cros-domain load
 15  *       proxy:"http://scenejs.org/cgi-bin/jsonp_proxy.pl" });
 16  *
 17  *       new SceneJS.LoadCollada({
 18  *                  uri: "http://foo.com/airplane.dae",
 19  *                  node: "propeller"
 20  *            })
 21  *  );
 22  *  </pre></code>
 23  *
 24  * @extends SceneJS.load
 25  */
 26 
 27 SceneJS.LoadCollada = function() {
 28     SceneJS.Load.apply(this, arguments);
 29     this._nodeType = "loadCollada";
 30 };
 31 
 32 SceneJS._inherit(SceneJS.LoadCollada, SceneJS.Load);
 33 
 34 // @private
 35 SceneJS.LoadCollada.prototype._init = function(params) {
 36     if (!params.uri) {
 37         SceneJS_errorModule.fatalError(new SceneJS.NodeConfigExpectedException
 38                 ("SceneJS.LoadCollada parameter expected: uri"));
 39     }
 40     this._uri = params.uri;
 41     this._serverParams = {
 42         format: "xml"
 43     };
 44     var modes = {
 45         loadCamera: params.loadCamera,
 46         loadLights: params.loadLights,
 47         showBoundingBoxes : params.showBoundingBoxes
 48     };
 49     this._parser = function(xml, onError) {
 50         return SceneJS_colladaParserModule.parse(
 51                 params.uri, // Used in paths to texture images
 52                 xml,
 53                 params.node, // Optional cherry-picked asset
 54                 modes
 55                 );
 56     };
 57 };
 58 
 59 
 60 /** Returns a new SceneJS.LoadCollada instance
 61  * @param {Arguments} args Variable arguments that are passed to the SceneJS.LoadCollada constructor
 62  * @returns {SceneJS.LoadCollada}
 63  */
 64 SceneJS.loadCollada = function() {
 65     var n = new SceneJS.LoadCollada();
 66     SceneJS.LoadCollada.prototype.constructor.apply(n, arguments);
 67     return n;
 68 };
