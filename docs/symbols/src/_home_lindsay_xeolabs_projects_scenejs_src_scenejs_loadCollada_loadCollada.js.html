1 SceneJS.loadCollada = function() {
  2 
  3     var cfg = SceneJS._utils.getNodeConfig(arguments || [
  4         {}
  5     ]);
  6 
  7     var errorBackend = SceneJS._backends.getBackend("error");
  8     var colladaBackend = SceneJS._backends.getBackend("collada");
  9 
 10     var params = cfg.getParams();
 11     if (!params.uri) {
 12         errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException
 13                 ("Mandatory SceneJS.assets.collada parameter missing: uri"));
 14     }
 15 
 16     var modes = {
 17         loadCamera: params.loadCamera,
 18         loadLights: params.loadLights,
 19         showBoundingBoxes : params.showBoundingBoxes
 20     };
 21 
 22     return SceneJS.load({
 23         uri: params.uri,
 24 
 25         /* Uniquely ID different assets loaded from different nodes of same COLLADA file
 26          */
 27         id: params.node ? (params.uri + ":" + params.node) : params.uri,
 28 
 29         serverParams: {
 30             format: "xml"
 31         },
 32 
 33         parser: function(xml, onError) {
 34             return colladaBackend.parse(
 35                     params.uri, // Used in paths to texture images
 36                     xml,
 37                     params.node, // Optional cherry-picked asset
 38                     modes
 39                     );
 40         }
 41     });
 42 
 43 };
