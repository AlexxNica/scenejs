1 /**
  2  * Scene node that asynchronously loads its subgraph, cross-domain from a COLLADA file. This node is configured with the
  3  * location of the COLLADA file. When first visited during scene traversal, it will begin the load and allow traversal
  4  * to cintinue at its next sibling node. When on a subsequent visit its subgraph has been loaded, it will then allow
  5  * traversal to descend into that subgraph to render it.
  6  *
  7  * @class SceneJS.load
  8  * @extends SceneJS.load
  9  */
 10 
 11 SceneJS.LoadCollada = function() {
 12     SceneJS.Load.apply(this, arguments);
 13     this._nodeType = "loadCollada";
 14 };
 15 
 16 SceneJS._utils.inherit(SceneJS.LoadCollada, SceneJS.Load);
 17 
 18 // @private
 19 SceneJS.LoadCollada.prototype._init = function(params) {
 20     if (!params.uri) {
 21         SceneJS_errorModule.fatalError(new SceneJS.exceptions.NodeConfigExpectedException
 22                 ("SceneJS.LoadCollada parameter expected: uri"));
 23     }
 24     this._uri = params.uri;
 25     this._serverParams = {
 26         format: "xml"
 27     };
 28     var modes = {
 29         loadCamera: params.loadCamera,
 30         loadLights: params.loadLights,
 31         showBoundingBoxes : params.showBoundingBoxes
 32     };
 33     this._parser = function(xml, onError) {
 34         return SceneJS_colladaParserModule.parse(
 35                 params.uri, // Used in paths to texture images
 36                 xml,
 37                 params.node, // Optional cherry-picked asset
 38                 modes
 39                 );
 40     };
 41 };
 42 
 43 
 44 /** Function wrapper to support functional scene definition
 45  */
 46 SceneJS.loadCollada = function() {
 47     var n = new SceneJS.LoadCollada();
 48     SceneJS.LoadCollada.prototype.constructor.apply(n, arguments);
 49     return n;
 50 };
