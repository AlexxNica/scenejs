1 /**
  2  * Backend for projection nodes, creates context that holds the current
  3  * scene projection state
  4  */
  5 SceneJS._backends.installBackend(
  6         new (function() {
  7 
  8             this.type = 'viewprojection';
  9 
 10             var ctx;
 11 
 12             this.install = function(_ctx) {
 13                 ctx = _ctx;
 14                 ctx.viewTransform = (function() {
 15 
 16                     var transform;
 17                     var loaded;
 18 
 19                     ctx.events.onEvent("scene-activated", function() {
 20                         transform = {
 21                             matrix : SceneJS_math_identityMat4(),
 22                             frustum : new SceneJS_math_Frustum(SceneJS_math_identityMat4()),
 23                             fixed: true
 24                         };
 25                         loaded = false;
 26                     });
 27 
 28                     /** When a new program is activated we will need to lazy-load our current matrix
 29                      */
 30                     ctx.events.onEvent("program-activated", function() {
 31                         loaded = false;
 32                     });
 33 
 34                     /** When a program is deactivated we may need to re-load into the previously active program
 35                      */
 36                     ctx.events.onEvent("program-deactivated", function() {
 37                         loaded = false;
 38                     });
 39 
 40                     /**
 41                      * When geometry is about to draw we load our matrix if not loaded already
 42                      */
 43                     ctx.events.onEvent("geo-drawing", function() {
 44                         if (!loaded) {
 45 
 46                             /* Lazy-compute WebGL array
 47                              */
 48                             if (!transform.matrixAsArray) {
 49                                 transform.matrixAsArray = new WebGLFloatArray(transform.matrix);
 50                             }
 51 
 52                             ctx.programs.setVar('scene_ViewProjectionMatrix', transform.matrixAsArray);
 53 
 54                             loaded = true;
 55                         }
 56                     });
 57 
 58                     return {
 59 
 60                         setTransform: function(t) {
 61                             transform = t;
 62                             loaded = false;
 63                         },
 64 
 65                         getTransform: function() {
 66                             return transform;
 67                         },
 68 
 69                         transformPoint3: function(v) {
 70                             return SceneJS_math_transformPoint3(transform.matrix, v);
 71                         }
 72                     };
 73                 })();
 74             };
 75 
 76             this.setTransform = function(transform) {
 77                 ctx.viewTransform.setTransform(transform);
 78             };
 79 
 80             this.getTransform = function() {
 81                 return ctx.viewTransform.getTransform();
 82             };
 83 
 84         })());
