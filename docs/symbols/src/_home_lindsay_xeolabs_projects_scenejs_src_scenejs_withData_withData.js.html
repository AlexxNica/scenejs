1 /**
  2  * @class A scene node that creates data in a scope for its subgraph.
  3  * @extends SceneJS.Node
  4  * <p>This node provides a simple yet flexible mechanism for passing data down into a scene graph at runtime, analogous to
  5  * creation of a closure's data scope in JavaScript .</p>.
  6  * <p>The data scope is implemented by a SceneJS.Data instance. On each render a SceneJS.Scene creates a global
  7  * SceneJS.Data populated with any properties that were given to the SceneJS.Scene's render method. That Data forms a
  8  * chain on which SceneJS.With nodes will push and pop as they are visited and departed from during scene traversal.</p>
  9  * <p>When some node, or node config callback, looks for a property on its local SceneJS.Data, it will hunt up the chain
 10  * to get the first occurance of that property it finds.</p>
 11  * <p><b>Example:</b></p><p>Creating data for a child SceneJS.Scale node, which has a callback to configure itself from
 12  * the data:</b></p><pre><code>
 13  * var wd = new SceneJS.WithData({
 14  *         sizeX: 5,
 15  *         sizeY: 10,
 16  *         sizeZ: 2
 17  *      },
 18  *      new SceneJS.Translate({ x: 100 },
 19  *
 20  *          new SceneJS.Scale(function(data) {        // Function in this case, instead of a config object
 21  *                   return {
 22  *                       x: data.get("sizeX"),
 23  *                       y: data.get("sizeY"),
 24  *                       z: data.get("sizeZ")
 25  *                   }
 26  *          },
 27  *
 28  *              new SceneJS.objects.Cube()
 29  *          )
 30  *      )
 31  *  )
 32  *
 33  * </code></pre>
 34  * @constructor
 35  * Create a new SceneJS.WithData
 36  * @param {Object} config The config object, followed by zero or more child nodes
 37  *
 38  */
 39 SceneJS.WithData = function() {
 40     SceneJS.Node.apply(this, arguments);
 41     this._nodeType = "withData";
 42     this._data = {};
 43     this._childData = {};
 44     if (this._fixedParams) {
 45         this._init(this._getParams());
 46     }
 47 };
 48 
 49 SceneJS._inherit(SceneJS.WithData, SceneJS.Node);
 50 
 51 /**
 52  Sets a property
 53  @param {String} key Name of property
 54  @param {Object} value Value of property
 55  @returns {SceneJS.WithData} this
 56  */
 57 SceneJS.WithData.prototype.setProperty = function(key, value) {
 58     this._data[key] = value;
 59     this._memoLevel = 0;
 60     return this;
 61 };
 62 
 63 /**
 64  * Returns the value of a property
 65  *
 66  * @param {String} key Name of property
 67  * @returns {Object} Value of property
 68  */
 69 SceneJS.WithData.prototype.getProperty = function(key) {
 70     return this._data[key];
 71 };
 72 
 73 
 74 /** Clears all properties
 75  *@returns {SceneJS.WithData} this
 76  */
 77 SceneJS.WithData.prototype.clearProperties = function() {
 78     this._data = {};
 79     this._memoLevel = 0;
 80     return this;
 81 };
 82 
 83 SceneJS.WithData.prototype._init = function(params) {
 84     for (var key in params) {
 85         this._data[key] = params[key];
 86     }
 87 };
 88 
 89 SceneJS.WithData.prototype._render = function(traversalContext, data) {
 90     if (this._memoLevel == 0) {
 91         if (!this._fixedParams) {
 92             this._init(this._getParams(data));
 93         } else {
 94             this._memoLevel = 1;
 95         }
 96     }
 97     if (this._memoLevel < 2) {
 98         if (this._memoLevel == 1 && data.isFixed() && !SceneJS_instancingModule.instancing()) {
 99             this._memoLevel = 2;
100         }
101     }
102     this._renderNodes(traversalContext, new SceneJS.Data(data, this._fixedParams, this._data));
103 };
104 
105 /** Function wrapper to support functional scene definition
106  */
107 SceneJS.withData = function() {
108     var n = new SceneJS.WithData();
109     SceneJS.WithData.prototype.constructor.apply(n, arguments);
110     return n;
111 };
