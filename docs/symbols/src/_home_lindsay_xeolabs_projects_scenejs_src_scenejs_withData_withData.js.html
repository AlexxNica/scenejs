1 /**
  2  * @class SceneJS.WithData
  3  * @extends SceneJS.Node
  4  * <p>Scene node that creates a child data scope containing the elements of its configuration.</p><p>This node provides
  5  * a simple yet flexible mechanism for passing data down into a scene graph at runtime.</p>.
  6  * <p><b>Example:</b></p><p>Creating data for a child SceneJS.Scale node, which configures itself from the data
  7  * with a configuration function:</b></p><pre><code>
  8  *
  9  * var wd new SceneJS.WithData({
 10  *         sizeX: 5,
 11  *         sizeY: 10,
 12  *         sizeZ: 2
 13  *      },
 14  *      new SceneJS.translate({ x: 100 },
 15  *
 16  *          new SceneJS.scale(function(data) {        // Function in this case, instead of a config object
 17  *                   return {
 18  *                       x: data.get("sizeX"),
 19  *                       y: data.get("sizeY"),
 20  *                       z: data.get("sizeZ")
 21  *                   }
 22  *          },
 23  *
 24  *              new SceneJS.objects.Cube()
 25  *          )
 26  *      )
 27  *  )
 28  *
 29  * </code></pre>
 30  * @constructor
 31  * Create a new SceneJS.WithData
 32  * @param {Object} config The config object, followed by zero or more child nodes
 33  *
 34  */
 35 SceneJS.WithData = function() {
 36     SceneJS.Node.apply(this, arguments);
 37     this._nodeType = "withData";
 38     this._data = {};
 39     this._childData = {};
 40     if (this._fixedParams) {
 41         this._init(this._getParams());
 42     }
 43 };
 44 
 45 SceneJS._utils.inherit(SceneJS.WithData, SceneJS.Node);
 46 
 47 /**
 48  Sets a property
 49  @param {String} key Name of property
 50  @param {Object} value Value of property
 51  @returns {SceneJS.WithData} this
 52  */
 53 SceneJS.WithData.prototype.setProperty = function(key, value) {
 54     this._data[key] = value;
 55     this._memoLevel = 0;
 56     return this;
 57 };
 58 
 59 /**
 60  * Returns the value of a property
 61  *
 62  * @param {String} key Name of property
 63  * @returns {Object} Value of property
 64  */
 65 SceneJS.WithData.prototype.getProperty = function(key) {
 66     return this._data[key];
 67 };
 68 
 69 
 70 /** Clears all properties
 71  *@returns {SceneJS.WithData} this
 72  */
 73 SceneJS.WithData.prototype.clearProperties = function() {
 74     this._data = {};
 75     this._memoLevel = 0;
 76     return this;
 77 };
 78 
 79 SceneJS.WithData.prototype._init = function(params) {
 80     for (var key in params) {
 81         this._data[key] = params[key];
 82     }
 83 };
 84 
 85 SceneJS.WithData.prototype._render = function(traversalContext, data) {
 86     if (this._memoLevel == 0) {
 87         if (!this._fixedParams) {
 88             this._init(this._getParams(data));
 89         } else {
 90             this._memoLevel = 1;
 91         }
 92     }
 93     if (this._memoLevel < 2) {
 94         this._childData = SceneJS._utils.newScope(data, this._fixedParams);
 95         for (var key in this._data) {
 96             this._childData.put(key, this._data[key]);
 97         }
 98         if (this._memoLevel == 1 && data.isfixed()) {
 99             this._memoLevel = 2;
100         }
101     }
102     this._renderNodes(traversalContext, this._childData);
103 };
104 
105 /** Function wrapper to support functional scene definition
106  */
107 SceneJS.withData = function() {
108     var n = new SceneJS.WithData();
109     SceneJS.WithData.prototype.constructor.apply(n, arguments);
110     return n;
111 };
