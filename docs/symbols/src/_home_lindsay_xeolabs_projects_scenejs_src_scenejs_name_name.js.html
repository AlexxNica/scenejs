1 /**
  2  * Scene node that specifies a name for the nodes in its subgraph. These may be nested to create hierarchical
  3  * names, effectively overlaying semantics onto a scene.
  4  */
  5 SceneJS.name = function() {
  6     var cfg = SceneJS._utils.getNodeConfig(arguments);
  7     var errorBackend = SceneJS._backends.getBackend('error');
  8     var nameBackend = SceneJS._backends.getBackend('naming');
  9 
 10     return SceneJS._utils.createNode(
 11             "name",
 12             cfg.children,
 13 
 14             new (function() {
 15 
 16                 var _name = "unnamed";
 17 
 18                 /**
 19                  * Sets the name value
 20                  * @param {string} name The name value
 21                  * @returns {SceneJS.name} This name node
 22                  */
 23                 this.setName = function(name) {
 24                     if (!name) {
 25                         errorBackend.fatalError(new SceneJS.exceptions.InvalidNodeConfigException("SceneJS.name name is undefined"));
 26                     }
 27                     name = name.replace(/^\s+|\s+$/g, ''); // Trim
 28                     if (name.length == 0) {
 29                         errorBackend.fatalError(new SceneJS.exceptions.InvalidNodeConfigException("SceneJS.name name cannot be empty string"));
 30                     }
 31                     _name = name;
 32                     return this;
 33                 };
 34 
 35                 /** Returns the name value
 36                  * @returns The name string
 37                  */
 38                 this.getName = function() {
 39                     return _name;
 40                 };
 41 
 42                 this._init = function (params) {
 43                     if (params.name) {
 44                         this.setName(params.name);
 45                     }
 46                 };
 47 
 48                 if (cfg.fixed) {
 49                     this._init(cfg.getParams());
 50                 }
 51 
 52                 this._render = function(traversalContext, data) {
 53                     if (!cfg.fixed) {
 54                         this._init(cfg.getParams(data));
 55                     }
 56                     nameBackend.pushName(_name);
 57                     this._renderChildren(traversalContext, data);
 58                     nameBackend.popName();
 59                 };
 60             })());
 61 };
