1 /**
  2  *@class A scene node that specifies a name for the nodes in its subgraph.
  3  *<p> These may be nested to create hierarchical names, effectively overlaying semantics onto a scene.</p>
  4  *<p><b>Example:</b></p><p>Two cubes, blue and green, assigned names "cubes.blue" and "cubes.green".</b></p><pre><code>
  5  * var n = SceneJS.Name({ name: "cubes" },
  6  *      new SceneJS.Name({ name: "blue" },
  7  *              new SceneJS.Material({
  8  *                  baseColor: { b: 0.9 }
  9  *              },
 10  *                      new SceneJS.Translate({x: -10.0 },
 11  *                              new SceneJS.objects.Cube()
 12  *                              )
 13  *                      )
 14  *              ),
 15  *
 16  *      new SceneJS.Name({ name: "green" },
 17  *              new SceneJS.Material({
 18  *                  baseColor: { g: 0.9 }
 19  *              },
 20  *                      new SceneJS.Translate({x: 10.0 },
 21  *                              new SceneJS.objects.Cube()
 22  *                              )
 23  *                      )
 24  *              )
 25  *      )
 26  *  // ...
 27  * </code></pre>
 28  * @extends SceneJS.Node
 29  *  @constructor
 30  * Create a new SceneJS.Name
 31  * @param {Object} config  Config object or function, followed by zero or more child nodes
 32  */
 33 SceneJS.Name = function() {
 34     SceneJS.Node.apply(this, arguments);
 35     this._nodeType = "name";
 36     this._name = "undefined";
 37     if (this._fixedParams) {
 38         this._init(this._getParams());
 39     }
 40 };
 41 
 42 SceneJS._inherit(SceneJS.Name, SceneJS.Node);
 43 
 44 /**
 45  * Sets the name value
 46  * @param {String} name The name value
 47  * @returns {SceneJS.Name} This name node
 48  */
 49 SceneJS.Name.prototype.setName = function(name) {
 50     if (!name) {
 51         SceneJS_errorModule.fatalError(new SceneJS.InvalidNodeConfigException("SceneJS.name name is undefined"));
 52     }
 53     name = name.replace(/^\s+|\s+$/g, ''); // Trim
 54     if (name.length == 0) {
 55         SceneJS_errorModule.fatalError(new SceneJS.InvalidNodeConfigException("SceneJS.name name cannot be empty string"));
 56     }
 57     this._name = name;
 58     return this;
 59 };
 60 
 61 /** Returns the name value
 62  * @returns {String} The name string
 63  */
 64 SceneJS.Name.prototype.getName = function() {
 65     return this._name;
 66 };
 67 
 68 // @private
 69 SceneJS.Name.prototype._init = function(params) {
 70     if (params.name) {
 71         this.setName(params.name);
 72     }
 73 };
 74 
 75 // @private
 76 SceneJS.Name.prototype._render = function(traversalContext, data) {
 77     if (!this._fixedParams) {
 78         this._init(this._getParams(data));
 79     }
 80     SceneJS_nameModule.pushName(this._name);
 81     this._renderNodes(traversalContext, data);
 82     SceneJS_nameModule.popName();
 83 };
 84 
 85 /** Returns a new SceneJS.Name instance
 86  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Name constructor
 87  * @returns {SceneJS.Name}
 88  */
 89 SceneJS.name = function() {
 90     var n = new SceneJS.Name();
 91     SceneJS.Name.prototype.constructor.apply(n, arguments);
 92     return n;
 93 };
