1 /**
  2  * Backend module for VRAM management. This module tries to ensure that SceneJS always has enough video memory
  3  * to keep things ticking over, at least slowly. Whenever any backend wants to load something into video RAM, it
  4  * will get the memory manager to mediate the allocation, passing in a callback that will attempt the actual allocation.
  5  * The memory manager will then try the callback and if no exception is thrown by it, all is good and that's that.
  6  *
  7  * However, if the callback throws an out-of-memory exception, the memory manager will poll each registered evictor to
  8  * evict something to free up some memory in order to satisfy the request. As soon as one of the evictors has
  9  * successfully evicted something, the memory manager will have another go with the  callback. It will repeat this
 10  * process, polling a different evictor each time, until the callback succeeds. For fairness, the memory manager
 11  * remembers the last evictor it polled, to continue with the next one when it needs to evict something again.
 12  *
 13  * This module is to be used only when there is an active canvas.
 14  *
 15  *  @private
 16  */
 17 var SceneJS_memoryModule = new (function() {
 18 
 19     var canvas;                 // Used for testing for error conditions
 20     var evictors = [];          // Eviction function for each client
 21     var iEvictor = 0;           // Fair eviction policy - don't keep starting polling at first evictor
 22 
 23     SceneJS_eventModule.onEvent(
 24             SceneJS_eventModule.CANVAS_ACTIVATED,
 25             function(c) {
 26                 canvas = c;
 27             });
 28 
 29     SceneJS_eventModule.onEvent(
 30             SceneJS_eventModule.CANVAS_DEACTIVATED,
 31             function() {
 32                 canvas = null;
 33             });
 34 
 35     SceneJS_eventModule.onEvent(// Framework reset - start next polling at first evictor
 36             SceneJS_eventModule.RESET,
 37             function() {
 38                 canvas = null;
 39                 iEvictor = 0;
 40             });
 41 
 42     /**
 43      * Polls each registered evictor backend to evict something. Stops on the first one to
 44      * comply. When called again, resumes at the next in sequence to ensure fairness.
 45      * @private
 46      */
 47     function evict() {
 48         if (evictors.length == 0) {
 49             return false;
 50         }
 51         var tries = 0;
 52         while (true) {
 53             if (iEvictor > evictors.length) {
 54                 iEvictor = 0;
 55             }
 56             if (evictors[iEvictor++]()) {
 57                 SceneJS_loggingModule.warn("Evicted least-used item from memory");
 58                 return true;
 59             } else {
 60                 tries++;
 61                 if (tries == evictors.length) {
 62                     return false;
 63                 }
 64             }
 65         }
 66     }
 67 
 68     // @private
 69     function outOfMemory(description) {
 70         SceneJS_loggingModule.error("Memory allocation failed");
 71         SceneJS_errorModule.fatalError(new SceneJS.OutOfVRAMException(
 72                 "Out of memory - failed to allocate memory for " + description));
 73     }
 74 
 75     /**
 76      * Volunteers the caller as an evictor that is willing to attempt to free some memory when polled
 77      * by this module as memory runs low. The given evict callback is to attempt to free some memory
 78      * held by the caller, and should return true on success else false.
 79      * @private
 80      */
 81     this.registerEvictor = function(evict) {
 82         evictors.push(evict);
 83     };
 84 
 85     /**
 86      * Attempt allocation of some memory for the caller. This method does not return anything - the
 87      * tryAllocate callback is to wrap the allocation attempt and provide the result to the caller via
 88      * a closure, IE. not return it.
 89      * @private
 90      */
 91     this.allocate = function(description, tryAllocate) {
 92         // SceneJS_loggingModule.debug("Allocating memory for: " + description);
 93         if (!canvas) {
 94             SceneJS_errorModule.fatalError(new SceneJS.NoCanvasActiveException
 95                     ("No canvas active - failed to allocate shader memory"));
 96         }
 97         var maxTries = 10; // TODO: Heuristic for this? Does this really need be greater than one?
 98         var context = canvas.context;
 99         if (context.getError() == context.OUT_OF_MEMORY) {
100             outOfMemory(description);
101         }
102         var tries = 0;
103         while (true) {
104             try {
105                 tryAllocate();
106                 if (context.getError() == context.OUT_OF_MEMORY) {
107                     outOfMemory(description);
108                 }
109                 return; // No errors, must have worked
110             } catch (e) {
111                 if (context.getError() != context.OUT_OF_MEMORY) {
112                     SceneJS_loggingModule.error(e.message || e);
113                     throw e; // We only handle out-of-memory error here
114                 }
115                 if (++tries > maxTries || !evict()) { // Too many tries or no cacher wants to evict
116                     outOfMemory(description);
117                 }
118             }
119         }
120     };
121 })();
