1 /**
  2  * @class A scene node that specifies values to set on nodes in it's subgraph as they are rendered .
  3  * <p>This node provides a simple yet flexible mechanism for pushing data down during scene traversal
  4  * into the setter methods of the nodes in its subgraph.</p>.
  5  * <p>The configuration for a {@link SceneJS.WithConfigs} is a hierarchical map of values for the setter methods of
  6  * nodes in the subgraph. As shown in the example below, its hierarchy maps to that of the subgraph. The keys beginning
  7  * with "#" map to the SIDs of nodes, while other keys map to setter methods on those nodes. </p>.
  8  *
  9  * <p><b>Example:</b></p><p>Configuring {@link SceneJS.Translation} and {@link SceneJS.Scale} nodes in the subgraph:</b></p><pre><code>
 10  * var wd = new SceneJS.WithConfigs({
 11  *
 12  *         "#myTranslate" : {    // Selects the SceneJS.Scale
 13  *              x: 5,            // Invokes the setX, setY and setZ methods of the SceneJS.Translate
 14  *              y: 10,
 15  *              z: 2,
 16  *
 17  *              "#myScale" : {   // Selects the SceneJS.Scale
 18  *                  x: 2.0,      // Invokes the setX and setY methods of the SceneJS.Scale
 19  *                  z: 1.5
 20  *               }
 21  *          }
 22  *      },
 23  *
 24  *      new SceneJS.Translate({
 25  *                     sid: "myTranslate",
 26  *                     x: 100,
 27  *                     y: 20,
 28  *                     z: 15
 29  *                  },
 30  *
 31  *          new SceneJS.Scale({
 32  *                        sid: "myScale",
 33  *                        x : 1.0
 34  *                     },
 35  *
 36  *                new SceneJS.objects.Cube()
 37  *          )
 38  *      )
 39  *  )
 40  *
 41  * </code></pre>
 42  * @extends SceneJS.Node
 43  * @since Version 0.7.6
 44  * @constructor
 45  * Create a new SceneJS.WithConfigs
 46  * @param {Object} [cfg] Static configuration object containing hierarchical map of values for sub-nodes
 47  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function returning the hierarchical map
 48  * @param {...SceneJS.Node} [childNodes] Child nodes
 49  */
 50 SceneJS.WithConfigs = function() {
 51     SceneJS.Node.apply(this, arguments);
 52     this._nodeType = "with-configs";
 53     this._configs = {};
 54     if (this._fixedParams) {
 55         this._init(this._getParams());
 56     }
 57 };
 58 
 59 SceneJS._inherit(SceneJS.WithConfigs, SceneJS.Node);
 60 
 61 /**
 62  Sets the configs map
 63  @param {Object} configs The configs map
 64  @returns {SceneJS.WithConfigs} this
 65  */
 66 SceneJS.WithConfigs.prototype.setConfigs = function(configs) {
 67     this._configs = configs;
 68     this._memoLevel = 0;
 69     return this;
 70 };
 71 
 72 /**
 73  * Returns the configs map
 74  *
 75  * @returns {Object} The configs map
 76  */
 77 SceneJS.WithConfigs.prototype.getConfigs = function() {
 78     return this._configs;
 79 };
 80 
 81 SceneJS.WithConfigs.prototype._init = function(params) {
 82     this._configs = params;
 83 };
 84 
 85 SceneJS.WithConfigs.prototype._render = function(traversalContext, data) {
 86     if (this._memoLevel == 0) {
 87         if (!this._fixedParams) {
 88             this._init(this._getParams(data));
 89         } else {
 90             this._memoLevel = 1;
 91         }
 92     }
 93     if (this._memoLevel < 2) {
 94         if (this._memoLevel == 1 && data.isFixed() && !SceneJS_instancingModule.instancing()) {
 95             this._memoLevel = 2;
 96         }
 97     }
 98     traversalContext = {
 99         appendix : traversalContext.appendix,
100         insideRightFringe: this._children.length > 1,
101         configs : this._configs
102     };
103     this._renderNodes(traversalContext, new SceneJS.Data(data, this._fixedParams, this._data));
104 };
105 
106 /**
107  * Factory function that returns a new {@link SceneJS.WithConfigs} instance
108  * @param {Object} [cfg] Static config map object
109  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function returning the config map
110  * @param {...SceneJS.Node} [childNodes] Child nodes
111  * @returns {SceneJS.WithConfigs}
112  * @since Version 0.7.6
113  */
114 SceneJS.withConfigs = function() {
115     var n = new SceneJS.WithConfigs();
116     SceneJS.WithConfigs.prototype.constructor.apply(n, arguments);
117     return n;
118 };
