1 /**
  2  * @class Invokes addXXX, removeXXX and setXXX methods on the nodes in its subgraph during traversal to set, delete or add
  3  * elements on them.</p>.
  4  * <p>The configuration for a {@link SceneJS.WithConfigs} is a hierarchical map of values for methods on
  5  * nodes in the subgraph. As shown in the examples below, its hierarchy maps to that of the subgraph. The keys beginning
  6  * with "#" map to the SIDs of nodes. In the first example, we have property keys that map to setter methods on
  7  * those nodes. In the second example we have a property key prefixed with "+", which maps it to a method beginning
  8  * with "add" (addNode). In the third example, we have a property key prefixed with "-", which maps it to a method
  9  * beginning with "remove" (removeNode).</p>.
 10  *
 11  * <p>The functionality of {@link SceneJS.WithConfigs} is also provided in the {@link SceneJS.Socket} node, which allows
 12  * a server to push a configuration map onto a scene subgraph through a WebSocket.</p>
 13  *
 14  * <p>Note that configs are applied to nodes just before they are rendered, so as long as your config map
 15  * resolves correctly, the nodes don't have to have initial configurations wherever you have specified them for "setXXX"
 16  * methods in the config map.</p>
 17  *
 18  * <h2>Example 1: Setting properties on subnodes</h2>
 19  * <p><b>Example 1:</b></p><p>Configuring properties on {@link SceneJS.Translation} and {@link SceneJS.Scale} nodes in the subgraph:</b></p><pre><code>
 20  * var wc = new SceneJS.WithConfigs({
 21  *
 22  *         // Have the WithConfigs throw a SceneJS.WithConfigsNodeNotFoundException if any target node
 23  *         // is not found at exactly the hierarchy position specified in our configs map.
 24  *         //
 25  *         // Default is false, which would allow unmatched nodes to be simply skipped as traversal
 26  *         // descends into the subgraph.
 27  *         //
 28  *         strictNodes : true,       // Default is false
 29  *
 30  *         // Have the WithConfigs throw a SceneJS.WithConfigsPropertyNotFoundException if any target property
 31  *         // is not found on its target node.
 32  *         //
 33  *         // Default is true.
 34  *         //
 35  *         strictProperties : true,  // Default is true
 36  *
 37  *         configs: {
 38  *             "#myTranslate" : {    // Selects the SceneJS.Scale
 39  *                  x: 5,            // Invokes the setX, setY and setZ methods of the SceneJS.Translate
 40  *                  y: 10,
 41  *                  z: 2,
 42  *
 43  *                  "#myScale" : {   // Selects the SceneJS.Scale
 44  *                      x: 2.0,      // Invokes the setX and setY methods of the SceneJS.Scale
 45  *                      z: 1.5
 46  *                   }
 47  *             }
 48  *          }
 49  *      },
 50  *
 51  *      new SceneJS.Translate({
 52  *                     sid: "myTranslate",
 53  *                     x: 100,
 54  *                     y: 20,
 55  *                     z: 15
 56  *                  },
 57  *
 58  *          new SceneJS.Scale({
 59  *                        sid: "myScale",
 60  *                        x : 1.0
 61  *                     },
 62  *
 63  *                new SceneJS.objects.Cube()
 64  *          )
 65  *      )
 66  *  )
 67  * </code></pre>
 68  *
 69  * <h2>Adding elements to subnodes</h2>
 70  * <p><b>Example:</b></p><p>In this example we're using a WithConfigs node to attach a subgraph to a target node
 71  * within it's own subgraph. Abstractly, a teapot is being attached to a chair at a table in a cafe. Note
 72  * the "+node" key in the map for the "chair2" node, which maps to the "addNode" method on that node.</b></p><pre><code>
 73  * var wc = SceneJS.withConfigs({
 74  *         strictNodes : true,       // Target node must exist - default is false
 75  *         strictProperties : true,  // Function addNode must exist on target node - default is true
 76  *         configs: {
 77  *             "#cafe": {
 78  *                  "#table5" : {
 79  *                      "#chair2" : {
 80  *
 81  *                          // Content we're attaching. "+node" directs the WithConfig
 82  *                          // to call the "addNode" method on the Node at the attachment
 83  *                          // point to attach the object
 84  *                          //
 85  *                          "+node" :
 86  *                               SceneJS.rotate({ // The object to attach - imagine
 87  *                                         angle: 0,    // this teapot is a person!
 88  *                                         y : 1.0
 89  *                                    },
 90  *                                    SceneJS.objects.teapot())
 91  *                               }
 92  *                          }
 93  *                      }
 94  *                  }
 95  *             },
 96  *             SceneJS.node({ sid: "cafe" },
 97  *                  SceneJS.translate({ x: -1, y: 0, z: -1},
 98  *                      SceneJS.node({ sid: "table5" },
 99  *                          SceneJS.translate({ x: 1, y: 0, z: 1},
100  *
101  *                             // Attachment point
102  *
103  *                             SceneJS.node({ sid: "chair2" }))))))
104  * </code></pre>
105  *
106  * <h2>Removing elements from subnodes</h2>
107  * <p><b>Example:</b></p><p>In this example we're using a WithConfigs node to delete a subgraph from a target node.
108  * Abstractly, a chair is being removed from a table in a cafe. Note the "-node" key in the map for the "chair2" node,
109  * which maps to the "removeNode" method on that node, and the value, which is the SID of the node to remove.</b></p><pre><code>
110  * var wc = SceneJS.withConfigs({
111  *         strictNodes : true,        // Target node must exist - default is false
112  *         strictProperties : true,   // Function addNode must exist on target node - default is true
113  *         configs: {
114  *             "#cafe": {
115  *                  "#table5" : {
116  *
117  *                      // Content we're removing. "-node" directs the WithConfig
118  *                      // to call the "removeNode" method on the "table5" target Node
119  *                      //
120  *                      "-node" : "#chair2"
121  *                  }
122  *             },
123  *             SceneJS.node({ sid: "cafe" },
124  *                  SceneJS.translate({ x: -1, y: 0, z: -1},
125  *
126  *                      // Target node
127  *
128  *                      SceneJS.node({ sid: "table5" },
129  *                          SceneJS.translate({ x: 1, y: 0, z: 1},
130  *
131  *                             // Removing this subnode:
132  *
133  *                             SceneJS.node({ sid: "chair2" }))))))
134  * </code></pre>
135  * @extends SceneJS.Node
136  * @since Version 0.7.6
137  * @constructor
138  * Create a new SceneJS.WithConfigs
139  * @param {Object} [cfg] Static configuration object containing hierarchical map of values for sub-nodes
140  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function returning the hierarchical map
141  * @param {...SceneJS.Node} [childNodes] Child nodes
142  */
143 SceneJS.WithConfigs = function() {
144     SceneJS.Node.apply(this, arguments);
145     this._nodeType = "with-configs";
146     this._configs = {};
147     this._configsModes = {
148         strictProperties : true,
149         strictNodes : false
150     };
151     if (this._fixedParams) {
152         this._init(this._getParams());
153     }
154 };
155 
156 SceneJS._inherit(SceneJS.WithConfigs, SceneJS.Node);

218 SceneJS.WithConfigs.prototype._render = function(traversalContext, data) {
219     if (this._memoLevel == 0) {
220         if (!this._fixedParams) {
221             this._init(this._getParams(data));
222             this._configs = this._preprocessConfigs(this._configs);
223         } else {
224             this._memoLevel = 1;
225             this._configs = this._preprocessConfigs(this._configs);
226         }
227     }
228 //    if (this._memoLevel < 2) {
229 //        if (this._memoLevel == 1 && data.isFixed() && !SceneJS._instancingModule.instancing()) {
230 //            this._memoLevel = 2;
231 //        }
232 //    }
233     traversalContext = {
234         appendix : traversalContext.appendix,
235         insideRightFringe: this._children.length > 1,
236         configs : this._configs,
237         configsModes : this._configsModes
238     };
239    // this._renderNodes(traversalContext, new SceneJS.Data(data, this._fixedParams, this._data));

244 /* Preprocess config map for faster application when rendering nodes
245  */
246 SceneJS.WithConfigs.prototype._preprocessConfigs = function(configs) {
247     var configAction;
248     var funcName;
249     var newConfigs = {};
250     for (var key in configs) {
251         if (configs.hasOwnProperty(key)) {
252             key = key.replace(/^\s*/, "").replace(/\s*$/, "");    // trim
253             if (key.length > 0) {
254                 configAction = key.substr(0, 1);
255                 if (configAction != "#") {  // Property reference
256                     if (configAction == "+") {
257                         funcName = "add" + key.substr(1, 1).toUpperCase() + key.substr(2);
258                     } else if (configAction == "-") {
259                         funcName = "remove" + key.substr(1, 1).toUpperCase() + key.substr(2);
260                     } else {
261                         funcName = "set" + key.substr(0, 1).toUpperCase() + key.substr(1);
262                     }
263                     newConfigs[funcName] = {
264                         isFunc : true,
265                         value : configs[key]
266                     };
267 
268                 } else {
269                     newConfigs[key.substr(1)] = this._preprocessConfigs(configs[key]);
270                 }
271             }
272         }
273     }
274     return newConfigs;
275 };
