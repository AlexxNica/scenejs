1 /**
  2  * Defines a set of lights in a scene. These may appear at multiple points, anywhere in a scene graph, to define
  3  * multiple sources of light. The number of lights is only limited by memory available to the GPU.
  4  *
  5  * TODO: comment
  6  */
  7 SceneJS.lights = function() {
  8     var cfg = SceneJS._utils.getNodeConfig(arguments);
  9     var backend = SceneJS._backends.getBackend('lights');
 10 
 11     return SceneJS._utils.createNode(
 12             "lights",
 13             cfg.children,
 14 
 15             new (function() {
 16                 this._render = function(traversalContext, data) {
 17                     if (SceneJS._utils.traversalMode == SceneJS._utils.TRAVERSAL_MODE_PICKING) {
 18                         SceneJS._utils.visitChildren(cfg, traversalContext, data);
 19                     } else {
 20                         var sources = cfg.getParams(data).sources;
 21                         backend.pushLights(sources);
 22                         this._renderChildren(traversalContext, data);
 23                         backend.popLights(sources.length);
 24                     }
 25                 };
 26             })());
 27 };
