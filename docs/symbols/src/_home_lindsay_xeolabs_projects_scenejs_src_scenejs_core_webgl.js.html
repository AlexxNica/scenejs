 77 function SceneJS_webgl_ProgramUniform(context, program, name, type, size, location, logging) {
 78     //  logging.debug("Program uniform found in shader: " + name);
 79     var func = null;
 80     if (type == context.BOOL) {
 81         func = function (v) {
 82             context.uniform1i(location, v);
 83         };
 84     } else if (type == context.BOOL_VEC2) {
 85         func = function (v) {
 86             context.uniform2iv(location, v);
 87         };
 88     } else if (type == context.BOOL_VEC3) {
 89         func = function (v) {
 90             context.uniform3iv(location, v);
 91         };
 92     } else if (type == context.BOOL_VEC4) {
 93         func = function (v) {
 94             context.uniform4iv(location, v);
 95         };
 96     } else if (type == context.INT) {
 97         func = function (v) {
 98             context.uniform1iv(location, v);
 99         };
100     } else if (type == context.INT_VEC2) {
101         func = function (v) {
102             context.uniform2iv(location, v);
103         };
104     } else if (type == context.INT_VEC3) {
105         func = function (v) {
106             context.uniform3iv(location, v);
107         };
108     } else if (type == context.INT_VEC4) {
109         func = function (v) {
110             context.uniform4iv(location, v);
111         };
112     } else if (type == context.FLOAT) {
113         func = function (v) {
114             context.uniform1f(location, v);
115         };
116     } else if (type == context.FLOAT_VEC2) {
117         func = function (v) {
118             context.uniform2fv(location, v);
119         };
120     } else if (type == context.FLOAT_VEC3) {
121         func = function (v) {
122             context.uniform3fv(location, v);
123         };
124     } else if (type == context.FLOAT_VEC4) {
125         func = function (v) {
126             context.uniform4fv(location, v);
127         };
128     } else if (type == context.FLOAT_MAT2) {
129         func = function (v) {
130             context.uniformMatrix2fv(location, context.FALSE, v);
131         };
132     } else if (type == context.FLOAT_MAT3) {
133         func = function (v) {
134             context.uniformMatrix3fv(location, context.FALSE, v);
135         };
136     } else if (type == context.FLOAT_MAT4) {
137         func = function (v) {
138             context.uniformMatrix4fv(location, context.FALSE, v);
139         };
140     } else {
141         throw "Unsupported shader uniform type: " + type;
142     }
143 
144     this.setValue = function(v) {
145         //   alert("setValue " + name + " = " + v);
146         func(v);
147     };
148 
149     this.getValue = function() {
150         return context.getUniform(program, location);
151     };
152 }

156     this.bindTexture = function(texture, unit) {
157         texture.bind(unit);

165     // logging.debug("Program attribute found in shader: " + name);

175 /**
176  * A vertex/fragment shader in a program
177  *
178  * @param context WebGL context
179  * @param gl.VERTEX_SHADER | gl.FRAGMENT_SHADER
180  * @param source Source code for shader
181  * @param logging Shader will write logging's debug channel as it compiles
182  */
183 function SceneJS_webgl_Shader(context, type, source, logging) {
184     this.handle = context.createShader(type);
185 
186     //  logging.debug("Creating " + ((type == context.VERTEX_SHADER) ? "vertex" : "fragment") + " shader");
187     this.valid = true;
188 
189     context.shaderSource(this.handle, source);
190     context.compileShader(this.handle);
191 
192     if (context.getShaderParameter(this.handle, context.COMPILE_STATUS) != 0) {
193         //    logging.debug("Shader compile succeeded:" + context.getShaderInfoLog(this.handle));
194     }
195     else {
196         this.valid = false;
197         logging.error("Shader compile failed:" + context.getShaderInfoLog(this.handle));
198     }
199     if (!this.valid) {
200         SceneJS._backends.getBackend("error").fatalError(
201                 new SceneJS.exceptions.ShaderCompilationFailureException("Shader program failed to compile"));
202     }
203 }

206 /**
207  * A program on an active WebGL context
208  *
209  * @param hash SceneJS-managed ID for program
210  * @param lastUsed Time program was lst activated, for LRU cache eviction
211  * @param context WebGL context
212  * @param vertexSources Source codes for vertex shaders
213  * @param fragmentSources Source codes for fragment shaders
214  * @param logging Program and shaders will write to logging's debug channel as they compile and link
215  */
216 function SceneJS_webgl_Program(hash, lastUsed, context, vertexSources, fragmentSources, logging) {
217     this.hash = hash;
218     this.lastUsed = lastUsed;
219 
220     /* Create shaders from sources
221      */
222     var shaders = [];
223     for (var i = 0; i < vertexSources.length; i++) {
224         shaders.push(new SceneJS_webgl_Shader(context, context.VERTEX_SHADER, vertexSources[i], logging));
225     }
226     for (var i = 0; i < fragmentSources.length; i++) {
227         shaders.push(new SceneJS_webgl_Shader(context, context.FRAGMENT_SHADER, fragmentSources[i], logging));
228     }
229 
230     /* Create program, attach shaders, link and validate program
231      */
232     var handle = context.createProgram();
233 
234     for (var i = 0; i < shaders.length; i++) {
235         var shader = shaders[i];
236         if (shader.valid) {
237             context.attachShader(handle, shader.handle);
238         }
239     }
240     context.linkProgram(handle);
241     context.validateProgram(handle);
242 
243     this.valid = true;
244     this.valid = this.valid && (context.getProgramParameter(handle, context.LINK_STATUS) != 0);
245     this.valid = this.valid && (context.getProgramParameter(handle, context.VALIDATE_STATUS) != 0);
246 
247     //     logging.debug("Creating shader program: '" + hash + "'");
248     if (this.valid) {
249         //  logging.debug("Program link succeeded: " + context.getProgramInfoLog(handle));
250     }
251     else {
252         logging.debug("Program link failed: " + context.getProgramInfoLog(handle));
253     }
254 
255     if (!this.valid) {
256         SceneJS._backends.getBackend("error").fatalError(
257                 new SceneJS.exceptions.ShaderLinkFailureException("Shader program failed to link"));
258     }
259 
260     /* Discover active uniforms and samplers
261      */
262     var uniforms = {};
263     var samplers = {};
264 
265     var numUniforms = context.getProgramParameter(handle, context.ACTIVE_UNIFORMS);
266 
267     for (var i = 0; i < numUniforms; ++i) {
268         var u = context.getActiveUniform(handle, i);
269         if (u) {
270             var location = context.getUniformLocation(handle, u.name);
271             if ((u.type == context.SAMPLER_2D) || (u.type == context.SAMPLER_CUBE) || (u.type == 35682)) {
272 
273                 samplers[u.name] = new SceneJS_webgl_ProgramSampler(
274                         context,
275                         handle,
276                         u.name,
277                         u.type,
278                         u.size,
279                         location,
280                         logging);
281             } else {
282                 uniforms[u.name] = new SceneJS_webgl_ProgramUniform(
283                         context,
284                         handle,
285                         u.name,
286                         u.type,
287                         u.size,
288                         location,
289                         logging);
290             }
291         }
292     }
293 
294     /* Discover attributes
295      */
296     var attributes = {};
297 
298     var numAttribs = context.getProgramParameter(handle, context.ACTIVE_ATTRIBUTES);
299     for (var i = 0; i < numAttribs; i++) {
300         var a = context.getActiveAttrib(handle, i);
301         if (a) {
302             var location = context.getAttribLocation(handle, a.name);
303             attributes[a.name] = new SceneJS_webgl_ProgramAttribute(
304                     context,
305                     handle,
306                     a.name,
307                     a.type,
308                     a.size,
309                     location,
310                     logging);
311         }
312     }
313 
314     this.bind = function() {
315         context.useProgram(handle);
316     };
317 
318 
319     this.setUniform = function(name, value) {
320         var u = uniforms[name];
321         if (u) {
322             u.setValue(value);
323         } else {
324             //    logging.warn("Shader uniform load failed - uniform not found in shader : " + name);
325         }
326     };
327 
328     this.bindFloatArrayBuffer = function(name, buffer) {
329         var attr = attributes[name];
330         if (attr) {
331             attr.bindFloatArrayBuffer(buffer);
332         } else {
333             //  logging.warn("Shader attribute bind failed - attribute not found in shader : " + name);
334         }
335     };
336 
337     this.bindTexture = function(name, texture, unit) {
338         var sampler = samplers[name];
339         if (sampler) {
340             sampler.bindTexture(texture, unit);
341         } else {
342             //  logging.warn("Sampler not found: " + name);
343         }
344     };
345 
346     this.unbind = function() {
347         //     context.useProgram(0);
348     };
349 
350     this.destroy = function() {
351         if (this.valid) {
352             //   logging.debug("Destroying shader program: '" + hash + "'");
353             context.deleteProgram(handle);
354             for (var s in shaders) {
355                 context.deleteShader(shaders[s].handle);
356             }
357             attributes = null;
358             uniforms = null;
359             samplers = null;
360             this.valid = false;
361         }
362     };
363 }

365 function SceneJS_webgl_Texture2D(context, cfg) {
366     //  cfg.logging.debug("Creating texture: '" + cfg.textureId + "'");
367     this.canvas = cfg.canvas;
368     this.textureId = cfg.textureId;
369     this.handle = context.createTexture();
370     this.target = context.TEXTURE_2D;
371     this.minFilter = cfg.minFilter;
372     this.magFilter = cfg.magFilter;
373     this.wrapS = cfg.wrapS;
374     this.wrapT = cfg.wrapT;
375 
376     context.bindTexture(this.target, this.handle);
377 
378     if (cfg.image) {
379 
380         /* Texture from image
381          */
382         context.texImage2D(context.TEXTURE_2D, 0, cfg.image, cfg.flipY);
383 
384         this.format = context.RGBA;
385         this.width = cfg.image.width;
386         this.height = cfg.image.height;
387         this.isDepth = false;
388         this.depthMode = 0;
389         this.depthCompareMode = 0;
390         this.depthCompareFunc = 0;
391 
392     } else {
393 
394         /* Texture from data
395          */
396         if (!cfg.texels) {
397             if (cfg.sourceType == context.FLOAT) {
398                 cfg.texels = new WebGLFloatArray(cfg.width * cfg.height * 4);
399             }
400             else {
401                 cfg.texels = new WebGLUnsignedByteArray(cfg.width * cfg.height * 4);
402             }
403         }
404 
405         context.texImage2D(context.TEXTURE_2D, 0, cfg.internalFormat, cfg.width, cfg.height, 0, cfg.sourceFormat, cfg.sourceType, cfg.texels);
406 
407         if (cfg.isDepth) {
408             if (cfg.depthMode) {
409                 context.texParameteri(context.TEXTURE_2D, context.DEPTH_TEXTURE_MODE, cfg.depthMode);
410             }
411             if (cfg.depthCompareMode) {
412                 context.texParameteri(context.TEXTURE_2D, context.TEXTURE_COMPARE_MODE, cfg.depthCompareMode);
413             }
414             if (cfg.depthCompareFunc) {
415                 context.texParameteri(context.TEXTURE_2D, context.TEXTURE_COMPARE_FUNC, cfg.depthCompareFunc);
416             }
417         }
418 
419         this.format = cfg.internalFormat;
420         this.width = cfg.width;
421         this.height = cfg.height;
422         this.isDepth = cfg.isDepth;
423         this.depthMode = cfg.depthMode;
424         this.depthCompareMode = cfg.depthCompareMode;
425         this.depthCompareFunc = cfg.depthCompareFunc;
426     }
427 
428     if (cfg.minFilter) {
429         context.texParameteri(// Filtered technique when scaling texture down
430                 context.TEXTURE_2D,
431                 context.TEXTURE_MIN_FILTER,
432                 cfg.minFilter);
433     }
434 
435     if (cfg.magFilter) {
436         context.texParameteri(// Filtering technique when scaling texture up
437                 context.TEXTURE_2D,
438                 context.TEXTURE_MAG_FILTER,
439                 cfg.magFilter);
440     }
441     if (cfg.wrapS) {
442         context.texParameteri(
443                 context.TEXTURE_2D,
444                 context.TEXTURE_WRAP_S,
445                 cfg.wrapS);
446     }
447 
448     if (cfg.wrapT) {
449         context.texParameteri(
450                 context.TEXTURE_2D,
451                 context.TEXTURE_WRAP_T,
452                 cfg.wrapT);
453     }
454 
455     /* Generate MIP map if required
456      */
457     if (cfg.minFilter == context.NEAREST_MIPMAP_NEAREST ||
458         cfg.minFilter == context.LINEAR_MIPMAP_NEAREST ||
459         cfg.minFilter == context.NEAREST_MIPMAP_LINEAR ||
460         cfg.minFilter == context.LINEAR_MIPMAP_LINEAR) {
461 
462         context.generateMipmap(context.TEXTURE_2D);
463     }
464 
465     context.bindTexture(this.target, null);
466 
467     //        gl.activeTexture(gl.TEXTURE0);
468     //  gl.bindTexture(gl.TEXTURE_2D, cubeTexture);
469     //  gl.uniform1i(gl.getUniformLocation(shaderProgram, "uSampler"), 0);
470 
471     this.bind = function(unit) {
472         context.activeTexture(context["TEXTURE" + unit]);
473         context.bindTexture(this.target, this.handle);
474 
475     };
476 
477     this.unbind = function(unit) {
478         context.activeTexture(context["TEXTURE" + unit]);
479         context.bindTexture(this.target, null);
480     };
481 
482     this.generateMipmap = function() {
483         context.generateMipmap(context.TEXTURE_2D);
484     };
485 
486     this.destroy = function() {
487         if (this.handle) {
488             // cfg.logging.debug("Destroying texture");
489             context.deleteTexture(this.handle);
490             this.handle = null;
491         }
492     };
493 }

495 /** Buffer for vertices and indices
496  *
497  * @param context  WebGL context
498  * @param type     Eg. ARRAY_BUFFER, ELEMENT_ARRAY_BUFFER
499  * @param values   WebGL array wrapper
500  * @param numItems Count of items in array wrapper
501  * @param itemSize Size of each item
502  * @param usage    Eg. STATIC_DRAW
503  */
504 function SceneJS_webgl_ArrayBuffer(context, type, values, numItems, itemSize, usage) {
505     this.handle = context.createBuffer();
506     context.bindBuffer(type, this.handle);
507     context.bufferData(type, values, usage);
508     context.bindBuffer(type, null);
509 
510     this.type = type;
511     this.numItems = numItems;
512     this.itemSize = itemSize;
513 
514     this.bind = function() {
515         context.bindBuffer(type, this.handle);
516     };
517 
518     this.unbind = function() {
519         context.bindBuffer(type, null);
520     };
521 
522     this.destroy = function() {
523         context.deleteBuffer(this.handle);
524     };
525 }
