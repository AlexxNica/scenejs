1 /**
  2  * Backend that manages the current projection transform matrix.
  3  *
  4  * Services the scene projection transform nodes, such as SceneJS.frustum, providing them with methods to set and
  5  * get the current projection matrix.
  6  *
  7  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
  8  * PROJECTION_TRANSFORM_EXPORTED to pass the projection matrix as a WebGLFloatArray to the shading backend.
  9  *
 10  * The WebGLFloatArray is lazy-computed and cached on export to avoid repeatedly regenerating it.
 11  *
 12  * Avoids redundant export of the matrix with a dirty flag; the matrix is only exported when the flag is set, which
 13  * occurs when the matrix is set by scene node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and SHADER_DEACTIVATED events.
 14  *
 15  * Whenever a scene node sets the matrix, this backend publishes it with a PROJECTION_TRANSFORM_UPDATED to allow other
 16  * dependent backends (such as "view-frustum") to synchronise their resources.
 17  */
 18 SceneJS._backends.installBackend(
 19 
 20         "projection",
 21 
 22         function(ctx) {
 23 
 24             var transform;
 25             var dirty;
 26 
 27             ctx.events.onEvent(
 28                     SceneJS._eventTypes.SCENE_ACTIVATED,
 29                     function() {
 30                         transform = {
 31                             matrix : SceneJS_math_identityMat4(),
 32                             fixed: true
 33                         };
 34                         dirty = true;
 35                     });
 36 
 37             ctx.events.onEvent(
 38                     SceneJS._eventTypes.SHADER_ACTIVATED,
 39                     function() {
 40                         dirty = true;
 41                     });
 42 
 43             ctx.events.onEvent(
 44                     SceneJS._eventTypes.SHADER_RENDERING,
 45                     function() {
 46                         if (dirty) {                
 47                             if (!transform.matrixAsArray) {
 48                                 transform.matrixAsArray = new WebGLFloatArray(transform.matrix);
 49                             }
 50                             ctx.events.fireEvent(
 51                                     SceneJS._eventTypes.PROJECTION_TRANSFORM_EXPORTED,
 52                                     transform);
 53                             dirty = false;
 54                         }
 55                     });
 56 
 57             ctx.events.onEvent(
 58                     SceneJS._eventTypes.SHADER_DEACTIVATED,
 59                     function() {
 60                         dirty = true;
 61                     });
 62 
 63             /* Node-facing API
 64              */
 65             return {
 66 
 67                 setTransform: function(t) {
 68                     transform = t;
 69                     dirty = true;
 70                     ctx.events.fireEvent(
 71                             SceneJS._eventTypes.PROJECTION_TRANSFORM_UPDATED,
 72                             transform);
 73                 },
 74 
 75                 getTransform: function() {
 76                     return transform;
 77                 }
 78             };
 79         });
