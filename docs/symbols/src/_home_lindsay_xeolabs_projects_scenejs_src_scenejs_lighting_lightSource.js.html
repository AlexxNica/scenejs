1 /** @class SceneJS.LightSource
  2 
  3  A light source for containment within a @link SceneJS.Lights} node.
  4 
  5  @constructor
  6  Create a new SceneJS.LightSource
  7  @param {Object} cfg The config object
  8  */
  9 SceneJS.LightSource = function(cfg) {
 10     this._type = "point";
 11     this._color = [1.0, 1.0, 1.0];
 12     this._diffuse = true;
 13     this._specular = true;
 14     this._pos = [0.0, 0.0, 0.0];
 15     this._viewPos = [0.0, 0.0, 0.0]; // Transformed view-space pos - accessed by lights module and shading module
 16     this._dir = [0.0, 0.0, -1.0];
 17     this._constantAttenuation = 1.0;
 18     this._linearAttenuation = 0.0;
 19     this._quadraticAttenuation = 0.0;
 20 
 21     if (cfg) {
 22         this._init(cfg);
 23     }
 24 };
 25 
 26 /** Sets the light source type
 27  * @param {String} type Light source type - "dir" or "point"
 28  * @return {SceneJS.LightSource} this
 29  */
 30 SceneJS.LightSource.prototype.setType = function(type) {
 31     if (type != "dir" && type != "point") {
 32         throw SceneJS_errorModule.fatalError(new SceneJS.InvalidNodeConfigException(
 33                 "SceneJS.LightSource unsupported type - should be 'dir' or 'point' or 'ambient'"));
 34     }
 35     this._type = type;
 36     return this;
 37 };
 38 
 39 /** Gets the light source type
 40  * @return {String} Light source type - "dir" or "point"
 41  */
 42 SceneJS.LightSource.prototype.getType = function() {
 43     return this._type;
 44 };

 46 /** Sets the light source color
 47  *
 48  * @param color {Object} - Eg. {r: 1.0, g: 1.0, b: 1.0 }
 49  * @return {SceneJS.LightSource} this
 50  */
 51 SceneJS.LightSource.prototype.setColor = function(color) {
 52     this._color = [
 53         color.r != undefined ? color.r : 1.0,
 54         color.g != undefined ? color.g : 1.0,
 55         color.b != undefined ? color.b : 1.0
 56     ];
 57     return this;
 58 };
 59 
 60 /** Gets the light source color
 61  * @return {Object} Eg. {r: 1.0, g: 1.0, b: 1.0 }
 62  */
 63 SceneJS.LightSource.prototype.getColor = function() {
 64     return {
 65         r: this._color[0],
 66         g: this._color[1],
 67         b: this._color[2] };
 68 };

106 /** Sets the light source object-space position.
107  * This is only used when the source is of type "point".

125 /** Sets the light source object-space direction vector.
126  * This is only used when the source is of type "dir".
127  * Components will fall back on defaults of { x: 0, y: 0, z: -1 } where not supplied;
128  * <pre><code>
129  * myLightSource.setDir({  });       // Sets direction of { x : 0.0, y: 0.0, z: -1.0 }
130  * myLightSource.setDir({ y: 2.0 }); // Sets direction of { x : 0.0, y: 2.0, z: -1.0 }
131  * </pre></code>
132  *
133  * @param dir {Object} - Eg. {x: 5.0, y: 5.0, z: 5.0 }
134  * @return {SceneJS.LightSource} this
135  */
136 SceneJS.LightSource.prototype.setDir = function(dir) {
137     this._dir = [ dir.x || 0.0, dir.y || 0.0, (dir.z == undefined || dir.z == null) ? -1 : dir.z ];
138     return this;
139 };
140

144  */

146     return { x: this._dir[0], y: this._dir[1], z: this._dir[2] };
147 };

207 // @private
208 SceneJS.LightSource.prototype._init = function(cfg) {
209     if (cfg) {
210         if (cfg.type) {
211             this.setType(cfg.type);
212         }
213         if (cfg.color) {
214             this.setColor(cfg.color);
215         }
216         if (cfg.diffuse != undefined) {
217             this._diffuse = cfg.diffuse;
218         }
219         if (cfg.specular != undefined) {
220             this._specular = cfg.specular;
221         }
222         if (cfg.pos) {
223             this.setPos(cfg.pos);
224         }
225         if (cfg.dir) {
226             this.setDir(cfg.dir);
227         }
228         if (cfg.constantAttenuation) {
229             this.setConstantAttenuation(cfg.constantAttenuation);
230         }
231         if (cfg.linearAttenuation) {
232             this.setLinearAttenuation(cfg.linearAttenuation);
233         }
234         if (cfg.quadraticAttenuation) {
235             this.setQuadraticAttenuation(cfg.quadraticAttenuation);
236         }
237     }
238 };
