1 /**
  2  * Backend that manages scene lighting.
  3  *
  4  * Holds the sources on a stack and provides the SceneJS.light node with methods to push and pop them.
  5  *
  6  * Tracks the view and modelling transform matrices through incoming VIEW_TRANSFORM_UPDATED and
  7  * MODEL_TRANSFORM_UPDATED events. As each light are pushed, its position and/or direction is multipled by the
  8  * matrices. The stack will therefore contain sources that are instanced in view space by different modelling
  9  * transforms, with positions and directions that may be animated,
 10  *
 11  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
 12  * LIGHTS_EXPORTED to pass the entire light stack to the shading backend.
 13  *
 14  * Avoids redundant export of the sources with a dirty flag; they are only exported when that is set, which occurs
 15  * when the stack is pushed or popped by the lights node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and
 16  * SHADER_DEACTIVATED events.
 17  *
 18  * Whenever a scene node pushes or pops the stack, this backend publishes it with a LIGHTS_UPDATED to allow other
 19  * dependent backends to synchronise their resources.
 20  */
 21 SceneJS._backends.installBackend(
 22 
 23         "lights",
 24 
 25         function(ctx) {
 26 
 27             var viewMat;
 28             var modelMat;
 29             var lightStack = [];
 30             var dirty;
 31 
 32             ctx.events.onEvent(
 33                     SceneJS._eventTypes.SCENE_ACTIVATED,
 34                     function() {
 35                         modelMat = viewMat = SceneJS_math_identityMat4();
 36                         lightStack = [];
 37                         dirty = true;
 38                     });
 39 
 40             ctx.events.onEvent(
 41                     SceneJS._eventTypes.SHADER_ACTIVATED,
 42                     function() {
 43                         dirty = true;
 44                     });
 45 
 46             ctx.events.onEvent(
 47                     SceneJS._eventTypes.VIEW_TRANSFORM_UPDATED,
 48                     function(params) {
 49                         viewMat = params.matrix;
 50                     });
 51 
 52             ctx.events.onEvent(
 53                     SceneJS._eventTypes.MODEL_TRANSFORM_UPDATED,
 54                     function(params) {
 55                         modelMat = params.matrix;
 56                     });
 57 
 58             ctx.events.onEvent(
 59                     SceneJS._eventTypes.SHADER_RENDERING,
 60                     function() {
 61                         if (dirty) {
 62                             ctx.events.fireEvent(
 63                                     SceneJS._eventTypes.LIGHTS_EXPORTED,
 64                                     lightStack);
 65                             dirty = false;
 66                         }
 67                     });
 68 
 69             ctx.events.onEvent(
 70                     SceneJS._eventTypes.SHADER_DEACTIVATED,
 71                     function() {
 72                         dirty = true;
 73                     });
 74 
 75             function vectorToArray(v, fallback) {
 76                 return v ? [ v.x || 0, v.y || 0, v.z || 0] : fallback;
 77             }
 78 
 79             function pointToArray(p, fallback) {
 80                 return p ? [ p.x || 0, p.y || 0, p.z || 0, p.w || 1] : fallback;
 81             }
 82 
 83             function colourToArray(v, fallback) {
 84                 return v ? [
 85                     v.r == undefined ? fallback[0] : v.r,
 86                     v.g == undefined ? fallback[1] : v.g,
 87                     v.b == undefined ? fallback[2] : v.b] : fallback;
 88             }
 89 
 90             /* Creates a view-space light
 91              */
 92             function createLight(light) {
 93                 if (light.type) {
 94                     if (light.type != "dir"
 95                             && light.type != "point") {
 96                         ctx.error.fatalError(new SceneJS.exceptions.InvalidNodeConfigException(
 97                                 "SceneJS.light node has a light of unsupported type - should be 'dir' or 'point'"));
 98                     }
 99                 } else {
100                     light.type = light.type || "point";
101                 }
102 
103                 if (light.type == "point") {
104                     return {
105                         type: light.type,
106                         color: colourToArray(light.color, [ 1.0, 1.0, 1.0 ]),
107                         diffuse : light.diffuse == undefined ? true : light.diffuse,
108                         specular : light.specular == undefined ? true : light.specular,
109                         pos : SceneJS_math_transformPoint3(
110                                 viewMat,
111                                 SceneJS_math_transformPoint3(
112                                         modelMat,
113                                         pointToArray(light.pos, [ 0,  0, 1.0, 1.0]))),
114                         constantAttenuation: light.constantAttenuation == undefined ? 1.0 : light.constantAttenuation,
115                         linearAttenuation: light.linearAttenuation == undefined ? 0.0 : light.linearAttenuation,
116                         quadraticAttenuation: light.quadraticAttenuation == undefined ? 0.0 : light.quadraticAttenuation
117                     };
118                 }
119 
120                 if (light.type == "dir") {
121                     return {
122                         type: light.type,
123                         color: colourToArray(light.color, [ 1.0, 1.0, 1.0 ]),
124                         diffuse : light.diffuse == undefined ? true : light.diffuse,
125                         specular : light.specular == undefined ? true : light.specular,
126                         dir : SceneJS_math_transformVector3(
127                                 viewMat,
128                                 SceneJS_math_transformVector3(
129                                         modelMat,
130                                         vectorToArray(light.dir, [ 0,  0,  1.0])))
131                     };
132                 }
133 
134                 //                if (light.type == "spot") {
135                 //                    return {
136                 //                        type: light.type,
137                 //                        color: colourToArray(light.color, [ 1.0, 1.0, 1.0 ]),
138                 //                        diffuse : light.diffuse,
139                 //                        specular : light.specular,
140                 //                        pos : SceneJS_math_transformPoint3(
141                 //                                viewMat,
142                 //                                SceneJS_math_transformPoint3(
143                 //                                        modelMat,
144                 //                                        pointToArray(light.pos, [ 0,  0,  1.0, 1.0]))),
145                 //
146                 //                        dir : SceneJS_math_transformVector3(
147                 //                                viewMat,
148                 //                                SceneJS_math_transformVector3(
149                 //                                        modelMat,
150                 //                                        vectorToArray(light.dir, [ 0,  0,  1.0]))),
151                 //                        spotExponent: light.spotExponent == undefined ? 1.0 : light.spotExponent,
152                 //                        spotCosCutOff: light.spotCosCutOff == undefined ? 20.0 : light.spotCosCutOff,
153                 //                        constantAttenuation: light.constantAttenuation == undefined ? 1.0 : light.constantAttenuation,
154                 //                        linearAttenuation: light.linearAttenuation == undefined ? 0.0 : light.linearAttenuation,
155                 //                        quadraticAttenuation: light.quadraticAttenuation == undefined ? 0.0 : light.quadraticAttenuation
156                 //                    };
157                 //                }
158             }
159 
160             /* Node-facing API
161              */
162             return {
163 
164                 pushLights : function(sources) {
165                     for (var i = 0; i < sources.length; i++) {
166                         lightStack.push(createLight(sources[i]));
167                     }
168                     dirty = true;
169                     ctx.events.fireEvent(
170                             SceneJS._eventTypes.LIGHTS_UPDATED,
171                             lightStack);
172                 },
173 
174                 popLights : function(numSources) {
175                     for (var i = 0; i < numSources; i++) {
176                         lightStack.pop();
177                     }
178                     dirty = true;
179                     ctx.events.fireEvent(
180                             SceneJS._eventTypes.LIGHTS_UPDATED,
181                             lightStack);
182                 }
183             };
184         });
