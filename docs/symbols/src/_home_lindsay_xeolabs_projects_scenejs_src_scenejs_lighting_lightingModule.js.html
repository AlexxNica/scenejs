1 /**
  2  * Backend that manages scene lighting.
  3  *
  4  * Holds the sources on a stack and provides the SceneJS.light node with methods to push and pop them.
  5  *
  6  * Tracks the view and modelling transform matrices through incoming VIEW_TRANSFORM_UPDATED and
  7  * MODEL_TRANSFORM_UPDATED events. As each light are pushed, its position and/or direction is multipled by the
  8  * matrices. The stack will therefore contain sources that are instanced in view space by different modelling
  9  * transforms, with positions and directions that may be animated,
 10  *
 11  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
 12  * LIGHTS_EXPORTED to pass the entire light stack to the shading backend.
 13  *
 14  * Avoids redundant export of the sources with a dirty flag; they are only exported when that is set, which occurs
 15  * when the stack is pushed or popped by the lights node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and
 16  * SHADER_DEACTIVATED events.
 17  *
 18  * Whenever a scene node pushes or pops the stack, this backend publishes it with a LIGHTS_UPDATED to allow other
 19  * dependent backends to synchronise their resources.
 20  *
 21  *  @private
 22  */
 23 var SceneJS_lightingModule = new (function() {
 24 
 25     var viewMat;
 26     var modelMat;
 27     var lightStack = [];
 28     var dirty;
 29 
 30     SceneJS_eventModule.onEvent(
 31             SceneJS_eventModule.SCENE_ACTIVATED,
 32             function() {
 33                 modelMat = viewMat = SceneJS_math_identityMat4();
 34                 lightStack = [];
 35                 dirty = true;
 36             });
 37 
 38     SceneJS_eventModule.onEvent(
 39             SceneJS_eventModule.SHADER_ACTIVATED,
 40             function() {
 41                 dirty = true;
 42             });
 43 
 44     SceneJS_eventModule.onEvent(
 45             SceneJS_eventModule.VIEW_TRANSFORM_UPDATED,
 46             function(params) {
 47                 viewMat = params.matrix;
 48             });
 49 
 50     SceneJS_eventModule.onEvent(
 51             SceneJS_eventModule.MODEL_TRANSFORM_UPDATED,
 52             function(params) {
 53                 modelMat = params.matrix;
 54             });
 55 
 56     SceneJS_eventModule.onEvent(
 57             SceneJS_eventModule.SHADER_RENDERING,
 58             function() {
 59                 if (dirty) {
 60                     SceneJS_eventModule.fireEvent(
 61                             SceneJS_eventModule.LIGHTS_EXPORTED,
 62                             lightStack);
 63                     dirty = false;
 64                 }
 65             });
 66 
 67     SceneJS_eventModule.onEvent(
 68             SceneJS_eventModule.SHADER_DEACTIVATED,
 69             function() {
 70                 dirty = true;
 71             });
 72 
 73     /**
 74      * @private
 75      */
 76     function instanceSources(sources) {
 77         for (var i = 0; i < sources.length; i++) {
 78             var source = sources[i];
 79             if (source._type == "point") {
 80                 source._viewPos =  SceneJS_math_transformPoint3(viewMat, SceneJS_math_transformPoint3(modelMat, source._pos));
 81             } else if (source._type == "dir") {
 82                 source._viewDir = SceneJS_math_transformVector3(viewMat, SceneJS_math_transformVector3(modelMat, source._dir));
 83             }
 84         }
 85     };
 86 
 87     // @private
 88     this.pushLightSources = function(sources) {
 89         instanceSources(sources);
 90         for (var i = 0; i < sources.length; i++) {
 91             lightStack.push(sources[i]);
 92         }
 93         dirty = true;
 94         SceneJS_eventModule.fireEvent(
 95                 SceneJS_eventModule.LIGHTS_UPDATED,
 96                 lightStack);
 97     };
 98 
 99     // @private
100     this.popLightSources = function(numSources) {
101         for (var i = 0; i < numSources; i++) {
102             lightStack.pop();
103         }
104         dirty = true;
105         SceneJS_eventModule.fireEvent(
106                 SceneJS_eventModule.LIGHTS_UPDATED,
107                 lightStack);
108     };
109 })();
