  4 SceneJS.stationary = function() {
  5     var cfg = SceneJS._utils.getNodeConfig(arguments);
  6     var backend = SceneJS._backends.getBackend('view-transform');
  7 
  8     return SceneJS._utils.createNode(
  9             "stationary",
 10             cfg.children,
 11 
 12             new (function() {
 13 
 14                 var xform;
 15 
 16                 this._render = function(traversalContext, data) {
 17                     var superXform = backend.getTransform();
 18                     var lookAt = superXform.lookAt;
 19                     if (lookAt) {
 20                         if (!xform || !superXform.fixed) {
 21                             xform = {
 22                                 matrix: SceneJS_math_mulMat4(
 23                                         superXform.matrix,
 24                                         SceneJS_math_translationMat4c(
 25                                                 lookAt.eye.x,
 26                                                 lookAt.eye.y,
 27                                                 lookAt.eye.z)),
 28                                 lookAt: lookAt,
 29                                 fixed: superXform.fixed
 30                             };
 31                         }
 32                         backend.setTransform(xform);
 33                         this._renderChildren(traversalContext, data);
 34                         backend.setTransform(superXform);
 35                     } else {
 36                         this._renderChildren(traversalContext, data);
 37                     }
 38                 };
 39             })());
 40 };
