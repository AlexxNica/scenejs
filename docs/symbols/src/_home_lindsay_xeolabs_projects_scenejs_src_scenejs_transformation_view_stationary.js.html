1 /**
  2  * @class SceneJS.Stationary
  3  * @extends SceneJS.Node
  4  *
  5  * <p>Scene node that defines a region within a SceneJS.LookAt in which the translations specified by that node
  6  * have no effect. As the parameters of the SceneJS.lookAt are modified, the content in the subgraph
  7  * of this node will rotate about the eye position, but will not translate as the eye position moves. You could therefore
  8  * define a skybox within the subgraph of this node, that will always stay in the distance.</p>
  9  *
 10  * <p><b>Example:</b></p><p>A box that the eye position never appears to move outside of</b></p><pre><code>
 11  * var l = new SceneJS.LookAt({
 12  *     eye  : { x: 0.0, y: 10.0, z: -15 },
 13  *     look : { y:1.0 },
 14  *     up   : { y: 1.0 },
 15  *
 16  *      new SceneJS.Stationary(
 17  *          new SceneJS.Scale({ x: 100.0, y: 100.0, z: 100.0 },
 18  *              new SceneJS.objects.Cube()
 19  *          )
 20  *      )
 21  *  )
 22  *
 23  * </pre></code>
 24  *
 25  *  @constructor
 26  * Create a new SceneJS.Stationary
 27  * @param {args} args Zero or more child nodes
 28  */
 29 SceneJS.Stationary = function() {
 30     SceneJS.Node.apply(this, arguments);
 31     this._nodeType = "stationary";
 32     this._xform = null;
 33 };
 34 
 35 SceneJS._utils.inherit(SceneJS.Stationary, SceneJS.Node);
 36 
 37 SceneJS.Stationary.prototype._render = function(traversalContext, data) {
 38     var superXform = SceneJS_viewTransformModule.getTransform();
 39     var lookAt = superXform.lookAt;
 40     if (lookAt) {
 41         if (this._memoLevel == 0) {
 42             this._xform = {
 43                 matrix: SceneJS_math_mulMat4(
 44                         superXform.matrix,
 45                         SceneJS_math_translationMat4c(
 46                                 lookAt.eye.x,
 47                                 lookAt.eye.y,
 48                                 lookAt.eye.z)),
 49                 lookAt: lookAt,
 50                 fixed: superXform.fixed
 51             };
 52             if (superXform.fixed) {
 53                 this._memoLevel = 1;
 54             }
 55         }
 56         SceneJS_viewTransformModule.setTransform(this._xform);
 57         this._renderNodes(traversalContext, data);
 58         SceneJS_viewTransformModule.setTransform(superXform);
 59     } else {
 60         this._renderNodes(traversalContext, data);
 61     }
 62 };
 63 
 64 /** Function wrapper to support functional scene definition
 65  */
 66 SceneJS.stationary = function() {
 67     var n = new SceneJS.Stationary();
 68     SceneJS.Stationary.prototype.constructor.apply(n, arguments);
 69     return n;
 70 };
