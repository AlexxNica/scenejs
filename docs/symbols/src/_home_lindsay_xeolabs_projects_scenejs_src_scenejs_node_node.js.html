1 /**
  2  @class The basic scene node type, providing the ability to connect nodes into parent-child relationships to form scene graphs.
  3  <p>Every node type has a specific type, which is its SceneJS-specific type name, along with methods for checking the
  4  * type like {@link #getType}. This is the list of all valid xtypes:</p>
  5  *
  6  * <table>
  7  * <tr><td>type</td><td>Class</td></tr>
  8  * <tr><td>----</td><td>-----</td></tr>
  9  * <tr><td>bounding-box</td><td>{@link SceneJS.BoundingBox}</td></tr>

 28  * <tr><td>rotate</td><td>{@link SceneJS.Rotate}</td></tr>
 29  * <tr><td>scale</td><td>{@link SceneJS.Scale}</td></tr>
 30  * <tr><td>scene</td><td>{@link SceneJS.Scene}</td></tr>
 31  * <tr><td>scalar-interpolator</td><td>{@link SceneJS.ScalerInterpolator}</td></tr>
 32  * <tr><td>selector</td><td>{@link SceneJS.Selector}</td></tr>
 33  * <tr><td>sphere</td><td>{@link SceneJS.objects.Sphere}</td></tr>
 34  * <tr><td>stationary</td><td>{@link SceneJS.Stationary}</td></tr>
 35  * <tr><td>symbol</td><td>{@link SceneJS.Symbol}</td></tr>
 36  * <tr><td>teapot</td><td>{@link SceneJS.objects.Teapot}</td></tr>
 37  * <tr><td>text</td><td>{@link SceneJS.Text}</td></tr>
 38  * <tr><td>texture</td><td>{@link SceneJS.Texture}</td></tr>
 39  * <tr><td>translate</td><td>{@link SceneJS.Translate}</td></tr>
 40  * <tr><td>view-matrix</td><td>{@link SceneJS.ViewMatrix}</td></tr>
 41  * <tr><td>with-data</td><td>{@link SceneJS.WithData}</td></tr>
 42  * </table>
 43  * @constructor
 44  * Create a new SceneJS.node
 45  * @param {SceneJS.node, ...} arguments Zero or more child nodes
 46  */
 47 SceneJS.Node = function() {
 48     this._nodeType = "node";
 49     this._children = [];
 50     this._fixedParams = true;
 51     this._parent = null;
 52 
 53     /* Used by many node types to track the level at which they can
 54      * memoise internal state. When rendered, a node increments
 55      * this each time it discovers that it can cache more state, so that
 56      * it knows not to recompute that state when next rendered.
 57      * Since internal state is usually dependent on the states of higher
 58      * nodes, this is reset whenever the node is attached to a new
 59      * parent.
 60      *
 61      * private
 62      */
 63     this._memoLevel = 0;
 64 
 65     /* Configure from variable args
 66      */
 67     if (arguments.length > 0) {
 68         for (var i = 0; i < arguments.length; i++) {
 69             var arg = arguments[i];
 70             if (arg._render) {                      // arg is node
 71                 this._children.push(arg);
 72             } else if (i == 0) {                    // arg is config
 73                 if (arg instanceof Function) {      // arg is config function
 74                     this._getParams = arg;
 75                     this._fixedParams = false;
 76                 } else {
 77                     var config = arg;               // arg is config object
 78                     this._fixedParams = true;
 79                     for (var key in arg) {
 80                         if (arg.hasOwnProperty(key)) {
 81                             if (this._fixedParams && arg[key] instanceof Function) {
 82                                 this._fixedParams = false;
 83                             }
 84                         }
 85                     }
 86                     this._getParams = function() {  // Wrap with function to return config object
 87                         return config;
 88                     };
 89                 }
 90             } else {
 91                 SceneJS_errorModule.fatalError(new SceneJS.InvalidNodeConfigException
 92                         ("Invalid node parameters - config should be first, IE. node(config, node, node,...)"));
 93             }
 94         }
 95         if (!this._getParams) {
 96             this._getParams = function() {
 97                 return {
 98                 };
 99             };
100         }
101     } else {
102         if (!this._getParams) {
103             this._getParams = function() {
104                 return {};
105             };
106         }
107     }
108 };

112 /**
113  * Resets memoization level to zero - used when moving nodes around in graph
114  * @private
115  */
116 SceneJS.Node.prototype._resetMemoLevel = function() {
117     this._memoLevel = 0;
118     for (var i = 0; i < this._children.length; i++) {
119         this._children[i]._resetMemoLevel();
120     }
121 };
122 
123 /** @private */
124 SceneJS.Node.prototype._renderNodes = function(traversalContext, data, children) {
125     var child;
126     children = children || this._children;
127     var len = children.length;
128     if (len) {
129         for (var i = 0; i < len; i++) {
130             child = children[i];
131             child._render.call(child, { // Traversal context
132                 appendix : traversalContext.appendix,
133                 insideRightFringe: traversalContext.insideRightFringe || (i < len - 1)
134             }, data);
135         }
136     } else {
137 
138         /* Leaf node - if on right fringe of tree then
139          * render appended nodes
140          */
141         if (traversalContext.appendix && (!traversalContext.insideRightFringe)) {
142             len = traversalContext.appendix.length;
143             for (var i = 0; i < len; i++) {
144                 child = traversalContext.appendix[i];
145                 child._render.call(child, { // Traversal context
146                     appendix : null,
147                     insideRightFringe: (i < len - 1)
148                 }, data);
149             }
150         }
151     }
152 };
153 
154 /** @private */
155 SceneJS.Node.prototype._renderNode = function(index, traversalContext, data) {
156     var child = this._children[index];
157     child._render.call(child, traversalContext, data);
158 };
159 
160 /** @private */
161 SceneJS.Node.prototype._render = function(traversalContext, data) {
162     this._renderNodes(traversalContext, data);
163 };

165 /**
166  * Returns the type ID of the node. For the SceneJS.Node base class, it is "node",
167  * which is overriden in sub-classes.
168  * @returns {string} Type ID

174 /**
175  * Returns the number of child nodes

177  */
178 SceneJS.Node.prototype.getNumNodes = function() {
179     return this._children.length;

182 /** Returns child nodes
183  * @returns {Array} Child nodes
184  */
185 SceneJS.Node.prototype.getNodes = function() {
186     var list = new Array(this._children.length);
187     var len = this._children.length;
188     for (var i = 0; i < len; i++) {
189         list[i] = this._children[i];
190     }
191     return list;
192 };

194 /** Returns child node at given index
195  * @returns {SceneJS.Node} Child node
196  */
197 SceneJS.Node.prototype.getNodeAt = function(index) {
198     return this._children[index];
199 };

201 /** Removes the child node at the given index
202  * @param {int} index Child node index
203  */
204 SceneJS.Node.prototype.removeNodeAt = function(index) {
205     var r = this._children.splice(index, 1);
206     if (r.length > 0) {
207         r[0]._parent = null;
208         return r[0];
209     } else {
210         return null;
211     }
212 };

214 /** Appends a child node
215  * @param {SceneJS.Node} node Child node
216  * @return {SceneJS.Node} The child node
217  */
218 SceneJS.Node.prototype.addNode = function(node) {
219     if (node._parent != null) {
220         SceneJS_errorModule.fatalError(
221                 new SceneJS.InvalidSceneGraphException(
222                         "Attempted to add a child to a node without " +
223                         "first removing the child from it's current parent"));
224     }
225     this._children.push(node);
226     node._parent = this;
227     node._resetMemoLevel();
228     return node;
229 };
230 
231 /** Inserts a child node
232  * @param {SceneJS.Node} node Child node
233  * @param {int} i Index for new child node
234  * @return {SceneJS.Node} The child node
235  */
236 SceneJS.Node.prototype.insertNode = function(node, i) {
237     if (node._parent != null) {
238          SceneJS_errorModule.fatalError(
239                 new SceneJS.InvalidSceneGraphException(
240                         "Attempted to insert a child to a node without " +
241                         "first removing the child from it's current parent"));
242     }
243     if (i == undefined || i <= 0) {
244         this._children.unshift(node);
245     } else if (i >= this._children.length) {
246         this._children.push(node);
247     } else {
248         this._children.splice(i, 0, node);
249     }
250     node._parent = this;
251     node._resetMemoLevel();
252     return node;
253 };

262 /** Returns either all child or all sub-nodes of the given type, depending on whether search is recursive or not.
263  * @param {string} type Node type
264  * @param {boolean} recursive When true, will return all matching nodes in subgraph, otherwise returns just children (default)
265  * @return {SceneJS.node[]} Array of matching nodes
266  */
267 SceneJS.Node.prototype.findNodesByType = function(type, recursive) {
268     return this._findNodesByType(type, [], recursive);
269 };
270 
271 /** @private */
272 SceneJS.Node.prototype._findNodesByType = function(type, list, recursive) {
273     for (var i = 0; i < this._children; i++) {
274         var node = this._children[i];
275         if (node.nodeType == type) {
276             list.add(node);
277         }
278     }
279     if (recursive) {
280         for (var i = 0; i < this._children; i++) {
281             this._children[i]._findNodesByType(type, list, recursive);
282         }
283     }
284     return list;
285 };

287 /** Returns a new SceneJS.Node instance
288  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Node constructor
289  * @returns {SceneJS.Node}
290  */
291 SceneJS.node = function() {
292     var n = new SceneJS.Node();
293     SceneJS.Node.prototype.constructor.apply(n, arguments);
294     return n;
