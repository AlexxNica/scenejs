 69             function createPickBuffer() { // TODO: Dont assume allocation succeeds!
 70                 var context = canvas.context;
 71                 var pickBuffer = {
 72                     frameBuffer : context.createFramebuffer(),
 73                     renderBuffer : context.createRenderbuffer(),
 74                     texture : context.createTexture()
 75                 };
 76                 context.bindTexture(context.TEXTURE_2D, pickBuffer.texture);
 77                 try {
 78                     context.texImage2D(
 79                             context.TEXTURE_2D, 0, context.RGB, 1, 1, 0, context.RGB, context.UNSIGNED_BYTE, null);
 80                 } catch (e) {
 81                     var texture = new WebUnsignedByteArray(3);
 82                     context.texImage2D(
 83                             context.TEXTURE_2D, 0, context.RGB, 1, 1, 0, context.RGB, context.UNSIGNED_BYTE, texture);
 84                 }
 85                 context.bindFramebuffer(context.FRAMEBUFFER, pickBuffer.frameBuffer);
 86                 context.bindRenderbuffer(context.RENDERBUFFER, pickBuffer.renderBuffer);
 87                 context.renderbufferStorage(context.RENDERBUFFER, context.COLOR_COMPONENT, 1, 1);
 88                 context.bindRenderbuffer(context.RENDERBUFFER, null);
 89                 context.framebufferTexture2D(
 90                         context.FRAMEBUFFER, context.COLOR_ATTACHMENT0, context.TEXTURE_2D, pickBuffer.texture, 0);
 91                 context.framebufferRenderbuffer(
 92                         context.FRAMEBUFFER, context.DEPTH_ATTACHMENT, context.RENDERBUFFER, pickBuffer.renderBuffer);
 93                 context.bindFramebuffer(context.FRAMEBUFFER, null);
 94                 return pickBuffer;
 95             }

109             function pick(x, y) {
110 
111                 var viewTransform = currentPickInfo.viewTransform;
112                 var projTransform = currentPickInfo.viewTransform;
113 
114                 var origViewMatrix = viewTransform.matrix;
115                 var origProjMatrix = projTransform.matrix;
116 
117                 /* Get camera space coordinates                
118                  */
119                 var xcoord = -( ( ( 2 * x ) / canvas.canvas.width ) - 1 ) / SceneJS_math_getCellMat4(viewTransform.matrix, 1, 1);
120                 var ycoord = ( ( ( 2 * y ) / canvas.canvas.height ) - 1 ) / SceneJS_math_getCellMat4(projTransform.matrix, 2, 2);
121                 var zcoord = 1;
122 
123                 if (projTransform.type == "perspective") {
124                     var coord = SceneJS_math_transformPoint3(
125                             SceneJS_math_inverseMat4(viewTransform.matrix),
126                             [xcoord,ycoord,zcoord,0]);
127 
128                     var cameraPos = viewTransform.lookAt.eye;
129 
130                     var zvec = coord.toUnitVector();
131                     var xvec = (new GLGE.Vec([0,0,1])).cross(zvec).toUnitVector();
132                     var yvec = zvec.cross(xvec).toUnitVector();
133                     this.camera.matrix = new GLGE.Mat([xvec.e(1), yvec.e(1), zvec.e(1), cameraPos.x,
134                         xvec.e(2), yvec.e(2), zvec.e(2), cameraPos.y,
135                         xvec.e(3), yvec.e(3), zvec.e(3), cameraPos.z,
136                         0, 0, 0, 1]).inverse();
137                 }
138                 if (this.camera.type == GLGE.C_ORTHO) {
139                     this.camera.matrix = this.camera.matrix.inv().x(GLGE.translateMatrix(-xcoord, -ycoord, 0)).inv();
140                 }
141                 this.camera.pMatrix = GLGE.makeOrtho(-0.0001, 0.0001, -0.0001, 0.0001, this.camera.near, this.camera.far);
142                 //render for picking
143                 var gl = this.renderer.gl;
144                 gl.bindFramebuffer(gl.FRAMEBUFFER, this.framePickBuffer);
145                 gl.viewport(0, 0, 1, 1);
146                 gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
147                 this.renderer.gl.disable(this.renderer.gl.BLEND);
148 
149                 for (var i = 0; i < this.objects.length; i++) {
150                     this.objects[i].GLRender(this.renderer.gl, GLGE.RENDER_PICK);
151                 }
152                 var data = gl.readPixels(0, 0, 1, 1, gl.RGBA, gl.UNSIGNED_BYTE);
