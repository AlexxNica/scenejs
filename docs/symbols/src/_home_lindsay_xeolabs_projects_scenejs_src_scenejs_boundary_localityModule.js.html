1 /**
  2  * Backend that maintains a model-space sphere centered about the current eye position, computed from the
  3  * current view transform matrix.
  4  *
  5  * Services queries on it from scene nodes (ie. intersections etc.).
  6  *
  7  * Tracks the matrix through incoming VIEW_TRANSFORM_UPDATED events.
  8  *
  9  * Lazy-computes the sphere on demand, caching it until the matrix is updated.
 10  *
 11  * Provides an interface through which scene nodes can test axis-aligned bounding boxes for intersection
 12  * with the sphere.
 13  *
 14  * @private
 15  */
 16 var SceneJS_localityModule = new (function() {
 17 
 18     var eye;
 19     var radii;
 20     var radii2;
 21 
 22     SceneJS_eventModule.onEvent(
 23             SceneJS_eventModule.SCENE_ACTIVATED,
 24             function() {
 25                 eye = { x: 0, y: 0, z: 0 };
 26                 radii = {
 27                     inner : 2000,
 28                     outer : 2500
 29                 };
 30                 radii2 = {
 31                     inner : radii.inner * radii.inner,
 32                     outer : radii.outer * radii.outer
 33                 };
 34             });
 35 
 36     SceneJS_eventModule.onEvent(
 37             SceneJS_eventModule.VIEW_TRANSFORM_UPDATED,
 38             function(transform) {
 39                 if (transform.lookAt) {
 40                     var e = transform.lookAt.eye;
 41                     eye = [e.x, e.y, e.z];
 42                 } else {
 43                     eye = [0,0,0];
 44                 }
 45             });
 46 
 47     /**
 48      * @private
 49      */
 50     function intersects(radius2, box) { // Simple Arvo method - TODO: Larsson-Arkenine-Moller-Lengyel method
 51         var dmin = 0;
 52         var e;
 53         for (var i = 0; i < 3; i++) {
 54             if (eye[i] < box.min[i]) {
 55                 e = eye[i] - box.min[i];
 56                 dmin += (e * e);
 57             } else {
 58                 if (eye[i] > box.max[i]) {
 59                     e = eye[i] - box.max[i];
 60                     dmin += (e * e);
 61                 }
 62             }
 63         }
 64         return (dmin <= radius2);
 65     }
 66 
 67     /** Sets radii of inner and outer locality spheres
 68      * @private
 69      */
 70     this.setRadii = function(r) {
 71         radii = {
 72             outer : r.inner,
 73             inner : r.outer
 74         };
 75         radii2 = {
 76             inner : r.inner * r.inner,
 77             outer : r.outer * r.outer
 78         };
 79     };
 80 
 81     /** Returns current inner and ouer sphere radii
 82      * @private
 83      */
 84     this.getRadii = function() {
 85         return radii;
 86     };
 87 
 88     /** Tests the given axis-aligned bounding box for intersection with the outer locality sphere
 89      *
 90      * @param box
 91      * @private
 92      */
 93     this.testAxisBoxIntersectOuterRadius = function(box) {
 94         return intersects(radii2.outer, box);
 95     };
 96 
 97     /** Tests the given axis-aligned bounding box for intersection with the inner locality sphere
 98      *
 99      * @param box
100      * @private
101      */
102     this.testAxisBoxIntersectInnerRadius = function(box) {
103         return intersects(radii2.inner, box);
104     };
105 })();
