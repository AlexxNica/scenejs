  4 /**
  5  * @class SceneJS.objects.Sphere
  6  * @extends SceneJS.Geometry
  7  * <p>A scene node that defines sphere geometry, complete with normals for shading and one layer of UV coordinates for
  8  * texture-mapping. A Sphere may be configured with an optional radius, which will be 1 by default.</p>
  9  * <p><b>Example Usage</b></p><p>Definition of sphere with a radius of 6 units:</b></p><pre><code>
 10  * var c = new SceneJS.objects.Sphere({
 11  *          slices: 30,     // Optional number of longitudinal slices (30 is default)
 12  *          rings: 30,      // Optional number of latitudinal slices (30 is default)
 13  *          radius : 6      // Optional radius (1 is default)
 14  *     })
 15  * </pre></code>
 16  * @constructor
 17  * Create a new SceneJS.objects.Sphere
 18  * @param {Object} config  Config object or function, followed by zero or more child nodes
 19  */
 20 SceneJS.objects.Sphere = function() {
 21     SceneJS.Geometry.apply(this, arguments);
 22     this._nodeType = "sphere";
 23 };
 24 
 25 SceneJS._utils.inherit(SceneJS.objects.Sphere, SceneJS.Geometry);
 26 
 27 // @private
 28 SceneJS.objects.Sphere.prototype._init = function(params) {
 29     var slices = params.slices || 30;
 30     var rings = params.rings || 30;
 31 
 32     /* Type ID ensures that we reuse any sphere that has already been created with
 33      * these parameters instead of wasting memory
 34      */
 35     this._type = "sphere_" + rings + "_" + slices;
 36 
 37     /* Callback that does the creation in case we can't find matching sphere to reuse     
 38      */
 39     this._create = function() {
 40         var radius = 1;
 41         var positions = [];
 42         var normals = [];
 43         var uv = [];
 44         for (var sliceNum = 0; sliceNum <= slices; sliceNum++) {
 45             var theta = sliceNum * Math.PI / slices;
 46             var sinTheta = Math.sin(theta);
 47             var cosTheta = Math.cos(theta);
 48 
 49             for (var ringNum = 0; ringNum <= rings; ringNum++) {
 50                 var phi = ringNum * 2 * Math.PI / rings;
 51                 var sinPhi = Math.sin(phi);
 52                 var cosPhi = Math.cos(phi);
 53 
 54                 var x = cosPhi * sinTheta;
 55                 var y = cosTheta;
 56                 var z = sinPhi * sinTheta;
 57                 var u = 1 - (ringNum / rings);
 58                 var v = sliceNum / slices;
 59 
 60                 normals.push(-x);
 61                 normals.push(-y);
 62                 normals.push(-z);
 63                 uv.push(u);
 64                 uv.push(v);
 65                 positions.push(radius * x);
 66                 positions.push(radius * y);
 67                 positions.push(radius * z);
 68             }
 69         }
 70 
 71         var indices = [];
 72         for (var sliceNum = 0; sliceNum < slices; sliceNum++) {
 73             for (var ringNum = 0; ringNum < rings; ringNum++) {
 74                 var first = (sliceNum * (rings + 1)) + ringNum;
 75                 var second = first + rings + 1;
 76                 indices.push(first);
 77                 indices.push(second);
 78                 indices.push(first + 1);
 79 
 80                 indices.push(second);
 81                 indices.push(second + 1);
 82                 indices.push(first + 1);
 83             }
 84         }
 85 
 86         return {
 87             primitive : "triangles",
 88             positions : positions,
 89             normals: normals,
 90             uv : uv,
 91             indices : indices
 92         };
 93     };
 94 };
