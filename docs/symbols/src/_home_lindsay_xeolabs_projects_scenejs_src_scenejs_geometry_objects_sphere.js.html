1 SceneJS._utils.ns("SceneJS.objects");
  2 
  3 /**
  4  * Provides a sphere geometry node by wrapping a call to the core SceneJS.geometry node.
  5  *
  6  * Example of use:
  7  *
  8  * SceneJS.objects.sphere({ rings: 30, slices: 30 })
  9  */
 10 SceneJS.objects.sphere = function() {
 11 
 12     var cfg = SceneJS._utils.getNodeConfig(arguments || [
 13         {}
 14     ]);
 15 
 16     /* Dynamic config OK, but only applies first time - good for assets
 17      */
 18     var params = cfg.getParams();
 19 
 20     var slices = params.slices || 30;
 21     var rings = params.rings || 30;
 22 
 23     /* A geometry node is normally configured with arrays of positions, normals, indices etc., but can instead be
 24      * configured with a "create" callback, as demonstrated here, that returns an object containing those arrays.
 25      *
 26      * Every geometry must get a type that globally identifies it within SceneJS. In this case, the
 27      * type is generated from the number of rings and slices.
 28      *
 29      * Since SceneJS caches geometry by type, it will attempt to reuse the geometry if it can find it in the cache,
 30      * otherwise it will use the callback to create it.
 31      */
 32     return SceneJS.geometry({
 33 
 34         /* Unique global ID for this geometry
 35          */
 36         type: "sphere_" + rings + "_" + slices,
 37 
 38         /* Callback to create sphere geometry
 39          */
 40         create: function() {
 41             var radius = 1;
 42             var positions = [];
 43             var normals = [];
 44             var uv = [];
 45             for (var sliceNum = 0; sliceNum <= slices; sliceNum++) {
 46                 var theta = sliceNum * Math.PI / slices;
 47                 var sinTheta = Math.sin(theta);
 48                 var cosTheta = Math.cos(theta);
 49 
 50                 for (var ringNum = 0; ringNum <= rings; ringNum++) {
 51                     var phi = ringNum * 2 * Math.PI / rings;
 52                     var sinPhi = Math.sin(phi);
 53                     var cosPhi = Math.cos(phi);
 54 
 55                     var x = cosPhi * sinTheta;
 56                     var y = cosTheta;
 57                     var z = sinPhi * sinTheta;
 58                     var u = 1 - (ringNum / rings);
 59                     var v = sliceNum / slices;
 60 
 61                     normals.push(-x);
 62                     normals.push(-y);
 63                     normals.push(-z);
 64                     uv.push(u);
 65                     uv.push(v);
 66                     positions.push(radius * x);
 67                     positions.push(radius * y);
 68                     positions.push(radius * z);
 69                 }
 70             }
 71 
 72             var indices = [];
 73             for (var sliceNum = 0; sliceNum < slices; sliceNum++) {
 74                 for (var ringNum = 0; ringNum < rings; ringNum++) {
 75                     var first = (sliceNum * (rings + 1)) + ringNum;
 76                     var second = first + rings + 1;
 77                     indices.push(first);
 78                     indices.push(second);
 79                     indices.push(first + 1);
 80 
 81                     indices.push(second);
 82                     indices.push(second + 1);
 83                     indices.push(first + 1);
 84                 }
 85             }
 86 
 87             return {
 88                 primitive : "triangles",
 89                 positions : positions,
 90                 normals: normals,
 91                 uv : uv,
 92                 indices : indices
 93             };
 94         }
 95     });
 96 };
