  4 SceneJS._backends.installBackend(
  5 
  6         "scene",
  7 
  8         function(ctx) {
  9 
 10             var initialised = false; // True as soon as first scene registered
 11             var scenes = {};
 12             var nScenes = 0;
 13             var activeSceneId;
 14 
 15             var projMat;
 16             var viewMat;
 17             var picking;
 18 
 19             ctx.events.onEvent(
 20                     SceneJS._eventTypes.RESET,
 21                     function() {
 22                         scenes = {};
 23                         nScenes = 0;
 24                         activeSceneId = null;
 25                     });
 26 
 27             function updatePick() {
 28 
 29             }
 30 
 31             ctx.events.onEvent(
 32                     SceneJS._eventTypes.PROJECTION_TRANSFORM_UPDATED,
 33                     function(params) {
 34                         projMat = params.matrix;
 35                     });
 36 
 37             ctx.events.onEvent(
 38                     SceneJS._eventTypes.VIEW_TRANSFORM_UPDATED,
 39                     function(params) {
 40                         viewMat = params.matrix;
 41                     });
 42 
 43             /** Locates canvas in DOM, finds WebGL context on it,
 44              *  sets some default state on the context, then returns
 45              *  canvas, canvas ID and context wrapped up in an object.
 46              *
 47              * If canvasId is null, will fall back on SceneJS_webgl_DEFAULT_CANVAS_ID
 48              */
 49             var findCanvas = function(canvasId) {
 50                 var canvas;
 51                 if (!canvasId) {
 52                     ctx.logging.info("SceneJS.scene config 'canvasId' omitted - looking for default canvas with ID '"
 53                             + SceneJS_webgl_DEFAULT_CANVAS_ID + "'");
 54                     canvasId = SceneJS_webgl_DEFAULT_CANVAS_ID;
 55                     canvas = document.getElementById(canvasId);
 56                     if (!canvas) {
 57                         ctx.error.fatalError(new SceneJS.exceptions.CanvasNotFoundException
 58                                 ("SceneJS.scene config 'canvasId' omitted and could not find default canvas with ID '"
 59                                         + SceneJS_webgl_DEFAULT_CANVAS_ID + "'"));
 60                     }
 61                 } else {
 62                     canvas = document.getElementById(canvasId);
 63                     if (!canvas) {
 64                         ctx.logging.info("SceneJS.scene config 'canvasId' unresolved - looking for default canvas with " +
 65                                          "ID '" + SceneJS_webgl_DEFAULT_CANVAS_ID + "'");
 66                         canvasId = SceneJS_webgl_DEFAULT_CANVAS_ID;
 67                         canvas = document.getElementById(canvasId);
 68                         if (!canvas) {
 69                             ctx.error.fatalError(new SceneJS.exceptions.CanvasNotFoundException
 70                                     ("SceneJS.scene config 'canvasId' does not match any elements in the page and no " +
 71                                      "default canvas found with ID '" + SceneJS_webgl_DEFAULT_CANVAS_ID + "'"));
 72                         }
 73                     }
 74                 }
 75                 var context;
 76                 var contextNames = SceneJS_webgl_contextNames;
 77                 for (var i = 0; (!context) && i < contextNames.length; i++) {
 78                     try {
 79 
 80                         context = canvas.getContext(contextNames[i]);
 81 
 82 
 83 //                                                                        alert("WebGL Trace enabled");
 84 //                                                                        context = WebGLDebugUtils.makeDebugContext(canvas.getContext(contextNames[i]));
 85 //                                                                        context.setTracing(true);
 86                     } catch (e) {
 87 
 88                     }
 89                 }
 90                 if (!context) {
 91                     ctx.error.fatalError(new SceneJS.exceptions.WebGLNotSupportedException
 92                             ('Canvas document element with ID \''
 93                                     + canvasId
 94                                     + '\' failed to provide a supported WebGL context'));
 95                 }
 96                 context.clearColor(0.0, 0.0, 0.0, 1.0);
 97                 context.clearDepth(1.0);
 98                 context.enable(context.DEPTH_TEST);
 99                 context.disable(context.CULL_FACE);
100                 context.disable(context.TEXTURE_2D);
101                 context.depthRange(0, 1);
102                 context.disable(context.SCISSOR_TEST);
103                 return {
104                     canvas: canvas,
105                     context: context,
106                     canvasId : canvasId
107                 };
108             };
109 
110            
111 
112             return { // Node-facing API
113 
114                 /** Registers a scene, finds it's canvas, and returns the ID under which the scene is registered
115                  */
116                 createScene : function(scene, params) {
117                     if (!initialised) {
118                         ctx.logging.info("SceneJS V" + SceneJS.version + " initialised");
119                         ctx.events.fireEvent(SceneJS._eventTypes.INIT);
120                     }
121                     var canvas = findCanvas(params.canvasId); // canvasId can be null
122                     var sceneId = SceneJS._utils.createKeyForMap(scenes, "scene");
123                     scenes[sceneId] = {
124                         sceneId: sceneId,
125                         scene:scene,
126                         canvas: canvas
127                     };
128                     nScenes++;
129                     ctx.events.fireEvent(SceneJS._eventTypes.SCENE_CREATED, {sceneId : sceneId });
130                     ctx.logging.info("Scene defined: " + sceneId);
131                     return sceneId;
132                 },
133 
134                 /** Deregisters scene
135                  */
136                 destroyScene :function(sceneId) {
137                     scenes[sceneId] = null;
138                     nScenes--;
139                     ctx.events.fireEvent(SceneJS._eventTypes.SCENE_DESTROYED, {sceneId : sceneId });
140                     if (activeSceneId == sceneId) {
141                         activeSceneId = null;
142                     }
143                     ctx.logging.info("Scene destroyed: " + sceneId);
144                     if (nScenes == 0) {
145                         ctx.logging.info("SceneJS reset");
146                         ctx.events.fireEvent(SceneJS._eventTypes.RESET);
147 
148                     }
149                 },
150 
151                 /** Specifies which registered scene is the currently active one
152                  */
153                 activateScene : function(sceneId) {
154                     var scene = scenes[sceneId];
155                     if (!scene) {
156                         ctx.error.fatalError("Scene not defined: '" + sceneId + "'");
157                     }
158                     activeSceneId = sceneId;
159                     ctx.events.fireEvent(SceneJS._eventTypes.SCENE_ACTIVATED, { sceneId: sceneId });
160                     ctx.events.fireEvent(SceneJS._eventTypes.CANVAS_ACTIVATED, scene.canvas);
161                 },
162 
163                 /** Returns the canvas element the given scene is bound to
164                  */
165                 getSceneCanvas : function(sceneId) {
166                     var scene = scenes[sceneId];
167                     if (!scene) {
168                         ctx.error.fatalError("Scene not defined: '" + sceneId + "'");
169                     }
170                     return scene.canvas.canvas;
171                 },
172                 //
173                 //                activatePick : function(sceneId) {
174                 //
175                 //                },
176 
177                 /** Returns all registered scenes
178                  */
179                 getAllScenes:function() {
180                     var list = [];
181                     for (var id in scenes) {
182                         var scene = scenes[id];
183                         if (scene) {
184                             list.push(scene.scene);
185                         }
186                     }
187                     return list;
188                 },
189 
190                 /** Finds a registered scene
191                  */
192                 getScene : function(sceneId) {
193                     return scenes[sceneId].scene;
194                 },
195 
196                 /** Deactivates the currently active scene and reaps destroyed and timed out processes
197                  */
198                 deactivateScene : function() {
199                     if (!activeSceneId) {
200                         ctx.error.fatalError("Internal error: no scene active");
201                     }
202                     var sceneId = activeSceneId;
203                     activeSceneId = null;
204                     var scene = scenes[sceneId];
205                     if (!scene) {
206                         ctx.error.fatalError("Scene not defined: '" + sceneId + "'");
207                     }
208                     ctx.events.fireEvent(SceneJS._eventTypes.CANVAS_DEACTIVATED, scene.canvas);
209                     ctx.events.fireEvent(SceneJS._eventTypes.SCENE_DEACTIVATED, {sceneId : sceneId });
210                     //ctx.logging.info("Scene deactivated: " + sceneId);
211 
212                 }
213             };
214         });
