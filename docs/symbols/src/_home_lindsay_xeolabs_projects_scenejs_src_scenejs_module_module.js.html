1 new (function() {
  2 
  3     const TICK_INTERVAL = 50;
  4     const TIMEOUT = 60000; // 60 seconds
  5 
  6     var moduleQueue = [];
  7     var moduleLoading = null;
  8     var modules = {};
  9     var moduleLoadTimer = 0;
 10 
 11     SceneJS.requireModule = function(url) {
 12         moduleQueue.unshift({ url : url });
 13     };
 14 
 15     SceneJS.installModule = function(name, module) {
 16         try {
 17            // alert("installing " + moduleLoading.url);
 18             module.init({ baseURL : SceneJS._getBaseURL(moduleLoading.url) });
 19             modules[name] = module;
 20         } catch (e) {
 21             throw SceneJS._errorModule.fatalError(
 22                     new SceneJS.ModuleInstallFailureException(
 23                             "Module install failed - " + moduleLoading.url + ": " + e));
 24         } finally {
 25             moduleLoading = null;
 26         }
 27     };
 28 
 29     SceneJS._moduleLoadTicker = function() {
 30         if (moduleLoading) {
 31             if (moduleLoadTimer > TIMEOUT) {
 32                 var url = moduleLoading.url;
 33                 moduleLoading = null;
 34                 moduleQueue = [];
 35                 throw SceneJS._errorModule.fatalError(
 36                         new SceneJS.ModuleLoadTimeoutException(
 37                                 "Module load timed out - SceneJS.requireModule(" + url + ") - check console for more info"));
 38             }
 39             moduleLoadTimer += TICK_INTERVAL;
 40             return;
 41         }
 42         if (moduleQueue.length == 0) {
 43             return;
 44         }
 45 
 46         /* Load next module
 47          */
 48         moduleLoading = moduleQueue.pop();
 49         moduleLoadTimer = 0;
 50 
 51         var headID = document.getElementsByTagName("head")[0];
 52         var newScript = document.createElement('script');
 53         newScript.type = 'text/javascript';
 54         newScript.src = moduleLoading.url;
 55                 // alert("loading " + moduleLoading.url);
 56         headID.appendChild(newScript);
 57     };
 58     window.setInterval("SceneJS._moduleLoadTicker()", TICK_INTERVAL);
 59 
 60 
 61     SceneJS.UseModule = function() {
 62         SceneJS.Node.apply(this, arguments);
 63         this._startTime = null;
 64         this._timer = null;
 65         this._nodeType = "usemodule";
 66         this._moduleName = null;
 67         this._moduleNode = null;
 68         this._moduleParams = null;
 69         if (this._fixedParams) {
 70             this._init(this._getParams());
 71         }
 72     };
 73 
 74     SceneJS._inherit(SceneJS.UseModule, SceneJS.Node);
 75 
 76     // @private
 77     SceneJS.UseModule.prototype._init = function(params) {
 78         if (params.name) {
 79             this._moduleName = params.name;
 80             this._moduleParams = params.params || {};
 81         }
 82     };
 83 
 84     // @private
 85     SceneJS.UseModule.prototype._render = function(traversalContext, data) {
 86         if (!this._fixedParams) {
 87             this._init(this._getParams(data));
 88         }
 89         if (!this._moduleNode) {
 90             var module = modules[this._moduleName];
 91             if (module) {
 92                 this._moduleNode = module.getNode(this._moduleParams);
 93             } else {
 94                 if (!this._startTime) {
 95                     this._startTime = (new Date()).getTime();
 96                 } else if (((new Date()).getTime() - this._startTime) > TIMEOUT) {
 97                     throw SceneJS._errorModule.fatalError(
 98                             new SceneJS.ModuleNotFoundException(
 99                                     "SceneJS.UseModule failed to find module '"
100                                             + this._moduleName + "' after waiting " + (TIMEOUT / 1000) + " seconds - check console for more info"));
101                 }
102             }
103         }
104         if (this._moduleNode) {
105             this._moduleNode._render(traversalContext, data);
106         }
107         //this._renderNodes(traversalContext, data);
108         ;
109     };
110 
111     SceneJS.useModule = function() {
112         var n = new SceneJS.UseModule();
113         SceneJS.UseModule.prototype.constructor.apply(n, arguments);
114         return n;
115     };
