1 /**
  2  * Backend that manages material texture layers.
  3  *
  4  * Manages asynchronous load of texture images.
  5  *
  6  * Caches textures with a least-recently-used eviction policy.
  7  *
  8  * Holds currently-applied textures as "layers". Each layer specifies a texture and a set of parameters for
  9  * how the texture is to be applied, ie. to modulate ambient, diffuse, specular material colors, geometry normals etc.
 10  *
 11  * Holds the layers on a stack and provides the SceneJS.texture node with methods to push and pop them.
 12  *
 13  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
 14  * TEXTURES_EXPORTED to pass the entire layer stack to the shading backend.
 15  *
 16  * Avoids redundant export of the layers with a dirty flag; they are only exported when that is set, which occurs
 17  * when the stack is pushed or popped by the texture node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and
 18  * SHADER_DEACTIVATED events.
 19  *
 20  * Whenever a texture node pushes or pops the stack, this backend publishes it with a TEXTURES_UPDATED to allow other
 21  * dependent backends to synchronise their resources.
 22  */
 23 SceneJS._backends.installBackend(
 24 
 25         "texture",
 26 
 27         function(ctx) {
 28 
 29             var time = (new Date()).getTime();      // Current system time for LRU caching
 30             var canvas;
 31             var textures = {};
 32             var layerStack = [];
 33             var dirty;
 34 
 35             ctx.events.onEvent(
 36                     SceneJS._eventTypes.TIME_UPDATED,
 37                     function(t) {
 38                         time = t;
 39                     });
 40 
 41             ctx.events.onEvent(
 42                     SceneJS._eventTypes.SCENE_ACTIVATED,
 43                     function() {
 44                         layerStack = [];
 45                         dirty = true;
 46                     });
 47 
 48             ctx.events.onEvent(
 49                     SceneJS._eventTypes.CANVAS_ACTIVATED,
 50                     function(c) {
 51                         canvas = c;
 52                         dirty = true;
 53                     });
 54 
 55             ctx.events.onEvent(
 56                     SceneJS._eventTypes.CANVAS_DEACTIVATED,
 57                     function() {
 58                         canvas = null;
 59                         dirty = true;
 60                     });
 61 
 62             ctx.events.onEvent(
 63                     SceneJS._eventTypes.SHADER_ACTIVATED,
 64                     function() {
 65                         dirty = true;
 66                     });
 67 
 68             ctx.events.onEvent(
 69                     SceneJS._eventTypes.SHADER_RENDERING,
 70                     function() {
 71                         if (dirty) {
 72                             ctx.events.fireEvent(
 73                                     SceneJS._eventTypes.TEXTURES_EXPORTED,
 74                                     layerStack
 75                                     );
 76                             dirty = false;
 77                         }
 78                     });
 79 
 80             ctx.events.onEvent(
 81                     SceneJS._eventTypes.SHADER_DEACTIVATED,
 82                     function() {
 83                         dirty = true;
 84                     });
 85 
 86             /** Removes texture from shader (if canvas exists in DOM) and deregisters it from backend
 87              */
 88             function deleteTexture(texture) {
 89                 textures[texture.textureId] = undefined;
 90                 if (document.getElementById(texture.canvas.canvasId)) {
 91                     texture.destroy();
 92                 }
 93             }
 94 
 95             /**
 96              * Deletes all textures from their GL contexts - does not attempt
 97              * to delete them when their canvases no longer exist in the DOM.
 98              */
 99             function deleteTextures() {
100                 for (var textureId in textures) {
101                     var texture = textures[textureId];
102                     deleteTexture(texture);
103                 }
104                 textures = {};
105                 layerStack = [];
106                 dirty = true;
107             }
108 
109             ctx.events.onEvent(
110                     SceneJS._eventTypes.RESET, // Framework reset - delete textures
111                     function() {
112                         deleteTextures();
113                     });
114 
115             /**
116              * Registers this backend module with the memory management module as willing
117              * to attempt to destroy a texture when asked, in order to free up memory. Eviction
118              * is done on a least-recently-used basis, where a texture may be evicted if the
119              * time that it was last used is the earliest among all textures, and after the current
120              * system time. Since system time is updated just before scene traversal, this ensures that
121              * textures previously or currently active during this traversal are not suddenly evicted.
122              */
123             ctx.memory.registerEvictor(
124                     function() {
125                         var earliest = time; // Doesn't evict textures that are current in layers
126                         var evictee;
127                         for (var id in textures) {
128                             if (id) {
129                                 var texture = textures[id];
130                                 if (texture.lastUsed < earliest) {
131                                     evictee = texture;
132                                     earliest = texture.lastUsed;
133                                 }
134                             }
135                         }
136                         if (evictee) { // Delete LRU texture
137                             ctx.logging.info("Evicting texture: " + id);
138                             deleteTexture(evictee);
139                             return true;
140                         }
141                         return false;   // Couldnt find suitable evictee
142                     });
143 
144             /**
145              * Translates a SceneJS param value to a WebGL enum value,
146              * or to default if undefined. Throws exception when defined
147              * but not mapped to an enum.
148              */
149             function getGLOption(name, context, cfg, defaultVal) {
150                 var value = cfg[name];
151                 if (value == undefined) {
152                     return defaultVal;
153                 }
154                 var glName = SceneJS_webgl_enumMap[value];
155                 if (glName == undefined) {
156                     ctx.error.fatalError(new SceneJS.exceptions.InvalidNodeConfigException(
157                             "Unrecognised value for SceneJS.texture node property '" + name + "' value: '" + value + "'"));
158                 }
159                 var glValue = context[glName];
160                 //                if (!glValue) {
161                 //                    throw new SceneJS.exceptions.WebGLUnsupportedNodeConfigException(
162                 //                            "This browser's WebGL does not support value of SceneJS.texture node property '" + name + "' value: '" + value + "'");
163                 //                }
164                 return glValue;
165             }
166 
167             /** Returns default value for when given value is undefined
168              */
169             function getOption(value, defaultVal) {
170                 return (value == undefined) ? defaultVal : value;
171             }
172 
173             return { // Node-facing API
174 
175                 /** Verifies that texture still cached - it may have been evicted after lack of recent use,
176                  * in which case client texture node will have to recreate it.
177                  */
178                 textureExists : function(texture) {
179                     return textures[texture.textureId];
180                 },
181 
182                 /**
183                  * Starts load of texture image
184                  *
185                  * @param uri Image location
186                  * @param onSuccess Callback returns image on success
187                  * @param onError Callback fired on failure
188                  * @param onAbort Callback fired when load aborted, eg. user hits "stop" button in browser
189                  */
190                 loadImage : function(uri, onSuccess, onError, onAbort) {
191                     var image = new Image();
192                     image.onload = function() {
193                         onSuccess(image);
194                     };
195                     image.onerror = function() {
196                         onError();
197                     };
198                     image.onabort = function() {
199                         onAbort();
200                     };
201                     image.src = uri;  // Starts image load
202                     return image;
203                 },
204 
205                
206 
207                 /**
208                  * Creates and returns a new texture, or re-uses existing one if possible
209                  */
210                 createTexture : function(image, cfg) {
211                     if (!canvas) {
212                         ctx.error.fatalError(new SceneJS.exceptions.NoCanvasActiveException("No canvas active"));
213                     }
214                     var context = canvas.context;
215                     var textureId = SceneJS._utils.createKeyForMap(textures, "tex");
216 
217                     ctx.memory.allocate(
218                             "texture '" + textureId + "'",
219                             function() {
220                                 textures[textureId] = new SceneJS_webgl_Texture2D(context, {
221                                     textureId : textureId,
222                                     canvas: canvas,
223                                     image : image,
224                                     texels :cfg.texels,
225                                     minFilter : getGLOption("minFilter", context, cfg, context.LINEAR),
226                                     magFilter :  getGLOption("magFilter", context, cfg, context.LINEAR),
227                                     wrapS : getGLOption("wrapS", context, cfg, context.CLAMP_TO_EDGE),
228                                     wrapT :   getGLOption("wrapT", context, cfg, context.CLAMP_TO_EDGE),
229                                     isDepth :  getOption(cfg.isDepth, false),
230                                     depthMode : getGLOption("depthMode", context, cfg, context.LUMINANCE),
231                                     depthCompareMode : getGLOption("depthCompareMode", context, cfg, context.COMPARE_R_TO_TEXTURE),
232                                     depthCompareFunc : getGLOption("depthCompareFunc", context, cfg, context.LEQUAL),
233                                     flipY : getOption(cfg.flipY, true),
234                                     width: getOption(cfg.width, 1),
235                                     height: getOption(cfg.height, 1),
236                                     internalFormat : getGLOption("internalFormat", context, cfg, context.LEQUAL),
237                                     sourceFormat : getGLOption("sourceType", context, cfg, context.ALPHA),
238                                     sourceType : getGLOption("sourceType", context, cfg, context.UNSIGNED_BYTE),
239                                     logging: ctx.logging
240                                 });
241                             });
242                        //   ctx.logging.info("texture created: '" + textureId + "'");
243                     return textures[textureId];
244                 },
245 
246                 pushLayer : function(texture, params) {
247                     if (!textures[texture.textureId]) {
248                         ctx.error.fatalError("No such texture loaded \"" + texture.textureId + "\"");
249                     }
250                     texture.lastUsed = time;
251 
252                     if (params.matrix && !params.matrixAsArray) {
253                         params.matrixAsArray = new WebGLFloatArray(params.matrix);
254                     }
255                     layerStack.push({
256                         texture: texture,
257                         params: params
258                     });
259                     dirty = true;
260                     ctx.events.fireEvent(SceneJS._eventTypes.TEXTURES_UPDATED, layerStack);
261                 },
262 
263                 popLayers : function(nLayers) {
264                     for (var i = 0; i < nLayers; i++) {
265                         layerStack.pop();
266                     }
267                     dirty = true;
268                     ctx.events.fireEvent(SceneJS._eventTypes.TEXTURES_UPDATED, layerStack);
269                 }
270             };
271         });
