1 /** Specifies logging for its sub-nodes
  2  */
  3 SceneJS.logging = function() {
  4     var cfg = SceneJS._utils.getNodeConfig(arguments);
  5     var backend = SceneJS._backends.getBackend('logging');
  6     var funcs;
  7 
  8     return SceneJS._utils.createNode(
  9             "logging",
 10             cfg.children,
 11 
 12             new (function() {
 13                 this._render = function(traversalContext, data) {
 14                     var prevFuncs = backend.getFuncs();
 15                     if (!funcs || !cfg.fixed) {
 16                         funcs = cfg.getParams(data);
 17                         var p = prevFuncs || {};
 18                         funcs.warn = funcs.warn || p.warn;
 19                         funcs.error = funcs.error || p.error;
 20                         funcs.debug = funcs.debug || p.debug;
 21                         funcs.info = funcs.info || p.info;
 22                     }
 23                     backend.setFuncs(funcs);
 24                     this._renderChildren(traversalContext, data);
 25                     backend.setFuncs(prevFuncs);
 26                 };
 27             })());
 28 };
