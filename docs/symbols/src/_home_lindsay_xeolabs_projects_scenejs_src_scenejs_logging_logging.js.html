1 /**
  2  * @class A scene node that allows you to intercept SceneJS logging at selected points within your scene graph.
  3  * <p><b>Example Usage</b></p><p>Routing all logging within a given subgraph to alert popups:</b></p><pre><code>
  4  * var node = new SceneJS.Node(
  5  *
  6  *    // ...some nodes ...
  7  *
  8  *    new SceneJS.Logging({
  9  *           error: function(msg) {
 10  *                alert("Error at global logger: " + msg);
 11  *           },
 12  *
 13  *           warn: function(msg) {
 14  *               alert("Warning at global logger: " + msg);
 15  *           },
 16  *
 17  *           debug: function(msg) {
 18  *               alert("Debug at global logger: " + msg);
 19  *           },
 20  *
 21  *           info: function(msg) {
 22  *               alert("Info at global logger: " + msg);
 23  *           }
 24  *       },
 25  *
 26  *       // ... sub-nodes we are in interesting
 27  *       //        intercepting logs from here ...
 28  *    )
 29  * );
 30  *
 31  * </pre></code>
 32  * @extends SceneJS.Node
 33  * @constructor
 34  * Create a new SceneJS.Logging
 35  * @param {Object} config  Config object or function, followed by zero or more child nodes
 36  */
 37 SceneJS.Logging = function() {
 38     SceneJS.Node.apply(this, arguments);
 39     this._nodeType = "logging";
 40     this._funcs = {};
 41     if (this._fixedParams) {
 42         this._init(this._getParams());
 43     }
 44 };
 45 
 46 SceneJS._inherit(SceneJS.Logging, SceneJS.Node);
 47 
 48 /**
 49  * Sets functions
 50  *
 51  * @param {object} funcs - object cntaining functions - see class comment for example
 52  * @returns {SceneJS.Logging} This logging node
 53  */
 54 SceneJS.Logging.prototype.setFuncs = function(funcs) {
 55     if (funcs.warn) {
 56         this._funcs.warn = funcs.warn;
 57     }
 58     if (funcs.error) {
 59         this._funcs.error = funcs.error;
 60     }
 61     if (funcs.debug) {
 62         this._funcs.debug = funcs.debug;
 63     }
 64     if (funcs.info) {
 65         this._funcs.info = funcs.info;
 66     }
 67     return this;
 68 };
 69 
 70 /**
 71  Returns logging functions
 72  @returns {object} The functions
 73  */
 74 SceneJS.Logging.prototype.getFuncs = function() {
 75     return {
 76         warn : this._funcs.warn,
 77         error : this._funcs.error,
 78         debug : this._funcs.debug,
 79         info : this._funcs.info
 80     };
 81 };
 82 
 83 // @private
 84 SceneJS.Logging.prototype._init = function(params) {
 85     this.setFuncs(params);
 86 };
 87 
 88 // @private
 89 SceneJS.Logging.prototype._render = function(traversalContext, data) {
 90     if (!this._fixedParams) {
 91         this._init(this._getParams(data));
 92     }
 93     var prevFuncs = SceneJS_loggingModule.getFuncs();
 94     SceneJS_loggingModule.setFuncs(this._funcs);
 95     this._renderNodes(traversalContext, data);
 96     SceneJS_loggingModule.setFuncs(prevFuncs);
 97 };
 98 
 99 /** Returns a new SceneJS.Logging instance
100  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Logging constructor
101  * @returns {SceneJS.Logging}
102  */
103 SceneJS.logging = function() {
104     var n = new SceneJS.Logging();
105     SceneJS.Logging.prototype.constructor.apply(n, arguments);
106     return n;
107 };
