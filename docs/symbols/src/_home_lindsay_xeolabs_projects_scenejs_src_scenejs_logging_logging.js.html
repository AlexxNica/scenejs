1 /** 
  2  * @class A scene node that routes messages logged by nodes in its subgraph through a given set of logging functions.
  3  * @extends SceneJS.node
  4  */
  5 SceneJS.logging = function() {
  6     var cfg = SceneJS.getNodeConfig(arguments);
  7 
  8     /* Augment the basic node type
  9      */
 10     return (function($) {
 11 
 12         var funcs;
 13 
 14         $._render = function(traversalContext, data) {
 15             var prevFuncs = SceneJS_loggingModule.getFuncs();
 16             if (!funcs || !cfg.fixed) {
 17                 funcs = cfg.getParams(data);
 18                 var p = prevFuncs || {};
 19                 funcs.warn = funcs.warn || p.warn;
 20                 funcs.error = funcs.error || p.error;
 21                 funcs.debug = funcs.debug || p.debug;
 22                 funcs.info = funcs.info || p.info;
 23             }
 24             SceneJS_loggingModule.setFuncs(funcs);
 25             $._renderNodes(traversalContext, data);
 26             SceneJS_loggingModule.setFuncs(prevFuncs);
 27         };
 28         return $;
 29     })(SceneJS.node.apply(this, arguments));
 30 };
