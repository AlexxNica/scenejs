1 /**
  2  *@class A scene node that defines inner and outer spheres of locality centered about the viewpoint.
  3  *<p>The subgraphs of contained {@link SceneJS.BoundingBox} nodes will only be rendered when their boundaries intersect
  4  *the inner radius.</p><p>The outer radius is used internally by SceneJS to support content staging strategies.</p>
  5  *<p>You can have as many of these as neccessary throughout your scene.</p>
  6  * <p>When you don't specify a Locality node, SceneJS has default inner and outer radii of 100000
  7  * and 200000, respectively.</p>
  8  *<p><b>Example:</b></p><p>Defining a locality</b></p><pre><code>
  9  *  var locality = new SceneJS.Locality({
 10  *      inner: 100000,  // Default node values, override these where needed
 11  *      outer: 200000
 12  *      },
 13  *
 14  *      // ... child nodes containing SceneJS.BoundingBox nodes ...
 15  *  )
 16  *</pre></code>
 17  * @extends SceneJS.Node
 18  * @since Version 0.7.3
 19  * @constructor
 20  * Create a new SceneJS.Locality
 21  * @param {Object} [cfg] Static configuration object
 22  * @param {double} [cfg.inner = 100000] Inner radius
 23  * @param {double} [cfg.outer = 200000] Outer radius
 24  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 25  * @param {...SceneJS.Node} [childNodes] Child nodes
 26  */
 27 SceneJS.Locality = function() {
 28     SceneJS.Node.apply(this, arguments);
 29     this._nodeType = "locality";
 30     this._radii = {
 31         inner : 100000,
 32         outer : 200000
 33     };
 34     if (this._fixedParams) {
 35         this._init(this._getParams());
 36     }
 37 };
 38 
 39 SceneJS._inherit(SceneJS.Locality, SceneJS.Node);
 40 
 41 /**
 42  Sets the inner radius
 43  @function setInner
 44  @param {double} inner
 45  @returns {SceneJS.Locality} this
 46  @since Version 0.7.4
 47  */
 48 SceneJS.Locality.prototype.setInner = function(inner) {
 49     this._radii.inner = inner;
 50     return this;
 51 };
 52 
 53 /**
 54  Returns the inner radius
 55  @function {double} getInner
 56  @returns {double} Inner radius
 57  @since Version 0.7.4
 58  */
 59 SceneJS.Locality.prototype.getInner = function() {
 60     return this._radii.inner;
 61 };
 62 
 63 /**
 64  Sets the outer radius
 65  @function setOuter
 66  @param {double} outer
 67  @returns {SceneJS.Locality} this
 68  @since Version 0.7.4
 69  */
 70 SceneJS.Locality.prototype.setOuter = function(outer) {
 71     this._radii.outer = outer;
 72     return this;
 73 };
 74 
 75 /**
 76  Returns the outer radius
 77  @function {double} getOuter
 78  @returns {double} Outer radius
 79  @since Version 0.7.4
 80  */
 81 SceneJS.Locality.prototype.getOuter = function() {
 82     return this._radii.outer;
 83 };
 84 
 85 // @private
 86 SceneJS.Locality.prototype._init = function(params) {
 87     if (params.inner) {
 88         this.setInner(params.inner);
 89     }
 90     if (params.outer) {
 91         this.setOuter(params.outer);
 92     }
 93 };
 94 
 95 // @private
 96 SceneJS.Locality.prototype._render = function(traversalContext, data) {
 97     if (!this._fixedParams) {
 98         this._init(this._getParams(data));
 99     }
100     var prevRadii = SceneJS_localityModule.getRadii();
101     SceneJS_localityModule.setRadii(this._radii);
102     this._renderNodes(traversalContext, data);
103     SceneJS_localityModule.setRadii(prevRadii);
104 };
105 
106 /** Returns a new SceneJS.Locality instance
107  * @param {Object} [cfg] Static configuration object
108  * @param {double} [cfg.inner = 100000] Inner radius
109  * @param {double} [cfg.outer = 200000] Outer radius
110  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
111  * @param {...SceneJS.Node} [childNodes] Child nodes
112  * @returns {SceneJS.Locality}
113  * @since Version 0.7.3
114  */
115 SceneJS.locality = function() {
116     var n = new SceneJS.Locality();
117     SceneJS.Locality.prototype.constructor.apply(n, arguments);
118     return n;
119 };
