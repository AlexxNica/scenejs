1 /**
  2  *@class A scene node that defines inner and outer spheres of locality centered about the viewpoint.
  3  *<p>The subgraphs of contained SceneJS.BoundingBox nodes will only be rendered when their boundaries intersect
  4  *the inner radius.</p><p>The outer radius is used internally by SceneJS to support content staging strategies.</p> 
  5  *<p>You can have as many of these as neccessary throughout your scene.</p>
  6  *<p><b>Example:</b></p><p>Defining a locality</b></p><pre><code>
  7  *  var locality = new SceneJS.Locality({
  8  *      inner: 1000,  // Default values, override these where needed
  9  *      outer: 2000
 10  *      },
 11  *
 12  *      // ... child nodes containing SceneJS.BoundingBox nodes ...
 13  *  )
 14  *</pre></code>
 15  *  @extends SceneJS.Node
 16  *@constructor
 17  *Create a new SceneJS.Locality
 18  * @param {Object} cfg  Config object or function, followed by zero or more child nodes
 19  */
 20 SceneJS.Locality = function() {
 21     SceneJS.Node.apply(this, arguments);
 22     this._nodeType = "locality";
 23     this._radii = {
 24         inner : 1000,
 25         outer : 2000
 26     };
 27     if (this._fixedParams) {
 28         this._init(this._getParams());
 29     }
 30 };
 31 
 32 SceneJS._inherit(SceneJS.Locality, SceneJS.Node);
 33 
 34 /**
 35  Sets the inner radius
 36  @function setInner
 37  @param {double} inner
 38  @returns {SceneJS.Locality} this
 39  */
 40 SceneJS.Locality.prototype.setInner = function(inner) {
 41     this._radii.inner = inner;
 42     return this;
 43 };
 44 
 45 /**
 46  Returns the inner radius
 47  @function {double} getInner
 48  @returns {double} Inner radius
 49  */
 50 SceneJS.Locality.prototype.getInner = function() {
 51     return this._radii.inner;
 52 };
 53 
 54 /**
 55  Sets the outer radius
 56  @function setOuter
 57  @param {int} outer
 58  @returns {SceneJS.Locality} this
 59  */
 60 SceneJS.Locality.prototype.setOuter = function(outer) {
 61     this._radii.outer = outer;
 62     return this;
 63 };
 64 
 65 /**
 66  Returns the outer radius
 67  @function {double} getOuter
 68  @returns {double} Outer radius
 69  */
 70 SceneJS.Locality.prototype.getOuter = function() {
 71     return this._radii.outer;
 72 };
 73 
 74 // @private
 75 SceneJS.Locality.prototype._init = function(params) {
 76     if (params.inner) {
 77         this.setInner(params.inner);
 78     }
 79     if (params.outer) {
 80         this.setOuter(params.outer);
 81     }
 82 };
 83 
 84 // @private
 85 SceneJS.Locality.prototype._render = function(traversalContext, data) {
 86     if (!this._fixedParams) {
 87         this._init(this._getParams(data));
 88     }
 89     var prevRadii = SceneJS_localityModule.getRadii();
 90     SceneJS_localityModule.setRadii(this._radii);
 91     $._renderNodes(traversalContext, data);
 92     SceneJS_localityModule.setRadii(prevRadii);
 93 };
 94 
 95 /** Returns a new SceneJS.Locality instance
 96  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Locality constructor
 97  * @returns {SceneJS.Locality}
 98  */
 99 SceneJS.locality = function() {
100     var n = new SceneJS.Locality();
101     SceneJS.Locality.prototype.constructor.apply(n, arguments);
102     return n;
103 };
