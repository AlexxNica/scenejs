1 /**
  2  * Backend for the axis-aligned boundary node, provides ability to instance the boundary extents into view space
  3  * and test for its intersection with the view volume.
  4  */
  5 
  6 SceneJS._backends.installBackend(
  7         new (function() {
  8 
  9             this.type = 'axis-boundary';
 10 
 11             var ctx;
 12 
 13             this.install = function(_ctx) {
 14                 ctx = _ctx;
 15 
 16             };
 17 
 18             /** Transforms coordinates by the current modelling matrix
 19              */
 20             this.modelTransform = function(coords) {
 21                 var tfunc = ctx.modelTransform.transformPoint3;
 22                 var coords2 = [];
 23                 for (var i = 0; i < coords.length; i++) {
 24                     coords2.push(tfunc(coords[i]));
 25                 }
 26                 return coords2;
 27             };
 28 
 29             this.viewAndProjectionTransform = function(coords) {
 30                 var tfuncView = ctx.viewTransform.transformPoint3;
 31                // var tfuncProj = ctx.projection.transformPoint3;
 32                 var coords2 = [];
 33                 for (var i = 0; i < coords.length; i++) {
 34                     coords2.push(tfuncView(coords[i]));
 35                 }
 36                 return coords2;
 37             };
 38 
 39             /** Returns true if the coordinate space defined by the current view matrices is
 40              * fixed, ie. not animated, at this level of scene traversal. If not, then it is not safe
 41              * for the client node to memoize its boundary once it is are transformed, because the transformation
 42              * is therefore likely to vary, causing the boundary to move around.
 43              */
 44             this.getSafeToCache = function() {
 45                 return ctx.modelTransform.isFixed();
 46             };
 47 
 48             this.getProjectedBoundary = function(coords) {
 49                 var tfuncView = ctx.viewTransform.transformPoint3;
 50                // var tfuncProj = ctx.projection.transformPoint3;
 51                 var pBound = {
 52                     xmin : 1000000000000,
 53                     ymin : 1000000000000,
 54                     xmax : -1000000000000,
 55                     ymax : -1000000000000
 56                 };
 57                 for (var i = 0; i < coords.length; i++) {
 58                     var p = tfuncView(coords[i]);
 59                     var x = p.x * p.w;
 60                     var y = p.y * p.w;
 61                     if (x < pBound.xmin) {
 62                         pBound.xmin = x;
 63                     }
 64                     if (y < pBound.ymin) {
 65                         pBound.ymin = y;
 66                     }
 67                     if (x > pBound.xmax) {
 68                         pBound.xmax = x;
 69                     }
 70                     if (y > pBound.ymax) {
 71                         pBound.ymax = y;
 72                     }
 73                 }
 74                 return pBound;
 75             };
 76 
 77             this.testViewVolumeCoordsIntersection = function(coords) {
 78                 return ctx.viewTransform.getTransform().frustum.pointsIntersection(coords);
 79             };
 80         })());
