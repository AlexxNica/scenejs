1 /**
  2  */
  3 (function() {
  4 
  5     /**
  6      * Defines fog within a scene. Fog is effectively a region on the Z-axis of the view coordinate system within which
  7      * the colour of elements will blend with the scene ambient colour in proportion to their depth. You can define the
  8      * points on the Z axis at which the fog region starts and ends, along with the proportion as a linear, exponential
  9      * or quadratic mode. Scene content falling in front of the start point will have no fog applied, while content
 10      * after the end point will be invisible, having blended completely into the ambient colour.
 11      *
 12      * @class SceneJS.fog
 13      * @extends SceneJS.node
 14      */
 15     SceneJS.fog = function() {
 16         var cfg = SceneJS._utils.getNodeConfig(arguments);
 17 
 18         return SceneJS._utils.createNode(
 19                 "fog",
 20                 cfg.children,
 21 
 22                 new (function() {
 23                     this._errorBackend = SceneJS._backends.getBackend('error');
 24                     this._fogBackend = SceneJS._backends.getBackend('fog');
 25 
 26                     this._mode = "linear";
 27                     this._color = { r: 0.5, g: 0.5, b: 0.5 };
 28                     this._density = 1.0;
 29                     this._start = 0;
 30                     this._end = 0;
 31 
 32                     /**
 33                      Sets the fogging mode
 34                      @function setMode
 35                      @param {string} mode - "disabled", "exp", "exp2" or "linear"
 36                      @returns {SceneJS.fog} This fog node
 37                      */
 38                     this.setMode = function(mode) {
 39                         if (mode != "disabled" && mode != "exp" && mode != "exp2" && mode != "linear") {
 40                             this._errorBackend.fatalError(new SceneJS.exceptions.InvalidNodeConfigException(
 41                                     "SceneJS.fog has a mode of unsupported type: '" + mode + " - should be 'none', 'exp', 'exp2' or 'linear'"));
 42                         }
 43                         this._mode = mode;
 44                         return this;
 45                     };
 46 
 47                     /**
 48                      Returns fogging mode
 49                      @function {string} getMode
 50                      @returns {string} The fog mode - "disabled", "exp", "exp2" or "linear"
 51                      */
 52                     this.getMode = function() {
 53                         return this._mode;
 54                     };
 55 
 56                     /**
 57                      Sets the fog color
 58                      @function setColor
 59                      @param {object} color - eg. bright red: {r: 1.0, g: 0, b: 0 }
 60                      @returns {SceneJS.fog} This fog node
 61                      */
 62                     this.setColor = function(color) {
 63                         this._color.r = color.r != undefined ? color.r : 0.5;
 64                         this._color.g = color.g != undefined ? color.g : 0.5;
 65                         this._color.b = color.b != undefined ? color.b : 0.5;
 66                         return this;
 67                     };
 68 
 69                     /**
 70                      Returns the fog color
 71                      @function getColor
 72                      @returns {object} Fog color - eg. bright red: {r: 1.0, g: 0, b: 0 }
 73                      */
 74                     this.getColor = function() {
 75                         return {
 76                             r: this._color.r,
 77                             g: this._color.g,
 78                             b: this._color.b
 79                         };
 80                     };
 81 
 82                     /**
 83                      Sets the fog density
 84                      @function setDensity
 85                      @param {int} density - density factor
 86                      @returns {SceneJS.fog} This fog node
 87                      */
 88                     this.setDensity = function(density) {
 89                         this._density = density || 1.0;
 90                         return this;
 91                     };
 92 
 93                     /**
 94                      Returns the fog density
 95                      @function {double} getDensity
 96                      @returns {double} Fog density factor
 97                      */
 98                     this.getDensity = function() {
 99                         return this._density;
100                     };
101 
102                     /**
103                      Sets the near point on the Z view-axis at which fog begins
104                      @function setStart
105                      @param {double} start - location on Z-axis
106                      @returns {SceneJS.fog} This fog node
107                      */
108                     this.setStart = function(start) {
109                         this._start = start || 0;
110                         return this;
111                     };
112 
113                     /**
114                      Returns the near point on the Z view-axis at which fog begins
115                      @function {double} getStart
116                      @returns {double} Position on Z view axis
117                      */
118                     this.getStart = function() {
119                         return this._start;
120                     };
121 
122                     /**
123                      Sets the farr point on the Z view-axis at which fog ends
124                      @function setEnd
125                      @param {double} end - location on Z-axis
126                      @returns {SceneJS.fog} This fog node
127                      */
128                     this.setEnd = function(end) {
129                         this._end = end || 0;
130                         return this;
131                     };
132 
133                     /**
134                      Returns the far point on the Z view-axis at which fog ends
135                      @function {double} getEnd
136                      @returns {double} Position on Z view axis
137                      */
138                     this.getEnd = function() {
139                         return this._end;
140                     };
141 
142                     this._init = function(params) {
143                         if (params.mode) {
144                             this.setMode(params.mode);
145                         }
146                         if (params.color) {
147                             this.setColor(params.color);
148                         }
149                         if (params.density) {
150                             this.setDensity(params.density);
151                         }
152                         if (params.start) {
153                             this.setStart(params.start);
154                         }
155                         if (params.end) {
156                             this.setEnd(params.end);
157                         }
158                     };
159 
160                     if (cfg.fixed) {
161                         this._init(cfg.getParams());
162                     }
163 
164                     this._render = function(traversalContext, data) {
165                         if (SceneJS._utils.traversalMode == SceneJS._utils.TRAVERSAL_MODE_PICKING) {
166                             SceneJS._utils.visitChildren(cfg, traversalContext, data);
167                         } else {
168                             if (!cfg.fixed) {
169                                 this._init(cfg.getParams(data));
170                             }
171                             var f = this._fogBackend.getFog();
172                             this._fogBackend.setFog({
173                                 mode: this._mode,
174                                 color: this._color,
175                                 density: this._density,
176                                 start: this._start,
177                                 end: this._end
178                             });
179                             this._renderChildren(traversalContext, data);
180                             this._fogBackend.setFog(f);
181                         }
182                     };
183                 })());
184     };
185 })();
