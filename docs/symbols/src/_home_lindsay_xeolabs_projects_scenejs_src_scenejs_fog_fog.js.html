1 /**
  2  * @class A scene node that defines fog for nodes in its sub graph.
  3 
  4  * <p>Fog is effectively a region on the Z-axis of the view coordinate system within which
  5  * the colour of elements will blend with the scene ambient colour in proportion to their depth. You can define the
  6  * points on the Z axis at which the fog region starts and ends, along with the proportion as a linear, exponential
  7  * or quadratic mode. Scene content falling in front of the start point will have no fog applied, while content
  8  * after the end point will be invisible, having blended completely into the ambient colour.</p>
  9  * 
 10  * <p><b>Example Usage</b></p><p>Definition of fog with parameters that happen to be the defaults -
 11  * starting at Z=1, extending until Z=1000, linear mode, gray colour. Objects beyond Z=1000 will be entirely merged
 12  * into the background.</b></p><pre><code>
 13  * var fog = new SceneJS.Fog({
 14  *         mode:"linear",
 15  *         color: { r: 0.5, g: 0.5, b: 0.5 },
 16  *         density: 1.0,
 17  *         start: 1,
 18  *         end: 1000
 19  *     },
 20  *
 21  *     // ... child nodes
 22  * )
 23  * </pre></code>
 24  * @extends SceneJS.Node
 25  * @since Version 0.7.4
 26  * @constructor
 27  * Creates a new SceneJS.Fog
 28  * @param {Object} [cfg] Static configuration object
 29  * @param {String} [cfg.mode = "linear"] The fog mode - "disabled", "exp", "exp2" or "linear"
 30  * @param {Object} [cfg.color = {r: 0.5, g: 0.5, b: 0.5 } The fog color
 31  * @param {double} [cfg.density = 1.0] The fog density factor
 32  * @param {double} [cfg.start = 1.0] Point on Z-axis at which fog effect begins
 33  * @param {double} [cfg.end = 1.0] Point on Z-axis at which fog effect ends
 34  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 35  * @param {...SceneJS.Node} [childNodes] Child nodes
 36  */
 37 SceneJS.Fog = function() {
 38     SceneJS.Node.apply(this, arguments);
 39     this._nodeType = "fog";
 40     this._mode = "linear";
 41     this._color = { r: 0.5, g: 0.5, b: 0.5 };
 42     this._density = 1.0;
 43     this._start = 0;
 44     this._end = 1000.0;
 45     if (this._fixedParams) {
 46         this._init(this._getParams());
 47     }
 48 };
 49 
 50 SceneJS._inherit(SceneJS.Fog, SceneJS.Node);
 51 
 52 /**
 53  Sets the fogging mode
 54  @function setMode
 55  @param {string} mode - "disabled", "exp", "exp2" or "linear"
 56  @returns {SceneJS.Fog} This fog node
 57  @since Version 0.7.4
 58  */
 59 SceneJS.Fog.prototype.setMode = function(mode) {
 60     if (mode != "disabled" && mode != "exp" && mode != "exp2" && mode != "linear") {
 61         throw SceneJS._errorModule.fatalError(new SceneJS.errors.InvalidNodeConfigException(
 62                 "SceneJS.fog has a mode of unsupported type: '" + mode + " - should be 'none', 'exp', 'exp2' or 'linear'"));
 63     }
 64     this._mode = mode;
 65     return this;
 66 };
 67 
 68 /**
 69  Returns fogging mode
 70  @function {string} getMode
 71  @returns {string} The fog mode - "disabled", "exp", "exp2" or "linear"
 72  @since Version 0.7.4
 73  */
 74 SceneJS.Fog.prototype.getMode = function() {
 75     return this._mode;
 76 };

 78 /**
 79  Sets the fog color
 80  @function setColor
 81  @param {object} color - eg. bright red: {r: 1.0, g: 0, b: 0 }
 82  @returns {SceneJS.Fog} This fog node
 83  @since Version 0.7.4
 84  */
 85 SceneJS.Fog.prototype.setColor = function(color) {
 86     this._color.r = color.r != undefined ? color.r : 0.5;
 87     this._color.g = color.g != undefined ? color.g : 0.5;
 88     this._color.b = color.b != undefined ? color.b : 0.5;
 89     return this;
 90 };
 91 
 92 /**
 93  Returns the fog color
 94  @function getColor
 95  @returns {object} Fog color - eg. bright red: {r: 1.0, g: 0, b: 0 }
 96  @since Version 0.7.4
 97  */
 98 SceneJS.Fog.prototype.getColor = function() {
 99     return {

101         g: this._color.g,

103     };
104 };

172 // @private
173 SceneJS.Fog.prototype._init = function(params) {
174     if (params.mode) {
175         this.setMode(params.mode);
176     }
177     if (params.color) {
178         this.setColor(params.color);
179     }
180     if (params.density != undefined) {
181         this.setDensity(params.density);
182     }
183     if (params.start != undefined) {
184         this.setStart(params.start);
185     }
186     if (params.end != undefined) {
187         this.setEnd(params.end);
188     }
189 };
190 
191 // @private
192 SceneJS.Fog.prototype._render = function(traversalContext, data) {
193     if (SceneJS._traversalMode == SceneJS.TRAVERSAL_MODE_PICKING) {
194         this._renderNodes(traversalContext, data);
195     } else {
196         if (!this._fixedParams) {
197             this._init(this._getParams(data));
198         }
199         var f = SceneJS._fogModule.getFog();
200         SceneJS._fogModule.setFog({
201             mode: this._mode,
202             color: this._color,
203             density: this._density,
204             start: this._start,
205             end: this._end
206         });
207         this._renderNodes(traversalContext, data);
208         SceneJS._fogModule.setFog(f);
209     }
210 };
