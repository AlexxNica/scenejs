1 /**
  2  * Backend module that services the SceneJS.assets.XXX nodes to manage the asynchronous cross-domain
  3  * load and caching of remotely-stored scene fragments.
  4  *
  5  * Uses the memory management backend to mediate cache management.
  6  *
  7  *  @private
  8  */
  9 var SceneJS_loadModule = new (function() {
 10 
 11     var time = (new Date()).getTime();
 12     var proxyUri = null;
 13     var assets = {};        // Asset content subgraphs for eviction, not reuse
 14 
 15     SceneJS_eventModule.onEvent(
 16             SceneJS_eventModule.TIME_UPDATED,
 17             function(t) {
 18                 time = t;
 19             });
 20 
 21     SceneJS_eventModule.onEvent(
 22             SceneJS_eventModule.RESET,
 23             function() {
 24                 assets = {};
 25             });
 26 
 27     /** @private */
 28     function _loadFile(url, onLoad, onError) {
 29         try {
 30             var x;
 31             var request = new XMLHttpRequest();
 32             request.onreadystatechange = function() {
 33                 if (request.readyState == 4) {
 34                     if (request.status == 200) {
 35                         onLoad(request.responseText);
 36                     } else {
 37                         onError(request.status, request.statusText);
 38                     }
 39                 }
 40             };
 41             request.open("GET", url, true);
 42             request.send(null);
 43         }
 44         catch (e) {
 45             onError(request.status, e.toString());
 46         }
 47     }
 48 
 49     /** @private */
 50     function _loadAssetSameDomain(uri, assetId, parser, onSuccess, onError) {
 51         _loadFile(uri,
 52                 function(data) {  // onLoad
 53                     if (!data) {
 54                         onError(new SceneJS.EmptyResponseException("loaded content is empty"));
 55                     } else {
 56                         var assetNode = parser(
 57                                 data,
 58                                 function(msg) {
 59                                     onError(msg);
 60                                 });
 61                         if (!assetNode) {
 62                             onError(new SceneJS.InternalException("parser returned null result"));
 63                         } else {
 64                             assets[assetId] = {
 65                                 assetId: assetId,
 66                                 uri: uri,
 67                                 node: assetNode,
 68                                 lastUsed: time
 69                             };
 70                             onSuccess(assetNode);
 71                         }
 72                     }
 73                 },
 74                 function(status, statusText) {
 75                     onError(new SceneJS.HttpException(status + " - " + (statusText || "")));
 76                 });
 77     }
 78 
 79     /** @private */
 80     function _jsonp(fullUri, callbackName, onLoad) {
 81         var head = document.getElementsByTagName("head")[0];
 82         var script = document.createElement("script");
 83         script.type = "text/javascript";
 84         window[callbackName] = function(data) {
 85             onLoad(data);
 86             window[callbackName] = undefined;
 87             try {
 88                 delete window[callbackName];
 89             } catch(e) {
 90             }
 91             head.removeChild(script);
 92         };
 93         head.appendChild(script);
 94         script.src = fullUri;  // Request fires now
 95     }
 96 
 97     /** Loads asset and caches it against uri
 98      * @private
 99      */
100     function _loadAssetCrossDomain(uri, assetId, serverParams, callbackName, parser, onSuccess, onError) {
101         var url = [proxyUri, "?callback=", callbackName , "&uri=" + uri];
102         for (var param in serverParams) { // TODO: memoize string portion that contains params
103             url.push("&", param, "=", serverParams[param]);
104         }
105         _jsonp(url.join(""),
106                 callbackName,
107                 function(data) {    // onLoad
108                     if (!data) {
109                         onError(new SceneJS.ProxyEmptyResponseException("proxy server response is empty"));
110                     } else if (data.error) {
111                         onError(new SceneJS.ProxyErrorResponseException("proxy server responded with error - " + data.error));
112                     } else {
113                         var assetNode = parser(
114                                 data,
115                                 function(msg) {
116                                     onError(msg);
117                                 });
118                         if (!assetNode) {
119                             onError(new SceneJS.InternalException("parser returned null result"));
120                         } else {
121                             assets[assetId] = {
122                                 assetId: assetId,
123                                 uri: uri,
124                                 node: assetNode,
125                                 lastUsed: time
126                             };
127                             onSuccess(assetNode);
128                         }
129                     }
130                 });
131     }

133     // @private
134     this.setProxy = function(_proxyUri) {
135         proxyUri = _proxyUri;
136     };

138     /** Attempts to get currently-loaded asset, which may have been evicted
139      * @private
140      */
141     this.getAsset = function(handle) {
142         var asset = assets[handle.assetId];
143         if (asset) {
144             asset.lastUsed = time;
145             return asset.node;
146         }
147         return null;
148     };

150     /**
151      * Triggers asynchronous JSONP load of asset, creates new process and handle; callback
152      * will fire with new child for the  client asset node. The asset node will have to then call assetLoaded
153      * to notify the backend that the asset has loaded and allow backend to kill the process.
154      *
155      * JSON does not handle errors, so the best we can do is manage timeouts withing SceneJS's process management.
156      *
157      * @private
158      * @uri Location of asset
159      * @serverParams Request parameters for proxy
160      * @parser Processes asset data on load
161      * @onSuccess Callback through which processed asset data is returned
162      * @onTimeout Callback invoked when no response from proxy
163      * @onError Callback invoked when error reported by proxy
164      */
165     this.loadAsset = function(uri, serverParams, parser, onSuccess, onTimeout, onError) {
166         //        if (!proxyUri) {
167         //            SceneJS_errorModule.fatalError(new SceneJS.ProxyNotSpecifiedException
168         //                    ("Scene definition error - SceneJS.load node expects a 'proxy' property on the SceneJS.scene node"));
169         //        }
170         if (proxyUri) {
171             SceneJS_loggingModule.debug("Loading asset cross-domain from " + uri);
172         } else {
173             SceneJS_loggingModule.debug("Loading asset from local domain " + uri);
174         }
175         var assetId = SceneJS._createKeyForMap(assets, "asset");
176         var process = SceneJS_processModule.createProcess({
177             onTimeout: function() {  // process killed automatically on timeout
178                 SceneJS_loggingModule.error(
179                         "Asset load failed - timed out waiting for a reply " +
180                         "(incorrect proxy URI?) - proxy: " + proxyUri +
181                         ", uri: " + uri);
182                 onTimeout();
183             },
184             description:"asset load: proxy = " + proxyUri + ", uri = " + uri,
185             timeoutSecs: 180 // Big timeout to allow files to parse
186         });
187         if (proxyUri) {
188             var callbackName = "callback" + process.id; // Process ID is globally unique
189             _loadAssetCrossDomain(
190                     uri,
191                     assetId,
192                     serverParams,
193                     callbackName,
194                     parser,
195                     onSuccess,
196                     function(msg) {  // onError
197                         SceneJS_processModule.killProcess(process);
198                         onError(msg);
199                     });
200         } else {
201             _loadAssetSameDomain(
202                     uri,
203                     assetId,
204                     parser,
205                     onSuccess,
206                     function(msg) {  // onError
207                         SceneJS_processModule.killProcess(process);
208                         onError(msg);
209                     });
210         }
211         var handle = {
212             process: process,
213             assetId : assetId
214         };
215         return handle;
216     };
217 
218     /** Notifies backend that load has completed; backend then kills the process.
219      * @private
220      */
221     this.assetLoaded = function(handle) {
222         SceneJS_processModule.killProcess(handle.process);
223     };
224 
225 })();
