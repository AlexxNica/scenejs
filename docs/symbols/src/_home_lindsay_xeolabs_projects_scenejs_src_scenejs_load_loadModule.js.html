1 /**
  2  * Backend module that services the SceneJS.Instance node to manage the asynchronous cross-domain
  3  * load and caching of remotely-stored scene fragments.
  4  *
  5  * Uses the memory management backend to mediate cache management.
  6  *
  7  *  @private
  8  */
  9 var SceneJS_loadModule = new (function() {
 10     var parsers = {};
 11     var time = (new Date()).getTime();
 12     var jsonpStrategy = null;
 13     var defaultLoadTimeoutSecs = 180;
 14     var assets = {};
 15     var pInterval;
 16 
 17     SceneJS_eventModule.addListener(
 18             SceneJS_eventModule.TIME_UPDATED,
 19             function(t) {
 20                 time = t;
 21             });
 22 
 23     SceneJS_eventModule.addListener(
 24             SceneJS_eventModule.INIT,
 25             function() {
 26                 assets = {};
 27                 if (!pInterval) {
 28                     pInterval = setInterval("SceneJS_loadModule._updateSceneJSLoads()", 100);
 29                 }
 30             });
 31 
 32     SceneJS_eventModule.addListener(
 33             SceneJS_eventModule.RESET,
 34             function() {
 35                 assets = {};
 36                 if (!pInterval) {
 37                     pInterval = setInterval("SceneJS_loadModule._updateSceneJSLoads()", 100);
 38                 }
 39             });
 40 
 41     /**
 42      * Periodically called to update statii of assets, reaps those that have failed
 43      * @private
 44      */
 45     this._updateSceneJSLoads = function() {
 46         var asset;
 47         for (var url in assets) {
 48             if (assets.hasOwnProperty(url)) {
 49                 asset = assets[url];
 50                 if (asset) {
 51                     asset.poll();
 52                     if (asset.status == SceneJS.Asset.STATUS_ERROR
 53                             || asset.status == SceneJS.Asset.STATUS_TIMED_OUT) {
 54                         assets[url] = null;
 55                     }
 56                 }
 57             }
 58         }
 59     };
 60 
 61     /** Registers a parser function to handle a file of the given extension type
 62      */
 63     this.registerParser = function (fileExtension, parser) {
 64         parsers[fileExtension.toLowerCase()] = parser;
 65     };
 66 
 67     function getParserForURL(url) {
 68         var ext = url.split('.').pop();
 69         if (!ext) {
 70             return defaultParser;
 71         }
 72         return parsers[ext.toLowerCase()] || defaultParser;
 73     }
 74 
 75     function defaultParser(cfg) {
 76         if (cfg.data._render) {  // Cross-domain JSONP
 77             return cfg.data;
 78         }
 79 //        if (data.error) {
 80 //            cfg.onError(cfg.data.error);
 81 //        }
 82         try {
 83             var evalData = eval(cfg.data);  // Same-domain
 84             if (evalData.error) {           // TODO: This is specific to the old SceneJS proxy - remove?
 85                 cfg.onError(evalData.error);
 86             }
 87             return evalData;
 88         } catch (e) {
 89             cfg.onError(new SceneJS.ParseException("Error parsing response: " + e));
 90         }
 91     }
 92 
 93     this.setLoadTimeoutSecs = function(loadTimeoutSecs) {
 94         defaultLoadTimeoutSecs = loadTimeoutSecs || 180;
 95     };
 96 
 97     /** Attempts to get currently-loaded asset, which may have been evicted
 98      */
 99     this.getAsset = function(uri) {
100         var asset = assets[uri];
101         if (asset) {
102             asset.lastUsed = time;
103             return asset.assetNode;
104         }
105         return null;
106     };
107 
108     /**
109      * Triggers asynchronous JSONP load of asset, creates new process and handle; callback
110      * will fire with new child for the  client asset node. The asset node will have to then call assetLoaded
111      * to notify the backend that the asset has loaded and allow backend to kill the process.
112      *
113      * JSON does not handle errors, so the best we can do is manage timeouts withing SceneJS's process management.
114      *
115      * @private
116      * @uri Location of asset
117      * @loadTimeoutSecs Seconds after which response and parsing times out
118      * @onLoaded Callback through which processed asset data is returned
119      * @onTimeout Callback invoked when no response from proxy
120      * @onError Callback invoked when error reported by proxy
121      */
122     this.loadAsset = function(uri, loadTimeoutSecs, onLoaded, onTimeout, onError) {
123         uri = getBaseURL(uri);
124         var asset = assets[uri];
125         if (asset) {
126             if (asset.status == SceneJS.Asset.STATUS_LOADING) {
127                 asset.addListener({ onLoaded : onLoaded, onTimeout: onTimeout, onError : onError });
128             }
129             return uri;
130         }
131         assets[uri] = new SceneJS.Asset({
132             time: time,
133             timeoutSecs: loadTimeoutSecs || defaultLoadTimeoutSecs,
134             uri: uri,
135             jsonpStrategy: jsonpStrategy,
136             parser: getParserForURL(uri),
137             onLoaded: onLoaded,
138             onTimeout: onTimeout,
139             onError: onError
140         });
141         return uri;
142     };
143 
144     function getBaseURL(url) {
145         var i = url.indexOf("#");
146         return (i == -1) ? url : url.substr(0, i);
147     }
148 
149     /**
150      * @private
151      */
152     this.setJSONPStrategy = function(strategy) {
153         jsonpStrategy = strategy;
154     };
155 
156 })();

158 /**
159  * Configures SceneJS with a strategy to allow it to use a particular Web service to proxy cross-domain requests
160  * by nodes such as {@link SceneJS.Instance} when their URIs are not in the local domain. As shown in the example below,
161  * the strategy must implement two methods, one to create the request URL and another to extract a string of data
162  * from the response.
163  * <p><b>Example strategy- </b>specifying a JSONP handler for the Yahoo! Query Language (YQL) API:</p>
164  * <pre><code>
165  * SceneJS.setJSONPStrategy({
166  *          request : function(url, format, callback) {
167  *                  return "http://query.yahooapis.com/v1/public/yql?q=select%20*%20from%20" + format + "%20where%20url='"
168  *                                   + url + "'&format=" + format + "&callback=" + callback;
169  *           },
170  *
171  *          response : function(data) {
172  *                  return data.results[0];
173  *          }
174  *  });
175  * </code></pre>
176  * @param strategy {{request:function(url:String, format:String, callback:String), String response: function(data:String) }}
177  */
178 SceneJS.setJSONPStrategy = function(strategy) {
179     SceneJS_loadModule.setJSONPStrategy(strategy);
180 };

183 SceneJS.Asset = function(cfg) {
184     this.timeoutSecs = cfg.timeoutSecs;
185     this.lastUsed = cfg.time;
186     this.uri = cfg.uri;
187     this._listeners = [];
188     this._jsonpStrategy = cfg.jsonpStrategy;
189     this._parser = cfg.parser;
190     this.assetNode = null;
191     this.exception = null;
192     this.addListener({ onLoaded: cfg.onLoaded, onError: cfg.onError, onTimeout: cfg.onTimeout });
193     this.status = SceneJS.Asset.STATUS_LOADING;
194     this._load();
195 };
196 
197 SceneJS.Asset.STATUS_LOADING = 0;
198 SceneJS.Asset.STATUS_LOADED = 1;
199 SceneJS.Asset.STATUS_ERROR = 2;
200 SceneJS.Asset.STATUS_TIMED_OUT = 3;
201 
202 SceneJS.Asset.prototype._load = function() {
203     var description = this._jsonpStrategy
204             ? "Instancing scene content cross-domain from " + this.uri
205             : "Instancing scene content at local domain " + this.uri;
206     SceneJS_loggingModule.debug(description);
207     var self = this;
208     this._process = SceneJS_processModule.createProcess({
209         onTimeout: function() {  // process killed automatically on timeout
210             self._handleTimeout();
211         },
212         description: description,
213         timeoutSecs: this.timeoutSecs
214     });
215     if (this._jsonpStrategy) {
216         this._loadCrossDomain();
217     } else {
218         this._loadLocal();
219     }
220 };
221 
222 SceneJS.Asset.prototype._loadCrossDomain = function() {
223     var format = this._parser.serverParams ? (this._parser.serverParams.format || "json") : "json";
224     var callbackName = "callback" + this._process.id;
225     var fullUri = this._jsonpStrategy.request(this.uri, format, callbackName);
226     var head = document.getElementsByTagName("head")[0];
227     var script = document.createElement("script");
228     script.type = "text/javascript";
229     var self = this;
230     window[callbackName] = function(data) {
231         window[callbackName] = undefined;   // Remove callback
232         try {
233             delete window[callbackName];
234         } catch(e) {
235         }
236         head.removeChild(script);          // Remove script
237         self.assetNode = self._parser.parse({
238             uri: self.uri,
239             data: data,
240             onError: function(msg) { ////////// TODO: Needed?
241                 self._handleException(new SceneJS.EmptyResponseException(msg));
242             }
243         });
244         if (!self.assetNode) {
245             self._handleException(new SceneJS.InternalException("parser returned null result"));
246         } else {
247             self._handleSuccess();
248         }
249     };
250     head.appendChild(script);
251     script.src = fullUri;  // Sends request
252 };
253 
254 SceneJS.Asset.prototype._loadLocal = function() {
255     var request;
256     try {
257         request = new XMLHttpRequest();
258         var self = this;
259         request.onreadystatechange = function() {
260             if (request.readyState == 4) {
261                 if (!request.responseText) {
262                     self.handleError(new SceneJS.EmptyResponseException("response content is empty"));
263                 } else {
264                     self._assetNode = self._parser.parse({
265                         data: request.responseText,
266                         onError: function(msg) { ////////// TODO: Needed?
267                             self._handleException(new SceneJS.EmptyResponseException(msg));
268                         }
269                     });
270                     if (!self._assetNode) {
271                         self._handleException(new SceneJS.InternalException("parser returned null result"));
272                     } else {
273                         self._handleSuccess();
274                     }
275                 }
276             }
277         };
278         request.open("GET", this._url, true);
279         request.send(null);
280     }
281     catch (e) {
282         this._handleException(new SceneJS.HttpException(request.status + " - " + e.toString()));
283     }
284 };
285 
286 SceneJS.Asset.prototype.addListener = function(listener) {
287     this._listeners.push(listener);
288 };
289 
290 SceneJS.Asset.prototype._handleException = function(e) {
291     this.status = SceneJS.Asset.STATUS_ERROR;
292     this.exception = e;
293     if (this._process) {
294         SceneJS_processModule.killProcess(this._process);
295     }
296 };
297 
298 SceneJS.Asset.prototype._handleSuccess = function() {
299     this.status = SceneJS.Asset.STATUS_LOADED;
300     SceneJS_processModule.killProcess(this._process);
301 };
302 
303 SceneJS.Asset.prototype._handleTimeout = function() {
304     this.status = SceneJS.Asset.STATUS_TIMED_OUT;
305     if (this._jsonpStrategy) {
306         SceneJS_loggingModule.error(
307                 "Load failed - timed out waiting for a reply from JSONP proxy for content at uri: " + this.uri);
308     } else {
309         SceneJS_loggingModule.error(
310                 "Load failed - timed out waiting for content at uri: " + this.uri);
311     }
312 };
313 
314 SceneJS.Asset.prototype.poll = function() {
315     if (this.status == SceneJS.Asset.STATUS_LOADING) {
316         return;
317     }
318     var listener;
319     while (this._listeners.length > 0) {  // Removes listeners
320         listener = this._listeners.pop();
321         switch (this.status) {
322             case SceneJS.Asset.STATUS_LOADED:
323                 if (listener.onLoaded) {
324                     listener.onLoaded(this.assetNode);
325                 }
326                 break;
327             case SceneJS.Asset.STATUS_ERROR:
328                 if (listener.onError) {
329                     listener.onError(this.exception);
330                 }
331                 break;
332             case SceneJS.Asset.STATUS_TIMED_OUT:
333                 if (listener.onTimeout) {
334                     listener.onTimeout();
335                 }
336                 break;
337         }
338     }
339 };
