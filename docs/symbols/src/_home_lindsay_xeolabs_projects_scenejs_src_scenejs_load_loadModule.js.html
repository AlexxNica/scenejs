1 /**
  2  * Backend module that services the SceneJS.assets.XXX nodes to manage the asynchronous cross-domain
  3  * load and caching of remotely-stored scene fragments.
  4  *
  5  * Uses the memory management backend to mediate cache management.
  6  *
  7  *  @private
  8  */
  9 var SceneJS_loadModule = new (function() {
 10 
 11     var time = (new Date()).getTime();
 12     var proxyUri = null;
 13     var assets = {};        // Asset content subgraphs for eviction, not reuse
 14 
 15     SceneJS_eventModule.onEvent(
 16             SceneJS_eventModule.TIME_UPDATED,
 17             function(t) {
 18                 time = t;
 19             });
 20 
 21     SceneJS_eventModule.onEvent(
 22             SceneJS_eventModule.RESET,
 23             function() {
 24                 assets = {};
 25             });
 26 
 27     // @private
 28     function jsonp(fullUri, callbackName, onLoad) {
 29         var head = document.getElementsByTagName("head")[0];
 30         var script = document.createElement("script");
 31         script.type = "text/javascript";
 32         window[callbackName] = function(data) {
 33             onLoad(data);
 34             window[callbackName] = undefined;
 35             try {
 36                 delete window[callbackName];
 37             } catch(e) {
 38             }
 39             head.removeChild(script);
 40         };
 41         head.appendChild(script);
 42         script.src = fullUri;  // Request fires now
 43     }
 44 
 45     /** Loads asset and caches it against uri
 46      * @private
 47      */
 48     function _loadAsset(uri, assetId, serverParams, callbackName, parser, onSuccess, onError) {
 49         var url = [proxyUri, "?callback=", callbackName , "&uri=" + uri];
 50         for (var param in serverParams) { // TODO: memoize string portion that contains params
 51             url.push("&", param, "=", serverParams[param]);
 52         }
 53         jsonp(url.join(""),
 54                 callbackName,
 55                 function(data) {    // onLoad
 56                     if (!data) {
 57                         onError("server response is empty");
 58                     } else {
 59                         var assetNode = parser(
 60                                 data,
 61                                 function(msg) {
 62                                     onError(msg);
 63                                 });
 64                         if (!assetNode) {
 65                             onError("asset node's parser returned null result");
 66                         } else {
 67                             assets[assetId] = {
 68                                 assetId: assetId,
 69                                 uri: uri,
 70                                 node: assetNode,
 71                                 lastUsed: time
 72                             };
 73                             onSuccess(assetNode);
 74                         }
 75                     }
 76                 });
 77 
 78 
 79     }
 80 
 81     // @private
 82     this.setProxy = function(_proxyUri) {
 83         proxyUri = _proxyUri;
 84     };
 85 
 86     /** Attempts to get currently-loaded asset, which may have been evicted
 87      * @private
 88      */
 89     this.getAsset = function(handle) {
 90         var asset = assets[handle.assetId];
 91         if (asset) {
 92             asset.lastUsed = time;
 93             return asset.node;
 94         }
 95         return null;
 96     };
 97 
 98     /**
 99      * Triggers asynchronous JSONP load of asset, creates new process and handle; callback
100      * will fire with new child for the  client asset node. The asset node will have to then call assetLoaded
101      * to notify the backend that the asset has loaded and allow backend to kill the process.
102      *
103      * JSON does not handle errors, so the best we can do is manage timeouts withing SceneJS's process management.
104      *
105      * @private
106      * @uri Location of asset
107      * @serverParams Request parameters for proxy
108      * @parser Processes asset data on load
109      * @onSuccess Callback through which processed asset data is returned
110      * @onTimeout Callback invoked when no response from proxy
111      * @onError Callback invoked when error reported by proxy
112      */
113     this.loadAsset = function(uri, serverParams, parser, onSuccess, onTimeout, onError) {
114         if (!proxyUri) {
115             SceneJS_errorModule.fatalError(new SceneJS.exceptions.ProxyNotSpecifiedException
116                     ("Scene definition error - SceneJS.load node expects a 'proxy' property on the SceneJS.scene node"));
117         }
118         SceneJS_loggingModule.debug("Loading asset from " + uri);
119         var assetId = SceneJS._utils.createKeyForMap(assets, "asset");
120         var process = SceneJS_processModule.createProcess({
121             onTimeout: function() {  // process killed automatically on timeout
122                 SceneJS_loggingModule.error(
123                         "Asset load failed - timed out waiting for a reply " +
124                         "(incorrect proxy URI?) - proxy: " + proxyUri +
125                         ", uri: " + uri);
126                 onTimeout();
127             },
128             description:"asset load: proxy = " + proxyUri + ", uri = " + uri,
129             timeoutSecs: 180 // Big timeout to allow files to parse
130         });
131         var callbackName = "callback" + process.id; // Process ID is globally unique
132         _loadAsset(
133                 uri,
134                 assetId,
135                 serverParams,
136                 callbackName,
137                 parser,
138                 onSuccess,
139                 function(msg) {  // onError
140                     SceneJS_processModule.killProcess(process);
141                     onError(msg);
142                 });
143         var handle = {
144             process: process,
145             assetId : assetId
146         };
147         return handle;
148     };
149 
150     /** Notifies backend that load has completed; backend then kills the process.
151      * @private
152      */
153     this.assetLoaded = function(handle) {
154         SceneJS_processModule.killProcess(handle.process);
155     };
156 
157 })();
