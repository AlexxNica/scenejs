1 /**
  2  * Backend module that services the SceneJS.Instance node to manage the asynchronous cross-domain
  3  * load and caching of remotely-stored scene fragments.
  4  *
  5  * Uses the memory management backend to mediate cache management.
  6  *
  7  *  @private
  8  */
  9 var SceneJS_loadModule = new (function() {
 10     var parsers = {};
 11     var time = (new Date()).getTime();
 12     var jsonpStrategy = null;
 13     var defaultLoadTimeoutSecs = 180;
 14     var assets = {};
 15     var sceneLibraries = {}; // Library asset set for each active scene
 16     var sceneLibrary;  // Library of currently active scene
 17     var pInterval; // For polling SceneJS.Asset to update their statii
 18 
 19     SceneJS_eventModule.addListener(
 20             SceneJS_eventModule.TIME_UPDATED,
 21             function(t) {
 22                 time = t;
 23             });
 24 
 25     /* When SceneJS initialised, start the assets status update loop
 26      */
 27     SceneJS_eventModule.addListener(
 28             SceneJS_eventModule.INIT,
 29             function() {
 30                 assets = {};
 31                 if (!pInterval) {
 32                     pInterval = setInterval("SceneJS_loadModule._updateSceneJSLoads()", 100);
 33                 }
 34             });
 35 
 36     /* When SceneJS reset, destroy all assets
 37      */
 38     SceneJS_eventModule.addListener(
 39             SceneJS_eventModule.RESET,
 40             function() {
 41                 assets = {};
 42                 //                if (!pInterval) {
 43                 //                    pInterval = setInterval("SceneJS_loadModule._updateSceneJSLoads()", 100);
 44                 //                }
 45             });
 46 
 47     /* When scene created, define a map of library assets for it
 48      */
 49     SceneJS_eventModule.addListener(
 50             SceneJS_eventModule.SCENE_CREATED,
 51             function(params) {
 52                 sceneLibraries[params.sceneId] = {};
 53             });
 54 
 55 
 56     /* When scene activated, activate the current scene library
 57      */
 58     SceneJS_eventModule.addListener(
 59             SceneJS_eventModule.SCENE_RENDERING,
 60             function(params) {
 61                 sceneLibrary = sceneLibraries[params.sceneId];
 62             });
 63 
 64     /* When canvas activated, render the library's assets to define their Symbols.
 65      * Note we must have a canvas, so we can drive WebGL
 66      */
 67     SceneJS_eventModule.addListener(
 68             SceneJS_eventModule.CANVAS_ACTIVATED,
 69             function() {
 70                 var dummyTraversalContext = {};
 71                 var dummyData = new SceneJS.Data(null, false, {});
 72                 for (var key in sceneLibrary) {
 73                     if (sceneLibrary.hasOwnProperty(key)) {
 74                         var asset = sceneLibrary[key];
 75                         if (asset.status == SceneJS.Asset.STATUS_LOADED) {
 76                             asset.assetNode._render(dummyTraversalContext, dummyData);
 77                         }
 78                     }
 79                 }
 80             }, 10000);
 81 
 82     /* When scene destroyed, destroy its library assets
 83      */
 84     SceneJS_eventModule.addListener(
 85             SceneJS_eventModule.SCENE_DESTROYED,
 86             function(params) {
 87                 sceneLibraries[params.sceneId] = null;
 88             });
 89 
 90     /**
 91      * Periodically called to update statii of assets, reaps those that have failed
 92      * @private
 93      */
 94     this._updateSceneJSLoads = function() {
 95         var asset;
 96         for (var url in assets) { // TODO: remove from this list when loaded
 97             if (assets.hasOwnProperty(url)) {
 98                 asset = assets[url];
 99                 if (asset) {
100                     asset.poll();
101                     if (asset.status == SceneJS.Asset.STATUS_ERROR ||
102                         asset.status == SceneJS.Asset.STATUS_TIMED_OUT) {
103                         assets[url] = null;
104                     }
105                 }
106             }
107         }
108     };
109 
110     /** Registers a parser function to handle a file of the given extension type
111      */
112     this.registerParser = function (fileExtension, parser) {
113         parsers[fileExtension.toLowerCase()] = parser;
114     };
115 
116     function getParserForURL(url) {
117         var ext = url.split('.').pop();
118         if (!ext) {
119             return defaultParser;
120         }
121         return parsers[ext.toLowerCase()] || defaultParser;
122     }
123 
124     var defaultParser = new (function() {
125         this.parse = function(cfg) {
126             if (cfg.data._render) {  // Cross-domain JSONP
127                 return cfg.data;
128             }
129             //        if (data.error) {
130             //            cfg.onError(cfg.data.error);
131             //        }
132             try {
133                 var evalData = eval(cfg.data);  // Same-domain
134                 if (evalData.error) {           // TODO: This is specific to the old SceneJS proxy - remove?
135                     // TODO: Throw
136                     //cfg.onError(evalData.error);
137                 }
138                 return evalData;
139             } catch (e) {
140                 cfg.onError(new SceneJS.ParseException("Error parsing response: " + e));
141             }
142         };
143         this.serverParams = {
144             format: "json"
145         };
146     })();
147 
148     this.setLoadTimeoutSecs = function(loadTimeoutSecs) {
149         defaultLoadTimeoutSecs = loadTimeoutSecs || 180;
150     };
151 
152     /** Attempts to get currently-loaded asset, which may have been evicted
153      */
154     this.getAsset = function(uri) {
155         var asset = assets[uri];
156         if (asset && asset.status == SceneJS.Asset.STATUS_LOADED) {
157             asset.lastUsed = time;
158             return asset.assetNode;
159         }
160         return null;
161     };
162 
163     /**
164      * Triggers asynchronous JSONP load of asset, creates new process; callback
165      * will fire with new child for the  client asset node. The asset node will have to then call assetLoaded
166      * to notify the backend that the asset has loaded and allow backend to kill the process.
167      *
168      * JSON does not handle errors, so the best we can do is manage timeouts withing SceneJS's process management.
169      *
170      * @private
171      * @uri Location of asset
172      * @loadTimeoutSecs Seconds after which response and parsing times out
173      * @onLoaded Callback through which processed asset data is returned
174      * @onTimeout Callback invoked when no response from proxy
175      * @onError Callback invoked when error reported by proxy
176      * @isLibrary True to indicate that asset is a library subgraph containing Symbols
177      */
178     this.loadAsset = function(uri, loadTimeoutSecs, onLoaded, onTimeout, onError, isLibrary) {
179         uri = getBaseURL(uri);
180         var asset = assets[uri];
181         if (asset) {
182             if (asset.status == SceneJS.Asset.STATUS_LOADING) {
183                 asset.addListener({ onLoaded : onLoaded, onTimeout: onTimeout, onError : onError });
184             }
185         } else {
186             asset = new SceneJS.Asset({
187                 time: time,
188                 timeoutSecs: loadTimeoutSecs || defaultLoadTimeoutSecs,
189                 uri: uri,
190                 jsonpStrategy: jsonpStrategy,
191                 parser: getParserForURL(uri),
192                 onLoaded: onLoaded,
193                 onTimeout: onTimeout,
194                 onError: onError,
195                 isLibrary : isLibrary
196             });
197             assets[uri] = asset;
198         }
199         if (isLibrary) {
200 
201             /* Register library asset for rendering at the
202              * beginning of next scene render as soon as it
203              * is at status Asset.STATUS_LOADED, so that it
204              * may render its Symbol nodes.
205              */
206             if (!sceneLibrary[uri]) {
207                 sceneLibrary[uri] = asset;
208             }
209         }
210     };
211 
212     function getBaseURL(url) {
213         var i = url.indexOf("#");
214         return (i == -1) ? url : url.substr(0, i);
215     }
216 
217     /**
218      * @private
219      */
220     this.setJSONPStrategy = function(strategy) {
221         jsonpStrategy = strategy;
222     };
223 
224 })();

226 /**
227  * Configures SceneJS with a strategy to allow it to use a particular Web service to proxy cross-domain requests
228  * by nodes such as {@link SceneJS.Instance} when their URIs are not in the local domain. As shown in the example below,
229  * the strategy must implement two methods, one to create the request URL and another to extract a string of data
230  * from the response.
231  * <p><b>Example strategy- </b>specifying a JSONP handler for the Yahoo! Query Language (YQL) API:</p>
232  * <pre><code>
233  * SceneJS.setJSONPStrategy({
234  *          request : function(url, format, callback) {
235  *                  return "http://query.yahooapis.com/v1/public/yql?q=select%20*%20from%20" + format + "%20where%20url='"
236  *                                   + url + "'&format=" + format + "&callback=" + callback;
237  *           },
238  *
239  *          response : function(data) {
240  *                  return data.results[0];
241  *          }
242  *  });
243  * </code></pre>
244  * @param strategy {{request:function(url:String, format:String, callback:String), String response: function(data:String) }}
245  */
246 SceneJS.setJSONPStrategy = function(strategy) {
247     SceneJS_loadModule.setJSONPStrategy(strategy);
248 };

251 SceneJS.Asset = function(cfg) {
252     this.timeoutSecs = cfg.timeoutSecs;
253     this.lastUsed = cfg.time;
254     this.uri = cfg.uri;
255     this._listeners = [];
256     this._jsonpStrategy = cfg.jsonpStrategy;
257     this._parser = cfg.parser;
258     this.assetNode = null;
259     this.exception = null;
260     this.addListener({ onLoaded: cfg.onLoaded, onError: cfg.onError, onTimeout: cfg.onTimeout });
261     this.status = SceneJS.Asset.STATUS_LOADING;
262     this.isLibrary = cfg.isLibrary;
263     this._load();
264 };
265 
266 SceneJS.Asset.STATUS_LOADING = 0;
267 SceneJS.Asset.STATUS_LOADED = 1;
268 SceneJS.Asset.STATUS_ERROR = 2;
269 SceneJS.Asset.STATUS_TIMED_OUT = 3;
270 
271 SceneJS.Asset.prototype._load = function() {
272     var description = this._jsonpStrategy
273             ? "Instancing scene content cross-domain from " + this.uri
274             : "Instancing scene content at local domain " + this.uri;
275     SceneJS_loggingModule.debug(description);
276     var self = this;
277     this._process = SceneJS_processModule.createProcess({
278         onTimeout: function() {  // process killed automatically on timeout
279             self._handleTimeout();
280         },
281         description: description,
282         timeoutSecs: this.timeoutSecs
283     });
284     if (this._jsonpStrategy) {
285         this._loadCrossDomain();
286     } else {
287         this._loadLocal();
288     }
289 };

291 SceneJS.Asset.prototype._loadCrossDomain = function() {
292     var format = this._parser.serverParams ? (this._parser.serverParams.format || "json") : "json";
293     var callbackName = "callback" + this._process.id;
294     var fullUri = this._jsonpStrategy.request(this.uri, format, callbackName);
295     var head = document.getElementsByTagName("head")[0];
296     var script = document.createElement("script");
297     script.type = "text/javascript";
298     var self = this;
299     window[callbackName] = function(data) {
300         window[callbackName] = undefined;   // Remove callback
301         try {
302             delete window[callbackName];
303         } catch(e) {
304         }
305         head.removeChild(script);          // Remove script
306         try {
307             data = self._jsonpStrategy.response(data);
308         } catch (e) {
309             self._handleException(new SceneJS.ProxyErrorResponseException(e));
310             return;
311         }
312         self.assetNode = self._parser.parse({
313             uri: self.uri,
314             data: data,
315             onError: function(msg) { ////////// TODO: Needed?
316                 self._handleException(new SceneJS.EmptyResponseException(msg));
317             }
318         });
319         if (this.status != SceneJS.Asset.STATUS_ERROR) {
320             if (!self.assetNode) {
321                 self._handleException(new SceneJS.InternalException("parser returned null result"));
322             } else {              
323                 self._handleSuccess();
324             }
325         }
326     };
327     head.appendChild(script);
328     script.src = fullUri;  // Sends request
329 };

331 SceneJS.Asset.prototype._loadLocal = function() {
332     var request;
333     try {
334         request = new XMLHttpRequest();
335         var self = this;
336         request.onreadystatechange = function() {
337             if (request.readyState == 4) {
338                 if (!request.responseText) {
339                     self.handleError(new SceneJS.EmptyResponseException("response content is empty"));
340                 } else {
341                     self._assetNode = self._parser.parse({
342                         data: request.responseText,
343                         onError: function(msg) { ////////// TODO: Needed?
344                             self._handleException(new SceneJS.EmptyResponseException(msg));
345                         }
346                     });
347                     if (this.status != SceneJS.Asset.STATUS_ERROR) {
348                         if (!self._assetNode) {
349                             self._handleException(new SceneJS.InternalException("parser returned null result"));
350                         } else {
351                             self._handleSuccess();
352                         }
353                     }
354                 }
355             }
356         };
357         request.open("GET", this._url, true);
358         request.send(null);
359     }
360     catch (e) {
361         this._handleException(new SceneJS.HttpException(request.status + " - " + e.toString()));
362     }
363 };

365 SceneJS.Asset.prototype.addListener = function(listener) {
366     this._listeners.push(listener);
367 };
368 
369 SceneJS.Asset.prototype._handleException = function(e) {
370     this.status = SceneJS.Asset.STATUS_ERROR;
371     this.exception = e;
372     if (this._process) {
373         SceneJS_processModule.killProcess(this._process);
374     }
375 };
376 
377 SceneJS.Asset.prototype._handleSuccess = function() {
378     this.status = SceneJS.Asset.STATUS_LOADED;
379     SceneJS_processModule.killProcess(this._process);
380 };
381 
382 SceneJS.Asset.prototype._handleTimeout = function() {
383     this.status = SceneJS.Asset.STATUS_TIMED_OUT;
384     if (this._jsonpStrategy) {
385         SceneJS_loggingModule.error(
386                 "Load failed - timed out waiting for a reply from JSONP proxy for content at uri: " + this.uri);
387     } else {
388         SceneJS_loggingModule.error(
389                 "Load failed - timed out waiting for content at uri: " + this.uri);
390     }
391 };
392 
393 SceneJS.Asset.prototype.poll = function() {
394     if (this.status == SceneJS.Asset.STATUS_LOADING) {
395         return;
396     }
397     var listener;
398     while (this._listeners.length > 0) {  // Removes listeners
399         listener = this._listeners.pop();
400         switch (this.status) {
401             case SceneJS.Asset.STATUS_LOADED:
402                 if (listener.onLoaded) {
403                     listener.onLoaded(this.assetNode);
404                 }
405                 break;
406             case SceneJS.Asset.STATUS_ERROR:
407                 if (listener.onError) {
408                     listener.onError(this.exception);
409                 }
410                 break;
411             case SceneJS.Asset.STATUS_TIMED_OUT:
412                 if (listener.onTimeout) {
413                     listener.onTimeout();
414                 }
415                 break;
416         }
417     }
418 };
