1 /**
  2  * Scene node that constructs a 'lookAt' view transformation matrix and sets it on the current shader.
  3  */
  4 
  5 (function() {
  6     var backend = SceneJS._backends.getBackend('view-transform');
  7     var errorBackend = SceneJS._backends.getBackend('error');
  8 
  9     /* Memoization levels
 10      */
 11     const NO_MEMO = 0;              // No memoization, assuming that node's configuration is dynamic
 12     const FIXED_CONFIG = 1;         // Node config is fixed, memoizing local object-space matrix
 13     const FIXED_SPACE = 2;    // Both node config and model-space are fixed
 14 
 15     SceneJS.lookAt = function() {
 16         var cfg = SceneJS._utils.getNodeConfig(arguments);
 17 
 18         return SceneJS._utils.createNode(
 19                 "lookAt",
 20                 cfg.children,
 21 
 22                 new (function() {
 23                     var _memoLevel = NO_MEMO;
 24 
 25                     var _mat;
 26                     var xform;
 27 
 28                     var _eyeX = 0;
 29                     var _eyeY = 0;
 30                     var _eyeZ = 1;
 31 
 32                     var _lookX = 0;
 33                     var _lookY = 0;
 34                     var _lookZ = 0;
 35 
 36                     var _upX = 0;
 37                     var _upY = 1;
 38                     var _upZ = 0;
 39 
 40                     this.setEye = function(eye) {
 41                         _eyeX = eye.x || 0;
 42                         _eyeY = eye.y || 0;
 43                         _eyeZ = eye.z || 0;
 44                         _memoLevel = NO_MEMO;
 45                         return this;
 46                     };
 47 
 48                     this.getEye = function() {
 49                         return {
 50                             x: _eyeX,
 51                             y: _eyeY,
 52                             z: _eyeZ
 53                         };
 54                     };
 55 
 56                     this.setLook = function(look) {
 57                         _lookX = look.x || 0;
 58                         _lookY = look.y || 0;
 59                         _lookZ = look.z || 0;
 60                         _memoLevel = NO_MEMO;
 61                         return this;
 62                     };
 63 
 64                     this.getLook = function() {
 65                         return {
 66                             x: _lookX,
 67                             y: _lookY,
 68                             z: _lookZ
 69                         };
 70                     };
 71 
 72                     this.setUp = function(up) {
 73                         var x = up.x || 0;
 74                         var y = up.y || 0;
 75                         var z = up.z || 0;
 76                         if (x + y + z == 0) {
 77                             errorBackend.fatalError(
 78                                     new SceneJS.exceptions.IllegalRotateConfigException(
 79                                             "SceneJS.lookAt up vector is zero length - at least one of its x,y and z components must be non-zero"));
 80                         }
 81                         _upX = x;
 82                         _upY = y;
 83                         _upZ = z;
 84                         _memoLevel = NO_MEMO;
 85                         return this;
 86                     };
 87 
 88                     this.getUp = function() {
 89                         return {
 90                             x: _upX,
 91                             y: _upY,
 92                             z: _upZ
 93                         };
 94                     };
 95 
 96                     this._init = function(params) {
 97                         if (params.eye) {
 98                             this.setEye(params.eye);
 99                         }
100                         if (params.look) {
101                             this.setLook(params.look);
102                         }
103                         if (params.up) {
104                             this.setUp(params.up);
105                         }
106                     };
107 
108                     if (cfg.fixed) {
109                         this._init(cfg.getParams());
110                     }
111 
112                     this._render = function(traversalContext, data) {
113                         if (_memoLevel == NO_MEMO) {
114                             if (!cfg.fixed) {
115                                 this._init(cfg.getParams(data));
116                             } else {
117                                 _memoLevel = FIXED_CONFIG;
118                             }
119                             _mat = SceneJS_math_lookAtMat4c(
120                                     _eyeX, _eyeY, _eyeZ,
121                                     _lookX, _lookY, _lookZ,
122                                     _upX, _upY, _upZ);
123                         }
124                         var superXform = backend.getTransform();
125                         if (_memoLevel < FIXED_SPACE) {
126                             var tempMat = SceneJS_math_mulMat4(superXform.matrix, _mat);
127                             xform = {
128                                 type: "lookat",
129                                 matrix: tempMat,
130                                 lookAt : {
131                                     eye: { x: _eyeX, y: _eyeY, z: _eyeZ },
132                                     look: { x: _lookX, y: _lookY, z: _lookZ },
133                                     up:  { x: _upX, y: _upY, z: _upZ }
134                                 },
135                                 fixed: superXform.fixed && cfg.fixed
136                             };
137                             if (_memoLevel == FIXED_CONFIG && superXform.fixed) {   // Bump up memoization level if space fixed
138                                 _memoLevel = FIXED_SPACE;
139                             }
140                         }
141                         backend.setTransform(xform);
142                         this._renderChildren(traversalContext, data);
143                         backend.setTransform(superXform);
144                     };
145                 })());
146     };
147 })();
