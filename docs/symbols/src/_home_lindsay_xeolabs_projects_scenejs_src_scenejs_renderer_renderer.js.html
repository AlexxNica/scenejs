1 /** Scene graph node that sets renderer state for nodes in its subtree. These nodes may
  2  * be nested, and the root renderer node must specify the ID of a WebGL canvas node in
  3  * the DOM. Nested renderes may then omit the canvas ID to reuse the current canvas, or
  4  * may specify a different canvas ID to activate a different canvas.
  5  */
  6 SceneJS.renderer = function() {
  7     var cfg = SceneJS._utils.getNodeConfig(arguments);
  8     var backend = SceneJS._backends.getBackend('renderer');
  9     var env;
 10 
 11     return SceneJS._utils.createNode(
 12             "renderer",
 13             cfg.children,
 14 
 15             new (function() {
 16                 this._render = function(traversalContext, data) {
 17                     if (!env || !cfg.fixed) {
 18                         var params = cfg.getParams(data);
 19                         env = backend.createRendererState(params);
 20                     }
 21                     backend.setRendererState(env);
 22                     this._renderChildren(traversalContext, data);
 23                     backend.restoreRendererState(env);
 24                 };
 25             })());
 26 };
