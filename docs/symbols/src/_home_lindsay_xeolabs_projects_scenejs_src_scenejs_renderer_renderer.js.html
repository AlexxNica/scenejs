1 /** @class A scene node that sets WebGL state for nodes in its subtree.
  2  * <p>This node basically exposes various WebGL state configurations through the SceneJS API.</p>
  3  * (TODO: more comments here!)
  4  
  5  * @extends SceneJS.Node
  6  */
  7 SceneJS.Renderer = function() {
  8     SceneJS.Node.apply(this, arguments);
  9     this._nodeType = "renderer";
 10 };
 11 
 12 SceneJS._inherit(SceneJS.Renderer, SceneJS.Node);
 13 
 14 // @private
 15 SceneJS.Renderer.prototype._render = function(traversalContext, data) {
 16     if (this._memoLevel == 0) {  // One-shot dynamic config               
 17         this._rendererState = SceneJS._rendererModule.createRendererState(this._getParams(data));
 18         if (this._fixedParams) {
 19             this._memoLevel = 1;
 20         }
 21     }
 22     SceneJS._rendererModule.setRendererState(this._rendererState);
 23     this._renderNodes(traversalContext, data);
 24     SceneJS._rendererModule.undoRendererState(this._rendererState);
 25 };
 26 
 27 /** Factory function that returns a new {@link SceneJS.Renderer} instance
 28  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Renderer constructor
 29  * @returns {SceneJS.Renderer}
 30  */
 31 SceneJS.renderer = function() {
 32     var n = new SceneJS.Renderer();
 33     SceneJS.Renderer.prototype.constructor.apply(n, arguments);
 34     return n;
 35 };
