1 /**
  2  @class A scene node that defines a perspective transformation for the nodes within its subgraph.
  3  @extends SceneJS.Node
  4  <p><b>Example:</b></p><p>Defining perspective, specifying parameters that happen to be the default values</b></p><pre><code>
  5  var p = new SceneJS.Perspective({
  6  fovy : 55.0,
  7  aspect : 1.0,
  8  near : 0.10,
  9  far : 5000.0 },
 10 
 11  // ... child nodes
 12  )
 13  </pre></code>
 14 
 15  @constructor
 16  Create a new SceneJS.Perspective
 17  @param {Object} cfg  Config object or function, followed by zero or more child nodes
 18  */
 19 SceneJS.Perspective = function() {
 20     SceneJS.Node.apply(this, arguments);
 21     this._nodeType = "perspective";
 22     this._fovy = 45.0;
 23     this._aspect = 1.0;
 24     this._near = 0.1;
 25     this._far = 1.0;
 26     if (this._fixedParams) {
 27         this._init(this._getParams());
 28     }
 29 };
 30 
 31 SceneJS._inherit(SceneJS.Perspective, SceneJS.Node);
 32 
 33 /** Sets the field-of-view angle in degrees
 34  *
 35  * @param {float} fovy
 36  * @returns {SceneJS.Perspective} this
 37  */
 38 SceneJS.Perspective.prototype.setFovY = function(fovy) {
 39     this._fovy = fovy;
 40     this._memoLevel = 0;
 41 };

 44  * @returns {float} field-of-view

 46 SceneJS.Perspective.prototype.getFovyY = function() {

103 SceneJS.Perspective.prototype._init = function(params) {
104     if (params.fovy != undefined) {
105         this.setFovY(params.fovy);
106     }
107     if (params.aspect != undefined) {
108         this.setAspect(params.aspect);
109     }
110     if (params.near != undefined) {
111         this.setNear(params.near);
112     }
113     if (params.far != undefined) {
114         this.setFar(params.far);
115     }
116 };
117 
118 // Override
119 SceneJS.Perspective.prototype._render = function(traversalContext, data) {
120     if (this._memoLevel == 0) {
121         if (!this._fixedParams) {
122             this._init(this._getParams(data));
123         } else {
124             this._memoLevel = 1;
125         }
126         var tempMat = SceneJS_math_perspectiveMatrix4(
127                 this._fovy * Math.PI / 180.0,
128                 this._aspect,
129                 this._near,
130                 this._far);
131         this._transform = {
132             type: "perspective",
133             matrix:tempMat
134         };
135     }
136     var prevTransform = SceneJS_projectionModule.getTransform();
137     SceneJS_projectionModule.setTransform(this._transform);
138     this._renderNodes(traversalContext, data);
139     SceneJS_projectionModule.setTransform(prevTransform);
140 };

145 SceneJS.perspective = function() {
146     var n = new SceneJS.Perspective();
147     SceneJS.Perspective.prototype.constructor.apply(n, arguments);
148     return n;
