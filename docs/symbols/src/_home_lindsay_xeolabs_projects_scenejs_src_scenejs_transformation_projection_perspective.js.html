  4 SceneJS.perspective = function() {
  5     var cfg = SceneJS._utils.getNodeConfig(arguments);
  6     var projectionBackend = SceneJS._backends.getBackend('projection');
  7 
  8     /* Memoization levels
  9      */
 10     const NO_MEMO = 0;              // No memoization, assuming that node's configuration is dynamic
 11     const FIXED_CONFIG = 1;         // Node config is fixed, memoizing matrix
 12 
 13     return SceneJS._utils.createNode(
 14             "perspective",
 15             cfg.children,
 16 
 17             new (function() {
 18                 var _memoLevel = NO_MEMO;
 19 
 20                 var _fovy = 60.0;
 21                 var _aspect = 1.0;
 22                 var _near = 0.1;
 23                 var _far = 400.0;
 24 
 25                 var _transform;
 26 
 27                 this.setFovY = function(fovy) {
 28                     _fovy = fovy;
 29                     _memoLevel = NO_MEMO;
 30                 };
 31 
 32                 this.getFovyY = function() {
 33                     return _fovy;
 34                 };
 35 
 36                 this.setAspect = function(aspect) {
 37                     _aspect = aspect;
 38                     _memoLevel = NO_MEMO;
 39                 };
 40 
 41                 this.getAspect = function() {
 42                     return _aspect;
 43                 };
 44 
 45                 this.setNear = function(near) {
 46                     _near = near;
 47                     _memoLevel = NO_MEMO;
 48                 };
 49 
 50                 this.getNear = function() {
 51                     return _near;
 52                 };
 53 
 54                 this.setFar = function(far) {
 55                     _far = far;
 56                     _memoLevel = NO_MEMO;
 57                 };
 58 
 59                 this.getFar = function() {
 60                     return _far;
 61                 };
 62 
 63                 this._init = function(params) {
 64                     if (params.fovy) {
 65                         this.setFovY(params.fovy);
 66                     }
 67                     if (params.aspect) {
 68                         this.setAspect(params.aspect);
 69                     }
 70                     if (params.near) {
 71                         this.setNear(params.near);
 72                     }
 73                     if (params.far) {
 74                         this.setFar(params.far);
 75                     }
 76                 };
 77 
 78                 if (cfg.fixed) {
 79                     this._init(cfg.getParams());
 80                 }
 81 
 82                 this._render = function(traversalContext, data) {
 83                     if (_memoLevel == NO_MEMO) {
 84                         if (!cfg.fixed) {
 85                             this._init(cfg.getParams(data));
 86                         } else {
 87                             _memoLevel = FIXED_CONFIG;
 88                         }
 89                         var tempMat = SceneJS_math_perspectiveMatrix4(
 90                                 _fovy * Math.PI / 180.0,
 91                                 _aspect,
 92                                 _near,
 93                                 _far);
 94                         _transform = {
 95                             type: "perspective",
 96                             matrix:tempMat
 97                         };
 98                     }
 99                     var prevTransform = projectionBackend.getTransform();
100                     projectionBackend.setTransform(_transform);
101                     SceneJS._utils.visitChildren(cfg, traversalContext, data);
102                     projectionBackend.setTransform(prevTransform);
103                 };
104             })());
105 };
