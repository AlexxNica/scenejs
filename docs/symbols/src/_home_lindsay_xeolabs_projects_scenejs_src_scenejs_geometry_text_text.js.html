1 /**
  2  * @class A scene node that defines vector text.
  3  * <p>.</p>
  4  * <p><b>Example Usage</b></p><p>Definition of text:</b></p><pre><code>
  5  * var c = new SceneJS.Text({
  6  *          text : "in morning sunlight\nrising steam from the cats yawn\n a smell of salmon"
  7  *     })
  8  * </pre></code>
  9  * @extends SceneJS.Geometry
 10  * @constructor
 11  * Create a new SceneJS.Text
 12  * @param {Object} config  Config object or function, followed by zero or more child nodes
 13  */
 14 SceneJS.Text = function() {
 15     SceneJS.Geometry.apply(this, arguments);
 16     this._nodeType = "text";
 17     if (this._fixedParams) {
 18         this._init(this._getParams());
 19     }
 20 };
 21 
 22 SceneJS._inherit(SceneJS.Text, SceneJS.Geometry);
 23 
 24 // @private
 25 SceneJS.Text.prototype._init = function(params) {
 26 
 27     /* Callback that creates the text geometry
 28      */
 29     this._create = function() {
 30         var geo = SceneJS_vectorTextModule.getGeometry(1, 0, 0, params.text); // Unit size
 31         return {
 32             primitive : "lines",
 33             positions : geo.positions,
 34             normals: [],
 35             uv : [],
 36             indices : geo.indices,
 37             colors:[]
 38         };
 39     };
 40 };
 41 
 42 /** Returns a new SceneJS.Text instance
 43  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Text constructor
 44  * @returns {SceneJS.Text}
 45  */
 46 SceneJS.text = function() {
 47     var n = new SceneJS.Text();
 48     SceneJS.Text.prototype.constructor.apply(n, arguments);
 49     return n;
 50 };
