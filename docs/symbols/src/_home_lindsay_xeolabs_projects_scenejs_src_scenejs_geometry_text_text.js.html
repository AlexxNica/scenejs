1 /**
  2  * @class A scene node that defines vector text.
  3  * <p>.</p>
  4  * <p><b>Example Usage</b></p><p>Definition of text:</b></p><pre><code>
  5  * var c = new SceneJS.Text({
  6  *          text : "in morning sunlight\nrising steam from the cats yawn\n a smell of salmon"
  7  *     })
  8  * </pre></code>
  9  * @extends SceneJS.Geometry
 10  * @since Version 0.7.4
 11  * @constructor
 12  * Create a new SceneJS.Text
 13  * @param {Object} [cfg] Static configuration object
 14  * @param {String} cfg.text The string of text
 15  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 16  * @param {...SceneJS.Node} [childNodes] Child nodes
 17  */
 18 SceneJS.Text = function() {
 19     SceneJS.Geometry.apply(this, arguments);
 20     this._nodeType = "text";
 21     if (this._fixedParams) {
 22         this._init(this._getParams());
 23     }
 24 };
 25 
 26 SceneJS._inherit(SceneJS.Text, SceneJS.Geometry);
 27 
 28 // @private
 29 SceneJS.Text.prototype._init = function(params) {
 30 
 31     /* Callback that creates the text geometry
 32      */
 33     this._create = function() {
 34         var geo = SceneJS_vectorTextModule.getGeometry(1, 0, 0, params.text); // Unit size
 35         return {
 36             primitive : "lines",
 37             positions : geo.positions,
 38             normals: [],
 39             uv : [],
 40             indices : geo.indices,
 41             colors:[]
 42         };
 43     };
 44 };
 45 
 46 /** Returns a new SceneJS.Text instance
 47  * @param {Object} [cfg] Static configuration object
 48  * @param {String} cfg.text The string of text
 49  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 50  * @param {...SceneJS.Node} [childNodes] Child nodes
 51  * @returns {SceneJS.Text}
 52  *  @since Version 0.7.3
 53  */
 54 SceneJS.text = function() {
 55     var n = new SceneJS.Text();
 56     SceneJS.Text.prototype.constructor.apply(n, arguments);
 57     return n;
 58 };
