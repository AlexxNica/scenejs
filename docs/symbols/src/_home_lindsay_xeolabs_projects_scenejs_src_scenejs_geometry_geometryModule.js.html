1 /**
  2  * Services geometry node requests to store and render elements of geometry.
  3  *
  4  * Stores geometry in vertex buffers in video RAM, caching them there under a least-recently-used eviction policy
  5  * mediated by the "memory" backend.
  6  *
  7  * Geometry elements are identified by type IDs, which may either be supplied by scene nodes, or automatically
  8  * generated by this backend.
  9  *
 10  * After creating geometry, the backend returns to the node the type ID for the node to retain. The node
 11  * can then pass in the type ID to test if the geometry still exists (perhaps it has been evicted) or to have the
 12  * backend render the geometry.
 13  *
 14  * The backend is free to evict whatever geometry it chooses between scene traversals, so the node must always check
 15  * the existence of the geometry and possibly request its re-creation each time before requesting the backend render it.
 16  *
 17  * A geometry buffer consists of positions, normals, optional texture coordinates, indices and a primitive type
 18  * (eg. "triangles").
 19  *
 20  * When rendering a geometry element, the backend will first fire a GEOMETRY_UPDATED to give the shader backend a
 21  * chance to prepare a shader script to render the geometry for current scene state. Then it will fire a SHADER_ACTIVATE
 22  * to prompt the shader backend to fire a SHADER_ACTIVATED to marshal resources from various backends (including this one)
 23  * for its shader script variables, which then provide their resources to the shader through XXX_EXPORTED events.
 24  * This backend then likewise provides its geometry buffers to the shader backend through a GEOMETRY_EXPORTED event,
 25  * then bind and draw the index buffer.
 26  *
 27  * The backend avoids needlessly re-exporting and re-binding geometry (eg. when rendering a bunch of cubes in a row)
 28  * by tracking the type of the last geometry rendered. That type is maintained until another either geoemetry is rendered,
 29  * the canvas switches, shader deactivates or scene deactivates.
 30  *
 31  *  @private
 32 
 33  */
 34 var SceneJS_geometryModule = new (function() {
 35 
 36     var time = (new Date()).getTime();  // For LRU caching
 37     var canvas;
 38     var geoMaps = {};                   // Geometry map for each canvas
 39     var currentGeoMap = null;
 40     var currentBoundGeoType;            // Type of geometry currently bound to shader
 41 
 42     SceneJS_eventModule.onEvent(
 43             SceneJS_eventModule.TIME_UPDATED,
 44             function(t) {
 45                 time = t;
 46             });
 47 
 48     SceneJS_eventModule.onEvent(
 49             SceneJS_eventModule.SCENE_ACTIVATED,
 50             function() {
 51                 canvas = null;
 52                 currentGeoMap = null;
 53                 currentBoundGeoType = null;
 54             });
 55 
 56     SceneJS_eventModule.onEvent(
 57             SceneJS_eventModule.CANVAS_ACTIVATED,
 58             function(c) {
 59                 if (!geoMaps[c.canvasId]) {      // Lazy-create geometry map for canvas
 60                     geoMaps[c.canvasId] = {};
 61                 }
 62                 canvas = c;
 63                 currentGeoMap = geoMaps[c.canvasId];
 64                 currentBoundGeoType = null;
 65             });
 66 
 67     SceneJS_eventModule.onEvent(
 68             SceneJS_eventModule.CANVAS_DEACTIVATED,
 69             function() {
 70                 canvas = null;
 71                 currentGeoMap = null;
 72                 currentBoundGeoType = null;
 73             });
 74 
 75     SceneJS_eventModule.onEvent(
 76             SceneJS_eventModule.SHADER_ACTIVATED,
 77             function() {
 78                 currentBoundGeoType = null;
 79             });
 80 
 81     SceneJS_eventModule.onEvent(
 82             SceneJS_eventModule.SHADER_DEACTIVATED,
 83             function() {
 84                 currentBoundGeoType = null;
 85             });
 86 
 87     SceneJS_eventModule.onEvent(
 88             SceneJS_eventModule.RESET,
 89             function() {
 90                 for (var canvasId in geoMaps) {    // Destroy geometries on all canvases
 91                     var geoMap = geoMaps[canvasId];
 92                     for (var type in geoMap) {
 93                         var geometry = geoMap[type];
 94                         destroyGeometry(geometry);
 95                     }
 96                 }
 97                 canvas = null;
 98                 geoMaps = {};
 99                 currentGeoMap = null;
100                 currentBoundGeoType = null;
101             });
102 
103     /**
104      * Destroys geometry, returning true if memory freed, else false
105      * where canvas not found and geometry was implicitly destroyed
106      * @private
107      */
108     function destroyGeometry(geo) {
109         //  SceneJS_loggingModule.debug("Destroying geometry : '" + geo.type + "'");
110         if (geo.type == currentBoundGeoType) {
111             currentBoundGeoType = null;
112         }
113         if (document.getElementById(geo.canvas.canvasId)) { // Context won't exist if canvas has disappeared
114             if (geo.vertexBuf) {
115                 geo.vertexBuf.destroy();
116             }
117             if (geo.normalBuf) {
118                 geo.normalBuf.destroy();
119             }
120             if (geo.normalBuf) {
121                 geo.indexBuf.destroy();
122             }
123             if (geo.uvBuf) {
124                 geo.uvBuf.destroy();
125             }
126             if (geo.uvBuf2) {
127                 geo.uvBuf2.destroy();
128             }
129         }
130         var geoMap = geoMaps[geo.canvas.canvasId];
131         if (geoMap) {
132             geoMap[geo.type] = null;
133         }
134     }
135 
136     /**
137      * Volunteer to attempt to destroy a geometry when asked to by memory module
138      *
139      */
140     SceneJS_memoryModule.registerEvictor(
141             function() {
142                 var earliest = time;
143                 var evictee;
144                 for (var canvasId in geoMaps) {
145                     var geoMap = geoMaps[canvasId];
146                     if (geoMap) {
147                         for (var type in geoMap) {
148                             var geometry = geoMap[type];
149                             if (geometry) {
150                                 if (geometry.lastUsed < earliest
151                                         && document.getElementById(geometry.canvas.canvasId)) { // Canvas must still exist
152                                     evictee = geometry;
153                                     earliest = geometry.lastUsed;
154                                 }
155                             }
156                         }
157                     }
158                 }
159                 if (evictee) {
160                     SceneJS_loggingModule.warn("Evicting geometry from memory: " + evictee.type);
161                     destroyGeometry(evictee);
162                     return true;
163                 }
164                 return false;  // Couldnt find a geometry we can delete
165             });
166 
167     /**
168      * Creates an array buffer
169      *
170      * @private
171      * @param context WebGL context
172      * @param bufType Eg. ARRAY_BUFFER
173      * @param values WebGL array
174      * @param numItems
175      * @param itemSize
176      * @param usage Eg. STATIC_DRAW
177      */
178     function createArrayBuffer(description, context, bufType, values, numItems, itemSize, usage) {
179         var buf;
180         SceneJS_memoryModule.allocate(
181                 description,
182                 function() {
183                     buf = new SceneJS_webgl_ArrayBuffer(context, bufType, values, numItems, itemSize, usage);
184                 });
185         return buf;
186     }
187 
188     /**
189      * Converts SceneJS primitive type string to WebGL constant
190      * @private
191      */
192     function getPrimitiveType(context, primitive) {
193         switch (primitive) {
194             case "points":
195                 return context.POINTS;
196             case "lines":
197                 return context.LINES;
198             case "line-loop":
199                 return context.LINE_LOOP;
200             case "line-strip":
201                 return context.LINE_STRIP;
202             case "triangles":
203                 return context.TRIANGLES;
204             case "triangle-strip":
205                 return context.TRIANGLE_STRIP;
206             case "triangle-fan":
207                 return context.TRIANGLE_FAN;
208             default:
209                 SceneJS_errorModule.fatalError(new SceneJS.InvalidNodeConfigException(// Logs and throws
210                         "SceneJS.geometry primitive unsupported: '" +
211                         primitive +
212                         "' - supported types are: 'points', 'lines', 'line-loop', " +
213                         "'line-strip', 'triangles', 'triangle-strip' and 'triangle-fan'"));
214         }
215     }

218     /**
219      * Tests if the given geometry type exists on the currently active canvas
220      * @private
221      */
222     this.testGeometryExists = function(type) {
223         return currentGeoMap[type] ? true : false;
224     };
225 
226     /**
227      * Creates geometry on the active canvas - can optionally take a type ID. On success, when ID given
228      * will return that ID, else if no ID given, will return a generated one.
229      * @private
230      */
231     this.createGeometry = function(type, data) {
232         if (!type) {
233             type = SceneJS._createKeyForMap(currentGeoMap, "type");
234         }
235 
236         //   SceneJS_loggingModule.debug("Creating geometry: '" + type + "'");
237 
238         if (!data.primitive) { // "points", "lines", "line-loop", "line-strip", "triangles", "triangle-strip" or "triangle-fan"
239             SceneJS_errorModule.fatalError(
240                     new SceneJS.NodeConfigExpectedException(
241                             "SceneJS.geometry node property expected : primitive"));
242         }
243         var context = canvas.context;
244         var usage = context.STATIC_DRAW;
245         //var usage = (!data.fixed) ? context.STREAM_DRAW : context.STATIC_DRAW;
246 
247         var vertexBuf;
248         var normalBuf;
249         var uvBuf;
250         var uvBuf2;
251         var indexBuf;
252 
253         try { // TODO: Modify usage flags in accordance with how often geometry is evicted
254 
255             vertexBuf = createArrayBuffer("geometry vertex buffer", context, context.ARRAY_BUFFER,
256                     new WebGLFloatArray(data.positions), data.positions.length, 3, usage);
257 
258             if (data.normals && data.normals.length > 0) {
259                 normalBuf = createArrayBuffer("geometry normal buffer", context, context.ARRAY_BUFFER,
260                         new WebGLFloatArray(data.normals), data.normals.length, 3, usage);
261             }
262 
263             if (data.uv && data.uv.length > 0) {
264                 if (data.uv) {
265                     uvBuf = createArrayBuffer("geometry UV buffer", context, context.ARRAY_BUFFER,
266                             new WebGLFloatArray(data.uv), data.uv.length, 2, usage);
267                 }
268             }
269 
270             if (data.uv2 && data.uv2.length > 0) {
271                 if (data.uv2) {
272                     uvBuf2 = createArrayBuffer("geometry UV2 buffer", context, context.ARRAY_BUFFER,
273                             new WebGLFloatArray(data.uv2), data.uv2.length, 2, usage);
274                 }
275             }
276 
277             indexBuf = createArrayBuffer("geometry index buffer", context, context.ELEMENT_ARRAY_BUFFER,
278                     new WebGLUnsignedShortArray(data.indices), data.indices.length, 1, usage);
279 
280             var geo = {
281                 fixed : true, // TODO: support dynamic geometry
282                 primitive: getPrimitiveType(context, data.primitive),
283                 type: type,
284                 lastUsed: time,
285                 canvas : canvas,
286                 context : context,
287                 vertexBuf : vertexBuf,
288                 normalBuf : normalBuf,
289                 indexBuf : indexBuf,
290                 uvBuf: uvBuf,
291                 uvBuf2: uvBuf2
292             };
293             currentGeoMap[type] = geo;
294             return type;
295         } catch (e) { // Allocation failure - delete whatever buffers got allocated
296 
297             if (vertexBuf) {
298                 vertexBuf.destroy();
299             }
300             if (normalBuf) {
301                 normalBuf.destroy();
302             }
303             if (uvBuf) {
304                 uvBuf.destroy();
305             }
306             if (uvBuf2) {
307                 uvBuf2.destroy();
308             }
309             if (indexBuf) {
310                 indexBuf.destroy();
311             }
312             throw e;
313         }
314     };
315 
316     /**
317      * Draws the geometry of the given ID that exists on the current canvas.
318      * Client node must ensure prior that the geometry exists on the canvas
319      * using findGeometry, and have created it if neccessary with createGeometry.
320      * @private
321      */
322     this.drawGeometry = function(type) {
323         if (!canvas) {
324             SceneJS_errorModule.fatalError(SceneJS.NoCanvasActiveException("No canvas active"));
325         }
326         var geo = currentGeoMap[type];
327 
328         SceneJS_eventModule.fireEvent(SceneJS_eventModule.GEOMETRY_UPDATED, geo);  // Gives shader backend a chance to generate a shader
329 
330         /* Prompt shader backend to in turn prompt for exports from all backends.
331          * This backend exports proactively however (see below), since it is the one
332          * which prompted the shader backend.
333          */
334         SceneJS_eventModule.fireEvent(SceneJS_eventModule.SHADER_ACTIVATE);
335 
336         geo.lastUsed = time;  // Geometry now not evictable in this scene traversal
337 
338         var context = canvas.context;
339 
340         /* Dont re-export and bind if already the last one exported and bound - this is the case when
341          * we're drawing a batch of the same object, Eg. a bunch of cubes in a row
342          */
343         if (currentBoundGeoType != type) {
344             for (var i = 0; i < 8; i++) {
345                 context.disableVertexAttribArray(i);
346             }
347             SceneJS_eventModule.fireEvent(
348                     SceneJS_eventModule.GEOMETRY_EXPORTED,
349                     geo);
350 
351             geo.indexBuf.bind(); // Bind index buffer
352 
353             currentBoundGeoType = type;
354         }
355 
356         /* Draw geometry
357          */
358         context.drawElements(geo.primitive, geo.indexBuf.numItems, context.UNSIGNED_SHORT, 0);
359         context.flush();
360 
361         /* Don't need to unbind buffers - only one is bound at a time anyway
362          */
363 
364         /* Destroy one-off geometry
365          */
366         //                    if (!geo.fixed) {
367         //                        destroyGeometry(geo);
368         //                        currentBoundGeoType = null;
369         //                    }
370     };
371 })();
