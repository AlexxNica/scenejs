1 SceneJS._utils.ns("SceneJS.objects");
  2 
  3 /**
  4  * Provides a cube geometry node by wrapping a call to the core SceneJS.geometry node.
  5  *
  6  * Cube geometry. This node type takes no parameters, and the cube is fixed to size [-1..1] on each axis,
  7  * so if you want to resize it you should wrap it in a scale node.
  8  */
  9 
 10 SceneJS.objects.cube = function() {
 11 
 12     var cfg = SceneJS._utils.getNodeConfig(arguments || [
 13         {}
 14     ]);
 15 
 16     /* Dynamic config OK, but only applies first time - good for assets
 17      */
 18     var params = cfg.getParams();
 19 
 20     var x = params.xSize || 1;
 21     var y = params.ySize || 1;
 22     var z = params.zSize || 1;
 23 
 24     var solid = true;
 25 
 26     if (params.solid != undefined) {
 27         solid = params.solid;
 28     }
 29 
 30     /* A geometry node is normally configured with arrays of positions, normals, indices etc., but can instead be
 31      * configured with a "create" callback, as demonstrated here, that returns an object containing those arrays.
 32      *
 33      * Every geometry must get a type that globally identifies it within SceneJS. In this case, the
 34      * type is generated from the number of rings and slices.
 35      *
 36      * Since SceneJS caches geometry by type, it will attempt to reuse the geometry if it can find it in the cache,
 37      * otherwise it will use the callback to create it.
 38      */
 39     return SceneJS.geometry({
 40 
 41         type: "cube_" + x + "_" + y + "_" + z,
 42 
 43         /* Callback to create sphere geometry
 44          */
 45         create: function() {
 46             var positions = [
 47                 x, y, z,
 48                 -x, y, z,
 49                 -x,-y, z,
 50                 x,-y, z,
 51                 // v0-v1-v2-v3 front
 52                 x, y, z,
 53                 x,-y, z,
 54                 x,-y,-z,
 55                 x, y,-z,
 56                 // v0-v3-v4-v5 right
 57                 x, y, z,
 58                 x, y,-z,
 59                 -x, y,-z,
 60                 -x, y, z,
 61                 // v0-v5-v6-v1 top
 62                 -x, y, z,
 63                 -x, y,-z,
 64                 -x,-y,-z,
 65                 -x,-y, z,
 66                 // v1-v6-v7-v2 left
 67                 -x,-y,-z,
 68                 x,-y,-z,
 69                 x,-y, z,
 70                 -x,-y, z,
 71                 // v7-v4-v3-v2 bottom
 72                 x,-y,-z,
 73                 -x,-y,-z,
 74                 -x, y,-z,
 75                 x, y,-z
 76             ];   // v4-v7-v6-v5 back
 77 
 78             var normals = [
 79                 0, 0, -1,
 80                 0, 0, -1,
 81                 0, 0, -1,
 82                 0, 0, -1,
 83                 // v0-v1-v2-v3 front
 84                 -1, 0, 0,
 85                 -1, 0, 0,
 86                 -1, 0, 0,
 87                 -1, 0, 0,
 88                 // v0-v3-v4-v5 right
 89                 0, -1, 0,
 90                 0, -1, 0,
 91                 0, -1, 0,
 92                 0, -1, 0,
 93                 // v0-v5-v6-v1 top
 94                 1, 0, 0,
 95                 1, 0, 0,
 96                 1, 0, 0,
 97                 1, 0, 0,
 98                 // v1-v6-v7-v2 left
 99                 0,1, 0,
100                 0,1, 0,
101                 0,1, 0,
102                 0,1, 0,
103                 // v7-v4-v3-v2 bottom
104                 0, 0,1,
105                 0, 0,1,
106                 0, 0,1,
107                 0, 0,1
108             ];    // v4-v7-v6-v5 back
109 
110             var uv = [
111                 x, y,
112                 0, y,
113                 0, 0,
114                 x, 0,
115                 // v0-v1-v2-v3 front
116                 0, y,
117                 0, 0,
118                 x, 0,
119                 x, y,
120                 // v0-v3-v4-v5 right
121                 x, 0,
122                 x, y,
123                 0, y,
124                 0, 0,
125                 // v0-v5-v6-v1 top
126                 x, y,
127                 0, y,
128                 0, 0,
129                 x, 0,
130                 // v1-v6-v7-v2 left
131                 0, 0,
132                 x, 0,
133                 x, y,
134                 0, y,
135                 // v7-v4-v3-v2 bottom
136                 0, 0,
137                 x, 0,
138                 x, y,
139                 0, y
140             ];   // v4-v7-v6-v5 back
141 
142             var indices = [
143                 0, 1, 2,
144                 0, 2, 3,
145                 // front
146                 4, 5, 6,
147                 4, 6, 7,
148                 // right
149                 8, 9,10,
150                 8,10,11,
151                 // top
152                 12,13,14,
153                 12,14,15,
154                 // left
155                 16,17,18,
156                 16,18,19,
157                 // bottom
158                 20,21,22,
159                 20,22,23
160             ] ;  // back
161 
162             return {
163                 primitive : solid ? "triangles" : "lines",
164                 positions : positions,
165                 normals: normals,
166                 uv : uv,
167                 indices : indices,
168                 colors:[]
169             };
170         }
171     });
172 };
