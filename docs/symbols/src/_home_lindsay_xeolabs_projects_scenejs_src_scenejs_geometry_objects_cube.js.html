  3 /**
  4  * @class SceneJS.objects.Cube
  5  * @extends SceneJS.Geometry
  6  * <p>A scene node that defines cube geometry. A Cube may be configured with an optional half-size for each axis. Where
  7  * not specified, the half-size on each axis will be 1 by default. It can also be configured as solid (default),
  8  * to construct it from triangles with normals for shading and one layer of UV coordinates for texture-mapping
  9  * one made of triangles. When not solid, it will be a wireframe drawn as line segments.</p>
 10  * <p><b>Example Usage</b></p><p>Definition of solid cube that is 6 units long on the X axis and 2 units long on the
 11  * Y and Z axis:</b></p><pre><code>
 12  * var c = new SceneJS.objects.Cube({
 13  *          xSize : 3,
 14  *          solid: true // Optional - when true (default) cube is solid, otherwise it is wireframe
 15  *     })
 16  * </pre></code>
 17  * @constructor
 18  * Create a new SceneJS.objects.Cube
 19  * @param {Object} config  Config object or function, followed by zero or more child nodes
 20  */
 21 SceneJS.objects.Cube = function() {
 22     SceneJS.Geometry.apply(this, arguments);
 23     this._nodeType = "cube";
 24     if (this._fixedParams) {
 25         this._init(this._getParams());
 26     }
 27 };
 28 
 29 SceneJS._utils.inherit(SceneJS.objects.Cube, SceneJS.Geometry);

 31 // @private
 32 SceneJS.objects.Cube.prototype._init = function(params) {
 33     var x = params.xSize || 1;
 34     var y = params.ySize || 1;
 35     var z = params.zSize || 1;
 36     var solid = (params.solid != undefined) ? params.solid : true;
 37 
 38     /* Type ID ensures that we reuse any scube that has already been created with
 39      * these parameters instead of wasting memory
 40      */
 41     this._type = "cube_" + x + "_" + y + "_" + z + (solid ? "_solid" : "wire");
 42 
 43     /* Callback that does the creation in case we can't find matching cube to reuse
 44      */
 45     this._create = function() {
 46         var positions = [
 47             x, y, z,
 48             -x, y, z,

 56             // v0-v3-v4-v5 right
 57             x, y, z,
 58             x, y,-z,
 59             -x, y,-z,
 60             -x, y, z,
 61             // v0-v5-v6-v1 top
 62             -x, y, z,
 63             -x, y,-z,
 64             -x,-y,-z,
 65             -x,-y, z,
 66             // v1-v6-v7-v2 left
 67             -x,-y,-z,
 68             x,-y,-z,
 69             x,-y, z,
 70             -x,-y, z,
 71             // v7-v4-v3-v2 bottom
 72             x,-y,-z,
 73             -x,-y,-z,
 74             -x, y,-z,
 75             x, y,-z
 76         ];   // v4-v7-v6-v5 back
 77 
 78         var normals = [
 79             0, 0, -1,
 80             0, 0, -1,
 81             0, 0, -1,
 82             0, 0, -1,
 83             // v0-v1-v2-v3 front
 84             -1, 0, 0,
 85             -1, 0, 0,
 86             -1, 0, 0,
 87             -1, 0, 0,
 88             // v0-v3-v4-v5 right
 89             0, -1, 0,
 90             0, -1, 0,
 91             0, -1, 0,
 92             0, -1, 0,
 93             // v0-v5-v6-v1 top
 94             1, 0, 0,
 95             1, 0, 0,
 96             1, 0, 0,
 97             1, 0, 0,
 98             // v1-v6-v7-v2 left
 99             0,1, 0,
100             0,1, 0,
101             0,1, 0,
102             0,1, 0,
103             // v7-v4-v3-v2 bottom
104             0, 0,1,
105             0, 0,1,
106             0, 0,1,
107             0, 0,1
108         ];    // v4-v7-v6-v5 back
109 
110         var uv = [
111             x, y,
112             0, y,
113             0, 0,
114             x, 0,
115             // v0-v1-v2-v3 front
116             0, y,
117             0, 0,
118             x, 0,
119             x, y,
120             // v0-v3-v4-v5 right
121             x, 0,
122             x, y,
123             0, y,
124             0, 0,
125             // v0-v5-v6-v1 top
126             x, y,
127             0, y,
128             0, 0,
129             x, 0,
130             // v1-v6-v7-v2 left
131             0, 0,
132             x, 0,
133             x, y,
134             0, y,
135             // v7-v4-v3-v2 bottom
136             0, 0,
137             x, 0,
138             x, y,
139             0, y
140         ];   // v4-v7-v6-v5 back
141 
142         var indices = [
143             0, 1, 2,
144             0, 2, 3,
145             // front
146             4, 5, 6,
147             4, 6, 7,
148             // right
149             8, 9,10,
150             8,10,11,
151             // top
152             12,13,14,
153             12,14,15,
154             // left
155             16,17,18,
156             16,18,19,
157             // bottom
158             20,21,22,
159             20,22,23
160         ] ;  // back
161 
162         return {
163             primitive : solid ? "triangles" : "lines",
164             positions : positions,
165             normals: normals,
166             uv : uv,
167             indices : indices,
168             colors:[]
169         };
170     };
171 };
