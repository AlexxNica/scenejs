1 /**
  2  * Scene node that constructs an ortographic projection matrix and sets it on the current shader.
  3  * @private
  4  */
  5 SceneJS.ortho = function() {
  6     var cfg = SceneJS.getNodeConfig(arguments);
  7    
  8     var transform;
  9 
 10     return SceneJS.createNode(
 11             "ortho",
 12             cfg.children,
 13 
 14             new (function() {
 15                 this._render = function(traversalContext, data) {
 16                     if (!transform || !cfg.fixed) {
 17                         var params = cfg.getParams(data);
 18                         var volume = {
 19                             left: params.left || -1.0,
 20                             right: params.right || 1.0,
 21                             bottom: params.bottom || -1.0,
 22                             top: params.top || 1.0,
 23                             near: params.near || 0.1,
 24                             far: params.far || 100.0
 25                         };
 26                         var tempMat = SceneJS_math_orthoMat4c(
 27                                 volume.left,
 28                                 volume.right,
 29                                 volume.bottom,
 30                                 volume.top,
 31                                 volume.near,
 32                                 volume.far
 33                                 );
 34                         transform = {
 35                             type: "ortho",
 36                             matrix: tempMat
 37                         };
 38                     }
 39                     var prevTransform = SceneJS_projectionModule.getTransform();
 40                     SceneJS_projectionModule.setTransform(transform);
 41                     this._renderNodes(traversalContext, data);
 42                     SceneJS_projectionModule.setTransform(prevTransform);
 43                 };
 44             })());
 45 };
