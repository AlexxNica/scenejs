1 /**
  2  * @class A scene node that defines an orthographic projection transformation for the nodes within its subgraph.
  3  *
  4  * <p>Orthographic, or parallel, projections consist of those that involve no perspective correction.
  5  * There is no adjustment for distance from the camera made in these projections, meaning objects on the screen
  6  * will appear the same size no matter how close or far away they are.</p>
  7  *
  8  * <p><b>Example:</b></p><p>Defining orthographic projection, specifying parameters that happen to be the default values.
  9  * The left and right parameters specify the x-coordinate clipping planes, bottom and top specify the y-coordinate
 10  * clipping planes, and near and far specify the distance to the z-coordinate clipping planes.</p><pre><code>
 11  * var p = new SceneJS.Ortho({
 12  *       this._left = -1.0;
 13  *       this._right = 1.0;
 14  *       this._bottom = -1.0;
 15  *       this._top = 1.0;
 16  *       this._near = 0.1;
 17  *       this._far = 1000.0;
 18  *    },
 19  *
 20  *    // ... child nodes
 21  * )
 22  * </pre></code>
 23  *
 24  * @extends SceneJS.Node
 25  * @constructor
 26  * Create a new SceneJS.Ortho
 27  * @param {Object} cfg  Config object or function, followed by zero or more child nodes
 28  */
 29 SceneJS.Ortho = function() {
 30     SceneJS.Node.apply(this, arguments);
 31     this._nodeType = "ortho";
 32     this._left = -1.0;
 33     this._right = 1.0;
 34     this._bottom = -1.0;
 35     this._top = 1.0;
 36     this._near = 0.1;
 37     this._far = 1000.0;
 38     if (this._fixedParams) {
 39         this._init(this._getParams());
 40     }
 41 };
 42 
 43 SceneJS._inherit(SceneJS.Ortho, SceneJS.Node);

211 // @private
212 SceneJS.Ortho.prototype._init = function(params) {
213     this._left = params.left || -1.0;
214     this._bottom = params.bottom || -1.0;
215     this._near = params.near || 0.1;
216     this._right = params.right || 1.00;
217     this._top = params.top || 1.0;
218     this._far = params.far || 1000.0;
219 };
220 
221 // Override
222 SceneJS.Ortho.prototype._render = function(traversalContext, data) {
223     if (this._memoLevel == 0) {
224         if (!this._fixedParams) {
225             this._init(this._getParams(data));
226         } else {
227             this._memoLevel = 1;
228         }
229         var tempMat = SceneJS_math_orthoMat4c(
230                 this._left,
231                 this._right,
232                 this._bottom,
233                 this._top,
234                 this._near,
235                 this._far);
236         this._transform = {
237             type: "ortho",
238             matrix:tempMat
239         };
240     }
241     var prevTransform = SceneJS_projectionModule.getTransform();
242     SceneJS_projectionModule.setTransform(this._transform);
243     this._renderNodes(traversalContext, data);
244     SceneJS_projectionModule.setTransform(prevTransform);
245 };
