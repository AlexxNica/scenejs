1 /** Sets material properties on the current shader for sub-nodes
  2  */
  3 SceneJS.material = function() {
  4     var errorBackend = SceneJS._backends.getBackend('error');
  5     var backend = SceneJS._backends.getBackend('material');
  6 
  7     var cfg = SceneJS._utils.getNodeConfig(arguments);
  8 
  9     return SceneJS._utils.createNode(
 10             "material",
 11             cfg.children,
 12 
 13             new (function() {
 14                 var _material = {
 15                     baseColor : [ 0.0, 0.0, 0.0 ],
 16                     specularColor: [ 0.0,  0.0,  0.0 ],
 17                     specular : 0,
 18                     shine : 0,
 19                     reflect : 0,
 20                     alpha : 1.0,
 21                     emit : 0.0,
 22                     blendMode : "multiply"
 23                 };
 24 
 25                 this.setBaseColor = function(color) {
 26                     _material.baseColor = [
 27                         color.r != undefined ? color.r : 0.0,
 28                         color.g != undefined ? color.g : 0.0,
 29                         color.b != undefined ? color.b : 0.0
 30                     ];
 31                     return this;
 32                 };
 33 
 34                 this.getBaseColor = function() {
 35                     return {
 36                         r: _material.baseColor,
 37                         g: _material.baseColor,
 38                         b: _material.baseColor
 39                     };
 40                 };
 41 
 42                 this.setSpecularColor = function(color) {
 43                     _material.specularColor = [
 44                         color.r != undefined ? color.r : 0.5,
 45                         color.g != undefined ? color.g : 0.5,
 46                         color.b != undefined ? color.b : 0.5
 47                     ];
 48                     return this;
 49                 };
 50 
 51                 this.getSpecularColor = function() {
 52                     return {
 53                         r: _material.specularColor[0],
 54                         g: _material.specularColor[1],
 55                         b: _material.specularColor[2]
 56                     };
 57                 };
 58 
 59                 this.setSpecular = function(specular) {
 60                     _material.specular = specular || 0;
 61                 };
 62 
 63                 this.getSpecular = function() {
 64                     return _material.specular;
 65                 };
 66 
 67                 this.setShine = function(shine) {
 68                     _material.shine = shine || 0;
 69                 };
 70 
 71                 this.getShine = function() {
 72                     return _material.shine;
 73                 };
 74 
 75                 this.setReflect = function(reflect) {
 76                     _material.reflect = reflect || 0;
 77                 };
 78 
 79                 this.getReflect = function() {
 80                     return _material.reflect;
 81                 };
 82 
 83                 this.setEmit = function(emit) {
 84                     _material.emit = emit || 0;
 85                 };
 86 
 87                 this.getEmit = function() {
 88                     return _material.emit;
 89                 };
 90 
 91                 this.setAlpha = function(alpha) {
 92                     _material.alpha = alpha == undefined ? 1.0 : alpha;
 93                 };
 94 
 95                 this.getAlpha = function() {
 96                     return _material.alpha;
 97                 };
 98 
 99                 this.setBlendMode = function(mode) {
100                     if (mode != "add" && mode != "multiply") {
101                         errorBackend.fatalError(new SceneJS.exceptions.InvalidNodeConfigException(
102                                 "SceneJS.material blendMode of unsupported type: '" + mode + "' - should be 'add' or 'multiply'"));
103                     }
104                     _material.blendMode = mode;
105                 };
106 
107                 this.getBlendMode = function() {
108                     return _material.blendMode;
109                 };
110 
111                 this._init = function(params) {
112                     if (params.baseColor) {
113                         this.setBaseColor(params.baseColor);
114                     }
115                     if (params.specularColor) {
116                         this.setSpecularColor(params.specularColor);
117                     }
118                     if (params.specular) {
119                         this.setSpecular(params.specular);
120                     }
121                     if (params.shine) {
122                         this.setShine(params.shine);
123                     }
124                     if (params.reflect) {
125                         this.setReflect(params.reflect);
126                     }
127                     if (params.emit) {
128                         this.setEmit(params.emit);
129                     }
130                     if (params.alpha) {
131                         this.setAlpha(params.alpha);
132                     }
133                     if (params.blendMode) {
134                         this.setAlpha(params.blendMode);
135                     }
136                 };
137 
138                 if (cfg.fixed) {
139                     this._init(cfg.getParams());
140                 }
141 
142                 this._render = function(traversalContext, data) {
143                     if (SceneJS._utils.traversalMode == SceneJS._utils.TRAVERSAL_MODE_PICKING) {
144                         this._renderChildren(traversalContext, data);
145                     } else {
146                         if (!cfg.fixed) {
147                             this._init(cfg.getParams(data));
148                         }
149                         var saveMaterial = backend.getMaterial();
150                         backend.setMaterial(_material);
151                         this._renderChildren(traversalContext, data);
152                         backend.setMaterial(saveMaterial);
153                     }
154                 };
155             })());
156 };
