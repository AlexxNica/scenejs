1 /**
  2  * @class A scene node that defines how light is reflected by the geometry within its subgraph.
  3  * <p> These may be defined anywhere within a scene graph and may be nested. When nested, the properties on an inner material
  4  * node will override those on outer material nodes for the inner node's subgraph. These nodes are to be defined either
  5  * above or below SceneJS.Lights nodes, which provide light for geometry to reflect.</p>
  6  * <p><b>Live Examples</b></p>
  7  * <ul><li><a target = "other" href="http://bit.ly/scenejs-material-example">Example 1</a></li></ul>
  8  * <p><b>Usage Example</b></p><p>A cube illuminated by a directional light source and wrapped
  9  * with material properties that define how it reflects the light.</b></p><pre><code>
 10  * var l = new SceneJS.Lights({
 11  *          sources: [
 12  *              {
 13  *                  type: "dir",
 14  *                  color: { r: 1.0, g: 1.0, b: 0.0 },
 15  *                  diffuse: true,
 16  *                  specular: true,
 17  *                  dir: { x: 1.0, y: 2.0, z: 0.0 } // Direction of light from coordinate space origin
 18  *              }
 19  *          ]
 20  *      },
 21  *
 22  *      new SceneJS.Material({
 23  *              baseColor:      { r: 0.9, g: 0.2, b: 0.2 },
 24  *              specularColor:  { r: 0.9, g: 0.9, b: 0.2 },
 25  *              emit:           0.0,
 26  *              specular:       0.9,
 27  *              shine:          6.0
 28  *          },
 29  *
 30  *          new SceneJS.objects.Cube()
 31  *     )
 32  * )
 33  * </pre></code>
 34  * @extends SceneJS.Node
 35  * @constructor
 36  * Create a new SceneJS.Material
 37  * @param {Object} config The config object or function, followed by zero or more child nodes
 38  *
 39  */
 40 SceneJS.Material = function() {
 41     SceneJS.Node.apply(this, arguments);
 42     this._nodeType = "material";
 43     this._material = {
 44         baseColor : [ 0.0, 0.0, 0.0 ],
 45         specularColor: [ 0.0,  0.0,  0.0 ],
 46         specular : 0,
 47         shine : 0,
 48         reflect : 0,
 49         alpha : 1.0,
 50         emit : 0.0
 51     };
 52     if (this._fixedParams) {
 53         this._init(this._getParams());
 54     }
 55 };
 56 
 57 SceneJS._inherit(SceneJS.Material, SceneJS.Node);

 59 /**
 60  * Sets the material base color
 61  * @function {SceneJS.Material} setBaseColor
 62  * @param {Object} color Eg. { r: 1.0, g: 1.0, b: 0.0 }
 63  * @returns {SceneJS.Material} this
 64  */
 65 SceneJS.Material.prototype.setBaseColor = function(color) {
 66     this._material.baseColor = [
 67         color.r != undefined ? color.r : 0.0,
 68         color.g != undefined ? color.g : 0.0,
 69         color.b != undefined ? color.b : 0.0
 70     ];
 71     return this;
 72 };
 73 
 74 /**
 75  Returns the base color
 76  @function {Object} getBaseColor
 77  @returns {Object} color Eg. { r: 1.0, g: 1.0, b: 0.0 }
 78  */
 79 SceneJS.Material.prototype.getBaseColor = function() {
 80     return {
 81         r: this._material.baseColor,
 82         g: this._material.baseColor,
 83         b: this._material.baseColor
 84     };
 85 };

 87 /**
 88  * Sets the material specular
 89  * @function {SceneJS.Material} setSpecularColor
 90  * @param {Object} color Eg. { r: 1.0, g: 1.0, b: 0.0 }

 92  */
 93 SceneJS.Material.prototype.setSpecularColor = function(color) {
 94     this._material.specularColor = [
 95         color.r != undefined ? color.r : 0.5,

 99     return this;
100 };

215   // @private
216 SceneJS.Material.prototype._init = function(params) {
217     if (params.baseColor) {
218         this.setBaseColor(params.baseColor);
219     }
220     if (params.specularColor) {
221         this.setSpecularColor(params.specularColor);
222     }
223     if (params.specular) {
224         this.setSpecular(params.specular);
225     }
226     if (params.shine) {
227         this.setShine(params.shine);
228     }
229     if (params.reflect) {
230         this.setReflect(params.reflect);
231     }
232     if (params.emit) {
233         this.setEmit(params.emit);
234     }
235     if (params.alpha) {
236         this.setAlpha(params.alpha);
237     }
238 };
239 
240 // @private
241 SceneJS.Material.prototype._render = function(traversalContext, data) {
242     if (SceneJS._traversalMode == SceneJS.TRAVERSAL_MODE_PICKING) {
243         this._renderNodes(traversalContext, data);
244     } else {
245         if (!this._fixedParams) {
246             this._init(this._getParams(data));
247         }
248         var saveMaterial = SceneJS_materialModule.getMaterial();
249         SceneJS_materialModule.setMaterial(this._material);
250         this._renderNodes(traversalContext, data);
251         SceneJS_materialModule.setMaterial(saveMaterial);
252     }
253 };

257  * @returns {SceneJS.Material}

259 SceneJS.material = function() {

261     SceneJS.Material.prototype.constructor.apply(n, arguments);
