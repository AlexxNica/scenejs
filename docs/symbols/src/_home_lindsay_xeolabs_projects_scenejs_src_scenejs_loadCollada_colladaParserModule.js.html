 35     /**
 36      * Parses the given XML string into the xmlDoc
 37      * @private
 38      */
 39     function loadDoc(xml) {
 40         if (window.DOMParser) {
 41             var parser = new DOMParser();
 42             xmlDoc = parser.parseFromString(xml, "text/xml");
 43         }
 44         else { // Internet Explorer
 45             xmlDoc = new ActiveXObject("Microsoft.XMLDOM");
 46             xmlDoc.async = "false";
 47             xmlDoc.loadXML(xml);
 48         }

 51     /**
 52      * Finds every element in the xmlDoc and maps the by IDs into the idMap
 53      * @private
 54      */
 55     function buildIdMap() {
 56         idMap = {};
 57         var elements = xmlDoc.getElementsByTagName("*");
 58         var id;
 59         for (var i = elements.length - 1; i >= 0; i--) {
 60             id = elements[i].getAttribute("id");
 61             if (id != "") {
 62                 idMap[id] = elements[i];
 63             }
 64         }
 65     }

 67     /**
 68      * Parses the xmlDoc, optionally constrained to the subtree identified by rootId
 69      * @private
 70      */
 71     function parseDoc(rootId) {
 72         if (rootId) {
 73             SceneJS_loggingModule.info("Parsing Collada asset '" + rootId + "'");
 74             var root = idMap[rootId];
 75             if (root) {
 76                 return parseNode(root);
 77             } else {
 78                 SceneJS_errorModule.error(new SceneJS.exceptions.ColladaRootNotFoundException(
 79                         "SceneJS.assets.collada root not found in COLLADA document: '" + rootId + "'"));
 80             }
 81         } else {
 82             SceneJS_loggingModule.info("Parsing Collada scene. Asset: " + rootId ? "'" + rootId + "'" : "default");
 83             var scene = xmlDoc.getElementsByTagName("scene");
 84             if (scene.length > 0) {
 85                 return parseNode(scene[0]);
 86             } else {
 87                 SceneJS_errorModule.error(new SceneJS.exceptions.ColladaRootRequiredException(
 88                         "SceneJS.assets.collada root needs to be specified for COLLADA document: " + uri));
 89             }
 90         }
 91     }

 93     // @private
 94     function parseFloatArray(node) {
 95         var result = [];
 96         var prev = "";
 97         var child = node.firstChild;
 98         var currArray;
 99         while (child) {
100             currArray = (prev + child.nodeValue).replace(/\s+/g, " ").replace(/^\s+/g, "").split(" ");
101             child = child.nextSibling;
102             if (currArray[0] == "") {
103                 currArray.unshift();
104             }
105             if (child) {
106                 prev = currArray.pop();
107             }
108             for (var i = 0; i < currArray.length; i++) {
109                 result.push(parseFloat(currArray[i]));
110             }
111         }
112         return result;
113     }

116     /**
117      * Returns the data for either a <vertices> or a <source>
118      *
119      * A <source> declares a data repository that provides values according
120      * to the semantics of an <input> element that refers to it.
121      * @private
122      */
123     function getSource(id) {
124         var source = sources[id];
125         if (source) {
126             return source;
127         }
128         var element = idMap[id];
129         if (element.tagName == "vertices") {
130 
131             /* Recurse to child <source> element
132              */
133             source = getSource(
134                     element
135                             .getElementsByTagName("input")[0]
136                             .getAttribute("source")
137                             .substr(1));
138         } else {
139 
140             /* Element is a <source>
141              */
142             var accessor = element
143                     .getElementsByTagName("technique_common")[0]
144                     .getElementsByTagName("accessor")[0];
145 
146             var stride = parseInt(accessor.getAttribute("stride"));         // Number of values per unit
147             var offset = parseInt(accessor.getAttribute("offset")) || 0;    // Index of first value
148             var count = parseInt(accessor.getAttribute("count"));           // Number of units
149 
150             /* Create mask that indicates what data types are in the
151              * source - int, float, Name, bool and IDREF.
152              *
153              * The number and type of the <param> elements define the
154              * output of the <accessor>. Parameters are bound to values
155              * in the order in which both are specified. A <param> wtihout
156              * a name attribute indicates that the value is not part of the
157              * input.
158              */
159             var params = accessor.getElementsByTagName("param");
160             var typeMask = [];
161             for (var i = 0; i < params.length; i++) {
162                 if (params[i].hasAttribute("name")) {
163                     typeMask.push(true);
164                 } else {
165                     typeMask.push(false);
166                 }
167             }
168 
169             source = {
170                 array:parseFloatArray(idMap[accessor.getAttribute("source").substr(1)]),
171                 stride:stride,
172                 offset:offset,
173                 count:count,
174                 typeMask: typeMask
175             };
176         }
177         sources[id] = source;
178         return source;
179     }

181     // @private
182     function getMaxOffset(inputs) {
183         var maxOffset = 0;
184         for (var n = 0; n < inputs.length; n++) {
185             var offset = inputs[n].getAttribute("offset");
186             if (offset > maxOffset) {
187                 maxOffset = offset;
188             }
189         }
190         return maxOffset;
191     }

193     // @private
194     function getTrianglesFromPolyList(polyList) {
195         var i, j, k;
196         var inputs = polyList.getElementsByTagName("input");
197         var maxOffset = getMaxOffset(inputs);
198         var vcount = parseFloatArray(polyList.getElementsByTagName("vcount")[0]);
199         var faces = parseFloatArray(polyList.getElementsByTagName("p")[0]);         // TODO: parseInt
200         var triangles = [];
201         var base = 0;
202         for (i = 0; i < vcount.length; i++) {
203             for (j = 0; j < vcount[i] - 2; j++) { // For each vertex
204                 for (k = 0; k <= maxOffset; k++) { // A
205                     triangles.push(faces[base + k]);
206                 }
207                 for (k = 0; k <= maxOffset; k++) { // B
208                     triangles.push(faces[base + (maxOffset + 1) * (j + 1) + k]);
209                 }
210                 for (k = 0; k <= maxOffset; k++) { // C
211                     triangles.push(faces[base + (maxOffset + 1) * (j + 2) + k]);
212                 }
213             }
214             base = base + (maxOffset + 1) * vcount[i];
215         }
216         return triangles;
217     }

219     /** Extracts list of triangles from the given <mesh>, merged from both the
220      * <triangles> and <polylist> child nodes of the <mesh>.
221      * @private
222      */
223     function getTrianglesList(geometryNode) {
224         var trianglesList = [];
225         var meshNode = geometryNode.getElementsByTagName("mesh")[0];
226 
227         /* Extract <polylist> children
228          */
229         var polyLists = meshNode.getElementsByTagName("polylist");
230         for (var i = 0; i < polyLists.length; i++) {
231             var polyList = polyLists[i];
232             polyList.getElementsByTagName("p")[0].data = getTrianglesFromPolyList(polyList);
233             trianglesList.push(polyList);
234         }
235 
236         var tris = meshNode.getElementsByTagName("triangles");
237         for (i = 0; i < tris.length; i++) {
238             trianglesList.push(tris[i]);
239         }
240         return trianglesList;
241     }

244     /** Parses a <geometry> and returns an array containing a SceneJS.geometry node for
245      * each <mesh> child
246      * @private
247      * @param id
248      */
249     function getGeometriesData(geometryNode) {
250         var geometriesData = [];
251         var trianglesList = getTrianglesList(geometryNode);
252 
253         for (var it = 0; it < trianglesList.length; it++) {
254             var triangle = trianglesList [it];
255             var inputs = triangle.getElementsByTagName("input");
256             var inputArray = [];
257             var outputData = {};
258 
259             for (var n = 0; n < inputs.length; n++) {
260                 inputs[n].data = getSource(inputs[n].getAttribute("source").substr(1));
261                 var group = inputs[n].getAttribute("semantic");
262                 if (group == "TEXCOORD") {
263                     group = group + inputs[n].getAttribute("set") || 0;
264                 }
265                 inputs[n].group = group;
266                 inputArray[inputs[n].getAttribute("offset")] = inputs[n];
267                 outputData[group] = [];
268             }
269 
270             var faces;
271             if (triangle.getElementsByTagName("p")[0].data) {
272                 faces = triangle.getElementsByTagName("p")[0].data;
273             }
274             else {
275                 faces = parseFloatArray(triangle.getElementsByTagName("p")[0]);
276             }
277 
278             for (var i = 0; i < faces.length; i = i + inputArray.length) {
279                 for (var n = 0; n < inputArray.length; n++) {
280                     var group = inputArray[n].group;
281                     var pCount = 0;
282                     for (var j = 0; j < inputArray[n].data.stride; j++) {
283                         if (inputArray[n].data.typeMask[j]) {
284                             outputData[group].push(
285                                     parseFloat(inputArray[n].data.array[faces[i + n]
286                                             * inputArray[n].data.stride + j
287                                             + inputArray[n].data.offset]));
288                             pCount++;
289                         }
290                     }
291 
292                     /* 1D
293                      */
294                     if (group == "VERTEX" && pCount == 1) {
295                         outputData[group].push(0);
296                     }
297 
298                     /* 2D
299                      */
300                     if (group == "VERTEX" && pCount == 2) {
301                         outputData[group].push(0);
302                     }
303 
304                     /* 2D textures
305                      */
306                     if (group == "TEXCOORD0" && pCount == 3) {
307                         outputData[group].pop();
308                     }
309                     if (group == "TEXCOORD1" && pCount == 3) {
310                         outputData[group].pop();
311                     }
312                 }
313             }
314 
315             faces = [];
316             for (n = 0; n < outputData.VERTEX.length / 3; n++) {
317                 faces.push(n);
318             }
319 
320             geometriesData.push({
321                 materialName : triangle.getAttribute("material"),
322                 positions: outputData.VERTEX,
323                 normals: outputData.NORMAL,
324                 uv : outputData.TEXCOORD0,
325                 uv2 : outputData.TEXCOORD1,
326                 indices: faces
327             });
328         }
329         return geometriesData;
330     }

332     /**
333      * Returns profile/newparam[sid="<sid>"]/sampler2D[0]/source[0].nodeValue
334      * @private
335      */
336     function getSamplerSource(profile, sid) {
337         var params = profile.getElementsByTagName("newparam");
338         for (var i = 0; i < params.length; i++) {
339             if (params[i].getAttribute("sid") == sid) {
340                 return params[i]
341                         .getElementsByTagName("sampler2D")[0]
342                         .getElementsByTagName("source")[0]
343                         .firstChild
344                         .nodeValue;
345             }
346         }
347         SceneJS_errorModule.fatalError(
348                 new SceneJS.exceptions.ColladaParseException
349                         ("COLLADA element expected: "
350                                 + profile.tagName
351                                 + "/newparam[sid == '"
352                                 + sid + "']/sampler2D[0]/source[0]"));
353     }

374                         + sid + "']/surface[0]/init_from[0]"));

379         var source = getSamplerSource(profileCommon, texture.getAttribute("texture"));
380         var imageId = getImageId(profileCommon, source);
381         var image = idMap[imageId];
382         var imageFileName = image.getElementsByTagName("init_from")[0].firstChild.nodeValue;

384         return {

386             applyTo: applyTo,
387             blendMode: (blendMode == "MULTIPLY") ? "multiply" : "add"

391     // @private
392     function getDiffuseMaterialData(profileCommon, technique, materialData) {
393         var diffuse = technique.getElementsByTagName("diffuse");
394         if (diffuse.length > 0) {
395             var child = diffuse[0].firstChild;
396             do{
397                 switch (child.tagName) {
398                     case "color":
399                         var color = child.firstChild.nodeValue.split(" ");
400                         materialData.baseColor = { r:parseFloat(color[0]), g:parseFloat(color[1]), b:parseFloat(color[2]) };
401                         break;
402 
403                     case "texture":
404                         materialData.texturesData.push(
405                                 getTextureData(profileCommon, child, "baseColor"));
406                         break;
407                 }
408             } while (child = child.nextSibling);
409         }
410     }

471         var materialNode = idMap[id];
472         var effectId = materialNode
473                 .getElementsByTagName("instance_effect")[0]
474                 .getAttribute("url")
475                 .substr(1);
476         var effect = idMap[effectId];
477         var profileCommon = effect.getElementsByTagName("profile_COMMON")[0];
478         var technique = profileCommon.getElementsByTagName("technique")[0];
479         var materialData = {
480             texturesData : []
481         };
482         getDiffuseMaterialData(profileCommon, technique, materialData);
483         getSpecularColorMaterialData(profileCommon, technique, materialData);
484         getShininessMaterialData(profileCommon, technique, materialData);
485         getBumpMapMaterialData(profileCommon, technique, materialData);
486         return materialData;

506     // @private
507     function wrapWithBoundingBox(e, child) {
508         if (modes.showBoundingBoxes) {
509             return SceneJS.boundingBox(e, SceneJS.renderer({
510                 lineWidth:2,
511                 enableTexture2D: false
512             }, SceneJS.material({baseColor: { r: 1, g: 0, b: 0 }},
513                     SceneJS.geometry({
514                         primitive: "lines",
515                         positions : [
516                             e.xmax, e.ymax, e.zmax,
517                             e.xmax, e.ymin, e.zmax,
518                             e.xmin, e.ymin, e.zmax,
519                             e.xmin, e.ymax, e.zmax,
520                             e.xmax, e.ymax, e.zmin,
521                             e.xmax, e.ymin, e.zmin,
522                             e.xmin, e.ymin, e.zmin,
523                             e.xmin, e.ymax, e.zmin
524                         ],
525                         indices : [ 0, 1, 1, 2, 2, 3, 3, 0, 4, 5, 5, 6, 6, 7, 7, 4, 0, 4, 1,5, 2, 6,3,7 ]
526                     }))), child);
527         } else {                                   
528             return SceneJS.boundingBox(e, child);
529         }

541    // @private
542     function expandExtentsByPositions(e, positions) {
543         for (var i = 0; i < positions.length - 2; i += 3) {
544             var x = positions[i];
545             var y = positions[i + 1];
546             var z = positions[i + 2];
547             if (x < e.xmin) e.xmin = x;
548             if (y < e.ymin) e.ymin = y;
549             if (z < e.zmin) e.zmin = z;
550             if (x > e.xmax) e.xmax = x;
551             if (y > e.ymax) e.ymax = y;
552             if (z > e.zmax) e.zmax = z;
553         }
554         return e;
555     }

557     // @private
558     function expandExtentsByExtents(e, e2) {
559         if (e2.xmin < e.xmin) e.xmin = e2.xmin;
560         if (e2.ymin < e.ymin) e.ymin = e2.ymin;
561         if (e2.zmin < e.zmin) e.zmin = e2.zmin;
562         if (e2.xmax > e.xmax) e.xmax = e2.xmax;
563         if (e2.ymax > e.ymax) e.ymax = e2.ymax;
564         if (e2.zmax > e.zmax) e.zmax = e2.zmax;
565         return e;
566     }

568     // @private
569     function parseInstanceGeometry(instanceGeometryNode) {
570         var geoUrl = instanceGeometryNode.getAttribute("url").substr(1);
571         var geometryNode = idMap[geoUrl];
572         var geometriesData = getGeometriesData(geometryNode);
573         var materialsData = getMaterialsData(instanceGeometryNode);
574 
575         var nodeArgList = []; // Args for SceneJS node result
576 
577         var extents = newExtents();
578 
579         for (var i = 0; i < geometriesData.length; i++) {
580             var geoData = geometriesData[i];
581             var geoType = uri + ":" + geoUrl + i;
582             var sceneNode = SceneJS.geometry({
583                 type: geoType,
584                 primitive: "triangles",
585                 positions: geoData.positions,
586                 normals: geoData.normals,
587                 uv : geoData.uv,
588                 uv2 : geoData.uv2,
589                 indices: geoData.indices
590             });
591 
592             if (geoData.materialName) {
593                 var materialData = materialsData[geoData.materialName];
594 
595                 /* Wrap in SceneJS.material
596                  */
597                 if (materialData) {
598                     sceneNode = SceneJS.material({
599                         baseColor: materialData.baseColor,
600                         specularColor: materialData.specularColor ,
601                         shine: 10.0,
602                         specular: 1
603                     }, sceneNode);
604 
605                     /* Wrap in SceneJS.texture
606                      */
607                     var textureLayers = materialData.texturesData;
608                     if (textureLayers.length > 0) {
609                         var layers = [];
610                         for (var j = 0; j < textureLayers.length; j++) {
611                             layers.push({
612                                 uri : textureLayers[j].uri,
613                                 applyTo: textureLayers[j].applyTo,
614                                 flipY : false,
615                                 blendMode: textureLayers[j].blendMode,
616 
617                                 wrapS: "repeat",
618                                 wrapT: "repeat" ,
619                                 minFilter: "linearMipMapLinear",
620                                 magFilter: "linear"
621                             });
622                         }
623                         sceneNode = SceneJS.texture({ layers: layers }, sceneNode);
624                     }
625                 }
626             }
627             nodeArgList.push(wrapWithBoundingBox(
628                     expandExtentsByPositions(newExtents(), geoData.positions),
629                     sceneNode));
630         }
631         return SceneJS.node.apply(this, nodeArgList);
632     }

661     /** Parses data from camera node
662      * @private
663      */
664     function parseCameraOptics(camera) {
665         var optics = camera.getElementsByTagName("optics")[0];
666         var techniqueCommon = optics.getElementsByTagName("technique_common")[0];
667         var perspective = techniqueCommon.getElementsByTagName("perspective")[0];
668 
669         var opticsData = {};
670 
671         if (perspective) {
672             var yfov = perspective.getElementsByTagName("yfov")[0];
673             var aspectRatio = perspective.getElementsByTagName("aspect_ratio")[0];
674             var znear = perspective.getElementsByTagName("znear")[0];
675             var zfar = perspective.getElementsByTagName("zfar")[0];
676 
677             opticsData = {
678                 perspective: {
679                     fovy: yfov ? parseFloat(yfov.textContent) : 60.0,
680                     aspect: aspectRatio ? parseFloat(aspectRatio.textContent) : 1.0,
681                     near: znear ? parseFloat(znear.textContent) : 0.1,
682                     far: zfar ? parseFloat(zfar.textContent) : 10000.0
683                 }
684             };
685 
686 //            SceneJS_loggingModule.info("fovy = " + opticsData.perspective.fovy);
