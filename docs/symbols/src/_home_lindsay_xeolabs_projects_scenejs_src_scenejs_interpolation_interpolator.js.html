1 /**
  2  * @class A scene node that animates interpolates a scalar value by interpolating within a sequence of key values.
  3  * <p>This nodes reads an <i>alpha</i> value from the current data scope and writes the output to a child data scope
  4  * for nodes in its subgraph to configure themselves with.</p>
  5  * <p><b>Live Examples</b></p>
  6  * <ul><li><a target = "other" href="http://bit.ly/scenejs-scalarinterpolator-example">Example 1</a></li></ul>
  7  * <p><b>Example Usage</b></p><p>This example defines a {@link SceneJS.objects.Cube} with rotation that is animated by
  8  * a SceneJS.Interpolator, which is in turn driven by an alpha value supplied by a higher {@link SceneJS.WithData}.
  9  * If we thought of <em>alpha</em> as elapsed seconds, then this cube will rotate 360 degrees over one second, then
 10  * rotate 180 in the reverse direction over the next 0.5 seconds. In this example however, the alpha is actually fixed,
 11  * where the cube is stuck at 180 degrees - you would need to vary the "alpha" property on the WithData node to actually
 12  * animate it.</p><pre><code>
 13  * var wd = new SceneJS.WithData({ "alpha" : 0.5 }, // Interpolates the rotation to 180 degrees
 14  *
 15  *      new SceneJS.Interpolator({
 16  *              type:"linear",   // or 'cosine', 'cubic' or 'constant'
 17  *              input:"alpha",
 18  *              output:"angle",
 19  *              keys: [0.0, 1.0, 1.5],
 20  *              values: [0.0, 360.0, 180.0]
 21  *          },
 22  *
 23  *          new SceneJS.Rotate(function(data) {
 24  *                 return { angle : data.get("angle"), y: 1.0 };
 25  *              },
 26  *
 27  *                  new SceneJS.objects.Cube()
 28  *              )
 29  *          )
 30  *      )
 31  *
 32  *  // Bump the rotation along a notch:
 33  *
 34  *  wd.setProperty("alpha", 0.6);
 35  *
 36  *  </pre></code>
 37  *
 38  * @extends SceneJS.Node
 39  * @since Version 0.7.4
 40  * @constructor
 41  * Create a new SceneJS.Interpolator
 42  * @param {Object} [cfg] Static configuration object
 43  * @param {String} [cfg.type="linear"] Interpolation type - "linear", "cosine", "cubic" or "constant"
 44  * @param {String} cfg.input Name of property on {@link SceneJS.Data} scope that will supply the interpolation <em>alpha</em> value
 45  * @param {String} cfg.output Name of property to create on child {@link SceneJS.Data} scope that provide the output value
 46  * @param {double[]} [cfg.keys=[]] Alpha key values
 47  * @param {double[]} [cfg.values=[]] Output key values
 48  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
 49  * @param {...SceneJS.Node} [childNodes] Child nodes
 50  */
 51 SceneJS.Interpolator = function() {
 52     SceneJS.Node.apply(this, arguments);
 53     this._nodeType = "interpolator";
 54     this._input = null;
 55     this._output = null;
 56     this._outputValue = null;
 57     this._keys = null;
 58     this._values = null;
 59     this._key1 = 0;
 60     this._key2 = 0;
 61     this._type = null;
 62 };
 63 
 64 SceneJS._inherit(SceneJS.Interpolator, SceneJS.Node);
 65 
 66 /* Interpolator attempts to track the pair of keys that enclose the current alpha value -
 67  * these are the node's current states with regard to that:
 68  */
 69 
 70 // @private
 71 SceneJS.Interpolator.prototype._NOT_FOUND = 0;        // Alpha outside of key sequence
 72 
 73 // @private
 74 SceneJS.Interpolator.prototype._BEFORE_FIRST = 1;     // Alpha before first key
 75 
 76 // @private
 77 SceneJS.Interpolator.prototype._AFTER_LAST = 2;       // Alpha after last key
 78 
 79 // @private
 80 SceneJS.Interpolator.prototype._FOUND = 3;            // Found keys before and after alpha
 81 
 82 // @private
 83 SceneJS.Interpolator.prototype._linearInterpolate = function(k) {
 84     var u = this._keys[this._key2] - this._keys[this._key1];
 85     var v = k - this._keys[this._key1];
 86     var w = this._values[this._key2] - this._values[this._key1];
 87     return this._values[this._key1] + ((v / u) * w);
 88 };
 89 
 90 // @private
 91 SceneJS.Interpolator.prototype._constantInterpolate = function(k) {
 92     if (Math.abs((k - this._keys[this._key1])) < Math.abs((k - this._keys[this._key2]))) {
 93         return this._keys[this._key1];
 94     } else {
 95         return this._keys[this._key2];
 96     }
 97 };
 98 
 99 // @private
100 SceneJS.Interpolator.prototype._cosineInterpolate = function(k) {
101     var mu2 = (1 - Math.cos(k * Math.PI) / 2.0);
102     return (this._keys[this._key1] * (1 - mu2) + this._keys[this._key2] * mu2);
103 };
104 
105 // @private
106 SceneJS.Interpolator.prototype._cubicInterpolate = function(k) {
107     if (this._key1 == 0 || this._key2 == (this._keys.length - 1)) {
108 
109         /* Between first or last pair of keyframes - need four keyframes for cubic, so fall back on cosine
110          */
111         return this._cosineInterpolate(k);
112     }
113     var y0 = this._keys[this._key1 - 1];
114     var y1 = this._keys[this._key1];
115     var y2 = this._keys[this._key2];
116     var y3 = this._keys[this._key2 + 1];
117     var mu2 = k * k;
118     var a0 = y3 - y2 - y0 + y1;
119     var a1 = y0 - y1 - a0;
120     var a2 = y2 - y0;
121     var a3 = y1;
122     return (a0 * k * mu2 + a1 * mu2 + a2 * k + a3);
123 };
124 
125 // @private
126 SceneJS.Interpolator.prototype._slerp = function(k) {
127     var u = this._keys[this._key2] - this._keys[this._key1];
128     var v = k - this._keys[this._key1];  
129     return SceneJS_math_slerp((v / u), this._values[this._key1], this._values[this._key2]);
130 };
131 
132 // @private
133 SceneJS.Interpolator.prototype._findEnclosingFrame = function(key) {
134     if (this._keys.length == 0) {
135         return this._NOT_FOUND;
136     }
137     if (key < this._keys[0]) {
138         return this._BEFORE_FIRST;
139     }
140     if (key > this._keys[this._keys.length - 1]) {
141         return this._AFTER_LAST;
142     }
143     while (this._keys[this._key1] > key) {
144         this._key1--;
145         this._key2--;
146     }
147     while (this._keys[this._key2] < key) {
148         this._key1++;
149         this._key2++;
150     }
151     return this._FOUND;
152 };
153 
154 // @private
155 SceneJS.Interpolator.prototype._interpolate = function(k) {
156     switch (this._type) {
157         case 'linear':
158             return this._linearInterpolate(k);
159         case 'cosine':
160             return this._cosineInterpolate(k);
161         case 'cubic':
162             return this._cubicInterpolate(k);
163         case 'constant':
164             return this._constantInterpolate(k);
165         case 'slerp':
166             return this._slerp(k);
167         default:
168             throw SceneJS_errorModule.fatalError(
169                     new SceneJS.InternalException("SceneJS.Interpolator internal error - interpolation type not switched: '"
170                             + this._type + "'"));
171     }
172 };
173 
174 // @private
175 SceneJS.Interpolator.prototype._update = function(key) {
176     switch (this._findEnclosingFrame(key)) {
177         case this._NOT_FOUND:
178             break;
179         case this._BEFORE_FIRST:
180             break; // time delay before interpolation begins
181         case this._AFTER_LAST:
182             this._outputValue = this._values[this._values.length - 1];
183             break;
184         case this._FOUND:
185             this._outputValue = this._interpolate((key));
186             break;
187         default:
188             break;
189     }
190 };
191 
192 // @private
193 SceneJS.Interpolator.prototype._init = function(params) {
194 
195     /* Name of input property in data scope
196      */
197     if (!params.input) {
198         throw SceneJS_errorModule.fatalError(
199                 new SceneJS.NodeConfigExpectedException(
200                         "SceneJS.Interpolator config property expected: input"));
201     }
202     this._input = params.input;
203 
204     /* Name of output property on child data scope
205      */
206     if (!params.output) {
207         throw SceneJS_errorModule.fatalError(
208                 new SceneJS.NodeConfigExpectedException(
209                         "SceneJS.Interpolator config property expected: output"));
210     }
211     this._output = params.output;
212     this._outputValue = null;
213 
214     /* Keys and values
215      */
216     if (params.keys) {
217         if (!params.values) {
218             throw SceneJS_errorModule.fatalError(
219                     new SceneJS.InvalidNodeConfigException(
220                             "SceneJS.Interpolator configuration incomplete: " +
221                             "keys supplied but no values - must supply a value for each key"));
222         }
223     } else if (params.values) {
224         throw SceneJS_errorModule.fatalError(
225                 new SceneJS.InvalidNodeConfigException(
226                         "SceneJS.Interpolator configuration incomplete: " +
227                         "values supplied but no keys - must supply a key for each value"));
228     }
229     for (var i = 1; i < params.keys.length; i++) {
230         if (params.keys[i - 1] >= params.keys[i]) {
231             throw SceneJS_errorModule.fatalError(
232                     new SceneJS.InvalidNodeConfigException(
233                             "SceneJS.Interpolator configuration invalid: " +
234                             "two invalid keys found ("
235                                     + i - 1 + " and " + i + ") - key list should contain distinct values in ascending order"));
236         }
237     }
238     this._keys = params.keys;
239     this._values = params.values;
240     this._key1 = 0;
241     this._key2 = 1;
242 
243     /* Interpolation type
244      */
245     params.type = params.type || 'linear';
246     switch (params.type) {
247         case 'linear':
248             break;
249         case 'constant':
250             break;
251         case 'cosine':
252             break;
253         case 'cubic':
254             if (params.keys.length < 4) {
255                 throw SceneJS_errorModule.fatalError(
256                         new SceneJS.InvalidNodeConfigException(
257                                 "SceneJS.Interpolator configuration invalid: minimum of four keyframes " +
258                                 "required for cubic - only "
259                                         + params.keys.length
260                                         + " are specified"));
261             }
262             break;
263         case 'slerp':
264             break;
265         default:
266             throw SceneJS_errorModule.fatalError(
267                     new SceneJS.InvalidNodeConfigException(
268                             "SceneJS.Interpolator configuration invalid:  type not supported - " +
269                             "only 'linear', 'cosine', 'cubic', 'constant' and 'slerp' are supported"));
270         /*
271 
272 
273          case 'hermite':
274          break;
275          */
276     }
277     this._type = params.type;
278 };
279 
280 // @private
281 SceneJS.Interpolator.prototype._render = function(traversalContext, data) {
282     if (!this.type) {
283         this._init(this._getParams(data));
284     }
285     var key = data.get(this._input);
286     if (key == undefined || key == null) {
287         throw SceneJS_errorModule.fatalError(
288                 new SceneJS.DataExpectedException(
289                         "SceneJS.Interpolator failed to find input on data: '" + params.input + "'"));
290     }
291     this._update(key);
292     var obj = {};
293     obj[this._output] = this._outputValue;
294     this._renderNodes(traversalContext, new SceneJS.Data(data, false, obj));
295 };

298 /**  Factory function that returns a new {@link SceneJS.Interpolator} instance
299  * @param {Object} [cfg] Static configuration object
300  * @param {String} [cfg.type="linear"] Interpolation type - "linear", "cosine", "cubic" or "constant"
301  * @param {String} cfg.input Name of property on {@link SceneJS.Data} scope that will supply the interpolation <em>alpha</em> value
302  * @param {String} cfg.output Name of property to create on child {@link SceneJS.Data} scope that provide the output value
303  * @param {double[]} [cfg.keys=[]] Alpha key values
304  * @param {double[]} [cfg.values=[]] Output key values
305  * @param {...SceneJS.Node} [childNodes] Child nodes
306  * @param {function(SceneJS.Data):Object} [fn] Dynamic configuration function
307  * @returns {SceneJS.objects.Interpolator}
308  * @since Version 0.7.0
309  */
310 SceneJS.interpolator = function() {
311     var n = new SceneJS.Interpolator();
312     SceneJS.Interpolator.prototype.constructor.apply(n, arguments);
313     return n;
314 };
