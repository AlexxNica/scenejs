1 /**
  2  * Backend module that services the SceneJS.Symbol and SceneJS.Instance nodes to manage instancing of scene
  3  * fragments called "symbols".
  4  *  @private
  5  */
  6 var SceneJS_instancingModule = new (function() {
  7 
  8     this._symbols = {};
  9     this._nameStack = [];
 10     this._namePath = null;
 11     var countInstances = 0;
 12 
 13     SceneJS_eventModule.addListener(
 14             SceneJS_eventModule.RESET,
 15             function() {
 16                 this._symbols = {};
 17                 this._nameStack = [];
 18                 this._namePath = null;
 19                 countInstances = 0;
 20             });
 21 
 22     SceneJS_eventModule.addListener(
 23             SceneJS_eventModule.SCENE_RENDERING,
 24             function() {
 25                 this._symbols = {};
 26                 this._nameStack = [];
 27                 this._namePath = null;
 28                 countInstances = 0;
 29             });
 30 
 31     this.clearName = function() {
 32         this._nameStack = [];
 33         this._namePath = null;
 34     };
 35 
 36     this.setName = function(restore) {
 37         this._nameStack = restore.nameStack.slice(0);
 38         this._namePath = restore.namePath;
 39     };
 40 
 41     this.pushName = function(name, node) {
 42         this._nameStack.push(name);
 43         this._namePath = null;
 44     };
 45 
 46     this.getName = function() {
 47         return {
 48             nameStack : this._nameStack.slice(0),
 49             namePath : this._namePath
 50         };
 51     };
 52 
 53     this.createSymbol = function(name, symbol) {
 54         if (!this._namePath) {
 55             this._namePath = this._nameStack.join("/");
 56         }
 57         this._symbols[this._namePath ? this._namePath + "/" + name : name] = symbol;
 58     };
 59 
 60     this.getSymbol = function(name) {
 61         if (!this._namePath) {
 62             this._namePath = this._nameStack.join("/");
 63         }
 64         return this._symbols[getPath(this._namePath, name)];
 65     };
 66 
 67     this.acquireInstance = function(name) {
 68         if (!this._namePath) {
 69             this._namePath = this._nameStack.join("/");
 70         }
 71         var symbol = this._symbols[getPath(this._namePath, name)];
 72         if (symbol) {
 73             countInstances++;
 74         }
 75         return symbol;
 76     };
 77 
 78     this.instancing = function() {
 79         return countInstances > 0;
 80     };
 81 
 82     this.releaseInstance = function() {
 83         countInstances--;
 84     };
 85 
 86     this.popName = function() {
 87         this._nameStack.pop();
 88         this._namePath = null;
 89     };
 90 
 91     /**
 92      * Returns concatenation of base and relative paths
 93      * following simplified UNIX-style format
 94      *
 95      * getPath(null, "../alpha") == "alpha"
 96      * getPath("bla", "../alpha") == "alpha"
 97      * getPath("boo/baa", "../alpha") == "boo/alpha"
 98      * getPath("boo/baa/foo", "../alpha") == "boo/baa/alpha"
 99      * getPath("boo/baa/foo", "../../alpha") == "boo/alpha"
100      * getPath("boo/baa/foo", "/alpha") == "alpha"
101      * getPath("boo/baa/foo", "alpha") == "boo/baa/foo/alpha"
102      *
103      * @private
104      */
105     function getPath(path, name) {
106         if (name.charAt(0) == "/") {      // absolute path begins with "/" - remove it
107             return name.substr(1);
108         } else if (name.substr(0, 3) == "../") {
109             name = name.substr(3);
110             if (path) {
111                 var i = path.lastIndexOf("/");
112                 if (i == 0 || i == -1) {
113                     return name;
114                 }
115                 return getPath(path.substr(0, i), name);
116             } else {
117                 return name;
118             }
119         } else if (path) {
120             return path + "/" + name;
121         } else {
122             return name;
123         }
124     }
125 })();
