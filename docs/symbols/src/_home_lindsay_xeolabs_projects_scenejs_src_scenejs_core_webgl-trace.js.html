100 /**
101  * Map of numbers to names.
102  * @type {Object}
103  */
104 var glEnums = null;
105 
106 /**
107  * Initializes this module. Safe to call more than once.
108  * @param {!WebGLRenderingContext} ctx A WebGL context. If
109  *    you have more than one context it doesn't matter which one
110  *    you pass in, it is only used to pull out constants.
111  */
112 function init(ctx) {
113   if (glEnums == null) {
114     glEnums = { };
115     for (var propertyName in ctx) {
116       if (typeof ctx[propertyName] == 'number') {
117         glEnums[ctx[propertyName]] = propertyName;
118       }
119     }
120   }
121 }
122 
123 /**
124  * Checks the utils have been initialized.
125  */
126 function checkInit() {
127   if (glEnums == null) {
128     throw 'WebGLDebugUtils.init(ctx) not called';
129   }
130 }
131 
132 /**
133  * Returns true or false if value matches any WebGL enum
134  * @param {*} value Value to check if it might be an enum.
135  * @return {boolean} True if value matches one of the WebGL defined enums
136  */
137 function mightBeEnum(value) {
138   checkInit();
139   return (glEnums[value] !== undefined);
140 }
141 
142 /**
143  * Returns true if 'value' matches any WebGL enum, and the i'th parameter
144  * of the WebGL function 'fname' is expected to be (any) enum. Does not
145  * check that 'value' is actually a valid i'th parameter to 'fname', as
146  * that will be checked by the WebGL implementation itself.
147  *
148  * @param {string} fname the GL function to use for screening the enum
149  * @param {integer} i the parameter index to use for screening the enum
150  * @param {any} value the value to check for being a valid i'th parameter to 'fname'
151  * @return {boolean} true if value matches one of the defined WebGL enums,
152  *         and the i'th parameter to 'fname' is expected to be an enum
153  *
154  * @author Tomi Aarnio
155  */
156 function mightBeValidEnum(fname, i, value) {
157        if (!mightBeEnum(value)) return false;
158        return (fname in glValidEnumContexts) && (i in glValidEnumContexts[fname]);
159 }
160 
161 /**
162  * Gets an string version of an WebGL enum.
163  *
164  * Example:
165  *   var str = WebGLDebugUtil.glEnumToString(ctx.getError());
166  *
167  * @param {number} value Value to return an enum for
168  * @return {string} The string version of the enum.
169  */
170 function glEnumToString(value) {
171   checkInit();
172   var name = glEnums[value];
173   return (name !== undefined) ? name :
174       ("*UNKNOWN WebGL ENUM (0x" + value.toString(16) + ")");
175 }

177 /**
178  * Given a WebGL context returns a wrapped context that calls
179  * gl.getError after every command and calls a function if the
180  * result is not gl.NO_ERROR.
181  *
182  * @param {!WebGLRenderingContext} ctx The webgl context to
183  *        wrap.
184  * @param {!function(err, funcName, args): void} opt_onErrorFunc
185  *        The function to call when gl.getError returns an
186  *        error. If not specified the default function calls
187  *        console.log with a message.
188  */
189 function makeDebugContext(ctx, opt_onErrorFunc) {
190   init(ctx);
191   function formatFunctionCall(functionName, args) {
192         // apparently we can't do args.join(",");
193         var argStr = "";
194         for (var ii = 0; ii < args.length; ++ii) {
195           argStr += ((ii == 0) ? '' : ', ') +
196               (mightBeEnum(args[ii]) ? glEnumToString(args[ii]) : args[ii]);
197         }
198         return functionName +  "(" + argStr + ")";
199       };
200 
201   opt_onErrorFunc = opt_onErrorFunc || function(err, functionName, args) {
202         log("WebGL error "+ glEnumToString(err) + " in "+
203             formatFunctionCall(functionName, args));
204       };
205 
206   // Holds booleans for each GL error so after we get the error ourselves
207   // we can still return it to the client app.
208   var glErrorShadow = { };
209 
210   var tracing = false;
211 
212   ctx.setTracing = function (newTracing) {
213       if (!tracing && newTracing) {
214         log('gl.setTracing(' + newTracing + ');');
215       }
216       tracing = newTracing;
217   }
218 
219   var escapeDict = {
220     '\'' : '\\\'',
221     '\"' : '\\\"',
222     '\\' : '\\\\',
223     '\b' : '\\b',
224     '\f' : '\\f',
225     '\n' : '\\n',
226     '\r' : '\\r',
227     '\t' : '\\t'
228   };
229 
230   function quote(s) {
231     var q = '\'';
232     var l = s.length;
233     for (var i = 0; i < l; i++) {
234         var c = s.charAt(i);
235         var d = s.charCodeAt(i);
236         var e = escapeDict[c];
237         if ( e != undefined ) {
238             q += e;
239         } else if ( d < 32 || d >= 128 ) {
240             var h = '000' + d.toString(16);
241             q += '\\u' + h.substring(h.length - 4);
242         } else {
243             q += s.charAt(i);
244         }
245     }
246     q += '\'';
247     return q;
248   }
249 
250   function genSymMaker(name) {
251       var counter = 0;
252       return function() {
253           var sym = name + counter;
254           counter++;
255           return sym;
256       }
257   }
258 
259   var constructorDict = {
260     "createBuffer" : genSymMaker("buffer"),
261     "createFrameBuffer": genSymMaker("frameBuffer"),
262     "createProgram": genSymMaker("program"),
263     "createRenderbuffer": genSymMaker("renderBuffer"),
264     "createShader": genSymMaker("shader"),
265     "createTexture": genSymMaker("texture"),
266     "getUniformLocation": genSymMaker("uniformLocation"),
267     "readPixels": genSymMaker("pixels")
268   };
269 
270   var objectNameProperty = '__webgl_trace_name__';
271 
272   var arrayTypeDict = {
273     "[object WebGLByteArray]" : "WebGLByteArray",
274     "[object WebGLUnsignedByteArray]" : "WebGLUnsignedByteArray",
275     "[object WebGLShortArray]" : "WebGLShortArray",
276     "[object WebGLUnsignedShortArray]" : "WebGLUnsignedShortArray",
277     "[object WebGLIntArray]" : "WebGLIntArray",
278     "[object WebGLUnsignedIntArray]" : "WebGLUnsignedIntArray",
279     "[object WebGLFloatArray]" : "WebGLFloatArray"
280   }
281 
282   function asWebGLArray(a) {
283     var arrayType = arrayTypeDict[a];
284     if (arrayType === undefined) {
285         return undefined;
286     }
287     var buf = 'new ' + arrayType + '( [';
288     for (var i = 0; i < a.length; i++) {
289         if (i > 0 ) {
290             buf += ', ';
291         }
292         buf += a.get(i);
293     }
294     buf += '] )';
295     return buf;
296   };
297 
298   function traceFunctionCall(functionName, args) {
299         var argStr = "";
300         for (var ii = 0; ii < args.length; ++ii) {
301             var arg = args[ii];
302             if ( ii > 0 ) {
303                 argStr += ', ';
304             }
305             var objectName;
306             try {
307             if (arg !== null) {
308                 objectName = arg[objectNameProperty];
309             }
310             } catch (e) {
311                 alert(functionName);
312                 throw e;
313             }
314             var webGLArray = asWebGLArray(arg);
315             if (objectName != undefined ) {
316                 argStr += objectName;
317             } else if (webGLArray != undefined) {
318                 argStr += webGLArray;
319             }else if (typeof(arg) == "string") {
320                 argStr += quote(arg);
321             } else if ( mightBeValidEnum(functionName, ii, arg) ) {
322                 argStr += 'gl.' + glEnumToString(arg);
323             } else {
324                 argStr += arg;
325             }
326         }
327         return "gl." + functionName +  "(" + argStr + ");";
328   };
329 
330   // Makes a function that calls a WebGL function and then calls getError.
331   function makeErrorWrapper(ctx, functionName) {
332     return function() {
333       var resultName;
334       if (tracing) {
335           var prefix = '';
336           // Should we remember the result for later?
337           objectNamer = constructorDict[functionName];
338           if (objectNamer != undefined) {
339               resultName = objectNamer();
340               prefix = 'var ' + resultName + ' = ';
341           }
342           log(prefix + traceFunctionCall(functionName, arguments));
343       }
344 
345       var result = ctx[functionName].apply(ctx, arguments);
346 
347       if (tracing && resultName != undefined) {
348           result[objectNameProperty] = resultName;
349       }
350 
351       var err = ctx.getError();
352       if (err != 0) {
353         glErrorShadow[err] = true;
354         opt_onErrorFunc(err, functionName, arguments);
355       }
356       return result;
357     };
358   }
359 
360   // Make a an object that has a copy of every property of the WebGL context
361   // but wraps all functions.
362   var wrapper = {};
363   for (var propertyName in ctx) {
364     if (typeof ctx[propertyName] == 'function') {
365       wrapper[propertyName] = makeErrorWrapper(ctx, propertyName);
366      } else {
367        wrapper[propertyName] = ctx[propertyName];
368      }
369   }
370 
371   // Override the getError function with one that returns our saved results.
372   wrapper.getError = function() {
373     for (var err in glErrorShadow) {
374       if (glErrorShadow[err]) {
375         glErrorShadow[err] = false;
376         return err;
377       }
378     }
379     return ctx.NO_ERROR;
380   };
381 
382   return wrapper;
383 }
