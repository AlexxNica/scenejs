1 /**
  2  * @class A scene node that inverts the transformations (IE. the model/view matrix) defined by the nodes within its subgraph.
  3  * @extends SceneJS.Node
  4  * <p><b>Example</b></p><p>Inverting the transformation defined by a {@link SceneJS.Matrix) child node:</b></p><pre><code>
  5  * var inverse = new SceneJS.Inverse(
  6  *     new SceneJS.Matrix({
  7  *           elements : [
  8  *                  1, 0, 0, 10,
  9  *                  0, 1, 0, 5,
 10  *                  0, 0, 1, 3,
 11  *                  0, 0, 0, 1
 12  *              ]
 13  *        })
 14  *   })
 15  * </pre></code>
 16  * @constructor
 17  * Create a new SceneJS.Inverse
 18  * @param {Object} config  Config object or function, followed by zero or more child nodes
 19  */
 20 SceneJS.Inverse = function() {
 21     SceneJS.Node.apply(this, arguments);
 22     this._nodeType = "inverse";
 23     this._mat = SceneJS._math_identityMat4();
 24     this._xform = null;
 25 };
 26 
 27 SceneJS._inherit(SceneJS.Inverse, SceneJS.Node);
 28 
 29 SceneJS.Inverse.prototype._render = function(traversalContext, data) {
 30     if (this._memoLevel == 0) {
 31         this._memoLevel = 1; // For consistency with other transform nodes
 32     }
 33     var superXform = SceneJS._modelViewTransformModule.getTransform();
 34     if (this._memoLevel < 2) {
 35         var instancing = SceneJS._instancingModule.instancing();
 36         var tempMat = SceneJS._math_inverseMat4(superXform.matrix, this._mat);
 37         this._xform = {
 38             localMatrix: this._mat,
 39             matrix: tempMat,
 40             fixed: superXform.fixed && this._fixedParams && !instancing
 41         };
 42         if (this._memoLevel == 1 && superXform.fixed && !instancing) {   // Bump up memoization level if model-space fixed
 43             this._memoLevel = 2;
 44         }
 45     }
 46     SceneJS._modelViewTransformModule.setTransform(this._xform);
 47     this._renderNodes(traversalContext, data);
 48     SceneJS._modelViewTransformModule.setTransform(superXform);
 49 };
 50 
 51 /** Factory function that returns a new {@link SceneJS.Inverse} instance
 52  */
 53 SceneJS.inverse = function() {
 54     var n = new SceneJS.Inverse();
 55     SceneJS.Inverse.prototype.constructor.apply(n, arguments);
 56     return n;
 57 };
