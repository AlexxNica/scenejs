 29     /**
 30      * @private
 31      */
 32     _inherit : function(DerivedClassName, BaseClassName) {
 33         DerivedClassName.prototype = new BaseClassName();
 34         DerivedClassName.prototype.constructor = DerivedClassName;
 35     },
 36 
 37     /** Creates a namespace
 38      * @private
 39      */
 40     _namespace : function() {
 41         var a = arguments, o = null, i, j, d, rt;
 42         for (i = 0; i < a.length; ++i) {
 43             d = a[i].split(".");
 44             rt = d[0];
 45             eval('if (typeof ' + rt + ' == "undefined"){' + rt + ' = {};} o = ' + rt + ';');
 46             for (j = 1; j < d.length; ++j) {
 47                 o[d[j]] = o[d[j]] || {};
 48                 o = o[d[j]];
 49             }
 50         }
 51     },
 52 
 53     /**
 54      * Returns a key for a vacant slot in the given map
 55      * @private
 56      */
 57     _createKeyForMap : function(keyMap, prefix) {
 58         var i = 0;
 59         while (true) {
 60             var key = prefix + i++;
 61             if (!keyMap[key]) {
 62                 return key;
 63             }
 64         }
 65     }
 66 };
