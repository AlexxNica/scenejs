124         /**
125          * Used within scene node functions to extract parameters and child nodes from arguments.
126          *
127          * Here are the four ways you can invoke a scene node function, along with the results extracted from the
128          * arguments in each case by this utility:
129          *
130          * 1. No arguments
131          *
132          *      SceneJS.objects.cube()
133          *
134          *      Result:
135          *
136          *          {
137          *              getParams : function() {
138          *                    return {};
139          *              },
140          *              fixed : true,                   // Parameters will never vary because there are none
141          *              children: []
142          *          }
143          *
144          * 2. Child nodes with no parameters object
145          *
146          *      SceneJS.scene(node, node..)
147          *
148          *      Result:
149          *
150          *          {
151          *              getParams : function() {
152          *                    return {};
153          *              },
154          *              fixed : true,                   // Parameters will never vary because there are none
155          *              children: [ node, node ]
156          *          }
157          *
158          * 3. Parameters object followed by zero or more child nodes
159          *
160          *      SceneJS.rotate({ angle: 45 }, node, node..)
161          *
162          *      Result:
163          *
164          *          {
165          *              getParams : function() {
166          *                    return {
167          *                          angle : 45
168          *                    };
169          *              },
170          *              fixed : true,                   // Paramaters will never vary because they are an object
171          *              children: [ node, node ],
172          *          }
173          *
174          * 4. Dynamic parameterisation function followed by zero or more child nodes
175          *
176          *      SceneJS.rotate(function(data) { return data.get("angle"); }, node, node..)
177          *
178          *      Result:
179          *
180          *          {
181          *              getParams : function(data) {    // Same function passed into the node function above
182          *                    return data.get("angle");
183          *              },
184          *              fixed : false,                   // Params may vary because they are provided through a function
185          *              children: [ node, node ]
186          *          }
187          *
188          * @param args
189          */
190         getNodeConfig : function(args) {
191             args = args || [];
192             var result = {
193                 children : [],
194                 fixed: true
195             };
196             for (var i = 0; i < args.length; i++) {
197                 var arg = args[i];
198                 if (arg._render) {
199                     result.children.push(arg);
200                 } else if (i == 0) {
201                     if (arg instanceof Function) {
202                         result.getParams = arg;
203                         result.fixed = false; // Configs generated by function - probably variable
204                     } else {
205                         var config = arg;
206 
207                         /* Config is dynamic if any property is a function
208                          */
209                         result.fixed = true;  // Config object - may be constant - node may override if config has function
210                         for (var key in arg) {
211                             if (result.fixed && arg[key] instanceof Function) {
212                                 result.fixed = false;
213                             }
214                         }
215                         result.getParams = function() {
216                             return config;
217                         };
218 
219                     }
220                 } else {
221                     SceneJS._backends.getBackend("error").fatalError(new SceneJS.exceptions.InvalidNodeConfigException
222                             ("Invalid node parameters - config should be first, IE. node(config, node, node,...)"));
223                 }
224             }
225             if (!result.getParams) {
226                 result.getParams = function() {
227                     return {
228                     };
229                 };
230             }
231             return result;
232         },

380         /**
381          * This function is used when you want to wrap a scene node function invokation it with your own node function,
382          * while adding/overriding members on the parameters argument.
383          *
384          * See SceneJS._utils.getNodeConfigs to see how node arguments are processed.
385          *
386          * Examples
387          * --------
388          *
389          * Wrapping SceneJS.someNode and augmenting the parameters argument with a new member, "myNewParam":
390          *
391          *      SceneJS.myNewNode = function() {
392          *          var extendedArgs = SceneJS._utils.extendNodeArgs(arguments, { myNewParam : "new param value" });
393          *          return SceneJS.someNode.apply(this, extendedArgs);
394          *      };
395          *
396          * If the parameters argument already has a "myNewParam" then this won't replace it, unless you force it's
397          * replacement with an override flag like this:
398          *
399          *      var extendedArgs =  SceneJS._utils.extendNodeArgs(arguments, { myNewParam : "new param value" }, true);
400          *
401          * @param args The JavaScript Arguments passed to the scene node function
402          * @param cfg Each member of these is attached to the config object extracted from the arguments, overriding any member already there
403          * @param override When true, the extending parameters will replace any already in the Argument's parameters - false by default
404          */
405         extendNodeArgs : function(args, cfg, override) {
406             if (!args || args.length == 0) {
407                 return [cfg];
408             }
409             args = Array.prototype.slice.call(args);
410             var arg = args[0];
411             if (arg._render) {
412 
413                 /* First arg is a scene graph node - push override configs in front of it
414                  */
415                 args.unshift(cfg);
416 
417             } else if (arg instanceof Function) {
418 
419                 /* First arg is a dynamic configuration function - wrap it and override members on its result
420                  */
421                 var func = arg;
422                 args[0] = function(data) {
423                     var result = func(data);
424                     for (var key in cfg) {
425                         if (override || !result[key]) {
426                             result[key] = cfg[key];
427                         }
428                     }
429                     return result;
430                 };
431             } else {
432 
433                 /* First arg is a configuration object - extend it
434                  */
435                 for (var key in cfg) {
436                     if (override || !arg[key]) {
437                         arg[key] = cfg[key];
438                     }
439                 }
440             }
441             return args;
442         },

509         /** Registry of modules that provide backend functionality for scene graph nodes
510          */
511             SceneJS._backends = new (function() {
512                 var status = {
513                     error: null
514                 };
515                 var backends = {};
516 
517                 /** Context shared by all backend modules
518                  */
519                 var ctx = {
520 
521                 };
522 
523                 /**
524                  * Installs a backend module. If an error occurrs while installing, it will available
525                  * via getStatus - we allow query for exceptions instead of throwing from this method
526                  * so that we can process them in one place, ie. when we create a scene.
527                  */
528                 this.installBackend = function(id, backend) {
529                     if (!status.error) {
530                         try {
531                             var be = backend(ctx);
532                             if (be) {
533 
534                                 /* Not all backends are registered and available to nodes - some just
535                                  * anonymously set up resources on the backend context, wire up event
536                                  * handlers etc.
537                                  */
538                                 backends[id] = be;
539                             }
540                         } catch (e) {
541                             status.error = new SceneJS.exceptions.NodeBackendInstallFailedException
542                                     ("Failed to install backend module for node type \"" + id + "\": " + (e.message || e), e);
543                             throw status.error;
544                         }
545                     }
546                 };
547 
548                 /** Runs the given function in the context of the currently-installed backends, useful for extending them
549                  */
550                 this.extend = function(func) {
551                     if (!status.error) {
552                         try {
553                             func(ctx);
554                         } catch (e) {
555                             status.error = new SceneJS.exceptions.NodeBackendInstallFailedException
556                                     ("Failed to extend backends", e);
557                             throw status.error;
558                         }
559                     }
560                 };
561 
562                 /** Returns true if a backend of the given type is installed
563                  */
564                 this.hasBackend = function(type) {
565                     var backend = backends[type] ;
566                     return (backend != undefined && backend != null);
567                 };
568 
569 
570                 /** Obtains the backend module of the given type
571                  */
572                 this.getBackend = function(type) {
573                     var backend = backends[type];
574                     if (!backend) {
575                         throw new SceneJS.exceptions.NodeBackendNotFoundException("No backend installed of type \"" + type + "\"");
576                     }
577                     return backend;
578                 };
579 
580                 /** Returns current SceneJS backend error status
581                  */
582                 this.getStatus = function() {
583                     return status;
584                 };
585 
586                 /** Clears current SceneJS backend error status
587                  */
588                 this.clearStatus = function() {
589                     status = {};
590                 };
591 
592                 /** Cleans up all resources currently held by backend modules
593                  */
594                 this.reset = function() {
595                     for (var type in backends) {
596                         var backend = backends[type];
597                         if (backend.reset) {
598                             backend.reset();
599                         }
600                     }
601                 };
602             })();
