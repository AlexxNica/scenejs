1 /** Axis-aligned bounding box
  2  *
  3  */
  4 SceneJS.axisBoundary = function() {
  5     var cfg = SceneJS.utils.getNodeConfig(arguments);
  6 
  7     var errorBackend = SceneJS._backends.getBackend("error");
  8     var backend = SceneJS._backends.getBackend('axis-boundary');
  9     var objectCoords;
 10     var modelCoords;
 11 
 12     return function(scope) {
 13 
 14         if (!modelCoords || !cfg.fixed) {
 15             var params = cfg.getParams(scope);
 16 
 17             /* First version - mandatory manual configuration
 18              */
 19             if (!params.xmin || !params.ymin || !params.zmin || !params.xmax || !params.ymax || !params.zmax) {
 20                errorBackend.fatalError(new SceneJS.exceptions.NodeConfigExpectedException
 21                                     ("SceneJS.axisBoundary propertyr missing: one or more of xmin, ymin, zmin, xmax, ymax or zmax"));
 22             }
 23 
 24             /* Create local object-space coordinates
 25              */
 26             objectCoords = [
 27                 [params.xmin, params.ymin, params.zmin],
 28                 [params.xmax, params.ymin, params.zmin],
 29                 [params.xmax, params.ymax, params.zmin],
 30                 [params.xmin, params.ymax, params.zmin],
 31 
 32                 [params.xmin, params.ymin, params.zmax],
 33                 [params.xmax, params.ymin, params.zmax],
 34                 [params.xmax, params.ymax, params.zmax],
 35                 [params.xmin, params.ymax, params.zmax]
 36             ];
 37         }
 38 
 39      //   if (!modelCoords || !cfg.fixed || !backend.getSafeToCache()) {  // TODO: free the model coords if possible
 40 
 41             /* Transform boundary to model space
 42              */
 43             modelCoords = backend.modelTransform(objectCoords);
 44       //  }
 45 
 46         /* Transform boundary to homogeneous view space
 47          */
 48         var viewCoords = backend.viewAndProjectionTransform(modelCoords);
 49 
 50         var r = backend.testViewVolumeCoordsIntersection(viewCoords);
 51         var x = r;
 52         switch (r) {
 53             case SceneJS_math_INTERSECT_FRUSTUM:
 54 
 55             case SceneJS_math_INSIDE_FRUSTUM:
 56 
 57                 SceneJS.utils.visitChildren(cfg, scope);
 58                     SceneJS.numTeapots++;
 59                 break;
 60 
 61             case SceneJS_math_OUTSIDE_FRUSTUM:
 62   
 63                 break;
 64         }
 65     };
 66 };
