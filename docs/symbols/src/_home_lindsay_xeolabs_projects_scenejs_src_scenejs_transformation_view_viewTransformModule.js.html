1 /**
  2  * Backend that manages the current view transform matrices (view and normal).
  3  *
  4  * Services the scene view transform nodes, such as SceneJS.lookAt, providing them with methods to set and
  5  * get the current view transform matrices.
  6  *
  7  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
  8  * MODEL_TRANSFORM_EXPORTED to pass the view matrix and normal matrix as WebGLFloatArrays to the
  9  * shading backend.
 10  *
 11  * Normal matrix and WebGLFloatArrays are lazy-computed and cached on export to avoid repeatedly regenerating them.
 12  *
 13  * Avoids redundant export of the matrices with a dirty flag; they are only exported when that is set, which occurs
 14  * when transform is set by scene node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and SHADER_DEACTIVATED events.
 15  *
 16  * Whenever a scene node sets the matrix, this backend publishes it with a VIEW_TRANSFORM_UPDATED to allow other
 17  * dependent backends (such as "view-frustum") to synchronise their resources.
 18  *
 19  *  @private
 20  */
 21 var SceneJS_viewTransformModule = new (function() {
 22 
 23     var transform;
 24     var dirty;
 25 
 26     SceneJS_eventModule.onEvent(
 27             SceneJS_eventModule.SCENE_ACTIVATED,
 28             function() {
 29                 transform = {
 30                     matrix : SceneJS_math_identityMat4(),
 31                     fixed: true
 32                 };
 33                 dirty = true;
 34             });
 35 
 36     SceneJS_eventModule.onEvent(
 37             SceneJS_eventModule.SHADER_ACTIVATED,
 38             function() {
 39                 dirty = true;
 40             });
 41 
 42     SceneJS_eventModule.onEvent(
 43             SceneJS_eventModule.SHADER_RENDERING,
 44             function() {
 45                 if (dirty) {
 46 
 47                     if (!transform.matrixAsArray) {
 48                         transform.matrixAsArray = new WebGLFloatArray(transform.matrix);
 49                     }
 50 
 51                     if (!transform.normalMatrixAsArray) {
 52                         transform.normalMatrixAsArray = new WebGLFloatArray(
 53                                 SceneJS_math_transposeMat4(
 54                                         SceneJS_math_inverseMat4(transform.matrix)));
 55                     }
 56 
 57                     SceneJS_eventModule.fireEvent(
 58                             SceneJS_eventModule.VIEW_TRANSFORM_EXPORTED,
 59                             transform);
 60 
 61                     dirty = false;
 62                 }
 63             });
 64 
 65     SceneJS_eventModule.onEvent(
 66             SceneJS_eventModule.SHADER_DEACTIVATED,
 67             function() {
 68                 dirty = true;
 69             });
 70 
 71     this.setTransform = function(t) {
 72         transform = t;
 73         dirty = true;
 74         SceneJS_eventModule.fireEvent(
 75                 SceneJS_eventModule.VIEW_TRANSFORM_UPDATED,
 76                 transform);
 77     };
 78 
 79     this.getTransform = function() {
 80         return transform;
 81     };
 82 
 83 })();
