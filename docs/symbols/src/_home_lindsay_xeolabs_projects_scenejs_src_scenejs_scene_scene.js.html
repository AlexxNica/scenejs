1 /**
  2  @class Root node of a SceneJS scene graph.
  3  <p>This is entry and exit point for execution when rendering one frame of a scene graph.</p>  
  4  @extends SceneJS.Node
  5  */
  6 SceneJS.Scene = function() {
  7     SceneJS.Node.apply(this, arguments);
  8     this._nodeType = "scene";
  9     if (!this._fixedParams) {
 10         SceneJS_errorModule.fatalError(
 11                 new SceneJS.InvalidNodeConfigException
 12                         ("Dynamic configuration of SceneJS.scene node is not supported"));
 13     }
 14     this._params = this._getParams();
 15     this._lastRenderedData = null;
 16     if (this._params.canvasId) {
 17         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS.DEFAULT_CANVAS_ID;
 18     } else {
 19         this._canvasId = SceneJS.DEFAULT_CANVAS_ID;
 20     }
 21 };
 22 
 23 SceneJS._inherit(SceneJS.Scene, SceneJS.Node);

 25 /** Returns the ID of the canvas element that this scene is to bind to. When no canvasId was configured, it will be the
 26  * the default ID of "_scenejs-default-canvas".
 27  */
 28 SceneJS.Scene.prototype.getCanvasId = function() {
 29     return this._canvasId;
 30 };

 32 /**
 33  * Renders the scene, passing in any properties required for dynamic configuration of its contained nodes.
 34  * <p><b>Example:</b></p><p>A Scene with a LookAt node whoe's "eye" property is dynamically configured with a callback. When the Scene is
 35  * rendered, a value for the property is injected into it. The Scene will put the property on a data scope (implemented by a SceneJS.Data)
 36  * that the LookAt's config callback then accesses.</b></p><pre><code>
 37  * var myScene = new SceneJS.Scene(
 38  *          {
 39  *              // .. scene configs .. //
 40  *          },
 41  *
 42  *          new SceneJS.LookAt(
 43  *              function(data) {
 44  *                  return {
 45  *                      eye: data.get("eye")
 46  *                  };
 47  *              },
 48  *
 49  *              // ..more scene nodes..
 50  *      );
 51  *
 52  * myScene.render({
 53  *          eye: {
 54  *             x: 0, y: 0, z: -100
 55  *          }
 56  *      });
 57  *
 58  * </pre></code>
 59  */
 60 SceneJS.Scene.prototype.render = function(paramOverrides) {
 61     if (!this._sceneId) {
 62         this._sceneId = SceneJS_sceneModule.createScene(this, this._getParams());
 63     }
 64     SceneJS_sceneModule.activateScene(this._sceneId);
 65     if (this._params.proxy) {
 66         SceneJS_loadModule.setProxy(this._params.proxy);
 67     }
 68     var traversalContext = {};
 69     this._renderNodes(traversalContext, new SceneJS.Data(null, false, paramOverrides));
 70     SceneJS_loadModule.setProxy(null);
 71     SceneJS_sceneModule.deactivateScene();
 72     this._lastRenderedData = paramOverrides;
 73 };

109 /**
110  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
111  * at least one more time to allow asynchronous processes to complete - since processes are
112  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
113  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
114  * a race condition.
115  */
116 SceneJS.Scene.prototype.getNumProcesses = function() {
117     return (this._sceneId) ? SceneJS_processModule.getNumProcesses(this._sceneId) : 0;
118 };

120 /** Destroys this scene. You should destroy
121  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
122  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
123  * becomes un-destroyed as soon as you render it again.
124  */
125 SceneJS.Scene.prototype.destroy = function() {
126     if (this._sceneId) {
127         SceneJS_sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
128         this._sceneId = null;
129     }
130 };
131 
132 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
133  * when you render it.
134  */
135 SceneJS.Scene.prototype.isActive = function() {
136     return (this._sceneId != null);
137 };

139 /** Returns a new SceneJS.Scene instance
140  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Scene constructor
141  * @returns {SceneJS.Scene}
142  */
143 SceneJS.scene = function() {
144     var n = new SceneJS.Scene();
145     SceneJS.Scene.prototype.constructor.apply(n, arguments);
146     return n;
147 };

149 /** Total SceneJS reset - destroys all scenes and cached resources.
150  */
151 SceneJS.reset = function() {
152     var scenes = SceneJS_sceneModule.getAllScenes();
153     var temp = [];
154     for (var i = 0; i < scenes.length; i++) {
155         temp.push(scenes[i]);
156     }
157     while (temp.length > 0) {
158 
159         /* Destroy each scene individually so it they can mark itself as destroyed.
160          * A RESET command will be fired after the last one is destroyed.
161          */
162         temp.pop().destroy();
163     }
164 };
