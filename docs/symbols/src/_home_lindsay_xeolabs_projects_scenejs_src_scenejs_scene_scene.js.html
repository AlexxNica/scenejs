1 /**
  2  *@class Root node of a SceneJS scene graph.
  3  *
  4  * <p>This is entry and exit point for execution when rendering one frame of a scene graph, which also configures certain
  5  * global scene parameters as listed below. </p>
  6  * <p><b>Binding to a canvas</b></p>
  7  * <p>The Scene node can be configured with a <b>canvasId</b> property to specify the ID of a WebGL compatible Canvas
  8  * element for the scene to render to. When that is omitted, the node will look for one with the default ID of
  9  * "_scenejs-default-canvas".</p>
 10  * <p><b>Timeout for {@link SceneJS.Instance} sub-nodes</b></p>
 11  * <p>The Scene node can be configured with a <b>loadTimeoutSecs</b> property to specify the number of seconds within
 12  * which {@link SceneJS.Instance} nodes within its subgraph must receive and parse their content. That may be overridden
 13  * in the configs of individual {@link SceneJS.Instance} nodes.
 14  * <p><b>Usage Example:</b></p><p>Shown below is Scene bound to a canvas and specifying a JSONP proxy, that contains a
 15  * {@link SceneJS.LookAt} node whose "eye" property is dynamically configured with a callback. A {@link SceneJS.LoadCollada}
 16  * node loads a Collada model cross-domain through the proxy. When the Scene is rendered, a value for the
 17  * {@link Scene.LookAt}'s property is injected into it. The Scene will put the property on a data scope (which is
 18  * implemented by a {@link SceneJS.Data}) that the {@link SceneJS.LookAt}'s config callback then accesses.</b></p>
 19  * <pre><code>
 20  * var myScene = new SceneJS.Scene({
 21  *
 22  *              // Bind scene to render to WebGL Canvas element with given ID.
 23  *              // Default is "_scenejs-default-canvas"
 24  *
 25  *              canvasId:        "myCanvas",
 26  *
 27  *              // Optionally write scene logging to a DIV:
 28  *
 29  *              loggingElementId: "myLoggingDiv",
 30  *
 31  *              // Optional default timeout for SceneJS.Instance<xxx> nodes, which may override
 32  *              // it individually - default is 180 seconds
 33  *
 34  *              loadTimeoutSecs: 180
 35  *          },
 36  *
 37  *          new SceneJS.LookAt(
 38  *              function(data) {
 39  *                  return {
 40  *                      eye: data.get("eye")
 41  *                  };
 42  *              },
 43  *
 44  *              new SceneJS.LoadCollada("http://foo.com/models/myModel.dae")
 45  *      );
 46  *
 47  * myScene.render({
 48  *          eye: {
 49  *             x: 0, y: 0, z: -100
 50  *          }
 51  *      });
 52  *
 53  * </pre></code>
 54  * @extends SceneJS.Node
 55  */
 56 SceneJS.Scene = function() {
 57     SceneJS.Node.apply(this, arguments);
 58     this._nodeType = "scene";
 59     if (!this._fixedParams) {
 60         throw SceneJS_errorModule.fatalError(
 61                 new SceneJS.InvalidNodeConfigException
 62                         ("Dynamic configuration of SceneJS.scene node is not supported"));
 63     }
 64     this._params = this._getParams();
 65     this._lastRenderedData = null;
 66     if (this._params.canvasId) {
 67         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS.Scene.DEFAULT_CANVAS_ID;
 68     } else {
 69         this._canvasId = SceneJS.Scene.DEFAULT_CANVAS_ID;
 70     }
 71 };
 72 
 73 SceneJS._inherit(SceneJS.Scene, SceneJS.Node);

 75 /** ID of canvas SceneJS looks for when {@link SceneJS.Scene} node does not supply one
 76  */
 77 SceneJS.Scene.DEFAULT_CANVAS_ID = "_scenejs-default-canvas";

 79 /** ID ("_scenejs-default-logging") of default element to which {@link SceneJS.Scene} node will log to, if found.
 80  */
 81 SceneJS.Scene.DEFAULT_LOGGING_ELEMENT_ID = "_scenejs-default-logging";

 83 /** Returns the ID of the canvas element that this scene is to bind to. When no canvasId was configured, it will be the
 84  * the default ID of "_scenejs-default-canvas".
 85  */
 86 SceneJS.Scene.prototype.getCanvasId = function() {
 87     return this._canvasId;
 88 };

 90 /**
 91  * Renders the scene, passing in any properties required for dynamic configuration of its contained nodes.
 92  *
 93  */
 94 SceneJS.Scene.prototype.render = function(paramOverrides) {
 95     if (!this._sceneId) {
 96         this._sceneId = SceneJS_sceneModule.createScene(this, this._getParams());
 97     }
 98     SceneJS_sceneModule.activateScene(this._sceneId);
 99     SceneJS_loadModule.setLoadTimeoutSecs(this._params.loadTimeoutSecs);  // null to use default     
100     var traversalContext = {};
101     this._renderNodes(traversalContext, new SceneJS.Data(null, false, paramOverrides));
102     SceneJS_loadModule.setLoadTimeoutSecs(null);
103     SceneJS_sceneModule.deactivateScene();
104     this._lastRenderedData = paramOverrides;
105 };

141 /**
142  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
143  * at least one more time to allow asynchronous processes to complete - since processes are
144  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
145  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
146  * a race condition.
147  */
148 SceneJS.Scene.prototype.getNumProcesses = function() {
149     return (this._sceneId) ? SceneJS_processModule.getNumProcesses(this._sceneId) : 0;
150 };
151 
152 /** Destroys this scene. You should destroy
153  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
154  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
155  * becomes un-destroyed as soon as you render it again.
156  */
157 SceneJS.Scene.prototype.destroy = function() {
158     if (this._sceneId) {
159         SceneJS_sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
160         this._sceneId = null;
161     }
162 };
163 
164 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
165  * when you render it.
166  */
167 SceneJS.Scene.prototype.isActive = function() {
168     return (this._sceneId != null);
169 };

171 /** Factory function that returns a new {@link SceneJS.Scene} instance
172  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Scene constructor
173  * @returns {SceneJS.Scene}
174  */
175 SceneJS.scene = function() {
176     var n = new SceneJS.Scene();
177     SceneJS.Scene.prototype.constructor.apply(n, arguments);
178     return n;
179 };

181 /** Total SceneJS reset - destroys all scenes and cached resources.
182  */
183 SceneJS.reset = function() {
184     var scenes = SceneJS_sceneModule.getAllScenes();
185     var temp = [];
186     for (var i = 0; i < scenes.length; i++) {
187         temp.push(scenes[i]);
188     }
189     while (temp.length > 0) {
190 
191         /* Destroy each scene individually so it they can mark itself as destroyed.
192          * A RESET command will be fired after the last one is destroyed.
193          */
194         temp.pop().destroy();
195     }
196 };
