1 /**
  2  *@class Root node of a SceneJS scene graph.
  3  *
  4  * <p>This is entry and exit point for execution when rendering one frame of a scene graph, while also providing
  5  * the means to configure global data scope values and configurations for each frame. </p>
  6  * <p><b>Binding to a canvas</b></p>
  7  * <p>The Scene node can be configured with a <b>canvasId</b> property to specify the ID of a WebGL compatible Canvas
  8  * element for the scene to render to. When that is omitted, the node will look for one with the default ID of
  9  * "_scenejs-default-canvas".</p>
 10  * <p><b>Usage Example:</b></p><p>Below is a minimal scene graph. To render the scene, SceneJS will traverse its nodes
 11  * in depth-first order. Each node will set some scene state on entry, then un-set it again before exit. In this graph,
 12  * the {@link SceneJS.Scene} node binds to a WebGL Canvas element, a {@link SceneJS.LookAt} defines the viewoint,
 13  * a {@link SceneJS.Camera} defines the projection, a {@link SceneJS.Lights} defines a light source,
 14  * a {@link SceneJS.Material} defines the current material properties, {@link SceneJS.Rotate} nodes orient the modeling
 15  * coordinate space, then a {@link SceneJS.objects.Cube} defines our cube.</p>
 16  * <pre><code>
 17  *
 18  * var myScene = new SceneJS.Scene({
 19  *     canvasId: 'theCanvas'
 20  *   },
 21  *
 22  *   new SceneJS.LookAt({
 23  *       eye  : { x: -1.0, y: 0.0, z: 15 },
 24  *       look : { x: -1.0, y: 0, z: 0 },
 25  *       up   : { y: 1.0 }
 26  *     },
 27  *
 28  *     new SceneJS.Camera({
 29  *         optics: {
 30  *           type: "perspective",
 31  *           fovy   : 55.0,
 32  *           aspect : 1.0,
 33  *           near   : 0.10,
 34  *           far    : 1000.0
 35  *         }
 36  *       },
 37  *
 38  *       new SceneJS.Lights({
 39  *           sources: [
 40  *             {
 41  *               type:  "dir",
 42  *               color: { r: 1.0, g: 1.0, b: 1.0 },
 43  *               dir:   { x: 1.0, y: -1.0, z: 1.0 }
 44  *             },
 45  *             {
 46  *               type:  "dir",
 47  *               color: { r: 1.0, g: 1.0, b: 1.0 },
 48  *               dir:   { x: -1.0, y: -1.0, z: -3.0 }
 49  *             }
 50  *           ]
 51  *         },
 52  *
 53  *         new SceneJS.Material({
 54  *                  baseColor:      { r: 0.9, g: 0.2, b: 0.2 },
 55  *                  specularColor:  { r: 0.9, g: 0.9, b: 0.2 },
 56  *                  emit:           0.0,
 57  *                  specular:       0.9,
 58  *                  shine:          6.0
 59  *             },
 60  *
 61  *             new SceneJS.Rotate(
 62  *                 function(data) {
 63  *                    return {
 64  *                      angle: data.get('yaw'), y : 1.0
 65  *                   };
 66  *                 },
 67  *
 68  *                 new SceneJS.Rotate(
 69  *                     function(data) {
 70  *                       return {
 71  *                         angle: data.get('pitch'), x : 1.0
 72  *                       };
 73  *                     },
 74  *
 75  *                     new SceneJS.objects.Cube()
 76  *                   )
 77  *                )
 78  *              )
 79  *            )
 80  *          )
 81  *       )
 82  *     );
 83  *
 84  *   myScene.setData({ yaw: 315, pitch: 20 });
 85  *   myScene.render();
 86  * </pre></code>
 87  * <p>Take a closer look at those rotate nodes. See how they can optionally take a function which feeds them their
 88  * parameters? You can do that for any node to dynamically evaluate parameters for them at traversal-time. The functions
 89  * take an immutable data object, which is SceneJS's mechanism for passing variables down into scene graphs. Using the
 90  * yaw and pitch properties on that data object, our functions create configurations that specify rotations about
 91  * the X and Y axis. See also how we inject those angles when we render the scene.</p>
 92  * @extends SceneJS.Node
 93  */
 94 SceneJS.Scene = function() {
 95     SceneJS.Node.apply(this, arguments);
 96     this._nodeType = "scene";
 97     if (!this._fixedParams) {
 98         throw SceneJS._errorModule.fatalError(
 99                 new SceneJS.errors.InvalidNodeConfigException
100                         ("Dynamic configuration of SceneJS.scene node is not supported"));
101     }
102     this._params = this._getParams();
103     this._data = {};
104     this._configs = {};
105     this._lastRenderedData = null;
106     if (this._params.canvasId) {
107         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS.Scene.DEFAULT_CANVAS_ID;
108     } else {
109         this._canvasId = SceneJS.Scene.DEFAULT_CANVAS_ID;
110     }
111 };

123 /** Returns the ID of the canvas element that this scene is to bind to. When no canvasId was configured, it will be the
124  * the default ID of "_scenejs-default-canvas".

126 SceneJS.Scene.prototype.getCanvasId = function() {

130 /**
131  * Sets a map of values to set on the global scene data scope. This data will then be available
132  * to any configuration callbacks that are used to configure nodes. The map is the same as that
133  * configured on a {@link SceneJS.WithData} and works the same way. The given values will be forgotten
134  * when the scene is next rendered with {@link #render}.
135  * @param {object} values Values for the global scene data scope, same format as that given to {@link SceneJS.WithData}
136  */
137 SceneJS.Scene.prototype.setData = function(values) {
138     this._data = values || {};
139     return this;
140 };

171 /**
172  * Renders the scene, applying any config and data scope values given to {@link #setData} and {#link setConfigs},
173  * then clearing those values afterwards.
174  */
175 SceneJS.Scene.prototype.render = function() {
176     if (!this._sceneId) {
177         this._sceneId = SceneJS._sceneModule.createScene(this, this._getParams());
178     }
179     SceneJS._sceneModule.activateScene(this._sceneId);
180     var traversalContext = {};
181     this._renderNodes(traversalContext, new SceneJS.Data(null, false, this._data));
182     SceneJS._sceneModule.deactivateScene();
183     this._lastRenderedData = this._data;
184     this._data = {};
185     this._configs = {};
186 };

222 /**
223  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
224  * at least one more time to allow asynchronous processes to complete - since processes are
225  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
226  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
227  * a race condition.
228  */
229 SceneJS.Scene.prototype.getNumProcesses = function() {
230     return (this._sceneId) ? SceneJS._processModule.getNumProcesses(this._sceneId) : 0;
231 };
232 
233 /** Destroys this scene. You should destroy
234  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
235  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
236  * becomes un-destroyed as soon as you render it again.
237  */
238 SceneJS.Scene.prototype.destroy = function() {
239     if (this._sceneId) {
240         SceneJS._sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
241         this._sceneId = null;
242     }
243 };
244 
245 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
246  * when you render it.
247  */
248 SceneJS.Scene.prototype.isActive = function() {
249     return (this._sceneId != null);
250 };

252 /** Factory function that returns a new {@link SceneJS.Scene} instance
253  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Scene constructor
254  * @returns {SceneJS.Scene}
255  */
256 SceneJS.scene = function() {
257     var n = new SceneJS.Scene();
258     SceneJS.Scene.prototype.constructor.apply(n, arguments);
259     return n;
260 };

262 /** Total SceneJS reset - destroys all scenes and cached resources.
263  */
264 SceneJS.reset = function() {
265     var scenes = SceneJS._sceneModule.getAllScenes();
266     var temp = [];
267     for (var i = 0; i < scenes.length; i++) {
268         temp.push(scenes[i]);
269     }
270     while (temp.length > 0) {
271 
272         /* Destroy each scene individually so it they can mark itself as destroyed.
273          * A RESET command will be fired after the last one is destroyed.
274          */
275         temp.pop().destroy();
276     }
277 };
