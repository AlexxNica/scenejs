1 /**
  2  *@class Root node of a SceneJS scene graph.
  3  *
  4  * <p>This is entry and exit point for execution when rendering one frame of a scene graph, which also configures certain
  5  * global scene parameters as listed below. </p>
  6  * <p><b>Binding to a Canvas</b></p>
  7  * <p>The Scene node can be configured with the ID of a WebGL compatible Canvas element for the scene to render to. When that
  8  * is omitted, it will look for one with ID "_scenejs-default-canvas" by default.</p>
  9  * <p><b>JSONP Proxy for Cross-Domain Content</b></p>
 10  * <p>The Scene node can be configured with a <b>proxy</b> property to specify the URL of a SceneJS JSON proxy server to enable
 11  * {@link SceneJS.Load} and {@link SceneJS.LoadCollada} nodes can do their content loading cross-domain. When that property
 12  * is omitted, those nodes can only load their content from the local server domain in order to not violate the browser's
 13  * same-domain security policy. <a target="other" href="http://scenejs.org/library/v0.7/proxies/jsonp_proxy.pl">Here is
 14  * a download of </a>an example of a SceneJS JSONP proxy script written in Perl.</p>
 15  * <p><b>Usage Example:</b></p><p>A Scene bound to a canvas and specifying a JSONP proxy, that contains a LookAt node
 16  * whose "eye" property is dynamically configured with a callback. A {@link SceneJS.LoadCollada} node loads a Collada
 17  * model cross-domain through the proxy. When the Scene is rendered, a value for the {@link Scene.LookAt}'s property is
 18  * injected into it. The Scene will put the property on a data scope (which is implemented by a {@link SceneJS.Data})
 19  * that the {@link SceneJS.LookAt}'s config callback then accesses.</b></p><pre><code>
 20  * var myScene = new SceneJS.Scene({
 21  *              canvasId: "myCanvas",
 22  *              proxy: "http://scenejs.org/cgi-bin/jsonp_proxy.pl"
 23  *          },
 24  *
 25  *          new SceneJS.LookAt(
 26  *              function(data) {
 27  *                  return {
 28  *                      eye: data.get("eye")
 29  *                  };
 30  *              },
 31  *
 32  *              new SceneJS.LoadCollada("http://foo.com/models/myModel.dae")
 33  *      );
 34  *
 35  * myScene.render({
 36  *          eye: {
 37  *             x: 0, y: 0, z: -100
 38  *          }
 39  *      });
 40  *
 41  * </pre></code>
 42  * @extends SceneJS.Node
 43  */
 44 SceneJS.Scene = function() {
 45     SceneJS.Node.apply(this, arguments);
 46     this._nodeType = "scene";
 47     if (!this._fixedParams) {
 48         SceneJS_errorModule.fatalError(
 49                 new SceneJS.InvalidNodeConfigException
 50                         ("Dynamic configuration of SceneJS.scene node is not supported"));
 51     }
 52     this._params = this._getParams();
 53     this._lastRenderedData = null;
 54     if (this._params.canvasId) {
 55         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS.DEFAULT_CANVAS_ID;
 56     } else {
 57         this._canvasId = SceneJS.DEFAULT_CANVAS_ID;
 58     }
 59 };
 60 
 61 SceneJS._inherit(SceneJS.Scene, SceneJS.Node);

 63 /** Returns the ID of the canvas element that this scene is to bind to. When no canvasId was configured, it will be the
 64  * the default ID of "_scenejs-default-canvas".
 65  */
 66 SceneJS.Scene.prototype.getCanvasId = function() {
 67     return this._canvasId;
 68 };

 70 /**
 71  * Renders the scene, passing in any properties required for dynamic configuration of its contained nodes.
 72  *
 73  */
 74 SceneJS.Scene.prototype.render = function(paramOverrides) {
 75     if (!this._sceneId) {
 76         this._sceneId = SceneJS_sceneModule.createScene(this, this._getParams());
 77     }
 78     SceneJS_sceneModule.activateScene(this._sceneId);
 79     if (this._params.proxy) {
 80         SceneJS_loadModule.setProxy(this._params.proxy);
 81     }
 82     var traversalContext = {};
 83     this._renderNodes(traversalContext, new SceneJS.Data(null, false, paramOverrides));
 84     SceneJS_loadModule.setProxy(null);
 85     SceneJS_sceneModule.deactivateScene();
 86     this._lastRenderedData = paramOverrides;
 87 };

123 /**
124  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
125  * at least one more time to allow asynchronous processes to complete - since processes are
126  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
127  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
128  * a race condition.
129  */
130 SceneJS.Scene.prototype.getNumProcesses = function() {
131     return (this._sceneId) ? SceneJS_processModule.getNumProcesses(this._sceneId) : 0;
132 };
133 
134 /** Destroys this scene. You should destroy
135  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
136  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
137  * becomes un-destroyed as soon as you render it again.
138  */
139 SceneJS.Scene.prototype.destroy = function() {
140     if (this._sceneId) {
141         SceneJS_sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
142         this._sceneId = null;
143     }
144 };
145 
146 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
147  * when you render it.
148  */
149 SceneJS.Scene.prototype.isActive = function() {
150     return (this._sceneId != null);
151 };

153 /** Returns a new SceneJS.Scene instance
154  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Scene constructor
155  * @returns {SceneJS.Scene}
156  */
157 SceneJS.scene = function() {
158     var n = new SceneJS.Scene();
159     SceneJS.Scene.prototype.constructor.apply(n, arguments);
160     return n;
161 };

163 /** Total SceneJS reset - destroys all scenes and cached resources.
164  */
165 SceneJS.reset = function() {
166     var scenes = SceneJS_sceneModule.getAllScenes();
167     var temp = [];
168     for (var i = 0; i < scenes.length; i++) {
169         temp.push(scenes[i]);
170     }
171     while (temp.length > 0) {
172 
173         /* Destroy each scene individually so it they can mark itself as destroyed.
174          * A RESET command will be fired after the last one is destroyed.
175          */
176         temp.pop().destroy();
177     }
178 };
