1 /**
  2  *@class Root node of a SceneJS scene graph.
  3  *
  4  * <p>This is entry and exit point for execution when rendering one frame of a scene graph, while also providing
  5  * the means to configure global data scope values and configurations for each frame. </p>
  6  * <p><b>Binding to a canvas</b></p>
  7  * <p>The Scene node can be configured with a <b>canvasId</b> property to specify the ID of a WebGL compatible Canvas
  8  * element for the scene to render to. When that is omitted, the node will look for one with the default ID of
  9  * "_scenejs-default-canvas".</p>
 10  * <p><b>Timeout for {@link SceneJS.Instance} sub-nodes</b></p>
 11  * <p>The Scene node can be configured with a <b>loadTimeoutSecs</b> property to specify the number of seconds within
 12  * which {@link SceneJS.Instance} nodes within its subgraph must receive and parse content when they load it from files.
 13  * That may be overridden in the configs of individual {@link SceneJS.Instance} nodes.
 14  * <p><b>Usage Example:</b></p><p>Shown below is Scene bound to a canvas and specifying a JSONP proxy, that contains a
 15  * {@link SceneJS.LookAt} node whose "eye" property is dynamically configured with a callback. A {@link SceneJS.Instance}
 16  * node loads a Collada model cross-domain through the proxy. When the Scene is rendered, a value for the
 17  * {@link Scene.LookAt}'s property is injected into it. The Scene will put the property on a data scope (which is
 18  * implemented by a {@link SceneJS.Data}) that the {@link SceneJS.LookAt}'s config callback then accesses.</b></p>
 19  * <pre><code>
 20  *
 21  * // To enable the COLLADA content to load cross-domain, we'll first configure SceneJS with a strategy to allow it
 22  * // to use a Web service to proxy the JSONP load request. As shown here, the strategy implements two methods, one to
 23  * // create the request URL for the service, and another to extract the data from the response.
 24  *
 25  * SceneJS.setJSONPStrategy({
 26  *     request : function(url, format, callback) {
 27  *        return "http://scenejs.org/cgi-bin/jsonp_proxy.pl?uri=" + url + "&format=" + format + "&callback=" + callback;
 28  *     },
 29  *
 30  *    response : function(data) {
 31  *
 32  *        // The SceneJS proxy will provide an error message like this when
 33  *        // it fails to service the request
 34  *
 35  *        if (data.error) {
 36  *            throw "Proxy server responded with error: " + data.error;
 37  *        }
 38  *        return data;
 39  *    }
 40  * });
 41  *
 42  * var myScene = new SceneJS.Scene({
 43  *
 44  *         // Bind scene to render to WebGL Canvas element with given ID.
 45  *         // Default is "_scenejs-default-canvas"
 46  *
 47  *         canvasId:        "myCanvas",
 48  *
 49  *         // Optionally write scene logging to a DIV:
 50  *
 51  *        loggingElementId: "myLoggingDiv",
 52  *
 53  *        // Optional default timeout for SceneJS.Instance<xxx> nodes, which may override
 54  *        // it individually - default is 180 seconds
 55  *
 56  *       loadTimeoutSecs: 180
 57  *   },
 58  *
 59  *   new SceneJS.LookAt(
 60  *       function(data) {
 61  *           return {
 62  *              eye: data.get("eye")
 63  *          };
 64  *       },
 65  *
 66  *       new SceneJS.Instance({ uri: "http://foo.com/models/myModel.dae" })
 67  * );
 68  *
 69  * myScene.setData({
 70  *          eye: {
 71  *             x: 0, y: 0, z: -100
 72  *          }
 73  *      }).render();
 74  *
 75  * </pre></code>
 76  * @extends SceneJS.Node
 77  */
 78 SceneJS.Scene = function() {
 79     SceneJS.Node.apply(this, arguments);
 80     this._nodeType = "scene";
 81     if (!this._fixedParams) {
 82         throw SceneJS._errorModule.fatalError(
 83                 new SceneJS.InvalidNodeConfigException
 84                         ("Dynamic configuration of SceneJS.scene node is not supported"));
 85     }
 86     this._params = this._getParams();
 87     this._data = {};
 88     this._configs = {};
 89     this._lastRenderedData = null;
 90     if (this._params.canvasId) {
 91         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS.Scene.DEFAULT_CANVAS_ID;
 92     } else {
 93         this._canvasId = SceneJS.Scene.DEFAULT_CANVAS_ID;
 94     }
 95 };

109  */

114 /**
115  * Sets a map of values to set on the global scene data scope. This data will then be available
116  * to any configuration callbacks that are used to configure nodes. The map is the same as that
117  * configured on a {@link SceneJS.WithData} and works the same way. The given values will be forgotten
118  * when the scene is next rendered with {@link #render}.
119  * @param {object} values Values for the global scene data scope, same format as that given to {@link SceneJS.WithData}
120  */
121 SceneJS.Scene.prototype.setData = function(values) {
122     this._data = values || {};
123     return this;
124 };

155 /**
156  * Renders the scene, applying any config and data scope values given to {@link #setData} and {#link setConfigs},
157  * then clearing those values afterwards.
158  */
159 SceneJS.Scene.prototype.render = function() {
160     if (!this._sceneId) {
161         this._sceneId = SceneJS._sceneModule.createScene(this, this._getParams());
162     }
163     SceneJS._sceneModule.activateScene(this._sceneId);
164     var traversalContext = {};
165     this._renderNodes(traversalContext, new SceneJS.Data(null, false, this._data));
166     SceneJS._sceneModule.deactivateScene();
167     this._lastRenderedData = this._data;
168     this._data = {};
169     this._configs = {};
170 };

206 /**
207  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
208  * at least one more time to allow asynchronous processes to complete - since processes are
209  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
210  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
211  * a race condition.
212  */
213 SceneJS.Scene.prototype.getNumProcesses = function() {
214     return (this._sceneId) ? SceneJS._processModule.getNumProcesses(this._sceneId) : 0;
215 };

217 /** Destroys this scene. You should destroy
218  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
219  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
220  * becomes un-destroyed as soon as you render it again.
221  */
222 SceneJS.Scene.prototype.destroy = function() {
223     if (this._sceneId) {
224         SceneJS._sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
225         this._sceneId = null;
226     }
227 };
228 
229 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
230  * when you render it.
231  */
232 SceneJS.Scene.prototype.isActive = function() {
233     return (this._sceneId != null);
234 };

237  * @param {Arguments} args Variable arguments that are passed to the SceneJS.Scene constructor

240 SceneJS.scene = function() {
241     var n = new SceneJS.Scene();
242     SceneJS.Scene.prototype.constructor.apply(n, arguments);
243     return n;

246 /** Total SceneJS reset - destroys all scenes and cached resources.
247  */
248 SceneJS.reset = function() {
249     var scenes = SceneJS._sceneModule.getAllScenes();
250     var temp = [];
251     for (var i = 0; i < scenes.length; i++) {
252         temp.push(scenes[i]);
253     }
254     while (temp.length > 0) {
255 
256         /* Destroy each scene individually so it they can mark itself as destroyed.
257          * A RESET command will be fired after the last one is destroyed.
258          */
259         temp.pop().destroy();
260     }
261 };
