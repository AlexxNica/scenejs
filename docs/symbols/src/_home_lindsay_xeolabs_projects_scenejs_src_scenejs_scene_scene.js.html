1 /**
  2  @class SceneJS.Scene
  3  @extends SceneJS.Node
  4  <p>Root node of a scene graph</p>
  5  @throws {SceneJS.exceptions.UnsupportedOperationException} If attempt made to configure scene with function
  6  */
  7 SceneJS.Scene = function() {
  8     SceneJS.Node.apply(this, arguments);
  9     this._nodeType = "scene";
 10     if (!this._fixedParams) {
 11         SceneJS_errorModule.fatalError(
 12                 new SceneJS.exceptions.UnsupportedOperationException
 13                         ("Dynamic configuration of SceneJS.scene node is not supported"));
 14     }
 15     this._params = this._getParams();
 16     this._lastRenderedData = null;
 17     if (this._params.canvasId) {
 18         this._canvasId = document.getElementById(this._params.canvasId) ? this._params.canvasId : SceneJS_webgl_DEFAULT_CANVAS_ID;
 19     } else {
 20         this._canvasId = SceneJS_webgl_DEFAULT_CANVAS_ID;
 21     }
 22 };

 24 SceneJS._utils.inherit(SceneJS.Scene, SceneJS.Node);

 26 /** Returns the ID of the canvas element that this scene is to bind to. When no canvasId was configured, it will be the
 27  * the default ID of "_scenejs-default-canvas".
 28  */
 29 SceneJS.Scene.prototype.getCanvasId = function() {
 30     return this._canvasId;

 33 /**
 34  * Renders the scene, passing in the given parameters to override any node parameters
 35  * that were set on the Scene config.
 36  * @private
 37  */
 38 SceneJS.Scene.prototype.render = function(paramOverrides) {
 39     if (!this._sceneId) {
 40         this._sceneId = SceneJS_sceneModule.createScene(this, this._getParams());
 41     }
 42     SceneJS_sceneModule.activateScene(this._sceneId);
 43     var data = SceneJS._utils.newScope(null, false); // TODO: how to determine fixed data for cacheing??
 44     if (paramOverrides) {        // Override with traversal params
 45         for (var key in paramOverrides) {
 46             data.put(key, paramOverrides[key]);
 47         }
 48     }
 49     if (this._params.proxy) {
 50         SceneJS_loadModule.setProxy(this._params.proxy);
 51     }
 52     var traversalContext = {
 53     };
 54     this._renderNodes(traversalContext, data);
 55     SceneJS_loadModule.setProxy(null);
 56     SceneJS_sceneModule.deactivateScene();
 57     this._lastRenderedData = data;
 58 };

 94 /**
 95  * Returns count of active processes. A non-zero count indicates that the scene should be rendered
 96  * at least one more time to allow asynchronous processes to complete - since processes are
 97  * queried like this between renders (ie. in the idle period), to avoid confusion processes are killed
 98  * during renders, not between, in order to ensure that this count doesnt change unexpectedly and create
 99  * a race condition.
100  */
101 SceneJS.Scene.prototype.getNumProcesses = function() {
102     return (this._sceneId) ? SceneJS_processModule.getNumProcesses(this._sceneId) : 0;
103 };

105 /** Destroys this scene. You should destroy
106  * a scene as soon as you are no longer using it, to ensure that SceneJS does not retain
107  * resources for it (eg. shaders, VBOs etc) that are no longer in use. A destroyed scene
108  * becomes un-destroyed as soon as you render it again.
109  */
110 SceneJS.Scene.prototype.destroy = function() {
111     if (this._sceneId) {
112         SceneJS_sceneModule.destroyScene(this._sceneId); // Last one fires RESET command
113         this._sceneId = null;
114     }
115 };
116 
117 /** Returns true if scene active, ie. not destroyed. A destroyed scene becomes active again
118  * when you render it.
119  */
120 SceneJS.Scene.prototype.isActive = function() {
121     return (this._sceneId != null);
122 };

133 /** Total SceneJS reset - destroys all scenes and cached resources.
134  */
135 SceneJS.reset = function() {
136     var scenes = SceneJS_sceneModule.getAllScenes();
137     var temp = [];
138     for (var i = 0; i < scenes.length; i++) {
139         temp.push(scenes[i]);
140     }
141     while (temp.length > 0) {
142 
143         /* Destroy each scene individually so it they can mark itself as destroyed.
144          * A RESET command will be fired after the last one is destroyed.
145          */
146         temp.pop().destroy();
147     }
148 };

150 SceneJS.onEvent = function(name, func) {
151     switch (name) {
152 
153         case "error" : SceneJS_eventModule.onEvent(
154                 SceneJS_eventModule.ERROR,
155                 function(params) {
156                     func({
157                         exception: params.exception,
158                         fatal: params.fatal
159                     });
160                 });
161             break;
162 
163         case "reset" : SceneJS_eventModule.onEvent(
164                 SceneJS_eventModule.RESET,
165                 function() {
166                     func();
167                 });
168             break;
169 
170         case "scene-created" : SceneJS_eventModule.onEvent(
171                 SceneJS_eventModule.SCENE_CREATED,
172                 function(params) {
173                     func({
174                         sceneId : params.sceneId
175                     });
176                 });
177             break;
178 
179         case "scene-activated" : SceneJS_eventModule.onEvent(
180                 SceneJS_eventModule.SCENE_ACTIVATED,
181                 function(params) {
182                     func({
183                         sceneId : params.sceneId
184                     });
185                 });
186             break;
187 
188         case "canvas-activated" : SceneJS_eventModule.onEvent(
189                 SceneJS_eventModule.CANVAS_ACTIVATED,
190                 function(params) {
191                     func({
192                         canvas: params.canvas
193                     });
194                 });
195             break;
196 
197         case "process-created" : SceneJS_eventModule.onEvent(
198                 SceneJS_eventModule.PROCESS_CREATED,
199                 function(params) {
200                     func(params);
201                 });
202             break;
203 
204         case "process-timed-out" : SceneJS_eventModule.onEvent(
205                 SceneJS_eventModule.PROCESS_TIMED_OUT,
206                 function(params) {
207                     func(params);
208                 });
209             break;
210 
211         case "process-killed" : SceneJS_eventModule.onEvent(
212                 SceneJS_eventModule.PROCESS_KILLED,
213                 function(params) {
214                     func(params);
215                 });
216             break;
217 
218         case "scene-deactivated" : SceneJS_eventModule.onEvent(
219                 SceneJS_eventModule.SCENE_DEACTIVATED,
220                 function(params) {
221                     func({
222                         sceneId : params.sceneId
223                     });
224                 });
225             break;
226 
227         case "scene-destroyed" : SceneJS_eventModule.onEvent(
228                 SceneJS_eventModule.SCENE_DESTROYED,
229                 function(params) {
230                     func({
231                         sceneId : params.sceneId
232                     });
233                 });
234             break;
235 
236         default:
237             throw "SceneJS.onEvent - this event type not supported: '" + name + "'";
238     }
239 };
