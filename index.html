Quick Start

First, include the SceneJS library in the <head> tag of your web app: 

<script src="http://xeolabs.github.com/scenejs/build/latest/scenejs.js"></script>

Then build a scene. We'll make a spinning blue teapot:

var scene = SceneJS.createScene({
    nodes:[
        {
            type:"material",
            baseColor: { r: 0.3, g: 0.3, b: 0.1 },

            nodes:[
                {
                    type: "rotate",
                    id: "myRotate",
                    y: 1.0,
                    angle: 0,

                    nodes: [
                        {
                            type:"geometry",
                            source:{
                                type:"teapot"
                            }
                        }
                    ]
                }
            ]
        }
    ]
});

And voil√†, one blue teapot:


Now let's start the teapot spinning:

var angle = 0;
var myRotate = scene.getNode("myRotate");

scene.on("tick",
    function() {
        myRotate.setAngle(angle += 0.5);
    });
